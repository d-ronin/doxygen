177 static void PIOS_UDP_TxStart(uint32_t udp_id, uint16_t tx_bytes_avail)
178 {
179  pios_udp_dev * udp_dev = find_udp_dev_by_id(udp_id);
180 
181  PIOS_Assert(udp_dev);
182 
183  int32_t length,len,rem;
184 
188  if (udp_dev->tx_out_cb) {
189  while (tx_bytes_avail>0) {
190  bool tx_need_yield = false;
191  length = (udp_dev->tx_out_cb)(udp_dev->tx_out_context, udp_dev->tx_buffer, PIOS_UDP_RX_BUFFER_SIZE, NULL, &tx_need_yield);
192  rem = length;
193  while (rem>0) {
194  len = sendto(udp_dev->socket, udp_dev->tx_buffer+length-rem, rem, 0,
195  (struct sockaddr *) &udp_dev->client,
196  sizeof(udp_dev->client));
197  if (len<=0) {
198  rem=0;
199  } else {
200  rem -= len;
201  }
202  }
203  tx_bytes_avail -= length;
204  }
205  }
