151 static int32_t PIOS_BMP280_Common_Init(const struct pios_bmp280_cfg *cfg)
152 {
153  dev->oversampling = cfg->oversampling;
154  dev->cfg = cfg;
155 
156  uint8_t data[24];
157  uint16_t ref_digT1;
158  if (PIOS_BMP280_Read(BMP280_CAL_ADDR, data, 2) != 0)
159  return -2;
160 
161  /* Ignore first result */
162  if (PIOS_BMP280_Read(BMP280_CAL_ADDR, data, 2) != 0)
163  return -2;
164 
165  ref_digT1 = (data[ 1] << 8) | data[ 0];
166 
167  if (PIOS_BMP280_Read(BMP280_CAL_ADDR, data, 24) != 0)
168  return -2;
169 
170  dev->digT1 = (data[ 1] << 8) | data[ 0];
171  dev->digT2 = (data[ 3] << 8) | data[ 2];
172  dev->digT3 = (data[ 5] << 8) | data[ 4];
173  dev->digP1 = (data[ 7] << 8) | data[ 6];
174  dev->digP2 = (data[ 9] << 8) | data[ 8];
175  dev->digP3 = (data[11] << 8) | data[10];
176  dev->digP4 = (data[13] << 8) | data[12];
177  dev->digP5 = (data[15] << 8) | data[14];
178  dev->digP6 = (data[17] << 8) | data[16];
179  dev->digP7 = (data[19] << 8) | data[18];
180  dev->digP8 = (data[21] << 8) | data[20];
181  dev->digP9 = (data[23] << 8) | data[22];
182 
183  if (ref_digT1 != dev->digT1) {
184  /* Values inconsistent, so no good device detect */
185  return -10;
186  }
187 
188  if ((ref_digT1 == dev->digT2) &&
189  (ref_digT1 == dev->digP1)) {
190  /* Values invariant, so no good device detect */
191  return -11;
192  }
193 
194  if (PIOS_BMP280_WriteCommand(BMP280_CONFIG, 0x00) != 0)
195  return -2;
196 
197  PIOS_SENSORS_RegisterCallback(PIOS_SENSOR_BARO, PIOS_BMP280_callback, dev);
198 
199  return 0;
200 }

295 static int32_t PIOS_BMP280_ReadADC()
296 {
297  if (PIOS_BMP280_Validate(dev) != 0)
298  return -1;
299 
300  uint8_t data[6];
301 
302  /* Read and store results */
303 
304  if (PIOS_BMP280_Read(BMP280_PRESS_MSB, data, 6) != 0)
305  return -1;
306 
307  static int32_t T = 0;
308 
309  int32_t raw_temperature = (int32_t)((((uint32_t)(data[3])) << 12) | (((uint32_t)(data[4])) << 4) | ((uint32_t)data[5] >> 4));
310 
311  int32_t varT1, varT2;
312 
313  varT1 = ((((raw_temperature >> 3) - ((int32_t)dev->digT1 << 1))) * ((int32_t)dev->digT2)) >> 11;
314  varT2 = (((((raw_temperature >> 4) - ((int32_t)dev->digT1)) * ((raw_temperature >> 4) - ((int32_t)dev->digT1))) >> 12) * ((int32_t)dev->digT3)) >> 14;
315 
316  /* Filter T ourselves */
317  if (!T) {
318  T = (varT1 + varT2) * 5;
319  } else {
320  T = (varT1 + varT2) + (T * 4) / 5; // IIR Gain=5
321  }
322 
323  dev->compensatedTemperature = T;
324 
325  int32_t raw_pressure = (int32_t)((((uint32_t)(data[0])) << 12) | (((uint32_t)(data[1])) << 4) | ((uint32_t)data[2] >> 4));
326 
327  if (raw_pressure == 0x80000) {
328  return -1;
329  }
330 
331  int64_t varP1, varP2, P;
332 
333  varP1 = ((int64_t)T / 5) - 128000;
334  varP2 = varP1 * varP1 * (int64_t)dev->digP6;
335  varP2 = varP2 + ((varP1 * (int64_t)dev->digP5) << 17);
336  varP2 = varP2 + (((int64_t)dev->digP4) << 35);
337  varP1= ((varP1 * varP1 * (int64_t)dev->digP3) >> 8) + ((varP1 * (int64_t)dev->digP2) << 12);
338  varP1 = (((((int64_t)1) << 47) + varP1)) * ((int64_t)dev->digP1) >> 33;
339  if (varP1 == 0)
340  {
341  return -1; // avoid exception caused by division by zero
342  }
343  P = 1048576 - raw_pressure;
344  P = (((P << 31) - varP2) * 3125) / varP1;
345  varP1 = (((int64_t)dev->digP9) * (P >> 13) * (P >> 13)) >> 25;
346  varP2 = (((int64_t)dev->digP8) * P) >> 19;
347  dev->compensatedPressure = (uint32_t)((P + varP1 + varP2) >> 8) + (((int64_t)dev->digP7) << 4);
348  return 0;
349 }

529 static bool PIOS_BMP280_callback(void *ctx, void *output,
530  int ms_to_wait, int *next_call)
531 {
532  struct bmp280_dev *dev = (struct bmp280_dev *)ctx;
533 
534  PIOS_Assert(dev);
535  PIOS_Assert(output);
536 
537  // Poll a bit faster than sampling rate
538  *next_call = PIOS_BMP280_GetDelay() * 3 / 5;
539 
540  int32_t read_adc_result = 0;
541 
542  PIOS_BMP280_StartADC();
543 
544  PIOS_BMP280_ClaimDevice();
545  read_adc_result = PIOS_BMP280_ReadADC();
546  PIOS_BMP280_ReleaseDevice();
547 
548  if (read_adc_result) {
549  return false;
550  }
551 
552  struct pios_sensor_baro_data *data = (struct pios_sensor_baro_data *)output;
553 
554  // Compute the altitude from the pressure and temperature and send it out
555  data->pressure = ((float) dev->compensatedPressure) / 256.0f / 1000.0f;
556  float calc_alt = 44330.0f * (1.0f - powf(data->pressure / BMP280_P0, (1.0f / 5.255f)));
557  data->temperature = ((float) dev->compensatedTemperature) / 256.0f / 100.0f;
558  data->altitude = calc_alt;
559 
560  return true;
561 }
