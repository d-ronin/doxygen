91 static int32_t PIOS_INTERNAL_ADC_PinConfig(uint32_t internal_adc_id)
92 {
93  struct pios_internal_adc_dev * adc_dev = (struct pios_internal_adc_dev *) internal_adc_id;
94  if(!PIOS_INTERNAL_ADC_validate(adc_dev)) {
95  return -1;
96  }
97 
98  /* Setup analog pins */
99  GPIO_InitTypeDef GPIO_InitStructure;
100  GPIO_StructInit(&GPIO_InitStructure);
101  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz;
102  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AIN;
103 
104  for(int32_t i = 0; i < adc_dev->number_of_used_pins; i++) {
105  if(adc_dev->cfg->adc_pins[i].port == NULL )
106  continue;
107  // This driver only supports ADC1
108  if(!adc_dev->cfg->adc_pins[i].is_master_channel)
109  return -2;
110  GPIO_InitStructure.GPIO_Pin = adc_dev->cfg->adc_pins[i].pin;
111  GPIO_Init(adc_dev->cfg->adc_pins[i].port, (GPIO_InitTypeDef*) &GPIO_InitStructure);
112  }
113 
114  return 0;
115 }
120 static void PIOS_INTERNAL_DMAConfig(uint32_t internal_adc_id)
121 {
122  struct pios_internal_adc_dev * adc_dev = (struct pios_internal_adc_dev *) internal_adc_id;
123  if(!PIOS_INTERNAL_ADC_validate(adc_dev)) {
124  return;
125  }
126 
127  /* Disable interrupts */
128  DMA_ITConfig(adc_dev->cfg->dma.rx.channel, adc_dev->cfg->dma.irq.flags, DISABLE);
129 
130  /* Configure DMA channel */
131  DMA_DeInit(adc_dev->cfg->dma.rx.channel);
132 
133  RCC_AHBPeriphClockCmd(adc_dev->cfg->dma.ahb_clk, ENABLE);
134 
135  DMA_InitTypeDef DMAInit = adc_dev->cfg->dma.rx.init;
136 
137  DMAInit.DMA_PeripheralBaseAddr = (uint32_t) &adc_dev->cfg->adc_dev_master->DR;
138  DMAInit.DMA_MemoryBaseAddr = (uint32_t) adc_dev->raw_data_buffer;
139  DMAInit.DMA_BufferSize = adc_dev->dma_transfer_size;
140  DMAInit.DMA_DIR = DMA_DIR_PeripheralSRC;
141  DMAInit.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
142  DMAInit.DMA_MemoryInc = DMA_MemoryInc_Enable;
143  DMAInit.DMA_PeripheralDataSize = DMA_PeripheralDataSize_HalfWord;
144  DMAInit.DMA_MemoryDataSize = DMA_MemoryDataSize_HalfWord;
145  DMAInit.DMA_Mode = DMA_Mode_Circular;
146  DMAInit.DMA_M2M = DMA_M2M_Disable;
147 
148  DMA_Init(adc_dev->cfg->dma.rx.channel, &DMAInit); /* channel is actually stream ... */
149 
150  /* enable DMA */
151  DMA_Cmd(adc_dev->cfg->dma.rx.channel, ENABLE);
152 }
157 static void PIOS_INTERNAL_ADC_Converter_Config(uint32_t internal_adc_id)
158 {
159  struct pios_internal_adc_dev * adc_dev = (struct pios_internal_adc_dev *) internal_adc_id;
160 
161  ADC_DeInit(adc_dev->cfg->adc_dev_master);
162 
163  if(adc_dev->cfg->adc_dev_master == ADC1)
164  RCC_APB2PeriphClockCmd(RCC_APB2Periph_ADC1, ENABLE);
165  else
166  return;
167 
168  RCC_ADCCLKConfig(RCC_PCLK2_Div8);
169 
170  for(int32_t i = 0; i < adc_dev->number_of_used_pins; i++) {
171  ADC_RegularChannelConfig(ADC1, adc_dev->cfg->adc_pins[i].adc_channel,
172  i + 1, ADC_SampleTime_239Cycles5);
173  }
174 
175  ADC_InitTypeDef ADC_InitStructure;
176  ADC_StructInit(&ADC_InitStructure);
177 
178  ADC_InitStructure.ADC_Mode = ADC_Mode_Independent;
179  ADC_InitStructure.ADC_ScanConvMode = ENABLE;
180  ADC_InitStructure.ADC_ContinuousConvMode = ENABLE;
181  ADC_InitStructure.ADC_ExternalTrigConv = ADC_ExternalTrigConv_None;
182  ADC_InitStructure.ADC_DataAlign = ADC_DataAlign_Right;
183  ADC_InitStructure.ADC_NbrOfChannel = adc_dev->number_of_used_pins;
184 
185  ADC_Init(adc_dev->cfg->adc_dev_master, &ADC_InitStructure);
186 
187  ADC_Cmd(adc_dev->cfg->adc_dev_master, ENABLE);
188 
189  PIOS_DELAY_WaituS(10);
190  ADC_ResetCalibration(adc_dev->cfg->adc_dev_master);
191  while(ADC_GetResetCalibrationStatus(adc_dev->cfg->adc_dev_master));
192  ADC_StartCalibration(adc_dev->cfg->adc_dev_master);
193  while (ADC_GetCalibrationStatus(adc_dev->cfg->adc_dev_master));
194 
195  /* Enable ADC->DMA request */
196  ADC_DMACmd(adc_dev->cfg->adc_dev_master, ENABLE);
197 
198  PIOS_DELAY_WaituS(10);
199  ADC_SoftwareStartConvCmd(adc_dev->cfg->adc_dev_master, ENABLE);
200 }

205 int32_t PIOS_INTERNAL_ADC_LIGHT_Init(uint32_t *internal_adc_id,
206  const struct pios_internal_adc_cfg *cfg,
207  uint16_t number_of_used_pins)
208 {
209  PIOS_DEBUG_Assert(internal_adc_id); PIOS_DEBUG_Assert(cfg);
210 
211  // Only ADC1 is supported by this driver
212  if(cfg->adc_dev_master != ADC1)
213  return -2;
214 
215  struct pios_internal_adc_dev * adc_dev;
216  adc_dev = PIOS_INTERNAL_ADC_Allocate();
217  if(adc_dev == NULL )
218  return -1;
219  adc_dev->cfg = cfg;
220  adc_dev->number_of_used_pins = number_of_used_pins;
221 
222  *internal_adc_id = (uint32_t) adc_dev;
223 
224  // DMA transfer size in units defined by DMA_PeripheralDataSize, 32bits for dual mode and 16bits for single mode
225  adc_dev->dma_transfer_size = adc_dev->number_of_used_pins;
226  adc_dev->raw_data_buffer = PIOS_malloc(adc_dev->dma_transfer_size * sizeof(uint16_t));
227  if (adc_dev->raw_data_buffer == NULL )
228  return -1;
229 
230  if(PIOS_INTERNAL_ADC_PinConfig((uint32_t) adc_dev) != 0)
231  return -3;
232 
233  PIOS_INTERNAL_DMAConfig((uint32_t) adc_dev);
234 
235  PIOS_INTERNAL_ADC_Converter_Config((uint32_t) adc_dev);
236  return 0;
237 }

275 static int32_t PIOS_INTERNAL_ADC_PinGet(uint32_t internal_adc_id, uint32_t pin)
276 {
277  struct pios_internal_adc_dev * adc_dev = (struct pios_internal_adc_dev *) internal_adc_id;
278 
279  /* Check if pin exists */

284  /* This results in a half-word load (2 cycles) of the volatile buffer location
