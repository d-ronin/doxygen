111 void TaskMonitorUpdateAll(void)
112 {
113 #if defined(DIAG_TASKS)
114  TaskInfoData data;
115  int n;
116 
117  // Lock
118  PIOS_Mutex_Lock(lock, PIOS_MUTEX_TIMEOUT_MAX);
119 
120  uint32_t currentTime = 0;
121  uint32_t deltaTime;
122 
123  /*
124  * Calculate the amount of elapsed run time between the last time we
125  * measured and now. Scale so that we can convert task run times
126  * directly to percentages.
127  */
128 #if defined(PIOS_INCLUDE_CHIBIOS)
129  currentTime = hal_lld_get_counter_value();
130 #endif /* defined(PIOS_INCLUDE_CHIBIOS) */
131  deltaTime = ((currentTime - lastMonitorTime) / 100) ? : 1; /* avoid divide-by-zero if the interval is too small */
132  lastMonitorTime = currentTime;
133 
134  // Update all task information
135  for (n = 0; n < TASKINFO_RUNNING_NUMELEM; ++n)
136  {
137  if (handles[n] != 0)
138  {
139  data.Running[n] = TASKINFO_RUNNING_TRUE;
140  data.StackRemaining[n] = PIOS_Thread_Get_Stack_Usage(handles[n]);
141  /* Generate run time stats */
142  data.RunningTime[n] = PIOS_Thread_Get_Runtime(handles[n]) / deltaTime;
143  }
144  else
145  {
146  data.Running[n] = TASKINFO_RUNNING_FALSE;
147  data.StackRemaining[n] = 0;
148  data.RunningTime[n] = 0;
149  }
150  }
151 
152  // Update object
153  TaskInfoSet(&data);
154 
155  // Done
156  PIOS_Mutex_Unlock(lock);
157 #endif
158 }
