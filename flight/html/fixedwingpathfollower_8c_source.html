177 static void pathfollowerTask(void *parameters)
178 {
179  SystemSettingsData systemSettings;
180  FlightStatusData flightStatus;
181 
182  uint32_t lastUpdateTime;
183 
184  AirspeedActualConnectCallback(airspeedActualUpdatedCb);
185  FixedWingPathFollowerSettingsConnectCallback(SettingsUpdatedCb);
186  FixedWingAirspeedsConnectCallback(SettingsUpdatedCb);

197  // Main task loop
198  lastUpdateTime = PIOS_Thread_Systime();
199  while (1) {
200 
201  // Conditions when this runs:
202  // 1. Must have FixedWing type airframe
203  // 2. Flight mode is PositionHold and PathDesired.Mode is Endpoint OR
204  // FlightMode is PathPlanner and PathDesired.Mode is Endpoint or Path
205 
206  SystemSettingsGet(&systemSettings);
207  if ( (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWING) &&
208  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWINGELEVON) &&
209  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWINGVTAIL) )
210  {
211  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_CRITICAL);
212  PIOS_Thread_Sleep(1000);
213  continue;
214  }
215 
216  // Continue collecting data if not enough time
217  PIOS_Thread_Sleep_Until(&lastUpdateTime, fixedwingpathfollowerSettings.UpdatePeriod);
218 
219  static uint8_t last_flight_mode;
220  FlightStatusGet(&flightStatus);
221  PathStatusGet(&pathStatus);
222 
223  PositionActualData positionActual;
224 
225  static enum {FW_FOLLOWER_IDLE, FW_FOLLOWER_RUNNING, FW_FOLLOWER_ERR} state = FW_FOLLOWER_IDLE;
226 
227  // Check whether an update to the path desired occured and we should
228  // process it. This makes sure that the follower alarm state is
229  // updated.
230  bool process_path_desired_update =
231  (last_flight_mode == FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER) &&
232  path_desired_updated;
233  path_desired_updated = false;
234 
235  // Process most of these when the flight mode changes
236  // except when in path following mode in which case
237  // each iteration must make sure this has the latest
238  // PathDesired
239  if (flightStatus.FlightMode != last_flight_mode ||
240  process_path_desired_update) {
241 
242  last_flight_mode = flightStatus.FlightMode;
243 
244  switch(flightStatus.FlightMode) {
245  case FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME:
246  state = FW_FOLLOWER_RUNNING;
247 
248  PositionActualGet(&positionActual);
249 
250  pathDesired.Mode = PATHDESIRED_MODE_CIRCLEPOSITIONRIGHT;
251  pathDesired.Start[0] = positionActual.North;
252  pathDesired.Start[1] = positionActual.East;
253  pathDesired.Start[2] = positionActual.Down;
254  pathDesired.End[0] = 0;
255  pathDesired.End[1] = 0;
256  pathDesired.End[2] = -30.0f;
257  pathDesired.ModeParameters = fixedwingpathfollowerSettings.OrbitRadius;
258  pathDesired.StartingVelocity = fixedWingAirspeeds.CruiseSpeed;
259  pathDesired.EndingVelocity = fixedWingAirspeeds.CruiseSpeed;
260  PathDesiredSet(&pathDesired);
261 
262  break;
263  case FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD:
264  state = FW_FOLLOWER_RUNNING;
265 
266  PositionActualGet(&positionActual);
267 
268  pathDesired.Mode = PATHDESIRED_MODE_CIRCLEPOSITIONRIGHT;
269  pathDesired.Start[0] = positionActual.North;
270  pathDesired.Start[1] = positionActual.East;
271  pathDesired.Start[2] = positionActual.Down;
272  pathDesired.End[0] = positionActual.North;
273  pathDesired.End[1] = positionActual.East;
274  pathDesired.End[2] = positionActual.Down;
275  pathDesired.ModeParameters = fixedwingpathfollowerSettings.OrbitRadius;
276  pathDesired.StartingVelocity = fixedWingAirspeeds.CruiseSpeed;
277  pathDesired.EndingVelocity = fixedWingAirspeeds.CruiseSpeed;
278  PathDesiredSet(&pathDesired);
279 
280  break;
281  case FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER:
282  case FLIGHTSTATUS_FLIGHTMODE_TABLETCONTROL:
283  state = FW_FOLLOWER_RUNNING;
284 
285  PathDesiredGet(&pathDesired);
286  switch(pathDesired.Mode) {
287  case PATHDESIRED_MODE_ENDPOINT:
288  case PATHDESIRED_MODE_VECTOR:
289  case PATHDESIRED_MODE_CIRCLERIGHT:
290  case PATHDESIRED_MODE_CIRCLELEFT:
291  break;
292  default:
293  state = FW_FOLLOWER_ERR;
294  pathStatus.Status = PATHSTATUS_STATUS_CRITICAL;
295  PathStatusSet(&pathStatus);
296  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_CRITICAL);
297  break;
298  }
299  break;
300  default:
301  state = FW_FOLLOWER_IDLE;
302  break;
303  }
304  }
305 
306  switch(state) {
307  case FW_FOLLOWER_RUNNING:
308  {
309  updatePathVelocity();
310  uint8_t result = updateFixedDesiredAttitude();
311  if (result) {
312  AlarmsClear(SYSTEMALARMS_ALARM_PATHFOLLOWER);
313  } else {
314  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_WARNING);
315  }
316  PathStatusSet(&pathStatus);
317  break;
318  }
319  case FW_FOLLOWER_IDLE:
320  // Be cleaner and get rid of global variables
321  northVelIntegral = 0;
322  eastVelIntegral = 0;
323  downVelIntegral = 0;
324  bearingIntegral = 0;
325  speedIntegral = 0;
326  accelIntegral = 0;
327  powerIntegral = 0;
328  airspeedErrorInt = 0;
329  AlarmsClear(SYSTEMALARMS_ALARM_PATHFOLLOWER);
330  break;
331  case FW_FOLLOWER_ERR:
332  default:
333  // Leave alarms set above
334  break;
335  }
336  }
337 }

345 static void updatePathVelocity()
346 {
347  PositionActualData positionActual;
348  PositionActualGet(&positionActual);
349  VelocityActualData velocityActual;
350  VelocityActualGet(&velocityActual);
351 
352  float cur[3] = {
353  positionActual.North +
354  velocityActual.North * fixedwingpathfollowerSettings.PositionFeedforward,
355  positionActual.East +
356  velocityActual.East * fixedwingpathfollowerSettings.PositionFeedforward,
357  positionActual.Down
358  };
359  struct path_status progress;
360 
361  path_progress(&pathDesired, cur, &progress);
362 
363  float groundspeed = 0;
364  float altitudeSetpoint = 0;
365  switch (pathDesired.Mode) {
366  case PATHDESIRED_MODE_CIRCLERIGHT:
367  case PATHDESIRED_MODE_CIRCLELEFT:
368  groundspeed = pathDesired.EndingVelocity;
369  altitudeSetpoint = pathDesired.End[2];
370  break;
371  case PATHDESIRED_MODE_ENDPOINT:
372  case PATHDESIRED_MODE_VECTOR:
373  default:
374  groundspeed = pathDesired.StartingVelocity + (pathDesired.EndingVelocity - pathDesired.StartingVelocity) *
375  bound_min_max(progress.fractional_progress,0,1);
376  altitudeSetpoint = pathDesired.Start[2] + (pathDesired.End[2] - pathDesired.Start[2]) *
377  bound_min_max(progress.fractional_progress,0,1);
378  break;
379  }
380  // this ensures a significant forward component at least close to the real trajectory
381  if (groundspeed<fixedWingAirspeeds.BestClimbRateSpeed/10.0f)
382  groundspeed=fixedWingAirspeeds.BestClimbRateSpeed/10.0f;
383 
384  // calculate velocity - can be zero if waypoints are too close
385  VelocityDesiredData velocityDesired;
386  velocityDesired.North = progress.path_direction[0] * groundspeed;
387  velocityDesired.East = progress.path_direction[1] * groundspeed;
388 
389  float error_speed = progress.error * fixedwingpathfollowerSettings.HorizontalPosP;
390 
391  // calculate correction - can also be zero if correction vector is 0 or no error present
392  velocityDesired.North += progress.correction_direction[0] * error_speed;
393  velocityDesired.East += progress.correction_direction[1] * error_speed;
394 
395  float downError = altitudeSetpoint - positionActual.Down;
396  velocityDesired.Down = downError * fixedwingpathfollowerSettings.VerticalPosP;
397 
398  // update pathstatus
399  pathStatus.error = progress.error;
400  pathStatus.fractional_progress = progress.fractional_progress;
401 
402  pathStatus.fractional_progress = progress.fractional_progress;
403  if (pathStatus.fractional_progress < 1)
404  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
405  else
406  pathStatus.Status = PATHSTATUS_STATUS_COMPLETED;
407 
408  pathStatus.Waypoint = pathDesired.Waypoint;
409 
410  VelocityDesiredSet(&velocityDesired);
411 }

420 static uint8_t updateFixedDesiredAttitude()
421 {
422 
423  uint8_t result = 1;
424 
425  float dT = fixedwingpathfollowerSettings.UpdatePeriod / 1000.0f; //Convert from [ms] to [s]
426 
427  VelocityDesiredData velocityDesired;
428  VelocityActualData velocityActual;
429  StabilizationDesiredData stabDesired;
430  AttitudeActualData attitudeActual;
431  FixedWingPathFollowerStatusData fixedwingpathfollowerStatus;
432  AirspeedActualData airspeedActual;
433 
434  float groundspeedActual;
435  float groundspeedDesired;
436  float indicatedAirspeedActual;
437  float indicatedAirspeedDesired;
438  float airspeedError;
439 
440  float pitchCommand;
441 
442  float descentspeedDesired;
443  float descentspeedError;
444  float powerError;
445  float powerCommand;
446 
447  float bearingError;
448  float bearingCommand;
449 
450  FixedWingPathFollowerStatusGet(&fixedwingpathfollowerStatus);
451 
452  VelocityActualGet(&velocityActual);
453  StabilizationDesiredGet(&stabDesired);
454  VelocityDesiredGet(&velocityDesired);
455  AttitudeActualGet(&attitudeActual);
456  AirspeedActualGet(&airspeedActual);
457 
458 
463  // Current ground speed
464  groundspeedActual = sqrtf( velocityActual.East*velocityActual.East + velocityActual.North*velocityActual.North );
465  // note that airspeedActualBias is ( calibratedAirspeed - groundSpeed ) at the time of measurement,
466  // but thanks to accelerometers, groundspeed reacts faster to changes in direction
467  // than airspeed and gps sensors alone
468  indicatedAirspeedActual = groundspeedActual + indicatedAirspeedActualBias;
469 
470  // Desired ground speed
471  groundspeedDesired = sqrtf(velocityDesired.North*velocityDesired.North + velocityDesired.East*velocityDesired.East);
472  indicatedAirspeedDesired = bound_min_max( groundspeedDesired + indicatedAirspeedActualBias,
473  fixedWingAirspeeds.BestClimbRateSpeed,
474  fixedWingAirspeeds.CruiseSpeed);
475 
476  // Airspeed error (positive means not enough airspeed)
477  airspeedError = indicatedAirspeedDesired - indicatedAirspeedActual;
478 
479  // Vertical speed error
480  descentspeedDesired = bound_min_max (
481  velocityDesired.Down,
482  -fixedWingAirspeeds.VerticalVelMax,
483  fixedWingAirspeeds.VerticalVelMax);
484  descentspeedError = descentspeedDesired - velocityActual.Down;
485 
486  // Error condition: wind speed is higher than maximum allowed speed. We are forced backwards!
487  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_WIND] = 0;
488  if (groundspeedDesired - indicatedAirspeedActualBias <= 0 ) {
489  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_WIND] = 1;
490  result = 0;
491  }
492  // Error condition: plane too slow or too fast
493  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_HIGHSPEED] = 0;
494  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWSPEED] = 0;
495  if ( indicatedAirspeedActual > fixedWingAirspeeds.AirSpeedMax) {
496  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_OVERSPEED] = 1;
497  result = 0;
498  }
499  if ( indicatedAirspeedActual > fixedWingAirspeeds.CruiseSpeed * 1.2f) {
500  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_HIGHSPEED] = 1;
501  result = 0;
502  }
503  if (indicatedAirspeedActual < fixedWingAirspeeds.BestClimbRateSpeed * 0.8f && 1) { //The next three && 1 are placeholders for UAVOs representing LANDING and TAKEOFF
504  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWSPEED] = 1;
505  result = 0;
506  }
507  if (indicatedAirspeedActual < fixedWingAirspeeds.StallSpeedClean && 1 && 1) { //Where the && 1 represents the UAVO that will control whether the airplane is prepped for landing or not
508  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_STALLSPEED] = 1;
509  result = 0;
510  }
511  if (indicatedAirspeedActual < fixedWingAirspeeds.StallSpeedDirty && 1) {
512  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_STALLSPEED] = 1;
513  result = 0;
514  }
515 
516  if (indicatedAirspeedActual<1e-6f) {
517  // prevent division by zero, abort without controlling anything. This guidance mode is not suited for takeoff or touchdown, or handling stationary planes
518  // also we cannot handle planes flying backwards, lets just wait until the nose drops
519  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWSPEED] = 1;
520  return 0;
521  }
522 
528  // compute proportional throttle response
529  powerError = -descentspeedError +
530  bound_min_max(
531  (airspeedError / fixedWingAirspeeds.BestClimbRateSpeed)* fixedwingpathfollowerSettings.AirspeedToVerticalCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_AIRSPEEDTOVERTICALCROSSFEED_KP] ,
532  -fixedwingpathfollowerSettings.AirspeedToVerticalCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_AIRSPEEDTOVERTICALCROSSFEED_MAX],
533  fixedwingpathfollowerSettings.AirspeedToVerticalCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_AIRSPEEDTOVERTICALCROSSFEED_MAX]
534  );
535 
536  // compute saturated integral error throttle response. Make integral leaky for better performance. Approximately 30s time constant.
537  if (fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KI] >0) {
538  powerIntegral = bound_min_max(powerIntegral + -descentspeedError * dT,
539  -fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_ILIMIT]/fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KI],
540  fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_ILIMIT]/fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KI]
541  )*(1.0f-1.0f/(1.0f+30.0f/dT));
542  } else
543  powerIntegral = 0;
544 
545  // Compute final throttle response
546  powerCommand = bound_min_max(
547  (powerError * fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KP] +
548  powerIntegral* fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KI]) + fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_NEUTRAL],
549  fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_MIN],
550  fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_MAX]);
551 
552  //Output internal state to telemetry
553  fixedwingpathfollowerStatus.Error[FIXEDWINGPATHFOLLOWERSTATUS_ERROR_POWER] = powerError;
554  fixedwingpathfollowerStatus.ErrorInt[FIXEDWINGPATHFOLLOWERSTATUS_ERRORINT_POWER] = powerIntegral;
555  fixedwingpathfollowerStatus.Command[FIXEDWINGPATHFOLLOWERSTATUS_COMMAND_POWER] = powerCommand;
556 
557  // set throttle
558  stabDesired.Thrust = powerCommand;
559 
560  // Error condition: plane cannot hold altitude at current speed.
561  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWPOWER] = 0;
562  if (
563  powerCommand == fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_MAX] // throttle at maximum
564  && velocityActual.Down > 0 // we ARE going down
565  && descentspeedDesired < 0 // we WANT to go up
566  && airspeedError > 0 // we are too slow already
567  )
568  {
569  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWPOWER] = 1;
570  result = 0;
571  }
572  // Error condition: plane keeps climbing despite minimum throttle (opposite of above)
573  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_HIGHPOWER] = 0;
574  if (
575  powerCommand == fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_MIN] // throttle at minimum
576  && velocityActual.Down < 0 // we ARE going up
577  && descentspeedDesired > 0 // we WANT to go down
578  && airspeedError < 0 // we are too fast already
579  )
580  {
581  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_HIGHPOWER] = 1;
582  result = 0;
583  }
584 
585 
590  if (fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KI] > 0){
591  //Integrate with saturation
592  airspeedErrorInt=bound_min_max(airspeedErrorInt + airspeedError * dT,
593  -fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_ILIMIT]/fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KI],
594  fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_ILIMIT]/fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KI]);
595  } else
596  airspeedErrorInt = 0;
597 
598  //Compute the cross feed from vertical speed to pitch, with saturation
599  float verticalSpeedToPitchCommandComponent=bound_min_max (-descentspeedError * fixedwingpathfollowerSettings.VerticalToPitchCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_VERTICALTOPITCHCROSSFEED_KP],
600  -fixedwingpathfollowerSettings.VerticalToPitchCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_VERTICALTOPITCHCROSSFEED_MAX],
601  fixedwingpathfollowerSettings.VerticalToPitchCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_VERTICALTOPITCHCROSSFEED_MAX]
602  );
603 
604  //Compute the pitch command as err*Kp + errInt*Ki + X_feed.
605  pitchCommand= -(airspeedError*fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KP]
606  + airspeedErrorInt*fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KI]
607  ) + verticalSpeedToPitchCommandComponent;
608 
609  fixedwingpathfollowerStatus.Error[FIXEDWINGPATHFOLLOWERSTATUS_ERROR_SPEED] = airspeedError;
610  fixedwingpathfollowerStatus.ErrorInt[FIXEDWINGPATHFOLLOWERSTATUS_ERRORINT_SPEED] = airspeedErrorInt;
611  fixedwingpathfollowerStatus.Command[FIXEDWINGPATHFOLLOWERSTATUS_COMMAND_SPEED] = pitchCommand;

617 
618  // Error condition: high speed dive
619  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_PITCHCONTROL] = 0;
620  if (
621  pitchCommand == fixedwingpathfollowerSettings.PitchLimit[FIXEDWINGPATHFOLLOWERSETTINGS_PITCHLIMIT_MAX] // pitch demand is full up
622  && velocityActual.Down > 0 // we ARE going down
623  && descentspeedDesired < 0 // we WANT to go up
624  && airspeedError < 0 // we are too fast already
625  ) {
626  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_PITCHCONTROL] = 1;
627  result = 0;
628  }
629 
633  if (groundspeedDesired> 1e-6f) {

635  } else {

637  bearingError = 0;
638  }

641  if (bearingError>180.0f) bearingError-=360.0f;

644  -fixedwingpathfollowerSettings.BearingPI[FIXEDWINGPATHFOLLOWERSETTINGS_BEARINGPI_ILIMIT],
645  fixedwingpathfollowerSettings.BearingPI[FIXEDWINGPATHFOLLOWERSETTINGS_BEARINGPI_ILIMIT]);
646  bearingCommand = (bearingError * fixedwingpathfollowerSettings.BearingPI[FIXEDWINGPATHFOLLOWERSETTINGS_BEARINGPI_KP] +
647  bearingIntegral);
648 
649  fixedwingpathfollowerStatus.Error[FIXEDWINGPATHFOLLOWERSTATUS_ERROR_BEARING] = bearingError;
650  fixedwingpathfollowerStatus.ErrorInt[FIXEDWINGPATHFOLLOWERSTATUS_ERRORINT_BEARING] = bearingIntegral;
651  fixedwingpathfollowerStatus.Command[FIXEDWINGPATHFOLLOWERSTATUS_COMMAND_BEARING] = bearingCommand;
652 
653  stabDesired.Roll = bound_min_max( fixedwingpathfollowerSettings.RollLimit[FIXEDWINGPATHFOLLOWERSETTINGS_ROLLLIMIT_NEUTRAL] +
654  bearingCommand,
655  fixedwingpathfollowerSettings.RollLimit[FIXEDWINGPATHFOLLOWERSETTINGS_ROLLLIMIT_MIN],
656  fixedwingpathfollowerSettings.RollLimit[FIXEDWINGPATHFOLLOWERSETTINGS_ROLLLIMIT_MAX] );
657 
658  // TODO: find a check to determine loss of directional control. Likely needs some check of derivative
659 
660 
664  // TODO implement raw control mode for yaw and base on Accels.Y
665  stabDesired.Yaw = 0;
666 
667  stabDesired.ReprojectionMode = STABILIZATIONDESIRED_REPROJECTIONMODE_NONE;
668 
669  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
670  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
671  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_MANUAL;
672 
673  if (isnan(stabDesired.Roll) || isnan(stabDesired.Pitch) || isnan(stabDesired.Yaw) || isnan(stabDesired.Thrust)) {
674  northVelIntegral = 0;
675  eastVelIntegral = 0;
676  downVelIntegral = 0;
677  bearingIntegral = 0;
678  speedIntegral = 0;
679  accelIntegral = 0;
680  powerIntegral = 0;
681  airspeedErrorInt = 0;
682 
683  result = 0;
684  } else {
685  StabilizationDesiredSet(&stabDesired);
686  }
687 
688  FixedWingPathFollowerStatusSet(&fixedwingpathfollowerStatus);
689 
690  return result;
