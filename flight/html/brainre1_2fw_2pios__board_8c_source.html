77 static void settingdUpdatedCb(UAVObjEvent * ev, void *ctx, void *obj, int len)
78 {
79  (void) ev; (void) ctx; (void) len;
80 
81  HwBrainRE1Data * hwre1data = (HwBrainRE1Data *)obj;
82 
83  /* Configure the IR emitter for lap timing */
84  uint8_t ir_data[6];
85 
86  switch (hwre1data->IRProtocol) {
87  case HWBRAINRE1_IRPROTOCOL_DISABLED:
88  PIOS_RE1FPGA_SetIRProtocol(PIOS_RE1FPGA_IR_PROTOCOL_OFF);
89  break;
90  case HWBRAINRE1_IRPROTOCOL_ILAP:
91  pios_ir_generate_ilap_packet(hwre1data->IRIDILap, ir_data, 6);
92  PIOS_RE1FPGA_SetIRData(ir_data, 6);
93  PIOS_RE1FPGA_SetIRProtocol(PIOS_RE1FPGA_IR_PROTOCOL_ILAP);
94  break;
95  case HWBRAINRE1_IRPROTOCOL_TRACKMATE:
96  pios_ir_generate_trackmate_packet(hwre1data->IRIDTrackmate, ir_data, 4);
97  PIOS_RE1FPGA_SetIRData(ir_data, 4);
98  PIOS_RE1FPGA_SetIRProtocol(PIOS_RE1FPGA_IR_PROTOCOL_TRACKMATE);
99  break;
100  }
101 
102  /* Configure the buzzer type */
103  if (hwre1data->BuzzerType == HWBRAINRE1_BUZZERTYPE_4KHZ_BUZZER) {
104  PIOS_RE1FPGA_SetBuzzerType(PIOS_RE1FPGA_BUZZER_AC);
105  }
106  else if (hwre1data->BuzzerType == HWBRAINRE1_BUZZERTYPE_DC_BUZZER) {
107  PIOS_RE1FPGA_SetBuzzerType(PIOS_RE1FPGA_BUZZER_DC);
108  }
109 
110  /* Set video sync detector threshold */
111  PIOS_RE1FPGA_SetSyncThreshold(hwre1data->VideoSyncDetectorThreshold);
112 }
