161 #define PIOS_SYSCLK 168000000
162 // Peripherals that belongs to APB1 are:
163 // DAC |PWR |CAN1,2
164 // I2C1,2,3 |UART4,5 |USART3,2
165 // I2S3Ext |SPI3/I2S3 |SPI2/I2S2
166 // I2S2Ext |IWDG |WWDG
167 // RTC/BKP reg
168 // TIM2,3,4,5,6,7,12,13,14
169 
170 // Calculated as SYSCLK / APBPresc * (APBPre == 1 ? 1 : 2)
171 // Default APB1 Prescaler = 4
172 #define PIOS_PERIPHERAL_APB1_CLOCK (PIOS_SYSCLK / 4)
173 
174 // Peripherals belonging to APB2
175 // SDIO |EXTI |SYSCFG |SPI1
176 // ADC1,2,3
177 // USART1,6
178 // TIM1,8,9,10,11
179 //
180 // Default APB2 Prescaler = 2
181 //
