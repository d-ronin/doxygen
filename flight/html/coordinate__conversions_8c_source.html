61 // ****** find roll, pitch, yaw from quaternion ********
62 void Quaternion2RPY(const float q[4], float rpy[3])
63 {
64  float R13, R11, R12, R23, R33;
65  float q0s = q[0] * q[0];
66  float q1s = q[1] * q[1];
67  float q2s = q[2] * q[2];
68  float q3s = q[3] * q[3];
69 
70  R13 = 2.0f * (q[1] * q[3] - q[0] * q[2]);
71  R11 = q0s + q1s - q2s - q3s;
72  R12 = 2.0f * (q[1] * q[2] + q[0] * q[3]);
73  R23 = 2.0f * (q[2] * q[3] + q[0] * q[1]);
74  R33 = q0s - q1s - q2s + q3s;
75 
76  rpy[1] = RAD2DEG * asinf(-R13); // pitch always between -pi/2 to pi/2
77  rpy[2] = RAD2DEG * atan2f(R12, R11);
78  rpy[0] = RAD2DEG * atan2f(R23, R33);
79 
80  //TODO: consider the cases where |R13| ~= 1, |pitch| ~= pi/2
81 }

83 // ****** find quaternion from roll, pitch, yaw ********
84 void RPY2Quaternion(const float rpy[3], float q[4])
85 {
86  float phi, theta, psi;
87  float cphi, sphi, ctheta, stheta, cpsi, spsi;
88 
89  phi = DEG2RAD * rpy[0] / 2;
90  theta = DEG2RAD * rpy[1] / 2;
91  psi = DEG2RAD * rpy[2] / 2;
92  cphi = cosf(phi);
93  sphi = sinf(phi);
94  ctheta = cosf(theta);
95  stheta = sinf(theta);
96  cpsi = cosf(psi);
97  spsi = sinf(psi);
98 
99  q[0] = cphi * ctheta * cpsi + sphi * stheta * spsi;
100  q[1] = sphi * ctheta * cpsi - cphi * stheta * spsi;
101  q[2] = cphi * stheta * cpsi + sphi * ctheta * spsi;
102  q[3] = cphi * ctheta * spsi - sphi * stheta * cpsi;
103 
104  if (q[0] < 0) { // q0 always positive for uniqueness
105  q[0] = -q[0];
106  q[1] = -q[1];
107  q[2] = -q[2];
108  q[3] = -q[3];
109  }
110 }

112 //** Find Rbe, that rotates a vector from earth fixed to body frame, from quaternion **
113 void Quaternion2R(float q[4], float Rbe[3][3])
114 {
115 
116  float q0s = q[0] * q[0], q1s = q[1] * q[1], q2s = q[2] * q[2], q3s = q[3] * q[3];
117 
118  Rbe[0][0] = q0s + q1s - q2s - q3s;
119  Rbe[0][1] = 2 * (q[1] * q[2] + q[0] * q[3]);
120  Rbe[0][2] = 2 * (q[1] * q[3] - q[0] * q[2]);
121  Rbe[1][0] = 2 * (q[1] * q[2] - q[0] * q[3]);
122  Rbe[1][1] = q0s - q1s + q2s - q3s;
123  Rbe[1][2] = 2 * (q[2] * q[3] + q[0] * q[1]);
124  Rbe[2][0] = 2 * (q[1] * q[3] + q[0] * q[2]);
125  Rbe[2][1] = 2 * (q[2] * q[3] - q[0] * q[1]);
126  Rbe[2][2] = q0s - q1s - q2s + q3s;
127 }

129 //** Find Rbe, that rotates a vector from earth fixed to body frame, from euler angles, using Tait-Bryan
130 //** convention, i.e. Rbe=Rot_x(roll)*Rot_y(pitch)*Rot_z(yaw) **
131 void Euler2R(float rpy[3], float Rbe[3][3])
132 {
133 
134  float sF = sinf(rpy[0]), cF = cosf(rpy[0]);
135  float sT = sinf(rpy[1]), cT = cosf(rpy[1]);
136  float sP = sinf(rpy[2]), cP = cosf(rpy[2]);
137 
138  Rbe[0][0] = cT*cP;
139  Rbe[0][1] = cT*sP;
140  Rbe[0][2] = -sT;
141  Rbe[1][0] = sF*sT*cP - cF*sP;
142  Rbe[1][1] = sF*sT*sP + cF*cP;
143  Rbe[1][2] = cT*sF;
144  Rbe[2][0] = cF*sT*cP + sF*sP;
145  Rbe[2][1] = cF*sT*sP - sF*cP;
146  Rbe[2][2] = cT*cF;
147 }

149 // ****** convert Rotation Matrix to Quaternion ********
150 // ****** if R converts from e to b, q is rotation from e to b ****
151 void R2Quaternion(float R[3][3], float q[4])
152 {
153  float m[4], mag;
154  uint8_t index,i;
155 
156  m[0] = 1 + R[0][0] + R[1][1] + R[2][2];
157  m[1] = 1 + R[0][0] - R[1][1] - R[2][2];
158  m[2] = 1 - R[0][0] + R[1][1] - R[2][2];
159  m[3] = 1 - R[0][0] - R[1][1] + R[2][2];
160 
161  // find maximum divisor
162  index = 0;
163  mag = m[0];
164  for (i=1;i<4;i++){
165  if (m[i] > mag){
166  mag = m[i];
167  index = i;
168  }
169  }
170  mag = 2*sqrtf(mag);
171 
172  if (index == 0) {
173  q[0] = mag/4;
174  q[1] = (R[1][2]-R[2][1])/mag;
175  q[2] = (R[2][0]-R[0][2])/mag;
176  q[3] = (R[0][1]-R[1][0])/mag;
177  }
178  else if (index == 1) {
179  q[1] = mag/4;
180  q[0] = (R[1][2]-R[2][1])/mag;
181  q[2] = (R[0][1]+R[1][0])/mag;
182  q[3] = (R[0][2]+R[2][0])/mag;
183  }
184  else if (index == 2) {
185  q[2] = mag/4;
186  q[0] = (R[2][0]-R[0][2])/mag;
187  q[1] = (R[0][1]+R[1][0])/mag;
188  q[3] = (R[1][2]+R[2][1])/mag;
189  }
190  else {
191  q[3] = mag/4;
192  q[0] = (R[0][1]-R[1][0])/mag;
193  q[1] = (R[0][2]+R[2][0])/mag;
194  q[2] = (R[1][2]+R[2][1])/mag;
195  }
196 
197  // q0 positive, i.e. angle between pi and -pi
198  if (q[0] < 0){
199  q[0] = -q[0];
200  q[1] = -q[1];
201  q[2] = -q[2];
202  q[3] = -q[3];
203  }
204 }

206 // ****** Rotation Matrix from Two Vector Directions ********
207 // ****** given two vector directions (v1 and v2) known in two frames (b and e) find Rbe ***
208 // ****** solution is approximate if can't be exact ***
209 uint8_t RotFrom2Vectors(const float v1b[3], const float v1e[3], const float v2b[3], const float v2e[3], float Rbe[3][3])
210 {
211  float Rib[3][3], Rie[3][3];
212  float mag;
213  uint8_t i,j,k;
214 
215  // identity rotation in case of error
216  for (i=0;i<3;i++){
217  for (j=0;j<3;j++)
218  Rbe[i][j]=0;
219  Rbe[i][i]=1;
220  }
221 
222  // The first rows of rot matrices chosen in direction of v1
223  mag = VectorMagnitude(v1b);
224  if (fabsf(mag) < 1e-30f)
225  return (-1);
226  for (i=0;i<3;i++)
227  Rib[0][i]=v1b[i]/mag;
228 
229  mag = VectorMagnitude(v1e);
230  if (fabsf(mag) < 1e-30f)
231  return (-1);
232  for (i=0;i<3;i++)
233  Rie[0][i]=v1e[i]/mag;
234 
235  // The second rows of rot matrices chosen in direction of v1xv2
236  CrossProduct(v1b,v2b,&Rib[1][0]);
237  mag = VectorMagnitude(&Rib[1][0]);
238  if (fabsf(mag) < 1e-30f)
239  return (-1);
240  for (i=0;i<3;i++)
241  Rib[1][i]=Rib[1][i]/mag;
242 
243  CrossProduct(v1e,v2e,&Rie[1][0]);
244  mag = VectorMagnitude(&Rie[1][0]);
245  if (fabsf(mag) < 1e-30f)
246  return (-1);
247  for (i=0;i<3;i++)
248  Rie[1][i]=Rie[1][i]/mag;
249 
250  // The third rows of rot matrices are XxY (Row1xRow2)
251  CrossProduct(&Rib[0][0],&Rib[1][0],&Rib[2][0]);
252  CrossProduct(&Rie[0][0],&Rie[1][0],&Rie[2][0]);
253 
254  // Rbe = Rbi*Rie = Rib'*Rie
255  for (i=0;i<3;i++)
256  for(j=0;j<3;j++){
257  Rbe[i][j]=0;
258  for(k=0;k<3;k++)
259  Rbe[i][j] += Rib[k][i]*Rie[k][j];
260  }
261 
262  return 1;
263 }

265 void Rv2Rot(float Rv[3], float R[3][3])
266 {
267  // Compute rotation matrix from a rotation vector
268  // To save .text space, uses Quaternion2R()
269  float q[4];
270 
271  float angle = VectorMagnitude(Rv);
272  if (angle <= 0.00048828125f) {
273  // angle < sqrt(2*machine_epsilon(float)), so flush cos(x) to 1.0f
274  q[0] = 1.0f;
275 
276  // and flush sin(x/2)/x to 0.5
277  q[1] = 0.5f*Rv[0];
278  q[2] = 0.5f*Rv[1];
279  q[3] = 0.5f*Rv[2];
280  // This prevents division by zero, while retaining full accuracy
281  }
282  else {
283  q[0] = cosf(angle*0.5f);
284  float scale = sinf(angle*0.5f) / angle;
285  q[1] = scale*Rv[0];
286  q[2] = scale*Rv[1];
287  q[3] = scale*Rv[2];
288  }
289 
290  Quaternion2R(q, R);
291 }
