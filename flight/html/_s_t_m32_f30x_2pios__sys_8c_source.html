204 int32_t PIOS_SYS_SerialNumberGet(char *str)
205 {
206  int i;
207 
208  /* Stored in the so called "electronic signature" */
209  for (i = 0; i < PIOS_SYS_SERIAL_NUM_ASCII_LEN; ++i) {
210  uint8_t b = MEM8(0x1ffff7ac + (i / 2));
211  if (!(i & 1))
212  b >>= 4;
213  b &= 0x0f;
214 
215  str[i] = ((b > 9) ? ('A' - 10) : '0') + b;
216  }
217  str[i] = '\0';
218 
219  /* No error */
220  return 0;
221 }
