77 static void exampleTask(void *parameters)
78 {
79  ExampleSettingsData settings;
80  ExampleObject2Data data;
81  int32_t step;
82  uint32_t lastSysTime;
83 
84  // Main task loop
85  lastSysTime = PIOS_Thread_Systime();
86  while (1) {
87  // Update settings with latest value
88  ExampleSettingsGet(&settings);
89 
90  // Get the object data
91  ExampleObject2Get(&data);
92 
93  // Determine how to update the data
94  if (settings.StepDirection == EXAMPLESETTINGS_STEPDIRECTION_UP) {
95  step = settings.StepSize;
96  } else {
97  step = -settings.StepSize;
98  }
99 
100  // Update the data
101  data.Field1 += step;
102  data.Field2 += step;
103  data.Field3 += step;
104  data.Field4[0] += step;
105  data.Field4[1] += step;
106 
107  // Update the ExampleObject, after this function is called
108  // notifications to any other modules listening to that object
109  // will be sent and the GCS object will be updated through the
110  // telemetry link. All operations will take place asynchronously
111  // and the following call will return immediately.
112  ExampleObject2Set(&data);
113 
114  // Since this module executes at fixed time intervals, we need to
115  // block the task until it is time for the next update.
116  PIOS_Thread_Sleep_Until(&lastSysTime, settings.UpdatePeriod);
117  }
118 }
