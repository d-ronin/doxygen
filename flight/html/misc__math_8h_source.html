88 static inline int16_t sin_approx(int32_t x)
89 {
90 #define s_qN 13
91 #define s_qP 15
92 #define s_qA 12
93  static const int
94  qR= 2*s_qN - s_qP,
95  qS= s_qN + s_qP + 1 - s_qA;
96 
97  x= x<<(30-s_qN); // shift to full s32 range (Q13->Q30)
98 
99  if( (x^(x<<1)) < 0) // test for quadrant 1 or 2
100  x= (1<<31) - x;
101 
102  x= x>>(30-s_qN);
103 
104  return x * ( (3<<s_qP) - (x*x>>qR) ) >> qS;
105 }

119 static inline void matrix_mul(const float *a, const float *b,
120  float *out, int arows, int acolsbrows, int bcols)
121 {
122  const float *restrict apos = a;
123  float *restrict opos = out;
124 
125  for (int ar = 0; ar < arows; ar++) {
126  for (int bc = 0 ; bc < bcols; bc++) {
127  float sum = 0;
128 
129  const float *restrict bpos = b + bc;
130 
131  int acbr;
132 
133  /* This is manually unrolled. -Os doesn't really want to
134  * unroll things, but this is a case where we want to
135  * encourage it to.
136  *
137  * This structure was chosen / experimented with across
138  * many compilers to always do the right things-- namely
139  *
140  * A) vectorize when appropriate on the architecture
141  * B) eliminate the loop when there's a static, small number
142  * of cols (10 or less)
143  * C) generate pretty good code when the dimensions aren't
144  * known.
145  */
146  for (acbr = 0; acbr < (acolsbrows & (~3)); acbr += 4) {
147  /*
148  sum += a[ar * acolsbrows + acbr] *
149  b[acbr * bcols + bc];
150  */
151 
152  sum += (apos[acbr]) * (*bpos);
153  bpos += bcols;
154  sum += (apos[acbr+1]) * (*bpos);
155  bpos += bcols;
156  sum += (apos[acbr+2]) * (*bpos);
157  bpos += bcols;
158  sum += (apos[acbr+3]) * (*bpos);
159  bpos += bcols;
160  }
161 
162  for (; acbr < acolsbrows; acbr++) {
163  /*
164  sum += a[ar * acolsbrows + acbr] *
165  b[acbr * bcols + bc];
166  */
167 
168  sum += (apos[acbr]) * (*bpos);
169  bpos += bcols;
170  }
171  /* out[ar * bcols + bc] = sum; */
172 
173  *opos = sum;
174 
175  opos++;
176  }
177 
178  apos += acolsbrows;
179  }
180 }

276  * a and b look like good inverses.
277  */
278 static inline bool matrix_pseudoinv_convergecheck(const float *a,
279  const float *b, const float *prod, int rows, int cols)
280 {
281  const float *pos = prod;
282  bool ret = true;
283 
284  /* Product (desired pseudo identity matrix) is cols*cols */
285  for (int i = 0; i < cols; i++) {
286  float sum = 0.0f;
287 
288  for (int j = 0; j < cols; j++) {
289  sum += *pos;
290 
291  pos++;
292  }
293 
294  /* It needs to be near 1 or 0 for us to be converged.
295  * First, return false if outside 0..1
296  */
297  if (sum > (1 + TOL_EPS)) {
298  ret = false;
299  break;
300  }
301 
302  if (sum < (0 - TOL_EPS)) {
303  ret = false;
304  break;
305  }
306 
307  /* Next, continue if very close to 1 or 0 */
308  if (sum > (1 - TOL_EPS)) {
309  continue;
310  }
311 
312  if (sum < (0 + TOL_EPS)) {
313  continue;
314  }
315 
316  /* We're between 0 and 1, outside of the tolerance */
317  ret = false;
318  break;
319  }
320 
321  if (ret) {
322  return true;
323  }
324 
325  /* Sometimes the pseudoidentity matrix doesn't meet this constraint; fall
326  * back to verifying elements.
327  * (Which in itself is numerically problematic)
328  */
329 
330  int elems = rows * cols;
331 
332  for (int i = 0; i < elems; i++) {
333  float diff = (a[i] - b[i]);
334 
335  /* This strange structure chosen because NaNs will return false
336  * to both of these...
337  */
338  if (diff < TOL_EPS) {
339  if (diff > -TOL_EPS) {
340  continue;
341  }
342  }
343 
344  return false;
345  }
346 
347  return true;
348 }

350 /* Not intended for outside use. Performs one step of the pseudoinverse
351  * approximation by calculating 2 * ainv - ainv * a * ainv */
352 static inline bool matrix_pseudoinv_step(const float *a, float *ainv,
353  int arows, int acols)
354 {
355  float prod[acols * acols];
356  float invcheck[acols * arows];
357 
358  /* Calculate 2 * ainv - ainv * a * ainv */
359 
360  /* prod = ainv * a */
361  matrix_mul(ainv, a, prod, acols, arows, acols);
362 
363  /* invcheck = prod * ainv */
364  matrix_mul(prod, ainv, invcheck, acols, acols, arows);
365 
366  if (matrix_pseudoinv_convergecheck(ainv, invcheck, prod, arows, acols)) {
367  return true;
368  }
369 
370  /* ainv_ = 2 * ainv */
371  matrix_mul_scalar(ainv, 2, ainv, acols, arows);
372 
373  /* ainv__ = ainv_ - invcheck */
374  /* AKA expanded ainv__ = 2 * ainv - ainv * a * ainv */
375  matrix_sub(ainv, invcheck, ainv, acols, arows);
376 
377  return false;
378 }

421 static inline bool matrix_pseudoinv(const float *a, float *out,
422  int arows, int acols)
423 {
424  matrix_transpose(a, out, arows, acols);
425 
426  float scale = matrix_getmaxabs(a, arows, acols);
427 
428  matrix_mul_scalar(out, 0.01f / scale, out, acols, arows);
429 
430  for (int i = 0; i < PSEUDOINV_CONVERGE_LIMIT; i++) {
431  if (matrix_pseudoinv_step(a, out, arows, acols)) {
432  /* Do one more step when we look pretty good! */
433  matrix_pseudoinv_step(a, out, arows, acols);
434 
435  return true;
436  }
437  }
438 
439  return false;
440 }
