81 // The well known 3rd order expansion of sine; 2^15 units/circle
82 // Based on http://www.coranac.com/2009/07/sines/
83 // Output is Q12
84 static inline int16_t sin_approx(int32_t x)
85 {
86 #define s_qN 13
87 #define s_qP 15
88 #define s_qA 12
89  static const int
90  qR= 2*s_qN - s_qP,
91  qS= s_qN + s_qP + 1 - s_qA;
92 
93  x= x<<(30-s_qN); // shift to full s32 range (Q13->Q30)
94 
95  if( (x^(x<<1)) < 0) // test for quadrant 1 or 2
96  x= (1<<31) - x;
97 
98  x= x>>(30-s_qN);
99 
100  return x * ( (3<<s_qP) - (x*x>>qR) ) >> qS;
101 }

152 static inline float
153 fastlog2 (float x)
154 {
155  union { float f; uint32_t i; } vx = { x };
156  union { uint32_t i; float f; } mx = { (vx.i & 0x007FFFFF) | (0x7e << 23) };
157  float y = vx.i;
158  y *= 1.0f / (1 << 23);
159 
160  return y - 124.22551499f
161  - 1.498030302f * mx.f
162  - 1.72587999f / (0.3520887068f + mx.f);
163 }
