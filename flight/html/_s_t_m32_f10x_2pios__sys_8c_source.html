133 int32_t PIOS_SYS_SerialNumberGet(char *str)
134 {
135  int i;
136 
137  /* Stored in the so called "electronic signature" */
138  for (i = 0; i < PIOS_SYS_SERIAL_NUM_ASCII_LEN; ++i) {
139  uint8_t b = MEM8(0x1ffff7e8 + (i / 2));
140  if (!(i & 1))
141  b >>= 4;
142  b &= 0x0f;
143 
144  str[i] = ((b > 9) ? ('A' - 10) : '0') + b;
145  }
146  str[i] = '\0';
147 
148  /* No error */
149  return 0;
