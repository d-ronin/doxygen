112 static struct mpu9250_dev *PIOS_MPU9250_alloc(const struct pios_mpu9250_cfg *cfg)
113 {
114  struct mpu9250_dev *mpu9250_dev;
115 
116  mpu9250_dev = (struct mpu9250_dev *)PIOS_malloc(sizeof(*mpu9250_dev));
117  if (!mpu9250_dev)
118  return NULL;
119 
120  mpu9250_dev->magic = PIOS_MPU9250_DEV_MAGIC;
121 
122  mpu9250_dev->accel_queue = PIOS_Queue_Create(PIOS_MPU9250_MAX_DOWNSAMPLE, sizeof(struct pios_sensor_accel_data));
123  if (mpu9250_dev->accel_queue == NULL) {
124  PIOS_free(mpu9250_dev);
125  return NULL;
126  }
127 
128  mpu9250_dev->gyro_queue = PIOS_Queue_Create(PIOS_MPU9250_MAX_DOWNSAMPLE, sizeof(struct pios_sensor_gyro_data));
129  if (mpu9250_dev->gyro_queue == NULL) {
130  PIOS_Queue_Delete(dev->accel_queue);
131  PIOS_free(mpu9250_dev);
132  return NULL;
133  }
134 
135  if (cfg->use_magnetometer) {
136  mpu9250_dev->mag_queue = PIOS_Queue_Create(PIOS_MPU9250_MAX_DOWNSAMPLE, sizeof(struct pios_sensor_mag_data));
137  if (mpu9250_dev->mag_queue == NULL) {
138  PIOS_Queue_Delete(dev->accel_queue);
139  PIOS_Queue_Delete(dev->gyro_queue);
140  PIOS_free(mpu9250_dev);
141  return NULL;
142  }
143  }
144 
145  mpu9250_dev->data_ready_sema = PIOS_Semaphore_Create();
146  if (mpu9250_dev->data_ready_sema == NULL) {
147  PIOS_Queue_Delete(dev->accel_queue);
148  PIOS_Queue_Delete(dev->gyro_queue);
149  if (cfg->use_magnetometer)
150  PIOS_Queue_Delete(dev->mag_queue);
151  PIOS_free(mpu9250_dev);
152  return NULL;
153  }
154 
155  return mpu9250_dev;

258 static int32_t PIOS_MPU9250_Mag_WriteReg(uint8_t reg, uint8_t data)
259 {
260  // we will use I2C SLV4 to manipulate with AK8963 control registers
261  if (PIOS_MPU9250_WriteReg(PIOS_MPU60X0_SLV4_REG_REG, reg) != 0)
262  return -1;
263  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_SLV4_ADDR_REG, PIOS_MPU9250_AK8963_ADDR);
264  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_SLV4_DO_REG, data);
265  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_SLV4_CTRL_REG, PIOS_MPU60X0_I2CSLV_EN);
266  uint32_t timeout = 0;
267 
268  // wait for I2C transaction done, use simple safety
269  // escape counter to prevent endless loop in case
270  // MPU9250 is broken
271  uint8_t status = 0;
272  do {
273  if (timeout++ > 50)
274  return -2;
275 
276  status = PIOS_MPU9250_ReadReg(PIOS_MPU60X0_I2C_MST_STATUS_REG);
277  } while ((status & PIOS_MPU60X0_I2C_MST_SLV4_DONE) == 0);
278 
279  if (status & PIOS_MPU60X0_I2C_MST_SLV4_NACK)
280  return -3;
281 
282  return 0;
283 }

317 static int32_t PIOS_MPU9250_Mag_Config(void)
318 {
319  uint8_t id = PIOS_MPU9250_Mag_ReadReg(AK8963_WHOAMI_REG);
320  if (id != AK8963_WHOAMI_ID)
321  return -2;
322 
323  // reset AK8963
324  if (PIOS_MPU9250_Mag_WriteReg(AK8963_CNTL2_REG, AK8963_CNTL2_SRST) != 0)
325  return -3;
326 
327  // give chip some time to initialize
328  PIOS_DELAY_WaitmS(2);
329 
330  // set magnetometer sampling rate to 100Hz and 16-bit resolution
331  PIOS_MPU9250_Mag_WriteReg(AK8963_CNTL1_REG, AK8963_MODE_CONTINUOUS_FAST_16B);
332 
333  // configure mpu9250 to read ak8963 data range from STATUS1 to STATUS2 at ODR
334  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_SLV0_REG_REG, AK8963_ST1_REG);
335  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_SLV0_ADDR_REG, PIOS_MPU9250_AK8963_ADDR | 0x80);
336  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_SLV0_CTRL_REG, PIOS_MPU60X0_I2CSLV_EN | 8);
337 
338  return 0;
339 }

347 static int32_t PIOS_MPU9250_Config(struct pios_mpu9250_cfg const *cfg)
348 {
349  // reset chip
350  if (PIOS_MPU9250_WriteReg(PIOS_MPU60X0_PWR_MGMT_REG, PIOS_MPU60X0_PWRMGMT_IMU_RST) != 0)
351  return -1;
352 
353  // give chip some time to initialize
354  PIOS_DELAY_WaitmS(50);
355 
356  uint8_t id = PIOS_MPU9250_ReadReg(PIOS_MPU60X0_WHOAMI);
357  switch (id) {
358  case MPU9250_WHOAMI_ID:
359  break;
360  case MPU6500_WHOAMI_ID:
361  case ICM20608G_WHOAMI_ID:
362  if (dev->cfg->use_magnetometer)
363  return -2;
364  break;
365  default:
366  return -3;
367  }
368 
369  // power management config
370  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_PWR_MGMT_REG, PIOS_MPU60X0_PWRMGMT_PLL_X_CLK);
371 
372  // user control
373  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_USER_CTRL_REG, PIOS_MPU60X0_USERCTL_DIS_I2C | PIOS_MPU60X0_USERCTL_I2C_MST_EN);
374 
375  if (dev->cfg->use_magnetometer)
376  if (PIOS_MPU9250_Mag_Config() != 0)
377  return -4;
378 
379  // Digital low-pass filter and scale
380  // set this before sample rate else sample rate calculation will fail
381  PIOS_MPU9250_SetAccelLPF(cfg->default_accel_filter);
382  PIOS_MPU9250_SetGyroLPF(cfg->default_gyro_filter);
383 
384  // Sample rate
385  if (PIOS_MPU9250_SetSampleRate(cfg->default_samplerate) != 0)
386  return -5;
387 
388  // Set the gyro scale
389  PIOS_MPU9250_SetGyroRange(PIOS_MPU60X0_SCALE_500_DEG);
390 
391  // Set the accel scale
392  PIOS_MPU9250_SetAccelRange(PIOS_MPU60X0_ACCEL_8G);
393 
394  // Interrupt configuration
395  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_INT_CFG_REG, cfg->interrupt_cfg);
396 
397  // Interrupt enable
398  PIOS_MPU9250_WriteReg(PIOS_MPU60X0_INT_EN_REG, PIOS_MPU60X0_INTEN_DATA_RDY);
399 
400  return 0;
401 }

407 int32_t PIOS_MPU9250_SPI_Init(uint32_t spi_id, uint32_t slave_num, const struct pios_mpu9250_cfg *cfg)
408 {
409  dev = PIOS_MPU9250_alloc(cfg);
410  if (dev == NULL)
411  return -1;
412 
413  dev->spi_id = spi_id;
414  dev->slave_num = slave_num;
415  dev->cfg = cfg;
416 
417  /* Configure the MPU9250 Sensor */
418  if (PIOS_MPU9250_Config(cfg) != 0)
419  return -2;
420 
421  /* Set up EXTI line */
422  PIOS_EXTI_Init(cfg->exti_cfg);
423 
424  // Wait 20 ms for data ready interrupt and make sure it happens
425  // twice
426  if ((PIOS_Semaphore_Take(dev->data_ready_sema, 20) != true) ||
427  (PIOS_Semaphore_Take(dev->data_ready_sema, 20) != true)) {
428  return -10;
429  }
430 
431  dev->TaskHandle = PIOS_Thread_Create(
432  PIOS_MPU9250_Task, "pios_mpu9250", MPU9250_TASK_STACK_BYTES, NULL, MPU9250_TASK_PRIORITY);
433  PIOS_Assert(dev->TaskHandle != NULL);
434 
435  PIOS_SENSORS_Register(PIOS_SENSOR_ACCEL, dev->accel_queue);
436  PIOS_SENSORS_Register(PIOS_SENSOR_GYRO, dev->gyro_queue);
437 
438  if (dev->cfg->use_magnetometer)
439  PIOS_SENSORS_Register(PIOS_SENSOR_MAG, dev->mag_queue);
440 
441  return 0;
442 }

511 int32_t PIOS_MPU9250_SetSampleRate(uint16_t samplerate_hz)
512 {
513  // mpu9250 ODR divider is unable to run from 8kHz clock like mpu60x0 :(
514  // check if someone want to use 250Hz DLPF and don't want 8kHz sampling
515  // and politely refuse him
516  if ((dev->gyro_filter == PIOS_MPU9250_GYRO_LOWPASS_250_HZ) && (samplerate_hz != 8000))
517  return -1;
518 
519  uint16_t filter_frequency = 1000;
520 
521  // limit samplerate to filter frequency
522  if (samplerate_hz > filter_frequency)
523  samplerate_hz = filter_frequency;
524 
525  // calculate divisor, round to nearest integeter
526  int32_t divisor = (int32_t)(((float)filter_frequency / samplerate_hz) + 0.5f) - 1;
527 
528  // limit resulting divisor to register value range
529  if (divisor < 0)
530  divisor = 0;
531 
532  if (divisor > 0xff)
533  divisor = 0xff;
534 
535  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_ACCEL, samplerate_hz);
536  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_GYRO, samplerate_hz);
537 
538  return PIOS_MPU9250_WriteReg(PIOS_MPU60X0_SMPLRT_DIV_REG, (uint8_t)divisor);
539 }
