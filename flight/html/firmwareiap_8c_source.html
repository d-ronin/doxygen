127 static uint8_t iap_state = IAP_STATE_READY;
128 static void FirmwareIAPCallback(UAVObjEvent* ev, void *ctx, void *obj, int len)
129 {
130  (void) ctx; (void) obj; (void) len;
131 
132  static uint32_t last_time = 0;
133  uint32_t this_time;
134  uint32_t delta;
135 
136  if(iap_state == IAP_STATE_RESETTING)
137  return;
138 
139  FirmwareIAPObjData data;
140  FirmwareIAPObjGet(&data);
141 
142  if ( ev->obj == FirmwareIAPObjHandle() ) {
143  // Get the input object data
144  FirmwareIAPObjGet(&data);
145  this_time = get_time();
146  delta = this_time - last_time;
147  last_time = this_time;
148 
149  switch(iap_state) {
150  case IAP_STATE_READY:
151  if( data.Command == IAP_CMD_STEP_1 ) {
152  iap_state = IAP_STATE_STEP_1;
153  } break;
154  case IAP_STATE_STEP_1:
155  if( data.Command == IAP_CMD_STEP_2 ) {
156  if( delta > iap_time_2_low_end && delta < iap_time_2_high_end ) {
157  iap_state = IAP_STATE_STEP_2;
158  } else {
159  iap_state = IAP_STATE_READY;
160  }
161  } else {
162  iap_state = IAP_STATE_READY;
163  }
164  break;
165  case IAP_STATE_STEP_2:
166  if( (data.Command == IAP_CMD_STEP_3) || (data.Command == IAP_CMD_STEP_3NB) ) {
167  if( delta > iap_time_3_low_end && delta < iap_time_3_high_end ) {
168 
169  FlightStatusData flightStatus;
170  FlightStatusGet(&flightStatus);
171 
172  if(flightStatus.Armed != FLIGHTSTATUS_ARMED_DISARMED) {
173  // Abort any attempts if not disarmed
174  iap_state = IAP_STATE_READY;
175  break;
176  }
177 
178  // we've met the three sequence of command numbers
179  // we've met the time requirements.
180  if(data.Command == IAP_CMD_STEP_3) {
181  PIOS_IAP_SetRequest1();
182  PIOS_IAP_SetRequest2();
183  }
184  else if(data.Command == IAP_CMD_STEP_3NB) {
185  PIOS_IAP_SetRequest1();
186  PIOS_IAP_SetRequest3();
187  }
188 
189  /* Note: Cant just wait timeout value, because first time is randomized */
190  reset_count = 0;
191  lastResetSysTime = PIOS_Thread_Systime();
192  UAVObjEvent * ev = PIOS_malloc_no_dma(sizeof(UAVObjEvent));
193  memset(ev,0,sizeof(UAVObjEvent));
194  EventPeriodicCallbackCreate(ev, resetTask, 100);
195  PIOS_Servo_PrepareForReset(); // Stop PWM
196  iap_state = IAP_STATE_RESETTING;
197  } else {
198  iap_state = IAP_STATE_READY;
199  }
200  } else {
201  iap_state = IAP_STATE_READY;
202  }
203  break;
204  case IAP_STATE_RESETTING:
205  // stay here permanentally, should reboot
206  break;
207  default:
208  iap_state = IAP_STATE_READY;
209  last_time = 0; // Reset the time counter, as we are not doing a IAP reset
210  break;
211  }
212  }
213 }
