86 int32_t vtol_follower_control_path(const PathDesiredData *pathDesired,
87  struct path_status *progress)
88 {
89  PositionActualData positionActual;
90  PositionActualGet(&positionActual);
91 
92  VelocityActualData velocityActual;
93  VelocityActualGet(&velocityActual);
94 
95  PathStatusData pathStatus;
96  PathStatusGet(&pathStatus);
97 
98  const float cur_pos_ned[3] = {
99  positionActual.North +
100  velocityActual.North * guidanceSettings.PositionFeedforward,
101  positionActual.East +
102  velocityActual.East * guidanceSettings.PositionFeedforward,
103  positionActual.Down };
104 
105  path_progress(pathDesired, cur_pos_ned, progress);
106 
107  // Check if we have already completed this leg
108  bool current_leg_completed =
109  (pathStatus.Status == PATHSTATUS_STATUS_COMPLETED) &&
110  (pathStatus.Waypoint == pathDesired->Waypoint);
111 
112  pathStatus.fractional_progress = progress->fractional_progress;
113  pathStatus.error = progress->error;
114  pathStatus.Waypoint = pathDesired->Waypoint;
115 
116  // Figure out how low (high) we should be and the error
117  const float altitudeSetpoint = interpolate_value(progress->fractional_progress,
118  pathDesired->Start[2], pathDesired->End[2]);
119 
120  const float downError = altitudeSetpoint - positionActual.Down;
121 
122  // If leg is completed signal this
123  if (current_leg_completed || pathStatus.fractional_progress > 1.0f) {
124  const bool criterion_altitude =
125  (downError > -guidanceSettings.WaypointAltitudeTol) ||
126  (!guidanceSettings.ThrottleControl);
127 
128  // Once we complete leg and hit altitude criterion signal this
129  // waypoint is done. Or if we're not controlling throttle,
130  // ignore height for completion.
131 
132  // Waypoint heights are thus treated as crossing restrictions-
133  // cross this point at or above...
134  if (criterion_altitude || current_leg_completed) {
135  pathStatus.Status = PATHSTATUS_STATUS_COMPLETED;
136  PathStatusSet(&pathStatus);
137  } else {
138  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
139  PathStatusSet(&pathStatus);
140  }
141 
142  // Wait here for new path segment
143  return vtol_follower_control_impl(pathDesired->End, 0, false);
144  }
145 
146  // Interpolate desired velocity and altitude along the path
147  float groundspeed = interpolate_value(progress->fractional_progress,
148  pathDesired->StartingVelocity, pathDesired->EndingVelocity);
149 
150  float error_speed = cubic_deadband(progress->error,
151  guidanceSettings.PathDeadbandWidth,
152  guidanceSettings.PathDeadbandCenterGain,
153  vtol_path_m, vtol_path_r) *
154  guidanceSettings.HorizontalPosPI[VTOLPATHFOLLOWERSETTINGS_HORIZONTALPOSPI_KP];
155 
156  /* Sum the desired path movement vector with the correction vector */
157  float commands_ned[3];
158  commands_ned[0] = progress->path_direction[0] * groundspeed +
159  progress->correction_direction[0] * error_speed;
160 
161  commands_ned[1] = progress->path_direction[1] * groundspeed +
162  progress->correction_direction[1] * error_speed;
163 
164  /* Limit the total velocity based on the configured value. */
165  vector2_clip(commands_ned, guidanceSettings.HorizontalVelMax);
166 
167  commands_ned[2] = pid_apply_antiwindup(&vtol_pids[DOWN_POSITION], downError,
168  -altitudeHoldSettings.MaxClimbRate * 0.1f,
169  altitudeHoldSettings.MaxDescentRate * 0.1f,
170  0);
171 
172  VelocityDesiredData velocityDesired;
173  VelocityDesiredGet(&velocityDesired);
174  velocityDesired.North = commands_ned[0];
175  velocityDesired.East = commands_ned[1];
176  velocityDesired.Down = commands_ned[2];
177  VelocityDesiredSet(&velocityDesired);
178 
179  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
180  PathStatusSet(&pathStatus);
181 
182  return 0;
183 }

192 static int32_t vtol_follower_control_impl(
193  const float *hold_pos_ned, float alt_rate, bool update_status)
194 {
195  PositionActualData positionActual;
196  VelocityDesiredData velocityDesired;
197 
198  PositionActualGet(&positionActual);
199 
200  VelocityActualData velocityActual;
201 
202  VelocityActualGet(&velocityActual);
203 
204  /* Where would we be in ___ second at current rates? */
205  const float cur_pos_ned[3] = {
206  positionActual.North +
207  velocityActual.North * guidanceSettings.PositionFeedforward,
208  positionActual.East +
209  velocityActual.East * guidanceSettings.PositionFeedforward,
210  positionActual.Down };
211 
212  float errors_ned[3];
213 
214  /* Calculate the difference between where we want to be and the
215  * above position */
216  vector3_distances(cur_pos_ned, hold_pos_ned, errors_ned, false);
217 
218  float horiz_error_mag = vectorn_magnitude(errors_ned, 2);
219  float scale_horiz_error_mag = 0;
220 
221  /* Apply a cubic deadband; if we are already really close don't work
222  * as hard to fix it as if we're far away. Prevents chasing high
223  * frequency noise in direction of correction.
224  *
225  * That is, when we're far, noise in estimated position mostly results
226  * in noise/dither in how fast we go towards the target. When we're
227  * close, there's a large directional / hunting component. This
228  * attenuates that.
229  */
230  if (horiz_error_mag > 0.00001f) {
231  scale_horiz_error_mag = cubic_deadband(horiz_error_mag,
232  guidanceSettings.EndpointDeadbandWidth,
233  guidanceSettings.EndpointDeadbandCenterGain,
234  vtol_end_m, vtol_end_r) / horiz_error_mag;
235  }
236 
237  float damped_ne[2] = { errors_ned[0] * scale_horiz_error_mag,
238  errors_ned[1] * scale_horiz_error_mag };
239 
240  float commands_ned[3];
241 
242  // Compute desired north command velocity from position error
243  commands_ned[0] = pid_apply_antiwindup(&vtol_pids[NORTH_POSITION], damped_ne[0],
244  -guidanceSettings.HorizontalVelMax, guidanceSettings.HorizontalVelMax, 0);
245 
246  // Compute desired east command velocity from position error
247  commands_ned[1] = pid_apply_antiwindup(&vtol_pids[EAST_POSITION], damped_ne[1],
248  -guidanceSettings.HorizontalVelMax, guidanceSettings.HorizontalVelMax, 0);
249 
250  if (fabsf(alt_rate) < 0.001f) {
251  // Compute desired down comand velocity from the position difference
252  commands_ned[2] = pid_apply_antiwindup(&vtol_pids[DOWN_POSITION], errors_ned[2],
253  -altitudeHoldSettings.MaxClimbRate * 0.1f,
254  altitudeHoldSettings.MaxDescentRate * 0.1f,
255  0);
256  } else {
257  // Just use the commanded rate
258  commands_ned[2] = alt_rate;
259  }
260 
261  // Limit the maximum horizontal velocity any direction (not north and east separately)
262  vector2_clip(commands_ned, guidanceSettings.HorizontalVelMax);
263 
264  velocityDesired.North = commands_ned[0];
265  velocityDesired.East = commands_ned[1];
266  velocityDesired.Down = commands_ned[2];
267 
268  VelocityDesiredSet(&velocityDesired);
269 
270  if (update_status) {
271  uint8_t path_status = PATHSTATUS_STATUS_INPROGRESS;
272 
273  if (fabsf(alt_rate) < 0.001f) {
274  const bool criterion_altitude =
275  (errors_ned[2]> -guidanceSettings.WaypointAltitudeTol) || (!guidanceSettings.ThrottleControl);
276 
277  // Indicate whether we are in radius of this endpoint
278  // And at/above the altitude requested
279  if ((vectorn_magnitude(errors_ned, 2) < guidanceSettings.EndpointRadius) && criterion_altitude) {
280  path_status = PATHSTATUS_STATUS_COMPLETED;
281  }
282  }
283  // Otherwise, we're not done-- we're in autoland or someone
284  // upstream is explicitly trimming our altitude
285 
286  PathStatusStatusSet(&path_status);
287  }
288 
289  return 0;
290 }

339 static int32_t vtol_follower_control_accel(float dT)
340 {
341  VelocityDesiredData velocityDesired;
342  VelocityActualData velocityActual;
343  AccelDesiredData accelDesired;
344  NedAccelData nedAccel;
345 
346  float north_error, north_acceleration;
347  float east_error, east_acceleration;
348  float down_error;
349 
350  static float last_north_velocity;
351  static float last_east_velocity;
352 
353  NedAccelGet(&nedAccel);
354  VelocityActualGet(&velocityActual);
355  VelocityDesiredGet(&velocityDesired);
356 
357  // Optionally compute the acceleration required component from a changing velocity desired
358  if (guidanceSettings.VelocityChangePrediction == VTOLPATHFOLLOWERSETTINGS_VELOCITYCHANGEPREDICTION_TRUE && dT > 0) {
359  north_acceleration = (velocityDesired.North - last_north_velocity) / dT;
360  east_acceleration = (velocityDesired.East - last_east_velocity) / dT;
361  last_north_velocity = velocityDesired.North;
362  last_east_velocity = velocityDesired.East;
363  } else {
364  north_acceleration = 0;
365  east_acceleration = 0;
366  }
367 
368  // Convert the max angles into the maximum angle that would be requested
369  const float MAX_ACCELERATION = GRAVITY * sinf(guidanceSettings.MaxRollPitch * DEG2RAD);
370 
371  // Compute desired north command from velocity error
372  north_error = velocityDesired.North - velocityActual.North;
373  north_acceleration += pid_apply_antiwindup(&vtol_pids[NORTH_VELOCITY], north_error,
374  -MAX_ACCELERATION, MAX_ACCELERATION, 0) +
375  velocityDesired.North * guidanceSettings.VelocityFeedforward +
376  -nedAccel.North * guidanceSettings.HorizontalVelPID[VTOLPATHFOLLOWERSETTINGS_HORIZONTALVELPID_KD];
377 
378  // Compute desired east command from velocity error
379  east_error = velocityDesired.East - velocityActual.East;
380  east_acceleration += pid_apply_antiwindup(&vtol_pids[EAST_VELOCITY], east_error,
381  -MAX_ACCELERATION, MAX_ACCELERATION, 0) +
382  velocityDesired.East * guidanceSettings.VelocityFeedforward +
383  -nedAccel.East * guidanceSettings.HorizontalVelPID[VTOLPATHFOLLOWERSETTINGS_HORIZONTALVELPID_KD];
384 
385  accelDesired.North = north_acceleration;
386  accelDesired.East = east_acceleration;
387 
388  // Note: vertical controller really isn't currently in units of acceleration and the anti-windup may
389  // not be appropriate. However, it is fine for now since it this output is just directly used on the
390  // output. To use this appropriately we need a model of throttle to acceleration.
391  // Compute desired down command. Using NED accel as the damping term
392  down_error = velocityDesired.Down - velocityActual.Down;
393  // Negative is critical here since throttle is negative with down
394  accelDesired.Down = -pid_apply_antiwindup(&vtol_pids[DOWN_VELOCITY], down_error, -1, 0, 0);
395 
396  // Store the desired acceleration
397  AccelDesiredSet(&accelDesired);
398 
399  return 0;
400 }

403 static float vtol_follower_control_altitude(float downCommand) {
404  AltitudeHoldStateData altitudeHoldState;
405  altitudeHoldState.VelocityDesired = downCommand;
406  altitudeHoldState.Integral = vtol_pids[DOWN_VELOCITY].iAccumulator;
407  altitudeHoldState.AngleGain = 1.0f;
408 
409  if (altitudeHoldSettings.AttitudeComp > 0) {
410  // Thrust desired is at this point the mount desired in the up direction, we can
411  // account for the attitude if desired
412  AttitudeActualData attitudeActual;
413  AttitudeActualGet(&attitudeActual);
414 
415  // Project a unit vector pointing up into the body frame and
416  // get the z component
417  float fraction = attitudeActual.q1 * attitudeActual.q1 -
418  attitudeActual.q2 * attitudeActual.q2 -
419  attitudeActual.q3 * attitudeActual.q3 +
420  attitudeActual.q4 * attitudeActual.q4;
421 
422  // Add ability to scale up the amount of compensation to achieve
423  // level forward flight
424  fraction = powf(fraction, (float) altitudeHoldSettings.AttitudeComp / 100.0f);
425 
426  // Dividing by the fraction remaining in the vertical projection will
427  // attempt to compensate for tilt. This acts like the thrust is linear
428  // with the output which isn't really true. If the fraction is starting
429  // to go negative we are inverted and should shut off throttle
430  downCommand = (fraction > 0.1f) ? (downCommand / fraction) : 0.0f;
431 
432  altitudeHoldState.AngleGain = 1.0f / fraction;
433  }
434 
435  altitudeHoldState.Thrust = downCommand;
436  AltitudeHoldStateSet(&altitudeHoldState);
437 
438  return downCommand;
439 }

450 int32_t vtol_follower_control_attitude(float dT, const float *att_adj)
451 {
452  vtol_follower_control_accel(dT);
453 
454  float default_adj[2] = {0,0};
455 
456  if (!att_adj) {
457  att_adj = default_adj;
458  }
459 
460  AccelDesiredData accelDesired;
461  AccelDesiredGet(&accelDesired);
462 
463  StabilizationSettingsData stabSet;
464  StabilizationSettingsGet(&stabSet);
465 
466  SystemSettingsData system_settings;
467  SystemSettingsGet( &system_settings );
468 
469  ManualControlCommandData manual_control_command;
470  ManualControlCommandGet( &manual_control_command );
471 
472  float northCommand = accelDesired.North;
473  float eastCommand = accelDesired.East;
474 
475  // Project the north and east acceleration signals into body frame
476  float yaw;
477  AttitudeActualYawGet(&yaw);
478  float forward_accel_desired = -northCommand * cosf(yaw * DEG2RAD) + -eastCommand * sinf(yaw * DEG2RAD);
479  float right_accel_desired = -northCommand * sinf(yaw * DEG2RAD) + eastCommand * cosf(yaw * DEG2RAD);
480 
481  StabilizationDesiredData stabDesired = {};
482 
483  // Set the angle that would achieve the desired acceleration given the thrust is enough for a hover
484  stabDesired.Pitch = bound_sym(RAD2DEG * atanf(forward_accel_desired / GRAVITY), guidanceSettings.MaxRollPitch) + att_adj[1];
485  stabDesired.Roll = bound_sym(RAD2DEG * atanf(right_accel_desired / GRAVITY), guidanceSettings.MaxRollPitch) + att_adj[0];
486 
487  // Re-bound based on maximum attitude settings
488  stabDesired.Pitch = bound_sym(stabDesired.Pitch,
489  stabSet.MaxLevelAngle[STABILIZATIONSETTINGS_MAXLEVELANGLE_PITCH]);
490  stabDesired.Roll = bound_sym(stabDesired.Roll,
491  stabSet.MaxLevelAngle[STABILIZATIONSETTINGS_MAXLEVELANGLE_ROLL]);
492 
493  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
494  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
495 
496  stabDesired.ThrustMode = STABILIZATIONDESIRED_THRUSTMODE_DIRECT;
497  stabDesired.ReprojectionMode = STABILIZATIONDESIRED_REPROJECTIONMODE_NONE;
498 
499  // Calculate the throttle setting or use pass through from transmitter
500  if (guidanceSettings.ThrottleControl == VTOLPATHFOLLOWERSETTINGS_THROTTLECONTROL_FALSE) {
501  stabDesired.Thrust = (system_settings.AirframeType == SYSTEMSETTINGS_AIRFRAMETYPE_HELICP) ? manual_control_command.Collective : manual_control_command.Throttle;
502  } else {
503  float downCommand = vtol_follower_control_altitude(accelDesired.Down);
504 
505  stabDesired.Thrust = bound_min_max(downCommand, 0, 1);
506  }
507 
508  // Various ways to control the yaw that are essentially manual passthrough. However, because we do not have a fine
509  // grained mechanism of manual setting the yaw as it normally would we need to duplicate that code here
510  switch(guidanceSettings.YawMode) {
511  case VTOLPATHFOLLOWERSETTINGS_YAWMODE_RATE:
512  /* This is awkward. This allows the transmitter to control the yaw while flying navigation */
513  stabDesired.Yaw = stabSet.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_YAW] * manual_control_command.Yaw;
514  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
515  break;
516  case VTOLPATHFOLLOWERSETTINGS_YAWMODE_AXISLOCK:
517  stabDesired.Yaw = stabSet.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_YAW] * manual_control_command.Yaw;
518  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK;
519  break;
520  case VTOLPATHFOLLOWERSETTINGS_YAWMODE_PATH:
521  {
522  // Face forward on the path
523  VelocityDesiredData velocityDesired;
524  VelocityDesiredGet(&velocityDesired);
525  float total_vel2 = velocityDesired.East*velocityDesired.East + velocityDesired.North*velocityDesired.North;
526  float path_direction = atan2f(velocityDesired.East, velocityDesired.North) * RAD2DEG;
527  if (total_vel2 > 1) {
528  stabDesired.Yaw = path_direction;
529  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
530  } else {
531  stabDesired.Yaw = 0;
532  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
533  }
534  }
535  break;
536  case VTOLPATHFOLLOWERSETTINGS_YAWMODE_POI:
537  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_POI;
538  break;
539  }
540 
541  StabilizationDesiredSet(&stabDesired);
542 
543  return 0;
544 }

546 static float loiter_deadband(float input, float threshold, float expoPercent) {
547  if (input > threshold) {
548  input -= threshold;
549  } else if (input < -threshold) {
550  input += threshold;
551  } else {
552  input = 0;
553  }
554 
555  input /= (1 - threshold); // Normalize to -1 to 1 range.
556 
557  /* Confine to desired range */
558  if (input > 1.0f) {
559  input = 1.0f;
560  } else if (input < -1.0f) {
561  input = -1.0f;
562  }
563 
564  return expo3(input, expoPercent);
565 }

577 bool vtol_follower_control_loiter(float dT, float *hold_pos, float *att_adj,
578  float *alt_adj) {
579  LoiterCommandData cmd;
580  LoiterCommandGet(&cmd);
581 
582  // XXX TODO reproject when we're not issuing body-centric commands
583  float commands_rp[2] = {
584  cmd.Roll,
585  cmd.Pitch
586  };
587 
588  const float CMD_THRESHOLD = 0.2f;
589 
590  float command_mag = vectorn_magnitude(commands_rp, 2);
591  float deadband_mag = loiter_deadband(command_mag, CMD_THRESHOLD, 40);
592 
593  float down_cmd = 0;
594 
595  if (guidanceSettings.ThrottleControl &&
596  guidanceSettings.LoiterAllowAltControl) {
597  // Inverted because we want units in "Down" frame
598  // loiter_deadband clips appropriately.
599  down_cmd = loiter_deadband(cmd.Thrust,
600  0, /* Deadbanded in manual control */
601  altitudeHoldSettings.Expo);
602  }
603 
604  // Peak detect and decay of the past command magnitude
605  static float historic_mag = 0.0f;
606 
607  // Initialize altitude command
608  *alt_adj = 0;
609 
610  // First reduce by decay constant
611  historic_mag *= loiter_brakealpha;
612 
613  // If our current magnitude is greater than the result, increase it.
614  if (deadband_mag > historic_mag) {
615  historic_mag = deadband_mag;
616  }
617 
618  // And if we haven't had any significant command lately, bug out and
619  // do nothing.
620  if ((historic_mag < 0.001f) && (fabsf(down_cmd) < 0.001f)) {
621  att_adj[0] = 0; att_adj[1] = 0;
622  return false;
623  }
624 
625  // Normalize our command magnitude. Command vectors from this
626  // point are normalized.
627  if (command_mag >= 0.001f) {
628  commands_rp[0] /= command_mag;
629  commands_rp[1] /= command_mag;
630  } else {
631  // Just pick a direction
632  commands_rp[0] = 0.0f;
633  commands_rp[1] = -1.0f;
634  }
635 
636  // Find our current position error
637  PositionActualData positionActual;
638  PositionActualGet(&positionActual);
639 
640  float cur_pos_ned[3] = { positionActual.North,
641  positionActual.East, positionActual.Down };
642 
643  float total_poserr_ned[3];
644  vector3_distances(cur_pos_ned, hold_pos, total_poserr_ned, false);
645 
646  if (deadband_mag >= 0.001f) {
647  float yaw;
648  AttitudeActualYawGet(&yaw);
649 
650  float commands_ne[2];
651  // 90 degrees here compensates for the above being in roll-pitch
652  // order vs. north-east (and where yaw is defined).
653  vector2_rotate(commands_rp, commands_ne, 90 + yaw);
654 
655  VelocityActualData velocityActual;
656  VelocityActualGet(&velocityActual);
657 
658  // Come up with a target velocity for us to fly the command
659  // at, considering our current momentum in that direction.
660  float target_vel = guidanceSettings.LoiterInitialMaxVel *
661  deadband_mag;
662 
663  // Plus whatever current velocity we're making good in
664  // that direction..
665  // find the portion of our current velocity vector parallel to
666  // cmd.
667  float parallel_sign =
668  velocityActual.North * commands_ne[0] +
669  velocityActual.East * commands_ne[1];
670 
671  if (parallel_sign > 0) {
672  float parallel_mag = sqrtf(
673  powf(velocityActual.North * commands_ne[0], 2) +
674  powf(velocityActual.East * commands_ne[1], 2));
675 
676  target_vel += deadband_mag * parallel_mag;
677  }
678 
679  // Feed the target velocity forward for our new desired position
680  hold_pos[0] = cur_pos_ned[0] +
681  commands_ne[0] * target_vel *
682  guidanceSettings.LoiterLookaheadTimeConstant;
683  hold_pos[1] = cur_pos_ned[1] +
684  commands_ne[1] * target_vel *
685  guidanceSettings.LoiterLookaheadTimeConstant;
686  }
687 
688  // Now put a portion of the error back in. At full stick
689  // deflection, decay error at specified time constant
690  float scaled_error_alpha = 1 - historic_mag * (1 - loiter_errordecayalpha);
691  hold_pos[0] -= scaled_error_alpha * total_poserr_ned[0];
692  hold_pos[1] -= scaled_error_alpha * total_poserr_ned[1];
693 
694  // Compute attitude feedforward
695  att_adj[0] = deadband_mag * commands_rp[0] *
696  guidanceSettings.LoiterAttitudeFeedthrough;
697  att_adj[1] = deadband_mag * commands_rp[1] *
698  guidanceSettings.LoiterAttitudeFeedthrough;
699 
700  // If we are being commanded to climb or descend...
701  if (fabsf(down_cmd) >= 0.001f) {
702  // Forgive all altitude error so when position controller comes
703  // back we do something sane
704 
705  hold_pos[2] = cur_pos_ned[2];
706 
707  // and output an adjustment for velocity control use
708  if (down_cmd < 0) { /* climbing*/
709  *alt_adj = down_cmd * altitudeHoldSettings.MaxClimbRate * 0.1f;
710  } else { /* descending */
711  *alt_adj = down_cmd * altitudeHoldSettings.MaxDescentRate * 0.1f;
712  }
713  }
714 
715  return true;
716 }
