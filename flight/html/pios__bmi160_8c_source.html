117 int32_t PIOS_BMI160_Init(pios_spi_t spi_id, uint32_t slave_num, const struct pios_bmi160_cfg *cfg, bool do_foc)
118 {
119  dev = PIOS_BMI160_alloc(cfg);
120  if (dev == NULL)
121  return -1;
122 
123  dev->spi_id = spi_id;
124  dev->slave_num = slave_num;
125  dev->cfg = cfg;
126 
127  /* Configure the scales */
128  switch (cfg->acc_range){
129  case PIOS_BMI160_RANGE_2G:
130  dev->accel_scale = GRAVITY / 16384.f;
131  break;
132  case PIOS_BMI160_RANGE_4G:
133  dev->accel_scale = GRAVITY / 8192.f;
134  break;
135  case PIOS_BMI160_RANGE_8G:
136  dev->accel_scale = GRAVITY / 4096.f;
137  break;
138  case PIOS_BMI160_RANGE_16G:
139  dev->accel_scale = GRAVITY / 2048.f;
140  break;
141  }
142 
143  switch (cfg->gyro_range){
144  case PIOS_BMI160_RANGE_125DPS:
145  dev->gyro_scale = 1.f / 262.4f;
146  PIOS_SENSORS_SetMaxGyro(125);
147  break;
148  case PIOS_BMI160_RANGE_250DPS:
149  dev->gyro_scale = 1.f / 131.2f;
150  PIOS_SENSORS_SetMaxGyro(250);
151  break;
152  case PIOS_BMI160_RANGE_500DPS:
153  dev->gyro_scale = 1.f / 65.6f;
154  PIOS_SENSORS_SetMaxGyro(500);
155  break;
156  case PIOS_BMI160_RANGE_1000DPS:
157  dev->gyro_scale = 1.f / 32.8f;
158  PIOS_SENSORS_SetMaxGyro(1000);
159  break;
160  case PIOS_BMI160_RANGE_2000DPS:
161  dev->gyro_scale = 1.f / 16.4f;
162  PIOS_SENSORS_SetMaxGyro(2000);
163  break;
164  }
165 
166  /* Read this address to acticate SPI (see p. 84) */
167  PIOS_BMI160_ReadReg(0x7F);
168  PIOS_DELAY_WaitmS(10); // Give SPI some time to start up
169 
170  /* Check the chip ID */
171  if (PIOS_BMI160_ReadReg(BMI160_REG_CHIPID) != 0xd1){
172  return -2;
173  }
174 
175  /* Configure the MPU9250 Sensor */
176  if (PIOS_BMI160_Config(cfg) != 0){
177  return -3;
178  }
179 
180  /* Perform fast offset compensation if requested */
181  if (do_foc) {
182  if (PIOS_BMI160_do_foc(cfg) != 0) {
183  return -4;
184  }
185  }
186 
187  /* Set up EXTI line */
188  PIOS_EXTI_Init(cfg->exti_cfg);
189 
190  // Wait 20 ms for data ready interrupt and make sure it happens
191  // twice
192  if ((PIOS_Semaphore_Take(dev->data_ready_sema, 20) != true) ||
193  (PIOS_Semaphore_Take(dev->data_ready_sema, 20) != true)) {
194  return -10;
195  }
196 
197  dev->TaskHandle = PIOS_Thread_Create(
198  PIOS_BMI160_Task, "pios_bmi160", PIOS_BMI160_TASK_STACK_BYTES, NULL, PIOS_BMI160_TASK_PRIORITY);
199  PIOS_Assert(dev->TaskHandle != NULL);
200 
201  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_ACCEL, 1600);
202  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_GYRO, 1600);
203 
204  PIOS_SENSORS_Register(PIOS_SENSOR_ACCEL, dev->accel_queue);
205  PIOS_SENSORS_Register(PIOS_SENSOR_GYRO, dev->gyro_queue);
206 
207  return 0;
208 }

214 static int32_t PIOS_BMI160_Config(const struct pios_bmi160_cfg *cfg)
215 {
216 
217  // Set normal power mode for gyro and accelerometer
218  if (PIOS_BMI160_WriteReg(BMI160_REG_CMD, BMI160_PMU_CMD_PMU_GYR_NORMAL) != 0){
219  return -1;
220  }
221  PIOS_DELAY_WaitmS(100); // can take up to 80ms
222 
223  if (PIOS_BMI160_WriteReg(BMI160_REG_CMD, BMI160_PMU_CMD_PMU_ACC_NORMAL) != 0){
224  return -2;
225  }
226  PIOS_DELAY_WaitmS(5); // can take up to 3.8ms
227 
228  // Verify that normal power mode was entered
229  uint8_t pmu_status = PIOS_BMI160_ReadReg(BMI160_REG_PMU_STAT);
230  if ((pmu_status & 0x3C) != 0x14){
231  return -3;
232  }
233 
234  // Set odr and ranges
235  // Set acc_us = 0 acc_bwp = 0b010 so only the first filter stage is used
236  if (PIOS_BMI160_WriteReg(BMI160_REG_ACC_CONF, 0x20 | cfg->odr) != 0){
237  return -3;
238  }
239  PIOS_DELAY_WaitmS(1);
240 
241  // Set gyr_bwp = 0b010 so only the first filter stage is used
242  if (PIOS_BMI160_WriteReg(BMI160_REG_GYR_CONF, 0x20 | cfg->odr) != 0){
243  return -4;
244  }
245  PIOS_DELAY_WaitmS(1);
246 
247  if (PIOS_BMI160_WriteReg(BMI160_REG_ACC_RANGE, cfg->acc_range) != 0){
248  return -5;
249  }
250  PIOS_DELAY_WaitmS(1);
251 
252  if (PIOS_BMI160_WriteReg(BMI160_REG_GYR_RANGE, cfg->gyro_range) != 0){
253  return -6;
254  }
255  PIOS_DELAY_WaitmS(1);
256 
257  // Enable offset compensation
258  uint8_t val = PIOS_BMI160_ReadReg(BMI160_REG_OFFSET_0);
259  if (PIOS_BMI160_WriteReg(BMI160_REG_OFFSET_0, val | 0xC0) != 0){
260  return -7;
261  }
262 
263  // Enable data ready interrupt
264  if (PIOS_BMI160_WriteReg(BMI160_REG_INT_EN1, BMI160_INT_EN1_DRDY) != 0){
265  return -8;
266  }
267  PIOS_DELAY_WaitmS(1);
268 
269  // Enable INT1 pin
270  if (PIOS_BMI160_WriteReg(BMI160_REG_INT_OUT_CTRL, BMI160_INT_OUT_CTRL_INT1_CONFIG) != 0){
271  return -9;
272  }
273  PIOS_DELAY_WaitmS(1);
274 
275  // Map data ready interrupt to INT1 pin
276  if (PIOS_BMI160_WriteReg(BMI160_REG_INT_MAP1, BMI160_REG_INT_MAP1_INT1_DRDY) != 0){
277  return -10;
278  }
279  PIOS_DELAY_WaitmS(1);
280 
281  return 0;
282 }

291 static int32_t PIOS_BMI160_do_foc(const struct pios_bmi160_cfg *cfg)
292 {
293  uint8_t val = 0;
294 
295  // Set the orientation: 0G for x and y, +/- 1G for z
296  switch (cfg->orientation) {
297  case PIOS_BMI160_TOP_0DEG:
298  case PIOS_BMI160_TOP_90DEG:
299  case PIOS_BMI160_TOP_180DEG:
300  case PIOS_BMI160_TOP_270DEG:
301  val = 0x7D;
302  break;
303  case PIOS_BMI160_BOTTOM_0DEG:
304  case PIOS_BMI160_BOTTOM_90DEG:
305  case PIOS_BMI160_BOTTOM_180DEG:
306  case PIOS_BMI160_BOTTOM_270DEG:
307  val = 0x7E;
308  }
309 
310  if (PIOS_BMI160_WriteReg(BMI160_REG_FOC_CONF, val) != 0) {
311  return -1;
312  }
313 
314  // Start FOC
315  if (PIOS_BMI160_WriteReg(BMI160_REG_CMD, BMI160_CMD_START_FOC) != 0) {
316  return -2;
317  }
318 
319  // Wait for FOC to complete
320  for (int i=0; i<50; i++) {
321  val = PIOS_BMI160_ReadReg(BMI160_REG_STATUS);
322  if (val & BMI160_REG_STATUS_FOC_RDY) {
323  break;
324  }
325  PIOS_DELAY_WaitmS(10);
326  PIOS_WDG_Clear();
327  }
328  if (!(val & BMI160_REG_STATUS_FOC_RDY)) {
329  return -3;
330  }
331 
332  // Program NVM
333  val = PIOS_BMI160_ReadReg(BMI160_REG_CONF);
334  if (PIOS_BMI160_WriteReg(BMI160_REG_CONF, val | BMI160_REG_CONF_NVM_PROG_EN) != 0) {
335  return -4;
336  }
337 
338  if (PIOS_BMI160_WriteReg(BMI160_REG_CMD, BMI160_CMD_PROG_NVM) != 0) {
339  return -5;
340  }
341 
342  // Wait for NVM programming to complete
343  for (int i=0; i<50; i++) {
344  val = PIOS_BMI160_ReadReg(BMI160_REG_STATUS);
345  if (val & BMI160_REG_STATUS_NVM_RDY) {
346  break;
347  }
348  PIOS_DELAY_WaitmS(10);
349  PIOS_WDG_Clear();
350  }
351  if (!(val & BMI160_REG_STATUS_NVM_RDY)) {
352  return -6;
353  }
354 
355  return 0;
356 }

361 static struct bmi160_dev *PIOS_BMI160_alloc(const struct pios_bmi160_cfg *cfg)
362 {
363  struct bmi160_dev *bmi160_dev;
364 
365  bmi160_dev = (struct bmi160_dev *)PIOS_malloc(sizeof(*bmi160_dev));
366  if (!bmi160_dev)
367  return NULL;
368 
369  bmi160_dev->magic = PIOS_BMI160_DEV_MAGIC;
370 
371  bmi160_dev->accel_queue = PIOS_Queue_Create(PIOS_BMI160_MAX_DOWNSAMPLE, sizeof(struct pios_sensor_accel_data));
372  if (bmi160_dev->accel_queue == NULL) {
373  PIOS_free(bmi160_dev);
374  return NULL;
375  }
376 
377  bmi160_dev->gyro_queue = PIOS_Queue_Create(PIOS_BMI160_MAX_DOWNSAMPLE, sizeof(struct pios_sensor_gyro_data));
378  if (bmi160_dev->gyro_queue == NULL) {
379  PIOS_Queue_Delete(dev->accel_queue);
380  PIOS_free(bmi160_dev);
381  return NULL;
382  }
383 
384  bmi160_dev->data_ready_sema = PIOS_Semaphore_Create();
385  if (bmi160_dev->data_ready_sema == NULL) {
386  PIOS_Queue_Delete(dev->accel_queue);
387  PIOS_Queue_Delete(dev->gyro_queue);
388  PIOS_free(bmi160_dev);
389  return NULL;
390  }
391 
392  return bmi160_dev;
393 }

498 static void PIOS_BMI160_Task(void *parameters)
499 {
500  float temperature = 0.f;
501  uint8_t temp_interleave_cnt = 0;
502 
503  while (1) {
504  //Wait for data ready interrupt
505  if (PIOS_Semaphore_Take(dev->data_ready_sema, PIOS_SEMAPHORE_TIMEOUT_MAX) != true)
506  continue;
507 
508  enum {
509  IDX_REG = 0,
510  IDX_GYRO_XOUT_L,
511  IDX_GYRO_XOUT_H,
512  IDX_GYRO_YOUT_L,
513  IDX_GYRO_YOUT_H,
514  IDX_GYRO_ZOUT_L,
515  IDX_GYRO_ZOUT_H,
516  IDX_ACCEL_XOUT_L,
517  IDX_ACCEL_XOUT_H,
518  IDX_ACCEL_YOUT_L,
519  IDX_ACCEL_YOUT_H,
520  IDX_ACCEL_ZOUT_L,
521  IDX_ACCEL_ZOUT_H,
522  BUFFER_SIZE,
523  };
524 
525  uint8_t bmi160_rec_buf[BUFFER_SIZE];
526  uint8_t bmi160_tx_buf[BUFFER_SIZE] = {BMI160_REG_GYR_DATA_X_LSB | 0x80, 0, 0, 0, 0, 0,
527  0, 0, 0, 0, 0, 0, 0};
528 
529  if (PIOS_BMI160_ClaimBus() != 0)
530  continue;
531 
532  if (PIOS_SPI_TransferBlock(dev->spi_id, bmi160_tx_buf, bmi160_rec_buf, BUFFER_SIZE) < 0) {
533  PIOS_BMI160_ReleaseBus();
534  continue;
535  }
536 
537  PIOS_BMI160_ReleaseBus();
538 
539  struct pios_sensor_accel_data accel_data;
540  struct pios_sensor_gyro_data gyro_data;
541 
542  float accel_x = (int16_t)(bmi160_rec_buf[IDX_ACCEL_XOUT_H] << 8 | bmi160_rec_buf[IDX_ACCEL_XOUT_L]);
543  float accel_y = (int16_t)(bmi160_rec_buf[IDX_ACCEL_YOUT_H] << 8 | bmi160_rec_buf[IDX_ACCEL_YOUT_L]);
544  float accel_z = (int16_t)(bmi160_rec_buf[IDX_ACCEL_ZOUT_H] << 8 | bmi160_rec_buf[IDX_ACCEL_ZOUT_L]);
545  float gyro_x = (int16_t)(bmi160_rec_buf[IDX_GYRO_XOUT_H] << 8 | bmi160_rec_buf[IDX_GYRO_XOUT_L]);
546  float gyro_y = (int16_t)(bmi160_rec_buf[IDX_GYRO_YOUT_H] << 8 | bmi160_rec_buf[IDX_GYRO_YOUT_L]);
547  float gyro_z = (int16_t)(bmi160_rec_buf[IDX_GYRO_ZOUT_H] << 8 | bmi160_rec_buf[IDX_GYRO_ZOUT_L]);
548 
549  /*
550  * Convert from sensor frame (x: forward y: left z: up) to
551  * TL convention (x: forward y: right z: down).
552  * See flight/Doc/imu_orientation.md for more detail
553  */
554  switch (dev->cfg->orientation) {
555  case PIOS_BMI160_TOP_0DEG:
556  accel_data.x = accel_x;
557  accel_data.y = -accel_y;
558  accel_data.z = -accel_z;
559  gyro_data.x = gyro_x;
560  gyro_data.y = -gyro_y;
561  gyro_data.z = -gyro_z;
562  break;
563  case PIOS_BMI160_TOP_90DEG:
564  accel_data.x = accel_y;
565  accel_data.y = accel_x;
566  accel_data.z = -accel_z;
567  gyro_data.x = gyro_y;
568  gyro_data.y = gyro_x;
569  gyro_data.z = -gyro_z;
570  break;
571  case PIOS_BMI160_TOP_180DEG:
572  accel_data.x = -accel_x;
573  accel_data.y = accel_y;
574  accel_data.z = -accel_z;
575  gyro_data.x = -gyro_x;
576  gyro_data.y = gyro_y;
577  gyro_data.z = -gyro_z;
578  break;
579  case PIOS_BMI160_TOP_270DEG:
580  accel_data.x = -accel_y;
581  accel_data.y = -accel_x;
582  accel_data.z = -accel_z;
583  gyro_data.x = -gyro_y;
584  gyro_data.y = -gyro_x;
585  gyro_data.z = -gyro_z;
586  break;
587  case PIOS_BMI160_BOTTOM_0DEG:
588  accel_data.x = accel_x;
589  accel_data.y = accel_y;
590  accel_data.z = accel_z;
591  gyro_data.x = gyro_x;
592  gyro_data.y = gyro_y;
593  gyro_data.z = gyro_z;
594  break;
595  case PIOS_BMI160_BOTTOM_90DEG:
596  accel_data.x = -accel_y;
597  accel_data.y = accel_x;
598  accel_data.z = accel_z;
599  gyro_data.x = -gyro_y;
600  gyro_data.y = gyro_x;
601  gyro_data.z = gyro_z;
602  break;
603  case PIOS_BMI160_BOTTOM_180DEG:
604  accel_data.x = -accel_x;
605  accel_data.y = -accel_y;
606  accel_data.z = accel_z;
607  gyro_data.x = -gyro_x;
608  gyro_data.y = -gyro_y;
609  gyro_data.z = gyro_z;
610  break;
611  case PIOS_BMI160_BOTTOM_270DEG:
612  accel_data.x = accel_y;
613  accel_data.y = -accel_x;
614  accel_data.z = accel_z;
615  gyro_data.x = gyro_y;
616  gyro_data.y = -gyro_x;
617  gyro_data.z = gyro_z;
618  break;
619  }
620 
621  // Apply sensor scaling
622  accel_data.x *= dev->accel_scale;
623  accel_data.y *= dev->accel_scale;
624  accel_data.z *= dev->accel_scale;
625 
626  gyro_data.x *= dev->gyro_scale;
627  gyro_data.y *= dev->gyro_scale;
628  gyro_data.z *= dev->gyro_scale;
629 
630  // Get the temperature
631  // NOTE: We do this down here so the chip-select has some time to go low. Strange things happen
632  // When this is done right after readin the accels / gyros
633  if (temp_interleave_cnt % dev->cfg->temperature_interleaving == 0){
634  if (PIOS_BMI160_ClaimBus() != 0)
635  continue;
