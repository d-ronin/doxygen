31 /*
32  * @note This is a stripped-down ADC driver intended primarily for sampling
33  * voltage and current values. Samples are averaged over the period between
34  * fetches so that relatively accurate measurements can be obtained without
35  * forcing higher-level logic to poll aggressively.
36  *
37  * @todo This module needs more work to be more generally useful. It should
38  * almost certainly grow callback support so that e.g. voltage and current readings
39  * can be shipped out for coulomb counting purposes. The F1xx interface presumes
40  * use with analog sensors, but that implementation largely dominates the ADC
41  * resources. Rather than commit to a new API without a defined use case, we
42  * should stick to our lightweight subset until we have a better idea of what's needed.
43  */

103  if (!PIOS_INTERNAL_ADC_validate(pios_adc_dev)) {
104  return;
105  }
106

111  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AIN;

121 static void init_dma(void)
122 {
123  if (!PIOS_INTERNAL_ADC_validate(pios_adc_dev)) {
124  return;
125  }
126 
127  /* Disable interrupts */
128  DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, pios_adc_dev->cfg->dma.irq.flags, DISABLE);
129 
130  /* Configure DMA channel */
131  DMA_DeInit(pios_adc_dev->cfg->dma.rx.channel);
132  DMA_InitTypeDef DMAInit = pios_adc_dev->cfg->dma.rx.init;
133  DMAInit.DMA_Memory0BaseAddr = (uint32_t)&adc_raw_buffer_0[0];
134  DMAInit.DMA_BufferSize = pios_adc_dev->max_samples * pios_adc_dev->cfg->adc_pin_count;
135  DMAInit.DMA_DIR = DMA_DIR_PeripheralToMemory;
136  DMAInit.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
137  DMAInit.DMA_MemoryInc = DMA_MemoryInc_Enable;
138  DMAInit.DMA_PeripheralDataSize = DMA_PeripheralDataSize_HalfWord;
139  DMAInit.DMA_MemoryDataSize = DMA_MemoryDataSize_HalfWord;
140  DMAInit.DMA_Mode = DMA_Mode_Circular;
141  DMAInit.DMA_Priority = DMA_Priority_Low;
142  DMAInit.DMA_FIFOMode = DMA_FIFOMode_Disable;
143  DMAInit.DMA_FIFOThreshold = DMA_FIFOThreshold_HalfFull;
144  DMAInit.DMA_MemoryBurst = DMA_MemoryBurst_Single;
145  DMAInit.DMA_PeripheralBurst = DMA_PeripheralBurst_Single;
146 
147  DMA_Init(pios_adc_dev->cfg->dma.rx.channel, &DMAInit); /* channel is actually stream ... */
148 
149  /* configure for double-buffered mode and interrupt on every buffer flip */
150  DMA_DoubleBufferModeConfig(pios_adc_dev->cfg->dma.rx.channel, (uint32_t)&adc_raw_buffer_1[0], DMA_Memory_0);
151  DMA_DoubleBufferModeCmd(pios_adc_dev->cfg->dma.rx.channel, ENABLE);
152  DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, DMA_IT_TC, ENABLE);
153  //DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, DMA_IT_HT, ENABLE);
154 
155  /* enable DMA */
156  DMA_Cmd(pios_adc_dev->cfg->dma.rx.channel, ENABLE);
157 
158  /* Configure DMA interrupt */
159  NVIC_InitTypeDef NVICInit = pios_adc_dev->cfg->dma.irq.init;
160  NVIC_Init(&NVICInit);
161 }

163 static void init_adc(void)
164 {
165  if (!PIOS_INTERNAL_ADC_validate(pios_adc_dev)) {
166  return;
167  }
168 
169  RCC_APB2PeriphClockCmd(RCC_APB2Periph_ADC1, ENABLE);
170 
171  ADC_DeInit();
172 
173  /* turn on VREFInt in case we need it */
174  ADC_TempSensorVrefintCmd(ENABLE);
175 
176  /* Do common ADC init */
177  ADC_CommonInitTypeDef ADC_CommonInitStructure;
178  ADC_CommonStructInit(&ADC_CommonInitStructure);
179  ADC_CommonInitStructure.ADC_Mode = ADC_Mode_Independent;
180  ADC_CommonInitStructure.ADC_Prescaler = ADC_Prescaler_Div8;
181  ADC_CommonInitStructure.ADC_DMAAccessMode = ADC_DMAAccessMode_Disabled;
182  ADC_CommonInitStructure.ADC_TwoSamplingDelay = ADC_TwoSamplingDelay_5Cycles;
183  ADC_CommonInit(&ADC_CommonInitStructure);
184 
185  ADC_InitTypeDef ADC_InitStructure;
186  ADC_StructInit(&ADC_InitStructure);
187  ADC_InitStructure.ADC_Resolution = ADC_Resolution_12b;
188  ADC_InitStructure.ADC_ScanConvMode = ENABLE;
189  ADC_InitStructure.ADC_ContinuousConvMode = ENABLE;
190  ADC_InitStructure.ADC_ExternalTrigConvEdge = ADC_ExternalTrigConvEdge_None;
191  ADC_InitStructure.ADC_DataAlign = ADC_DataAlign_Right;
192  ADC_InitStructure.ADC_NbrOfConversion = pios_adc_dev->cfg->adc_pin_count;
193  ADC_Init(pios_adc_dev->cfg->adc_dev_master, &ADC_InitStructure);
194 
195  /* Enable DMA request */
196  ADC_DMACmd(pios_adc_dev->cfg->adc_dev_master, ENABLE);
197 
198  /* Configure input scan */
199  for (int32_t i = 0; i < pios_adc_dev->cfg->adc_pin_count; i++) {
200  ADC_RegularChannelConfig(pios_adc_dev->cfg->adc_dev_master,
201  pios_adc_dev->cfg->adc_pins[i].adc_channel,
202  i+1,
203  ADC_SampleTime_56Cycles); /* XXX this is totally arbitrary... */
204  }
205 
206  ADC_DMARequestAfterLastTransferCmd(pios_adc_dev->cfg->adc_dev_master, ENABLE);
207 
208  /* Finally start initial conversion */
209  ADC_Cmd(pios_adc_dev->cfg->adc_dev_master, ENABLE);
210  ADC_ContinuousModeCmd(pios_adc_dev->cfg->adc_dev_master, ENABLE);
211  ADC_SoftwareStartConv(pios_adc_dev->cfg->adc_dev_master);
212 }

222 static struct pios_internal_adc_dev * PIOS_INTERNAL_ADC_Allocate(const struct pios_internal_adc_cfg * cfg)
223 {
224  struct pios_internal_adc_dev * adc_dev;
225 
226  adc_dev = (struct pios_internal_adc_dev *)PIOS_malloc(sizeof(*adc_dev));
227  if (!adc_dev) {
228  return NULL;
229  }
230 
231  // Maxmimum number of samples (XXX: not sure where the dependency on the ADC used comes from..)
232  bool use_adc_2 = cfg->adc_dev_master == ADC2;
233  adc_dev->max_samples = (((cfg->adc_pin_count + use_adc_2) >> use_adc_2) << use_adc_2) * PIOS_ADC_MAX_OVERSAMPLING * 2;
234 
235  accumulator = (struct adc_accumulator *)PIOS_malloc_no_dma(cfg->adc_pin_count * sizeof(struct adc_accumulator));
236  if (!accumulator) {
237  PIOS_free(adc_dev);
238  return NULL;
239  }
240 
241  adc_raw_buffer_0 = (uint16_t *)PIOS_malloc(adc_dev->max_samples * cfg->adc_pin_count * sizeof(uint16_t));
242  if (!adc_raw_buffer_0) {
243  PIOS_free(adc_dev);
244  PIOS_free(accumulator);
245  return NULL;
246  }
247 
248  adc_raw_buffer_1 = (uint16_t *)PIOS_malloc(adc_dev->max_samples * cfg->adc_pin_count * sizeof(uint16_t));
249  if (!adc_raw_buffer_1) {
250  PIOS_free(adc_dev);
251  PIOS_free(accumulator);
252  PIOS_free(adc_raw_buffer_0);
253  return NULL;
254  }
255 
256  adc_dev->magic = PIOS_INTERNAL_ADC_DEV_MAGIC;
257  return(adc_dev);
258 }

298 static int32_t PIOS_INTERNAL_ADC_PinGet(uint32_t internal_adc_id, uint32_t pin)
299 {
300 #if defined(PIOS_INCLUDE_ADC)
301  int32_t result;
302 
303  if (!PIOS_INTERNAL_ADC_validate(pios_adc_dev)) {
304  return -1;
305  }
306 
307  /* Check if pin exists */
308  if (pin >= pios_adc_dev->cfg->adc_pin_count) {
309  return -2;
310  }
311 
312  if (accumulator[pin].accumulator <= 0) {
313  return -3;
314  }
315 
316  /* return accumulated result and clear accumulator */
317  result = accumulator[pin].accumulator / (accumulator[pin].count ?: 1);
318  accumulator[pin].accumulator = result;
319  accumulator[pin].count = 1;
320 
321  return result;
322 #endif
323  return -1;
324 }

369 void accumulate(uint16_t *buffer, uint32_t count)
370 {
371 #if defined(PIOS_INCLUDE_ADC)
372  uint16_t *sp = buffer;
373  if (!PIOS_INTERNAL_ADC_validate(pios_adc_dev)) {
374  return;
375  }
376 
377  /*
378  * Accumulate sampled values.
379  */
380  while (count--) {
381  for (int i = 0; i < pios_adc_dev->cfg->adc_pin_count; i++) {
382  accumulator[i].accumulator += *sp++;
383  accumulator[i].count++;
384  /*
385  * If the accumulator reaches half-full, rescale in order to
386  * make more space.
387  */
388  if (accumulator[i].accumulator >= (1 << 31)) {
389  accumulator[i].accumulator /= 2;
390  accumulator[i].count /= 2;
391  }
392  }
393  }
394 
395 #endif
396 }

403 void PIOS_INTERNAL_ADC_DMA_Handler(void)
404 {
405  if (!PIOS_INTERNAL_ADC_validate(pios_adc_dev))
406  return;
407 
408 #if defined(PIOS_INCLUDE_ADC)
409  /* terminal count, buffer has flipped */
410  if (DMA_GetITStatus(pios_adc_dev->cfg->dma.rx.channel, pios_adc_dev->cfg->full_flag)) {
411  DMA_ClearITPendingBit(pios_adc_dev->cfg->dma.rx.channel, pios_adc_dev->cfg->full_flag);
412 
413  /* accumulate results from the buffer that was just completed */
414  if (DMA_GetCurrentMemoryTarget(pios_adc_dev->cfg->dma.rx.channel) == 0) {
415  accumulate(adc_raw_buffer_0, pios_adc_dev->max_samples);
416  }
417  else {
418  accumulate(adc_raw_buffer_1, pios_adc_dev->max_samples);
419  }
420  }
421 #endif
422 }
