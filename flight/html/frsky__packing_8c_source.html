153 bool frsky_encode_cells(struct frsky_settings *frsky, uint32_t *value, bool test_presence_only, uint32_t arg)
154 {
155  if ((frsky->batt_cell_count == 0) || (frsky->batt_cell_count - 1) < (arg * 2))
156  return false;
157  if (test_presence_only)
158  return true;
159 
160  uint8_t cellCount = 0;
161  FlightBatteryStateDetectedCellCountGet(&cellCount);
162  if(cellCount)
163  frsky->batt_cell_count = cellCount;
164 
165  float voltage = 0;
166  FlightBatteryStateVoltageGet(&voltage);
167 
168  uint32_t cell_voltage = (uint32_t)((voltage * 500.0f) / frsky->batt_cell_count);
169  *value = ((cell_voltage & 0xfff) << 8) | ((arg * 2) & 0x0f) | ((frsky->batt_cell_count << 4) & 0xf0);
170  if (((int16_t)frsky->batt_cell_count - 1) >= (arg * 2 + 1))
171  *value |= ((cell_voltage & 0xfff) << 20);
172 
173  return true;

578 uint8_t frsky_insert_byte(uint8_t *obuff, uint16_t *chk, uint8_t byte)
579 {
580  /* checksum calculation is based on data before byte-stuffing */
581  *chk += byte;
582  *chk += (*chk) >> 8;
583  *chk &= 0x00ff;
584 
585  if (byte == 0x7e || byte == 0x7d) {
586  obuff[0] = 0x7d;
587  obuff[1] = byte &= ~0x20;
588  return 2;
589  }
590 
591  obuff[0] = byte;
592  return 1;
593 }

600 int32_t frsky_send_frame(uintptr_t com, enum frsky_value_id id, uint32_t value,
601  bool send_prelude)
602 {
603  /* each call of frsky_insert_byte can add 2 bytes to the buffer at maximum
604  * and therefore the worst-case is 17 bytes total (the first byte 0x10 won't be
605  * escaped) */
606  uint8_t tx_data[17];
607  uint16_t chk = 0;
608  uint8_t cnt = 0;
609 
610  if (send_prelude) {
611  tx_data[0] = 0x7e;
612  tx_data[1] = 0x98;
613  cnt = 2;
614  }
615 
616  cnt += frsky_insert_byte(&tx_data[cnt], &chk, 0x10);
617  cnt += frsky_insert_byte(&tx_data[cnt], &chk, (uint16_t)id & 0xff);
618  cnt += frsky_insert_byte(&tx_data[cnt], &chk, ((uint16_t)id >> 8) & 0xff);
619  cnt += frsky_insert_byte(&tx_data[cnt], &chk, value & 0xff);
620  cnt += frsky_insert_byte(&tx_data[cnt], &chk, (value >> 8) & 0xff);
621  cnt += frsky_insert_byte(&tx_data[cnt], &chk, (value >> 16) & 0xff);
622  cnt += frsky_insert_byte(&tx_data[cnt], &chk, (value >> 24) & 0xff);
623  cnt += frsky_insert_byte(&tx_data[cnt], &chk, 0xff - chk);
624 
625  PIOS_COM_SendBuffer(com, tx_data, cnt);
626 
627  return cnt;
628 }
