353 static void go_starting(struct pios_i2c_adapter *i2c_adapter)
354 {
355  PIOS_DEBUG_Assert(i2c_adapter->active_txn);
356  PIOS_DEBUG_Assert(i2c_adapter->active_txn <= i2c_adapter->last_txn);
357 
358  i2c_adapter->active_byte = &(i2c_adapter->active_txn->buf[0]);
359  i2c_adapter->last_byte = &(i2c_adapter->active_txn->buf[i2c_adapter->active_txn->len - 1]);
360 
361  I2C_GenerateSTART(i2c_adapter->cfg->regs, ENABLE);
362  if (i2c_adapter->active_txn->rw == PIOS_I2C_TXN_READ) {
363  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_BUF | I2C_IT_ERR, ENABLE);
364  } else {
365  // For write operations, do not enable the IT_BUF events.
366  // The current driver does not act when the TX data register is not full, only when the complete byte is sent.
367  // With the IT_BUF enabled, we constantly get IRQs, See OP-326
368  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_ERR, ENABLE);
369  }
370 }

533 static void i2c_adapter_inject_event(struct pios_i2c_adapter *i2c_adapter, enum i2c_adapter_event event)
534 {
535  PIOS_IRQ_Disable();
536 
537 #if defined(PIOS_I2C_DIAGNOSTICS)
538  i2c_adapter->i2c_state_event_history[i2c_adapter->i2c_state_event_history_pointer] = event;
539  i2c_adapter->i2c_state_event_history_pointer = (i2c_adapter->i2c_state_event_history_pointer + 1) % I2C_LOG_DEPTH;
540 
541  i2c_adapter->i2c_state_history[i2c_adapter->i2c_state_history_pointer] = i2c_adapter->curr_state;
542  i2c_adapter->i2c_state_history_pointer = (i2c_adapter->i2c_state_history_pointer + 1) % I2C_LOG_DEPTH;
543 
544  if (i2c_adapter_transitions[i2c_adapter->curr_state].next_state[event] == I2C_STATE_FSM_FAULT)
545  i2c_adapter_log_fault(i2c_adapter, PIOS_I2C_ERROR_FSM);
546 #endif
547  /*
548  * Move to the next state
549  *
550  * This is done prior to calling the new state's entry function to
551  * guarantee that the entry function never depends on the previous
552  * state. This way, it cannot ever know what the previous state was.
553  */
554  enum i2c_adapter_state prev_state = i2c_adapter->curr_state;
555  if (prev_state) ;
556 
557  i2c_adapter->curr_state = i2c_adapter_transitions[i2c_adapter->curr_state].next_state[event];
558 
559  /* Call the entry function (if any) for the next state. */
560  if (i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn) {
561  i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn(i2c_adapter);
562  }
563 
564  /* Process any AUTO transitions in the FSM */
565  i2c_adapter_process_auto(i2c_adapter);
566 
567  PIOS_IRQ_Enable();
568 }

571 {
572  PIOS_IRQ_Disable();
573 
574  enum i2c_adapter_state prev_state = i2c_adapter->curr_state;
575  if (prev_state) ;
576 
577  while (i2c_adapter_transitions[i2c_adapter->curr_state].next_state[I2C_EVENT_AUTO]) {
578  i2c_adapter->curr_state = i2c_adapter_transitions[i2c_adapter->curr_state].next_state[I2C_EVENT_AUTO];
579

581  if (i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn) {
582  i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn(i2c_adapter);
583  }
584  }
585

595 static bool i2c_adapter_wait_for_stopped(struct pios_i2c_adapter *i2c_adapter)
596 {
597  uint32_t guard;
598 
599  /*
600  * Wait for the bus to return to the stopped state.
601  * This was pulled out of the FSM due to occasional
602  * failures at this transition which previously resulted
603  * in spinning on this bit in the ISR forever.
604  */
605 #define I2C_CR1_STOP_REQUESTED 0x0200
606  for (guard = 1e6; /* FIXME: should use the configured bus timeout */
607  guard && (i2c_adapter->cfg->regs->CR1 & I2C_CR1_STOP_REQUESTED); guard--)
608  continue;
609  if (!guard) {
610  /* We timed out waiting for the stop condition */
611  return false;
612  }
613 
614  return true;
615 }

617 static void i2c_adapter_reset_bus(struct pios_i2c_adapter *i2c_adapter)
618 {
619  static uint8_t retry_count = 0;
620  static uint8_t retry_count_clk = 0;
621 
622  /* Reset the I2C block */
623  I2C_DeInit(i2c_adapter->cfg->regs);
624 
625  /* Make sure the bus is free by clocking it until any slaves release the bus. */
626  GPIO_InitTypeDef scl_gpio_init;
627  scl_gpio_init = i2c_adapter->cfg->scl.init;
628  scl_gpio_init.GPIO_Mode = GPIO_Mode_Out_OD;
629  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
630  GPIO_Init(i2c_adapter->cfg->scl.gpio, &scl_gpio_init);
631 
632  GPIO_InitTypeDef sda_gpio_init;
633  sda_gpio_init = i2c_adapter->cfg->sda.init;
634  sda_gpio_init.GPIO_Mode = GPIO_Mode_Out_OD;
635  GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
636  GPIO_Init(i2c_adapter->cfg->sda.gpio, &sda_gpio_init);
637 
638  /* Check SDA line to determine if slave is asserting bus and clock out if so, this may */
639  /* have to be repeated (due to futher bus errors) but better than clocking 0xFF into an */
640  /* ESC */
641  //bool sda_hung = GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) == Bit_RESET;
642  retry_count_clk = 0;
643  while(GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) == Bit_RESET &&
644  (retry_count_clk++ < MAX_I2C_RETRY_COUNT))
645  {
646  retry_count = 0;
647  /* Set clock high and wait for any clock stretching to finish. */
648  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
649  while (GPIO_ReadInputDataBit(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin) == Bit_RESET &&
650  retry_count++ < MAX_I2C_RETRY_COUNT)
651  PIOS_DELAY_WaituS(1);
652  PIOS_DELAY_WaituS(2);
653 
654  /* Set clock low */
655  GPIO_ResetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
656  PIOS_DELAY_WaituS(2);
657 
658  /* Clock high again */
659  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
660  PIOS_DELAY_WaituS(2);
661  }
662 
663  /* Generate a start then stop condition */
664  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
665  PIOS_DELAY_WaituS(2);
666  GPIO_ResetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
667  PIOS_DELAY_WaituS(2);
668  GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
669  PIOS_DELAY_WaituS(2);
670 
671  /* Set data and clock high and wait for any clock stretching to finish. */
672  GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
673  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
674 
675  retry_count = 0;
676  while (GPIO_ReadInputDataBit(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin) == Bit_RESET &&
677  retry_count++ < MAX_I2C_RETRY_COUNT)
678  PIOS_DELAY_WaituS(1);
679  /* Wait for data to be high */
680  retry_count = 0;
681  while (GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) != Bit_SET &&
682  retry_count++ < MAX_I2C_RETRY_COUNT)
683  PIOS_DELAY_WaituS(1);
684 
685 
686  /* Bus signals are guaranteed to be high (ie. free) after this point */
687  /* Initialize the GPIO pins to the peripheral function */
688  GPIO_Init(i2c_adapter->cfg->scl.gpio, (GPIO_InitTypeDef *)&(i2c_adapter->cfg->scl.init));
689  GPIO_Init(i2c_adapter->cfg->sda.gpio, (GPIO_InitTypeDef *)&(i2c_adapter->cfg->sda.init));
690 
691  /* Reset the I2C block */
692  I2C_DeInit(i2c_adapter->cfg->regs);
693 
694  /* Initialize the I2C block */
695  I2C_Init(i2c_adapter->cfg->regs, (I2C_InitTypeDef *)&(i2c_adapter->cfg->init));
696 
697 #define I2C_BUSY 0x20
698  if (i2c_adapter->cfg->regs->SR2 & I2C_BUSY) {
699  /* Reset the I2C block */
700  I2C_SoftwareResetCmd(i2c_adapter->cfg->regs, ENABLE);
701  I2C_SoftwareResetCmd(i2c_adapter->cfg->regs, DISABLE);
702  }
703 }

857 int32_t PIOS_I2C_Transfer(uint32_t i2c_id, const struct pios_i2c_txn txn_list[], uint32_t num_txns)
858 {
859  struct pios_i2c_adapter * i2c_adapter = (struct pios_i2c_adapter *)i2c_id;
860 
861  bool valid = PIOS_I2C_validate(i2c_adapter);
862  PIOS_Assert(valid)
863 
864  PIOS_DEBUG_Assert(txn_list);
865  PIOS_DEBUG_Assert(num_txns);
866 
867  if (PIOS_Mutex_Lock(i2c_adapter->lock, i2c_adapter->cfg->transfer_timeout_ms) == false)
868  return -2;
869 
870  PIOS_DEBUG_Assert(i2c_adapter->curr_state == I2C_STATE_STOPPED);
871 
872  i2c_adapter->last_txn = &txn_list[num_txns - 1];
873  i2c_adapter->active_txn = &txn_list[0];
874 
875  /* Make sure the done/ready semaphore is consumed before we start */
876  PIOS_Semaphore_Take(i2c_adapter->sem_ready, 0);
877 
878  i2c_adapter->bus_error = false;
879  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_START);
880 
881  /* Wait for the transfer to complete */
882  bool semaphore_success = (PIOS_Semaphore_Take(i2c_adapter->sem_ready, i2c_adapter->cfg->transfer_timeout_ms) == true);
883 
884  /* Spin waiting for the transfer to finish */
885  while (!i2c_adapter_fsm_terminated(i2c_adapter)) ;
886 
887  if (i2c_adapter_wait_for_stopped(i2c_adapter)) {
888  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STOPPED);
889  } else {
890  i2c_adapter_fsm_init(i2c_adapter);
891  }
892 
893 #if defined(PIOS_I2C_DIAGNOSTICS)
894  if (!semaphore_success)
895  i2c_adapter->i2c_timeout_counter++;
896 #endif
897 
898  int32_t result = !semaphore_success ? -2 :
899  i2c_adapter->bus_error ? -1 :
900  0;
901 
902  PIOS_Mutex_Unlock(i2c_adapter->lock);
903 
904  return result;
905 }

908 void PIOS_I2C_EV_IRQ_Handler(uint32_t i2c_id)
909 {
910  struct pios_i2c_adapter * i2c_adapter = (struct pios_i2c_adapter *)i2c_id;
911 
912  bool valid = PIOS_I2C_validate(i2c_adapter);
913  PIOS_Assert(valid)
914 
915  uint32_t event = I2C_GetLastEvent(i2c_adapter->cfg->regs);
916 
917 #if defined(PIOS_I2C_DIAGNOSTICS)
918  /* Store event for diagnostics */
919  i2c_adapter->i2c_evirq_history[i2c_adapter->i2c_evirq_history_pointer] = event;
920  i2c_adapter->i2c_evirq_history_pointer = (i2c_adapter->i2c_evirq_history_pointer + 1) % I2C_LOG_DEPTH;
921 #endif
922 
923 #define EVENT_MASK 0x000700FF
924  event &= EVENT_MASK;
925 
926 
927  switch (event) { /* Mask out all the bits we don't care about */
928  case (I2C_EVENT_MASTER_MODE_SELECT | 0x40):
929  /* Unexplained event: EV5 + RxNE : Extraneous Rx. Probably a late NACK from previous read. */
930  /* Clean up the extra Rx until the root cause is identified and just keep going */
931  (void)I2C_ReceiveData(i2c_adapter->cfg->regs);
932  /* Fall through */
933  case I2C_EVENT_MASTER_MODE_SELECT: /* EV5 */
934  switch (i2c_adapter->active_txn->rw) {
935  case PIOS_I2C_TXN_READ:
936  if (i2c_adapter->active_txn == i2c_adapter->last_txn) {
937  /* Final transaction */
938  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_LAST_TXN_READ);
939  } else if (i2c_adapter->active_txn < i2c_adapter->last_txn) {
940  /* More transactions follow */
941  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_MORE_TXN_READ);
942  } else {
943  PIOS_DEBUG_Assert(0);
944  }
945  break;
946  case PIOS_I2C_TXN_WRITE:
947  if (i2c_adapter->active_txn == i2c_adapter->last_txn) {
948  /* Final transaction */
949  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_LAST_TXN_WRITE);
950  } else if (i2c_adapter->active_txn < i2c_adapter->last_txn) {
951  /* More transactions follow */
952  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_MORE_TXN_WRITE);
953  } else {
954  PIOS_DEBUG_Assert(0);
955  }
956  break;
957  default:
958  PIOS_DEBUG_Assert(0);
959  break;
960  }
961  break;
962  case I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED: /* EV6 */
963  case I2C_EVENT_MASTER_RECEIVER_MODE_SELECTED: /* EV6 */
964  switch (i2c_adapter->last_byte - i2c_adapter->active_byte + 1) {
965  case 0:
966  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_ADDR_SENT_LEN_EQ_0);
967  break;
968  case 1:
969  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_ADDR_SENT_LEN_EQ_1);
970  break;
971  case 2:
