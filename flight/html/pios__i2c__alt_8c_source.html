33 /*
34  * == DO NOT USE FreeRTOS API in PIOS_I2C_EV_IRQ_Handler ==
35  *
36  * Important STM32 Errata:
37  *
38  * See STM32F10xx8 and STM32F10xxB Errata sheet (Doc ID 14574 Rev 8),
39  * Section 2.11.1, 2.11.2.
40  *
41  * 2.11.1:
42  * When the EV7, EV7_1, EV6_1, EV6_3, EV2, EV8, and EV3 events are not
43  * managed before the current byte is being transferred, problems may be
44  * encountered such as receiving an extra byte, reading the same data twice
45  * or missing data.
46  *
47  * 2.11.2:
48  * In Master Receiver mode, when closing the communication using
49  * method 2, the content of the last read data can be corrupted.
50  *
51  * If the user software is not able to read the data N-1 before the STOP
52  * condition is generated on the bus, the content of the shift register
53  * (data N) will be corrupted. (data N is shifted 1-bit to the left).
54  *
55  * ----------------------------------------------------------------------
56  *
57  * In order to ensure that events are not missed, the i2c interrupt must
58  * not be preempted. We set the i2c interrupt priority to be the highest
59  * interrupt in the system (priority level 0 / PIOS_IRQ_PRIO_EXTREME).
60  *
61  * Since priority level 0 is above configMAX_SYSCALL_INTERRUPT_PRIORITY,
62  * we cannot use any API functions in the I2C interrupt handlers (see FreeRTOS doc)
63  *
64  * As a result, we cannot use a semaphore (sem_ready) in PIOS_I2C_EV_IRQ_Handler
65  */

105 static void i2c_adapter_reset_bus(struct pios_i2c_adapter *i2c_adapter)
106 {
107  GPIO_InitTypeDef scl_gpio_init;
108  GPIO_InitTypeDef sda_gpio_init;
109 
110  scl_gpio_init = i2c_adapter->cfg->scl.init;
111  scl_gpio_init.GPIO_Mode = GPIO_Mode_Out_OD;
112  GPIO_Init(i2c_adapter->cfg->scl.gpio, &scl_gpio_init);
113 
114  sda_gpio_init = i2c_adapter->cfg->sda.init;
115  sda_gpio_init.GPIO_Mode = GPIO_Mode_Out_OD;
116  GPIO_Init(i2c_adapter->cfg->sda.gpio, &sda_gpio_init);
117 
118  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
119  GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
120 
121  for (uint8_t i = 0; i < 8; i++) {
122  // Wait for any clock stretching to finish
123  while (!GPIO_ReadInputDataBit(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin)) {
124  PIOS_DELAY_WaituS(10);
125  }
126 
127  // Pull low
128  GPIO_ResetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin); //Set bus low
129  PIOS_DELAY_WaituS(10);
130  // Release high again
131  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin); //Set bus high
132  PIOS_DELAY_WaituS(10);
133  }
134 
135  // Generate a start then stop condition
136  GPIO_ResetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin); // Set bus data low
137  PIOS_DELAY_WaituS(10);
138  GPIO_ResetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin); // Set bus scl low
139  PIOS_DELAY_WaituS(10);
140  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin); // Set bus scl high
141  PIOS_DELAY_WaituS(10);
142  GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin); // Set bus sda high
143 
144  // Init pins
145  GPIO_Init(i2c_adapter->cfg->scl.gpio, (GPIO_InitTypeDef*)&(i2c_adapter->cfg->scl.init));
146  GPIO_Init(i2c_adapter->cfg->sda.gpio, (GPIO_InitTypeDef*)&(i2c_adapter->cfg->sda.init));
147 }

154 {

168  // Enable I2C peripheral
169  I2C_Cmd(i2c_adapter->cfg->regs, ENABLE);

207 {
208  PIOS_DEBUG_Assert(i2c_id);
209  PIOS_DEBUG_Assert(cfg);
210 
211  struct pios_i2c_adapter * i2c_adapter;
212 
213  i2c_adapter = (struct pios_i2c_adapter *) PIOS_I2C_alloc();
214  if (!i2c_adapter) {
215  return(-1);
216  }
217 
218  /* Bind the configuration to the device instance */
219  i2c_adapter->cfg = cfg;
220 
221  i2c_adapter->sem_busy = PIOS_Semaphore_Create();
222  i2c_adapter->sem_ready = NULL;
223 
224  /* Enable the associated peripheral clock */
225  switch ((uint32_t) i2c_adapter->cfg->regs) {
226  case (uint32_t) I2C1:
227  /* Enable I2C peripheral clock (APB1 == slow speed) */
228  RCC_APB1PeriphClockCmd(RCC_APB1Periph_I2C1, ENABLE);
229  break;
230  case (uint32_t) I2C2:
231  /* Enable I2C peripheral clock (APB1 == slow speed) */
232  RCC_APB1PeriphClockCmd(RCC_APB1Periph_I2C2, ENABLE);
233  break;
234  }
235 
236  if (i2c_adapter->cfg->remap) {
237  GPIO_PinRemapConfig(i2c_adapter->cfg->remap, ENABLE);
238  }
239 
240  /* Initialize the state machine */
241  i2c_adapter_fsm_init(i2c_adapter);
242 
243  *i2c_id = (uint32_t)i2c_adapter;
244 
245  /* Configure and enable I2C interrupts */
246  NVIC_Init((NVIC_InitTypeDef*)&(i2c_adapter->cfg->event.init));
247  NVIC_Init((NVIC_InitTypeDef*)&(i2c_adapter->cfg->error.init));
248 
249  /* No error */
250  return 0;
251 }

259 int32_t PIOS_I2C_Transfer(uint32_t i2c_id, const struct pios_i2c_txn txn_list[], uint32_t num_txns)
260 {
261  struct pios_i2c_adapter * i2c_adapter = (struct pios_i2c_adapter *)i2c_id;
262 
263  bool valid = PIOS_I2C_validate(i2c_adapter);
264  PIOS_Assert(valid)
265 
266  struct i2c_internal_state* state = (struct i2c_internal_state*)i2c_adapter->active_txn;
267 
268  PIOS_DEBUG_Assert(txn_list);
269  PIOS_DEBUG_Assert(num_txns);
270 
271  PIOS_DEBUG_Assert(num_txns == 1 || num_txns == 2);
272  if (num_txns < 1 || num_txns > 2) {
273  return -1;
274  }
275 
276  if (PIOS_Semaphore_Take(i2c_adapter->sem_busy, i2c_adapter->cfg->transfer_timeout_ms) == false) {
277 #if defined(I2C_HALT_ON_ERRORS)
278  PIOS_DEBUG_Assert(0);
279 #endif /*I2C_HALT_ON_ERRORS*/
280  return -2;
281  }
282 
283  i2c_adapter->bus_error = false;
284 
285  state->addr = txn_list[0].addr << 1;
286  state->reg = txn_list[0].buf[0];
287 
288  // Reading
289  if (num_txns == 2) {
290  state->dir = I2C_Direction_Receiver;
291  state->buf_p = txn_list[1].buf; // buf
292  state->bytes = txn_list[1].len; // len
293  }
294  // Writing
295  else if (num_txns == 1) {
296  state->dir = I2C_Direction_Transmitter;
297  state->buf_p = &(txn_list[0].buf[1]); // buf
298  state->bytes = txn_list[0].len - 1; // len
299  }
300 
301  state->busy = true;
302  state->subaddress_sent = false;
303 
304  PIOS_DEBUG_Assert(!(i2c_adapter->cfg->regs->CR2 & I2C_IT_EVT));
305  PIOS_DEBUG_Assert(!(i2c_adapter->cfg->regs->CR1 & 0x0100));
306 
307  // wait for any stop to finish sending
308  while (i2c_adapter->cfg->regs->CR1 & 0x0200) {
309  PIOS_DELAY_WaituS(10);
310  }
311 
312  // send the start for the new job
313  I2C_GenerateSTART(i2c_adapter->cfg->regs, ENABLE);
314 
315  // allow the interrupts to fire off again
316  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_ERR, ENABLE);
317 
318  uint32_t timeout = I2C_DEFAULT_TIMEOUT;
319  while (state->busy && --timeout > 0);
320  if (timeout == 0) {
321  state->error_count++;
322  // reinit peripheral + clock out garbage
323  i2c_adapter_fsm_init(i2c_adapter);
324 #if defined(I2C_HALT_ON_ERRORS)
325  PIOS_DEBUG_Assert(0);
326 #endif /*I2C_HALT_ON_ERRORS*/
327  }
328 
329 #if defined(I2C_HALT_ON_ERRORS)
330  PIOS_DEBUG_Assert(!i2c_adapter->bus_error);
331 #endif
332 
333  PIOS_Semaphore_Give(i2c_adapter->sem_busy);
334 
335  return (timeout == 0) ? -2 :
336  i2c_adapter->bus_error ? -1 :
337  0;
338 }

364 void PIOS_I2C_EV_IRQ_Handler(uint32_t i2c_id)
365 {
366  struct pios_i2c_adapter * i2c_adapter = (struct pios_i2c_adapter *)i2c_id;
367 
368  bool valid = PIOS_I2C_validate(i2c_adapter);
369  PIOS_Assert(valid);
370 
371  struct i2c_internal_state* state = (struct i2c_internal_state*)i2c_adapter->active_txn;
372 
373  uint8_t SReg_1 = i2c_adapter->cfg->regs->SR1; //read the status register here
374 
375  if (SReg_1 & 0x0001) { //we just sent a start - EV5 in ref manual
376  i2c_adapter->cfg->regs->CR1 &= ~0x0800; //reset the POS bit so ACK/NACK applied to the current byte
377  I2C_AcknowledgeConfig(i2c_adapter->cfg->regs, ENABLE); //make sure ACK is on
378  state->index = 0; //reset the index
379  if ((state->dir == I2C_Direction_Receiver) && (state->subaddress_sent || 0xFF == state->reg)) { //we have sent the subaddr
380  state->subaddress_sent = true; //make sure this is set in case of no subaddress, so following code runs correctly
381  if (state->bytes == 2)
382  i2c_adapter->cfg->regs->CR1 |= 0x0800; //set the POS bit so NACK applied to the final byte in the two byte read
383  I2C_Send7bitAddress(i2c_adapter->cfg->regs, state->addr, I2C_Direction_Receiver); //send the address and set hardware mode
384  } else { //direction is Tx, or we havent sent the sub and rep start
385  I2C_Send7bitAddress(i2c_adapter->cfg->regs, state->addr, I2C_Direction_Transmitter); //send the address and set hardware mode
386  if (state->reg != 0xFF) //0xFF as subaddress means it will be ignored, in Tx or Rx mode
387  state->index = -1; //send a subaddress
388  }
389  } else if (SReg_1 & 0x0002) { //we just sent the address - EV6 in ref manual
390  // Read SR1,2 to clear ADDR
391  __DMB(); // memory fence to control hardware
392  if (state->bytes == 1 && (state->dir == I2C_Direction_Receiver) && state->subaddress_sent) { // we are receiving 1 byte - EV6_3
393  I2C_AcknowledgeConfig(i2c_adapter->cfg->regs, DISABLE); // turn off ACK
394  __DMB();
395  (void)i2c_adapter->cfg->regs->SR2; // clear ADDR after ACK is turned off
396  I2C_GenerateSTOP(i2c_adapter->cfg->regs, ENABLE); // program the stop
397  state->final_stop = true;
398  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, ENABLE); // allow us to have an EV7
399  } else { // EV6 and EV6_1
400  (void)i2c_adapter->cfg->regs->SR2; // clear the ADDR here
401  __DMB();
402  if (state->bytes == 2 && (state->dir == I2C_Direction_Receiver) && state->subaddress_sent) { //rx 2 bytes - EV6_1
403  I2C_AcknowledgeConfig(i2c_adapter->cfg->regs, DISABLE); //turn off ACK
404  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, DISABLE); //disable TXE to allow the buffer to fill
405  } else if (state->bytes == 3 && (state->dir == I2C_Direction_Receiver) && state->subaddress_sent) //rx 3 bytes
406  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, DISABLE); //make sure RXNE disabled so we get a BTF in two bytes time
407  else //receiving greater than three bytes, sending subaddress, or transmitting
408  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, ENABLE);
409  }
410  } else if (SReg_1 & 0x004) { //Byte transfer finished - EV7_2, EV7_3 or EV8_2
411  state->final_stop = true;
412  if ((state->dir == I2C_Direction_Receiver) && state->subaddress_sent) { //EV7_2, EV7_3
413  if (state->bytes > 2) { //EV7_2
414  I2C_AcknowledgeConfig(i2c_adapter->cfg->regs, DISABLE); //turn off ACK
415  state->buf_p[state->index++] = I2C_ReceiveData(i2c_adapter->cfg->regs); //read data N-2
416  I2C_GenerateSTOP(i2c_adapter->cfg->regs, ENABLE); //program the Stop
417  state->final_stop = true; //reuired to fix hardware
418  state->buf_p[state->index++] = I2C_ReceiveData(i2c_adapter->cfg->regs); //read data N-1
419  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, ENABLE); //enable TXE to allow the final EV7
420  } else { //EV7_3
421  if (state->final_stop)
422  I2C_GenerateSTOP(i2c_adapter->cfg->regs, ENABLE); //program the Stop
423  else
424  I2C_GenerateSTART(i2c_adapter->cfg->regs, ENABLE); //program a rep start
425  state->buf_p[state->index++] = I2C_ReceiveData(i2c_adapter->cfg->regs); //read data N-1
426  state->buf_p[state->index++] = I2C_ReceiveData(i2c_adapter->cfg->regs); //read data N
427  state->index++; //to show job completed
428  }
429  } else { //EV8_2, which may be due to a subaddress sent or a write completion
430  if (state->subaddress_sent || ((state->dir == I2C_Direction_Transmitter))) {
431  if (state->final_stop)
432  I2C_GenerateSTOP(i2c_adapter->cfg->regs, ENABLE); //program the Stop
433  else
434  I2C_GenerateSTART(i2c_adapter->cfg->regs, ENABLE); //program a rep start
435  state->index++; //to show that the job is complete
436  } else { //We need to send a subaddress
437  I2C_GenerateSTART(i2c_adapter->cfg->regs, ENABLE); //program the repeated Start
438  state->subaddress_sent = true; //this is set back to zero upon completion of the current task
439  }
440  }
441  //we must wait for the start to clear, otherwise we get constant BTF
442  while (i2c_adapter->cfg->regs->CR1 & 0x0100) { ; }
443  } else if (SReg_1 & 0x0040) { //Byte received - EV7
444  state->buf_p[state->index++] = I2C_ReceiveData(i2c_adapter->cfg->regs);
445  if (state->bytes == (state->index + 3))
446  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, DISABLE); //disable TXE to allow the buffer to flush so we can get an EV7_2
447  if (state->bytes == state->index) //We have completed a final EV7
448  state->index++; //to show job is complete
449  } else if (SReg_1 & 0x0080) { //Byte transmitted -EV8/EV8_1
450  if (state->index != -1) { //we dont have a subaddress to send
451  I2C_SendData(i2c_adapter->cfg->regs, state->buf_p[state->index++]);
452  if (state->bytes == state->index) //we have sent all the data
453  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, DISABLE); //disable TXE to allow the buffer to flush
454  } else {
455  state->index++;
456  I2C_SendData(i2c_adapter->cfg->regs, state->reg); //send the subaddress
457  if ((state->dir == I2C_Direction_Receiver) || !state->bytes) //if receiving or sending 0 bytes, flush now
458  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, DISABLE); //disable TXE to allow the buffer to flush
459  }
460  }
461  if (state->index == state->bytes + 1) { //we have completed the current job
462  //Completion Tasks go here
463  //End of completion tasks
464  state->subaddress_sent = false; //reset this here
465  // i2c_adapter->cfg->regs->CR1 &= ~0x0800; //reset the POS bit so NACK applied to the current byte
466  if (state->final_stop) //If there is a final stop and no more jobs, bus is inactive, disable interrupts to prevent BTF
467  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_ERR, DISABLE); //Disable EVT and ERR interrupts while bus inactive
468  state->busy = false;
469  }
470 }

476 void PIOS_I2C_ER_IRQ_Handler(uint32_t i2c_id)
477 {
478  struct pios_i2c_adapter * i2c_adapter = (struct pios_i2c_adapter *)i2c_id;
479 
480  bool valid = PIOS_I2C_validate(i2c_adapter);
481  PIOS_Assert(valid);
482 
483  struct i2c_internal_state* state = (struct i2c_internal_state*)i2c_adapter->active_txn;
484 
485  volatile uint32_t SR1Register;
486  // Read the I2C1 status register
487  SR1Register = i2c_adapter->cfg->regs->SR1;
488  if (SR1Register & 0x0F00) { //an error
489  i2c_adapter->bus_error = true;
490  // I2C1error.error = ((SR1Register & 0x0F00) >> 8); //save error
491  // I2C1error.job = job; //the task
492  }
493  // If AF, BERR or ARLO, abandon the current job and commence new if there are jobs
494  if (SR1Register & 0x0700) {
495  (void)i2c_adapter->cfg->regs->SR2; // read second status register to clear ADDR if it is set (note that BTF will not be set after a NACK)
496  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_BUF, DISABLE); // disable the RXNE/TXE interrupt - prevent the ISR tailchaining onto the ER (hopefully)
497  if (!(SR1Register & 0x0200) && !(i2c_adapter->cfg->regs->CR1 & 0x0200)) { // if we dont have an ARLO error, ensure sending of a stop
498  if (i2c_adapter->cfg->regs->CR1 & 0x0100) { // We are currently trying to send a start, this is very bad as start,stop will hang the peripheral
499  while (i2c_adapter->cfg->regs->CR1 & 0x0100); // wait for any start to finish sending
