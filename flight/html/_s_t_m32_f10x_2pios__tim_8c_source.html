188 static void PIOS_TIM_generic_irq_handler(TIM_TypeDef * timer)
189 {
190  uint16_t flags = timer->SR;
191 
192  /* While the datasheet is not 100% clear that this (very common
193  * paradigm) is correct TIM_ClearITPendingBit from stdperiph shows
194  * that software is unable to set bits in this register.
195  *
196  * Therefore, clear only the events we're going to process.
197  * Otherwise we could lose an overflow just after a rising edge,
198  * which would offset our count.
199  */
200  timer->SR = ~flags;
201 
202  /* Check for an overflow event on this timer */
203  bool overflow_event;
204  uint16_t overflow_count;
205  if (flags & TIM_IT_Update) {
206  overflow_count = timer->ARR;
207  overflow_event = true;
208  } else {
209  overflow_count = 0;
210  overflow_event = false;
211  }
212 
213  /* Iterate over all registered clients of the TIM layer to find channels on this timer */
214  for (uint8_t i = 0; i < pios_tim_num_devs; i++) {
215  const struct pios_tim_dev * tim_dev = &pios_tim_devs[i];
216 
217  if (!tim_dev->channels || tim_dev->num_channels == 0) {
218  /* No channels to process on this client */
219  continue;
220  }
221 
222  for (uint8_t chan_num = 0; chan_num < tim_dev->num_channels; chan_num++) {
223  const struct pios_tim_channel *chan = &tim_dev->channels[chan_num];
224 
225  if (chan->timer != timer) {
226  /* channel is not on this timer */
227  continue;
228  }
229 
230  /* Figure out which interrupt bit we should be looking at */
231  uint16_t timer_it;
232  switch (chan->timer_chan) {
233  case TIM_Channel_1:
234  timer_it = TIM_IT_CC1;
235  break;
236  case TIM_Channel_2:
237  timer_it = TIM_IT_CC2;
238  break;
239  case TIM_Channel_3:
240  timer_it = TIM_IT_CC3;
241  break;
242  case TIM_Channel_4:
243  timer_it = TIM_IT_CC4;
244  break;
245  default:
246  PIOS_Assert(0);
247  break;
248  }
249 
250  bool edge_event;
251  uint16_t edge_count;
252  if (flags & timer_it) {
253  /* Read the current counter */
254  switch(chan->timer_chan) {
255  case TIM_Channel_1:
256  edge_count = TIM_GetCapture1(chan->timer);
257  break;
258  case TIM_Channel_2:
259  edge_count = TIM_GetCapture2(chan->timer);
260  break;
261  case TIM_Channel_3:
262  edge_count = TIM_GetCapture3(chan->timer);
263  break;
264  case TIM_Channel_4:
265  edge_count = TIM_GetCapture4(chan->timer);
266  break;
267  default:
268  PIOS_Assert(0);
269  break;
270  }
271  edge_event = true;
272  } else {
273  edge_event = false;
274  edge_count = 0;
275  }
276 
277  if (!tim_dev->callbacks) {
278  /* No callbacks registered, we're done with this channel */
279  continue;
280  }
281 
282  /* Generate the appropriate callbacks */
283  if (overflow_event & edge_event) {
284  /*
285  * When both edge and overflow happen in the same interrupt, we
286  * need a heuristic to determine the order of the edge and overflow
287  * events so that the callbacks happen in the right order. If we
288  * get the order wrong, our pulse width calculations could be off by up
289  * to ARR ticks. That could be bad.
290  *
291  * Heuristic: If the edge_count is in the last half of the timer period,
292  * assume it happened before the overflow.
293  */
294 
295  if (edge_count < chan->timer->ARR / 2) {
296  /* Call the overflow callback first */
297  if (tim_dev->callbacks->overflow) {
298  (*tim_dev->callbacks->overflow)((uintptr_t)tim_dev,
299  tim_dev->context,
300  chan_num,
301  overflow_count);
302  }
303  /* Call the edge callback second */
304  if (tim_dev->callbacks->edge) {
305  (*tim_dev->callbacks->edge)((uintptr_t)tim_dev,
306  tim_dev->context,
307  chan_num,
308  edge_count);
309  }
310  } else {
311  /* Call the edge callback first */
312  if (tim_dev->callbacks->edge) {
313  (*tim_dev->callbacks->edge)((uintptr_t)tim_dev,
314  tim_dev->context,
315  chan_num,
316  edge_count);
317  }
318  /* Call the overflow callback second */
319  if (tim_dev->callbacks->overflow) {
320  (*tim_dev->callbacks->overflow)((uintptr_t)tim_dev,
321  tim_dev->context,
322  chan_num,
323  overflow_count);
324  }
325  }
326  } else if (overflow_event && tim_dev->callbacks->overflow) {
327  (*tim_dev->callbacks->overflow)((uintptr_t)tim_dev,
328  tim_dev->context,
329  chan_num,
330  overflow_count);
331  } else if (edge_event && tim_dev->callbacks->edge) {
332  (*tim_dev->callbacks->edge)((uintptr_t)tim_dev,
333  tim_dev->context,
334  chan_num,
335  edge_count);
336  }
337  }
338  }
339 }
