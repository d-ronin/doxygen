129 static bool processUsbActivity(bool seen_active)
130 {
131  static volatile uint32_t usb_last_active;
132 
133  if (seen_active) {
134  usb_last_active = PIOS_Thread_Systime();
135 
136  return true;
137  }
138 
139  if (usb_last_active) {
140  if (!PIOS_Thread_Period_Elapsed(usb_last_active,
141  USB_ACTIVITY_TIMEOUT_MS)) {
142  return true;
143  } else {
144  /* "Latch" expiration so it doesn't become true
145  * again.
146  */
147  usb_last_active = 0;
148  }
149  }

151  return false;

223 static int32_t RadioComBridgeInitialize(void)
224 {
225  // allocate and initialize the static data storage only if module is enabled
226  data = PIOS_malloc(sizeof(RadioComBridgeData));

234

304 static void radioRxTask( __attribute__ ((unused))
305  void *parameters)
306 {
307  uint32_t time_since_inject = PIOS_Thread_Systime();
308 
309  // Task loop
310  while (1) {
311 #ifdef PIOS_INCLUDE_WDG
312  PIOS_WDG_UpdateFlag(PIOS_WDG_RADIORX);
313 #endif
314  if (PIOS_COM_RADIOBRIDGE &&
315  PIOS_COM_Available(PIOS_COM_RADIOBRIDGE)) {
316  uint8_t serial_data[32];
317  uint16_t bytes_to_process =
318  PIOS_COM_ReceiveBuffer(PIOS_COM_RADIOBRIDGE,
319  serial_data,
320  sizeof(serial_data),
321  MAX_PORT_DELAY);
322  if (bytes_to_process > 0) {
323  // Pass the data through the UAVTalk parser.
324  for (uint8_t i = 0;
325  i < bytes_to_process; i++) {
326  ProcessRadioStream(data->radioUAVTalkCon,
327  data->telemUAVTalkCon,
328  serial_data[i]);
329  }
330  }
331 
332  /* periodically inject ComBridgeStats to downstream */
333  if (PIOS_Thread_Period_Elapsed(time_since_inject,
334  COMSTATS_INJECT)) {
335  time_since_inject = PIOS_Thread_Systime();
336 
337  updateRadioComBridgeStats();
338 
339  UAVTalkSendObject(data->telemUAVTalkCon,
340  RadioComBridgeStatsHandle(),
341  0, false);
342  }
343  } else {
344  /* hand port to telemetry */
345  data->have_port = false;
346  PIOS_Thread_Sleep(MAX_PORT_DELAY + 5);
347  telemetry_set_inhibit(false);
348  }
349  }
350 }

357 static void telemetryRxTask( __attribute__ ((unused))
358  void *parameters)
359 {
360  // Task loop
361  while (1) {
362  uint32_t inputPort = getComPort();
363 
364 #ifdef PIOS_INCLUDE_WDG
365  PIOS_WDG_UpdateFlag(PIOS_WDG_TELEMETRY);
366 #endif
367 
368  if ((!PIOS_COM_Available(PIOS_COM_RADIOBRIDGE)) ||
369  (!data->have_port)) {
370 #ifndef PIPXTREME
371  PIOS_Thread_Sleep(5);
372  continue;
373 #else
374  /* PIPXtreme telemetry never touches serial
375  * port for safety, so we can continue to
376  * use it happily.
377  */
378  inputPort = PIOS_COM_TELEM_SER;
379 #endif
380  }
381 
382  if (inputPort) {
383  uint8_t serial_data[32];
384  uint16_t bytes_to_process =
385  PIOS_COM_ReceiveBuffer(inputPort, serial_data,
386  sizeof(serial_data),
387  MAX_PORT_DELAY);
388 
389  if (bytes_to_process > 0) {
390  if (inputPort == PIOS_COM_TELEM_USB) {
391  processUsbActivity(true);
392  }
393 
394  for (uint8_t i = 0; i < bytes_to_process;
395  i++) {
396  ProcessLocalStream(
397  data->telemUAVTalkCon,
398  data->radioUAVTalkCon,
399  serial_data[i]);
400  }
401  }
402  } else {
403  PIOS_Thread_Sleep(5);
404  }
405  }
406 }

464 static void ProcessLocalStream(UAVTalkConnection inConnectionHandle,
465  UAVTalkConnection outConnectionHandle,
466  uint8_t rxbyte)
467 {
468  // Keep reading until we receive a completed packet.
469  UAVTalkRxState state =
470  UAVTalkProcessInputStreamQuiet(inConnectionHandle, rxbyte);
471 
472  if (state == UAVTALK_STATE_COMPLETE) {
473  PIOS_ANNUNC_Toggle(PIOS_LED_RX);
474 
475  uint32_t objId = UAVTalkGetPacketObjId(inConnectionHandle);
476  switch (objId) {
477  // Ignore object...
478  // These objects are shadowed and are not sent over the
479  // transmitted over the radio link.
480  // - HWTauLink : No reconfiguring remote HW over radio
481  // link
482  // - UAVTalkReceiver : We generate this ourselves on
483  // the RX side and shouldn't forward it.
484  case HWTAULINK_OBJID:
485  case MetaObjectId(HWTAULINK_OBJID):
486  case UAVTALKRECEIVER_OBJID:
487  case MetaObjectId(UAVTALKRECEIVER_OBJID):
488  return;
489  default:
490  break;
491  }
492 
493  // all packets are transparently relayed to the remote modem
494  // Incidentally this means that we fail requests for the
495  // radio stats and only telemeter them, but I consider this
496  // okee-dokee.
497  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
498  }
499 }

508 static void ProcessRadioStream(UAVTalkConnection inConnectionHandle,
509  UAVTalkConnection outConnectionHandle,
510  uint8_t rxbyte)
511 {
512  // Keep reading until we receive a completed packet.
513  UAVTalkRxState state =
514  UAVTalkProcessInputStreamQuiet(inConnectionHandle, rxbyte);
515 
516  if (state == UAVTALK_STATE_COMPLETE) {
517  if (!data->have_port) {
518  telemetry_set_inhibit(true);
519  data->have_port = true;
520  }
521 
522  // We only want to unpack certain objects from the remote modem
523  // Similarly we only want to relay certain objects to the telemetry port
524  uint32_t objId = UAVTalkGetPacketObjId(inConnectionHandle);
525  switch (objId) {
526  case HWTAULINK_OBJID:
527  case MetaObjectId(HWTAULINK_OBJID):
528  case UAVTALKRECEIVER_OBJID:
529  case MetaObjectId(UAVTALKRECEIVER_OBJID):
530  break;
531  case FLIGHTBATTERYSTATE_OBJID:
532  case FLIGHTSTATUS_OBJID:
533  case POSITIONACTUAL_OBJID:
534  case VELOCITYACTUAL_OBJID:
535  case BAROALTITUDE_OBJID:
536  // process / store locally for relaying to taranis
537  UAVTalkReceiveObject(inConnectionHandle);
538  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
539  break;
540  default:
541  // all other packets are relayed to the telemetry port
542  UAVTalkRelayPacket(inConnectionHandle,
543  outConnectionHandle);
544  break;
545  }
546  }
