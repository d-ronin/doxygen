225 static void vtol_fsm_timer_set(int32_t ms)
226 {
227  if (ms) {
228  uint32_t now = PIOS_Thread_Systime();
229  vtol_fsm_timer_expiration = ms+now;
230 
231  // If we wrap and get very unlucky, make sure we still
232  // have a timer 1ms later.
233  if (vtol_fsm_timer_expiration == 0) {
234  vtol_fsm_timer_expiration++;
235  }
236  } else {
237  vtol_fsm_timer_expiration = 0;
238  }
239 }
240 
245 static bool vtol_fsm_timer_expired() {
246  /* If there's a timer... */
247  if (vtol_fsm_timer_expiration > 0) {
248  /* See if it expires. */
249 
250  uint32_t now = PIOS_Thread_Systime();
251  uint32_t interval = vtol_fsm_timer_expiration - now;
252 
253  /* If it has expired, this will wrap around and be a big
254  * number. Use a windowed scheme to detect this:
255  * Assume we will run at least every 0x10000000 us
256  * (3 days) and timeouts can't exceed 0xf0000000 us (46 days)
257  */
258  if (interval > 0xf0000000) {
259  return true;
260  }
261  }
262 
263  return false;
264 }

285 static int vtol_fsm_enter_state(enum vtol_fsm_state state)
286 {
287  if (state != FSM_STATE_UNCHANGED) {
288  curr_state = state;
289 
290  /* Call the entry function (if any) for the next state. */
291  if (current_goal[curr_state].entry_fn) {
292  current_goal[curr_state].entry_fn();
293  }
294 
295  /* 0 disables any pending timeout, otherwise it's set to the
296  * value for this state */
297  vtol_fsm_timer_set(current_goal[curr_state].timeout);
298 
299  return 1;
300  }
301 
302  return 0;
303 }

310 static bool vtol_fsm_process_event(enum vtol_fsm_event event)
311 {
312  enum vtol_fsm_state next = current_goal[curr_state].next_state[event];
313 
314  /* Special if condition to not have to explicitly define auto
315  * (don't do fault transitions on auto) */
316  if ((event != FSM_EVENT_AUTO) || (next != FSM_STATE_FAULT)) {
317  return vtol_fsm_enter_state(next);
318  }
319 
320  return false;
321 }

495 static int32_t do_loiter()
496 {
497  float att_adj[2] = { 0, 0 };
498  float hold_pos[3] = {
499  vtol_hold_position_ned[0],

501  vtol_hold_position_ned[2]
502  };
503 
504  float alt_adj = 0;
505 
506  if (vtol_follower_control_loiter(DT, hold_pos, att_adj, &alt_adj)) {
507  // If hold position changed, use it!
508  // We follow this conditional just to avoid unnecessarily
509  // spamming updates to the PositionDesired object.

511  hold_position(hold_pos[0], hold_pos[1], hold_pos[2]);

515  alt_adj) == 0) {
516  if (vtol_follower_control_attitude(DT, att_adj) == 0) {
517  return 0;

549 static int32_t do_ph_climb()
550 {
551  float cur_down;
552  PositionActualDownGet(&cur_down);
553 
554  int32_t ret_val;
555 
556  const float err = fabsf(cur_down - vtol_hold_position_ned[2]);
557 
558  if (err < RTH_ALT_ERROR) {
559  // If we're close to desired altitude, use poshold logic
560  // until timer expires.
561  ret_val = do_hold();
562 
563  if (vtol_fsm_timer_expired()) {
564  vtol_fsm_inject_event(FSM_EVENT_HIT_TARGET);
565  }
566  } else {
567  // If we are low, control for altitude change speed.
568  // If we are high, use the normal control loop.
569  if (cur_down > vtol_hold_position_ned[2]) {
570  ret_val = do_slow_altitude_change(-RTH_CLIMB_SPEED);
571  } else {
572  ret_val = do_hold();
573  }
574  }
575 
576  return ret_val;
577 }

668 static void go_enable_fly_home()
669 {
670  PositionActualData positionActual;
671  PositionActualGet(&positionActual);
672 
673  // Set start position at current position
674  vtol_fsm_path_desired.Start[0] = positionActual.North;
675  vtol_fsm_path_desired.Start[1] = positionActual.East;
676  vtol_fsm_path_desired.Start[2] = positionActual.Down;
677 
678  // Set end position above home
679  vtol_fsm_path_desired.End[0] = 0;
680  vtol_fsm_path_desired.End[1] = 0;
681  vtol_fsm_path_desired.End[2] = positionActual.Down;
682 
683  if (positionActual.Down > -RTH_MIN_ALTITUDE) {
684  vtol_fsm_path_desired.Start[2] = -RTH_MIN_ALTITUDE;
685  vtol_fsm_path_desired.End[2] = -RTH_MIN_ALTITUDE;
686  }
687 
688  /* Paranoia: set to 3.0f just in case calls below fail. TODO: change
689  * UAVO semantics to be guaranteed to either succeed or crash */
690  float rth_vel = 3.0f;
691  VtolPathFollowerSettingsReturnToHomeVelGet(&rth_vel);
692 
693  vtol_fsm_path_desired.StartingVelocity = rth_vel;
694  vtol_fsm_path_desired.EndingVelocity = rth_vel;
695 
696  vtol_fsm_path_desired.Mode = PATHDESIRED_MODE_VECTOR;
697  vtol_fsm_path_desired.ModeParameters = 0;
698 
699  /* It's necessary that this increment so that we don't end up
700  * latching completion status. Wraparound, etc, is OK.
701  * This is for compatibility with the waypoint handshaking in the
702  * path planner.
703  */
704  vtol_fsm_path_desired.Waypoint++;
705 
706  PathDesiredSet(&vtol_fsm_path_desired);
707 }
