159 int32_t PIOS_LIS3MDL_SPI_Init(lis3mdl_dev_t *dev, uint32_t spi_id,
160  uint32_t slave_mag, const struct pios_lis3mdl_cfg *cfg)
161 {
162  lis3mdl_dev_t lis_dev = PIOS_LIS_Alloc(cfg);
163  if (lis_dev == NULL)
164  return -1;
165  *dev = lis_dev;
166 
167  lis_dev->spi_id = spi_id;
168  lis_dev->spi_slave_mag = slave_mag;
169  lis_dev->cfg = cfg;
170 
171  int32_t ret;
172 
173  /* Unfortunately can't check mag right away, so proceed with
174  * init
175  */
176 
177  /* Verify expected chip id. */
178  ret = AssertReg(lis_dev, LIS_REG_MAG_WHO_AM_I,
179  LIS_WHO_AM_I_VAL);
180 
181  if (ret) {
182  return ret;
183  }
184 
185  /* Startup sequence, per ST AN4602 */
186 
187  /* First, write CTRL_REG2 to set scale */
188  ret = PIOS_LIS_WriteReg(lis_dev, LIS_REG_MAG_CTRL2,
189  LIS_CTRL2_FS_12GAU);
190 
191  if (ret) {
192  return ret;
193  }
194 
195  /* Next, set UHP on X/Y and ODR=80. Enable temp sensor */
196  ret = PIOS_LIS_WriteReg(lis_dev, LIS_REG_MAG_CTRL1,
197  LIS_CTRL1_OPMODE_UHP |
198  LIS_CTRL1_ODR_80 |
199  LIS_CTRL1_TEMPEN);
200 
201  if (ret) {
202  return ret;
203  }
204 
205  /* Next, enable UHP on Z axis */
206  ret = PIOS_LIS_WriteReg(lis_dev, LIS_REG_MAG_CTRL4,
207  LIS_CTRL4_OPMODEZ_UHP);
208 
209  if (ret) {
210  return ret;
211  }
212 
213  /* Enable "Block data update" mode which prevents skew between
214  * LSB / MSB bytes of sensor value */
215  ret = PIOS_LIS_WriteReg(lis_dev, LIS_REG_MAG_CTRL5,
216  LIS_CTRL5_BDU);
217 
218  if (ret) {
219  return ret;
220  }
221 
222  /* Finally, enable continuous measurement mode with these parameters */
223  ret = PIOS_LIS_WriteReg(lis_dev, LIS_REG_MAG_CTRL3,
224  LIS_CTRL3_MODE_CONTINUOUS);
225 
226  if (ret) {
227  return ret;
228  }
229 
230  lis_dev->task_handle = PIOS_Thread_Create(
231  PIOS_LIS_Task, "pios_lis", PIOS_LIS_TASK_STACK,
232  lis_dev, PIOS_LIS_TASK_PRIORITY);
233 
234  PIOS_SENSORS_Register(PIOS_SENSOR_MAG, lis_dev->mag_queue);
235 
236  return 0;
237 }

264 static int32_t PIOS_LIS_ReadReg(lis3mdl_dev_t lis_dev,
265  uint8_t address, uint8_t *buffer)
266 {
267  PIOS_Assert(!(address & LIS_ADDRESS_READ));
268  PIOS_Assert(!(address & LIS_ADDRESS_AUTOINCREMENT));
269 
270  if (PIOS_LIS_ClaimBus(lis_dev) != 0)
271  return -1;
272 
273  PIOS_SPI_TransferByte(lis_dev->spi_id, LIS_ADDRESS_READ |
274  address); // request byte
275  *buffer = PIOS_SPI_TransferByte(lis_dev->spi_id, 0); // receive response
276 
277  PIOS_LIS_ReleaseBus(lis_dev);
278 
279  return 0;
280 }

282 static int32_t PIOS_LIS_WriteReg(lis3mdl_dev_t lis_dev,
283  uint8_t address, uint8_t buffer)
284 {
285  PIOS_Assert(!PIOS_LIS_Validate(lis_dev));
286 
287  PIOS_Assert(!(address & LIS_ADDRESS_READ));
288  PIOS_Assert(!(address & LIS_ADDRESS_AUTOINCREMENT));
289 
290  if (PIOS_LIS_ClaimBus(lis_dev) != 0)
291  return -1;
292 
293  PIOS_SPI_TransferByte(lis_dev->spi_id, 0x7f & address);
294  PIOS_SPI_TransferByte(lis_dev->spi_id, buffer);
295 
296  PIOS_LIS_ReleaseBus(lis_dev);
297 
298  return 0;

301 static void PIOS_LIS_Task(void *parameters)
302 {
303  lis3mdl_dev_t lis_dev = (lis3mdl_dev_t) parameters;
304 
305  PIOS_Assert(!PIOS_LIS_Validate(lis_dev));
306 
307  while (true) {
308  /* Output rate of 80 implies 12.5ms between samples.
309  * We sleep 2 at the top of loop, and 9 at the bottom so
310  * that we poll twice before the sample typically
311  */
312  PIOS_Thread_Sleep(2);
313 
314  uint8_t status;
315 
316  if (PIOS_LIS_ReadReg(lis_dev, LIS_REG_MAG_STATUS, &status))
317  continue;
318 
319  /* Go back, sleep 2 more if we have no new data */
320  if (!(status & LIS_STATUS_ZYXDA))
321  continue;
322 
323  if (PIOS_LIS_ClaimBus(lis_dev) != 0)
324  continue;
325 
326  // TODO: Consider temperature in future
327  int16_t sensor_buf[
328  (LIS_REG_MAG_OUTZ_H - LIS_REG_MAG_OUTX_L + 1) / 2];
329 
330  PIOS_SPI_TransferByte(lis_dev->spi_id,
331  LIS_ADDRESS_READ | LIS_ADDRESS_AUTOINCREMENT |
332  LIS_REG_MAG_OUTX_L);
333 
334  if (PIOS_SPI_TransferBlock(lis_dev->spi_id, NULL,
335  (uint8_t *) sensor_buf,
336  sizeof(sensor_buf)) < 0) {
337  PIOS_LIS_ReleaseBus(lis_dev);
338  continue;
339  }
340 
341  PIOS_LIS_ReleaseBus(lis_dev);
342 
343 #define GET_SB_FROM_REGNO(x) \
344  (sensor_buf[ ((x) - LIS_REG_MAG_OUTX_L) / 2])
345 
346  float mag_x = GET_SB_FROM_REGNO(LIS_REG_MAG_OUTX_L);
347  float mag_y = GET_SB_FROM_REGNO(LIS_REG_MAG_OUTY_L);
348  float mag_z = GET_SB_FROM_REGNO(LIS_REG_MAG_OUTZ_L);
349 
350  struct pios_sensor_mag_data mag_data;
351 
352  switch (lis_dev->cfg->orientation) {
353  case PIOS_LIS_TOP_0DEG:
354  mag_data.y = mag_x;
355  mag_data.x = mag_y;
356  mag_data.z = -mag_z;
357  break;
358  case PIOS_LIS_TOP_90DEG:
359  mag_data.y = -mag_y;
360  mag_data.x = mag_x;
361  mag_data.z = -mag_z;
362  break;
363  case PIOS_LIS_TOP_180DEG:
364  mag_data.y = -mag_x;
365  mag_data.x = -mag_y;
366  mag_data.z = -mag_z;
367  break;
368  case PIOS_LIS_TOP_270DEG:
369  mag_data.y = mag_y;
370  mag_data.x = -mag_x;
371  mag_data.z = -mag_z;
372  break;
373  case PIOS_LIS_BOTTOM_0DEG:
374  mag_data.y = -mag_x;
375  mag_data.x = mag_y;
376  mag_data.z = mag_z;
377  break;
378  case PIOS_LIS_BOTTOM_90DEG:
379  mag_data.y = mag_y;
380  mag_data.x = mag_x;
381  mag_data.z = mag_z;
382  break;
383  case PIOS_LIS_BOTTOM_180DEG:
384  mag_data.y = mag_x;
385  mag_data.x = -mag_y;
386  mag_data.z = mag_z;
387  break;
388  case PIOS_LIS_BOTTOM_270DEG:
389  mag_data.y = -mag_y;
390  mag_data.x = -mag_x;
391  mag_data.z = mag_z;
392  break;
393  }
394 
395  /* Adjust from 2's comp integer value to appropriate
396  * range */
397  float mag_scale = LIS_RANGE_12GAU_COUNTS_PER_MGAU;
398 
399  mag_data.x *= mag_scale;
400  mag_data.y *= mag_scale;
401  mag_data.z *= mag_scale;
402 
403  PIOS_Queue_Send(lis_dev->mag_queue, &mag_data, 0);
404 
405  PIOS_Thread_Sleep(9);
406  }
407 }
