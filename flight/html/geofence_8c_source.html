124 static void checkPosition(UAVObjEvent* ev, void *ctx, void *obj, int len)
125 {
126  (void) ev; (void) ctx; (void) obj; (void) len;
127  if (PositionActualHandle()) {
128  PositionActualData positionActual;
129  PositionActualGet(&positionActual);
130 
131  const float distance2 = powf(positionActual.North, 2) + powf(positionActual.East, 2);
132 
133  // ErrorRadius is squared when it is fetched, so this is correct
134  if (distance2 > geofenceSettings->ErrorRadius) {
135  AlarmsSet(SYSTEMALARMS_ALARM_GEOFENCE, SYSTEMALARMS_ALARM_ERROR);
136  } else if (distance2 > geofenceSettings->WarningRadius) {
137  AlarmsSet(SYSTEMALARMS_ALARM_GEOFENCE, SYSTEMALARMS_ALARM_WARNING);
138  } else {
139  AlarmsClear(SYSTEMALARMS_ALARM_GEOFENCE);
140  }
141  }
142 }
