488 static int32_t updateAttitudeComplementary(float dT, bool first_run, bool secondary, bool raw_gps)
489 {
490  UAVObjEvent ev;
491  GyrosData gyrosData;
492  AccelsData accelsData;
493 
494  // If this is the primary estimation filter, wait until the accel and
495  // gyro objects are updated. If it timeouts then go to failsafe.
496  if (!secondary) {
497  bool gyroTimeout = PIOS_Queue_Receive(gyroQueue, &ev, FAILSAFE_TIMEOUT_MS) != true;
498  bool accelTimeout = PIOS_Queue_Receive(accelQueue, &ev, 1) != true;
499 
500  // When one of these is updated so should the other.
501  if (gyroTimeout || accelTimeout) {
502  // Do not set attitude timeout warnings in simulation mode
503  if (!AttitudeActualReadOnly()) {
504  if (gyroTimeout)
505  set_state_estimation_error(SYSTEMALARMS_STATEESTIMATION_GYROQUEUENOTUPDATING);
506  else if (accelTimeout)
507  set_state_estimation_error(SYSTEMALARMS_STATEESTIMATION_ACCELEROMETERQUEUENOTUPDATING);
508 
509  return -1;
510  }
511  }
512  }
513 
514  AccelsGet(&accelsData);
515 
516  // When this algorithm is first run force it to a known condition
517  if(first_run) {
518  MagnetometerData magData;
519  magData.x = 100;
520  magData.y = 0;
521  magData.z = 0;
522 
523  // Wait for a mag reading if a magnetometer was registered
524  if (PIOS_SENSORS_GetQueue(PIOS_SENSOR_MAG) != NULL) {
525  if (!secondary && PIOS_Queue_Receive(magQueue, &ev, 20) != true) {
526  return -1;
527  }
528  MagnetometerGet(&magData);
529  }
530 
531  /* Initialization is tricky. We're using two reference vectors, one which
532  * is the Earth's gravitational field and the other the Earth's magnetic field
533  * While we know that gravity is always almost perfectly straight down, and
534  * thus has trivially large x and y components, the magnetic field has
535  * components in all directions. Toward the equator the Z component is small,
536  * but toward the poles it grows very quickly. For instance, near Boston, USA,
537  * the magnetic field strength is stronger in the Z direction than the X and
538  * Y combined.
539  *
540  * The upshot is that while we can initialize roll and pitch from the gravity
541  * field alone, we cannot get yaw without taking into account the vehicle's
542  * roll and pitch.
543  */
544  float RPY_D[3];
545 
546  float theta = atan2f(accelsData.x, -accelsData.z);
547  RPY_D[1] = theta * RAD2DEG;
548  RPY_D[0] = atan2f(-accelsData.y, -accelsData.z / cosf(theta)) * RAD2DEG;
549 
550  // See above note about why we rotate the magnetic field before continuing.
551  float RPY_R[3] = {RPY_D[0] * DEG2RAD, RPY_D[1] * DEG2RAD, 0};
552  float Rbe[3][3];
553  float mag_body_frame[3] = {magData.x, magData.y, magData.z};
554  float mag_earth_frame[3];
555  Euler2R(RPY_R, Rbe);
556  rot_mult(Rbe, mag_body_frame, mag_earth_frame, true);
557 
558  // Now that we have the magnetic field in the Earth frame, extract yaw.
559  RPY_D[2] = atan2f(-mag_earth_frame[1], mag_earth_frame[0]) * RAD2DEG;
560 
561  // Convert Euler angles into quaternion
562  RPY2Quaternion(RPY_D, cf_q);
563 
564  complementary_filter_state.initialization = CF_POWERON;
565  complementary_filter_state.reset_timeval = PIOS_DELAY_GetRaw();
566 
567  complementary_filter_state.arming_count = 0;
568 
569  float baro;
570  BaroAltitudeAltitudeGet(&baro);
571  cfvert_reset(&cfvert, baro, attitudeSettings.VertPositionTau);
572 
573  return 0;
574  }

632  } else if (complementary_filter_state.initialization == CF_ARMING ||
633  complementary_filter_state.initialization == CF_INITIALIZING) {
634

636  complementary_filter_state.accel_filter_enabled = true;
637

639  // samples. Compute new bias.
640  if (complementary_filter_state.initialization == CF_ARMING) {
641  accumulate_gyro_compute();
642  complementary_filter_state.accumulating_gyro = false;
643  complementary_filter_state.arming_count = 0;
644  }
645

647  complementary_filter_state.initialization = CF_NORMAL;
648

650  float baro;
651  BaroAltitudeAltitudeGet(&baro);
652  cfvert_reset(&cfvert, baro, attitudeSettings.VertPositionTau);

659  float accel_err[3];

661  float *accels_filtered = complementary_filter_state.accels_filtered;

664  apply_accel_filter(&accelsData.x,accels_filtered);

667  grot[0] = -(2 * (cf_q[1] * cf_q[3] - cf_q[0] * cf_q[2]));
668  grot[1] = -(2 * (cf_q[2] * cf_q[3] + cf_q[0] * cf_q[1]));

675  CrossProduct((const float *) accels_filtered, (const float *) grot_filtered, accel_err);

679  grot_mag = sqrtf(grot_filtered[0]*grot_filtered[0] + grot_filtered[1]*grot_filtered[1] + grot_filtered[2]*grot_filtered[2]);

690  accel_err[2] /= (accel_mag * grot_mag);
691  } else {

693  accel_err[1] = 0;
694  accel_err[2] = 0;
695  }

704  // normally only arise due to a bad magnetometer calibration.
705  if (!(IS_NOT_FINITE(mag.x) || IS_NOT_FINITE(mag.y) || IS_NOT_FINITE(mag.z))) {
706  float bmag = 1.0f;
707  float brot[3];
708  float Rbe[3][3];
709 
710  // Get rotation to bring earth magnetic field into body frame
711  Quaternion2R(cf_q, Rbe);
712 
713  if (homeLocation.Set == HOMELOCATION_SET_TRUE) {
714  rot_mult(Rbe, homeLocation.Be, brot, false);
715  bmag = sqrtf(brot[0] * brot[0] + brot[1] * brot[1] + brot[2] * brot[2]);
716  brot[0] /= bmag;
717  brot[1] /= bmag;
718  brot[2] /= bmag;
719  } else {
720  const float Be[3] = {1.0f, 0.0f, 0.0f};
721  rot_mult(Rbe, Be, brot, false);
722  }
723 
724  float mag_len = sqrtf(mag.x * mag.x + mag.y * mag.y + mag.z * mag.z);
725  mag.x /= mag_len;
726  mag.y /= mag_len;
727  mag.z /= mag_len;
728 
729  // Only compute if neither vector is null
730  if (bmag < 1 || mag_len < 1)
731  mag_err[0] = mag_err[1] = mag_err[2] = 0;
732  else
733  CrossProduct((const float *) &mag.x, (const float *) brot, mag_err);
734 
735  if (mag_err[2] != mag_err[2])
736  mag_err[2] = 0;
737  } else
738  mag_err[2] = 0;
739  } else {
740  mag_err[2] = 0;
741  }
742 
743  // Accumulate integral of error. Scale here so that units are (deg/s) but Ki has units of s
744  GyrosBiasData gyrosBias;
745  GyrosBiasGet(&gyrosBias);
746  gyrosBias.x -= accel_err[0] * accKi * dT;
747  gyrosBias.y -= accel_err[1] * accKi * dT;
748  gyrosBias.z -= mag_err[2] * mgKi * dT;
749  GyrosBiasSet(&gyrosBias);
750 
751  // Correct rates based on error, integral component dealt with in updateSensors
752  gyrosData.x += accel_err[0] * accKp;
753  gyrosData.y += accel_err[1] * accKp;
754  gyrosData.z += accel_err[2] * accKp + mag_err[2] * mgKp;
755 
756  // Work out time derivative from INSAlgo writeup
757  // Also accounts for the fact that gyros are in deg/s
758  float qdot[4];
759  qdot[0] = (-cf_q[1] * gyrosData.x - cf_q[2] * gyrosData.y - cf_q[3] * gyrosData.z) * dT * DEG2RAD / 2;
760  qdot[1] = (cf_q[0] * gyrosData.x - cf_q[3] * gyrosData.y + cf_q[2] * gyrosData.z) * dT * DEG2RAD / 2;
761  qdot[2] = (cf_q[3] * gyrosData.x + cf_q[0] * gyrosData.y - cf_q[1] * gyrosData.z) * dT * DEG2RAD / 2;
762  qdot[3] = (-cf_q[2] * gyrosData.x + cf_q[1] * gyrosData.y + cf_q[0] * gyrosData.z) * dT * DEG2RAD / 2;
763 
764  // Take a time step
765  cf_q[0] = cf_q[0] + qdot[0];
766  cf_q[1] = cf_q[1] + qdot[1];
767  cf_q[2] = cf_q[2] + qdot[2];
768  cf_q[3] = cf_q[3] + qdot[3];
769 
770  if(cf_q[0] < 0) {
771  cf_q[0] = -cf_q[0];
772  cf_q[1] = -cf_q[1];
773  cf_q[2] = -cf_q[2];
774  cf_q[3] = -cf_q[3];
775  }
776 
777  // Renomalize
778  float qmag;
779  qmag = sqrtf(cf_q[0]*cf_q[0] + cf_q[1]*cf_q[1] + cf_q[2]*cf_q[2] + cf_q[3]*cf_q[3]);
780  cf_q[0] = cf_q[0] / qmag;
781  cf_q[1] = cf_q[1] / qmag;
782  cf_q[2] = cf_q[2] / qmag;
783  cf_q[3] = cf_q[3] / qmag;
784 
785  // If quaternion has become inappropriately short or has become Nan reinit.
786  // THIS SHOULD NEVER ACTUALLY HAPPEN
787  if((fabsf(qmag) < 1.0e-3f) || IS_NOT_FINITE(qmag)) {
788  cf_q[0] = 1;
789  cf_q[1] = 0;
790  cf_q[2] = 0;
791  cf_q[3] = 0;
792  }
793 
794  if (!secondary) {
795  static bool got_baro_pt = false;
796 
797  // Calculate the NED acceleration and get the z-component
798  float z_accel = calc_ned_accel(cf_q, &accelsData.x);
799 
800  // When this is the only filter compute th vertical state from baro data
801  // Reset the filter for barometric data
802  if (PIOS_Queue_Receive(baroQueue, &ev, 0) == true) {
803  float baro;
804  BaroAltitudeAltitudeGet(&baro);
805 
806  cfvert_predict_pos(&cfvert, z_accel, dT);
807  cfvert_update_baro(&cfvert, baro, dT);
808 
809  got_baro_pt = true;
810  } else if (!got_baro_pt) {
811  /* If we've never heard from the baro hold alt at 0 */
812  cfvert.position_z = 0;
813  cfvert.velocity_z = 0;
814  } else {
815  cfvert_predict_pos(&cfvert, z_accel, dT);
816  }
817  }
818 
819  if (!secondary && !raw_gps) {
820  // When in raw GPS mode, it will set the error to none if
821  // reception is good
822  set_state_estimation_error(SYSTEMALARMS_STATEESTIMATION_NONE);
823  }
824 
825  return 0;
826 }

867 static void cfvert_predict_pos(struct cfvert *cf, float z_accel, float dt)
868 {
869  float k1_z = 3 / cf->time_constant_z;
870  float k2_z = 3 / powf(cf->time_constant_z, 2);
871  float k3_z = 1 / powf(cf->time_constant_z, 3);
872 
873  cf->accel_correction_z += cf->position_error_z * k3_z * dt;
874  cf->velocity_z += cf->position_error_z * k2_z * dt;
875  cf->position_correction_z += cf->position_error_z * k1_z * dt;
876 
877  float velocity_increase;
878  velocity_increase = (z_accel + cf->accel_correction_z) * dt;
879  cf->position_base_z += (cf->velocity_z + velocity_increase * 0.5f) * dt;
880  cf->position_z = cf->position_base_z + cf->position_correction_z;
881  cf->velocity_z += velocity_increase;
