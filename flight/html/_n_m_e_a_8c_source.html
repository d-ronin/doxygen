105 int parse_nmea_stream (uint8_t c, char *gps_rx_buffer, GPSPositionData *GpsData, struct GPS_RX_STATS *gpsRxStats)
106 {
107  static uint8_t rx_count = 0;
108  static bool start_flag = false;
109  static bool found_cr = false;
110 
111  // detect start while acquiring stream
112  if (!start_flag && (c == '$')) // NMEA identifier found
113  {
114  start_flag = true;
115  found_cr = false;
116  rx_count = 0;
117  }
118  else
119  if (!start_flag)
120  return PARSER_ERROR;
121 
122  if (rx_count >= NMEA_MAX_PACKET_LENGTH)
123  {
124  // The buffer is already full and we haven't found a valid NMEA sentence.
125  // Flush the buffer and note the overflow event.
126  gpsRxStats->gpsRxOverflow++;
127  start_flag = false;
128  found_cr = false;
129  rx_count = 0;
130  return PARSER_OVERRUN;
131  }
132  else
133  {
134  gps_rx_buffer[rx_count] = c;
135  rx_count++;
136  }
137 
138  // look for ending '\r\n' sequence
139  if (!found_cr && (c == '\r') )
140  found_cr = true;
141  else
142  if (found_cr && (c != '\n') )
143  found_cr = false; // false end flag
144  else
145  if (found_cr && (c == '\n') )
146  {
147  // The NMEA functions require a zero-terminated string
148  // As we detected \r\n, the string as for sure 2 bytes long, we will also strip the \r\n
149  gps_rx_buffer[rx_count-2] = 0;
150 
151  // prepare to parse next sentence
152  start_flag = false;
153  found_cr = false;
154  rx_count = 0;
155  // Our rxBuffer must look like this now:
156  // [0] = '$'
157  // ... = zero or more bytes of sentence payload
158  // [end_pos - 1] = '\r'
159  // [end_pos] = '\n'
160  //
161  // Prepare to consume the sentence from the buffer
162 
163  // Validate the checksum over the sentence
164  if (!NMEA_checksum(&gps_rx_buffer[1]))
165  { // Invalid checksum. May indicate dropped characters on Rx.
166  //PIOS_DEBUG_PinHigh(2);
167  gpsRxStats->gpsRxChkSumError++;
168  //PIOS_DEBUG_PinLow(2);
169  return PARSER_ERROR;
170  }
171  else
172  { // Valid checksum, use this packet to update the GPS position
173  if (!NMEA_update_position(&gps_rx_buffer[1], GpsData)) {
174  //PIOS_DEBUG_PinHigh(2);
175  gpsRxStats->gpsRxParserError++;
176  //PIOS_DEBUG_PinLow(2);
177  }
178  else
179  gpsRxStats->gpsRxReceived++;;
180 
181  return PARSER_COMPLETE;
182  }
183  }
184  return PARSER_INCOMPLETE;
185 }

213 bool NMEA_checksum(char *nmea_sentence)
214 {
215  uint8_t checksum_computed = 0;
216  uint8_t checksum_received;
217 
218  while (*nmea_sentence != '\0' && *nmea_sentence != '*') {
219  checksum_computed ^= *nmea_sentence;
220  nmea_sentence++;
221  }
222 
223  /* Make sure we're now pointing at the checksum */
224  if (*nmea_sentence == '\0') {
225  /* Buffer ran out before we found a checksum marker */
226  return false;
227  }
228 
229  /* Load the checksum from the buffer */
230  checksum_received = strtol(nmea_sentence + 1, NULL, 16);
231 
232  //PIOS_COM_SendFormattedStringNonBlocking(COM_DEBUG_USART,"$%d=%d\r\n",checksum_received,checksum_computed);
233 
234  return (checksum_computed == checksum_received);
235 }

244 /* Parse a number encoded in a string of the format:
245  * [-]NN.nnnnn

247  * The fract_units field indicates the units of the fractional part as

250 static bool NMEA_parse_real(int32_t * whole, uint32_t * fract, uint8_t * fract_units, char *field)
251 {
252  char *s = field;
253  char *field_w;
254  char *field_f;
255 
256  PIOS_DEBUG_Assert(whole);
257  PIOS_DEBUG_Assert(fract);
258  PIOS_DEBUG_Assert(fract_units);
259 
260  field_w = strsep(&s, ".");
261  field_f = s;
262 
263  *whole = strtol(field_w, NULL, 10);
264 
265  if (field_f) {
266  /* decimal was found so we may have a fractional part */
267  *fract = strtoul(field_f, NULL, 10);
268  *fract_units = strlen(field_f);
269  } else {
270  /* no decimal was found, fractional part is zero */
271  *fract = 0;
272  *fract_units = 0;
273  }
274 
275  return true;
276 }

295 /*
296  * Parse a field in the format:
297  * DD[D]MM.mmmm[mm]
298  * into a fixed-point representation in units of (degrees * 1e-7)
299  */
300 static bool NMEA_latlon_to_fixed_point(int32_t * latlon, char *nmea_latlon, bool negative)
301 {
302  int32_t num_DDDMM;
303  uint32_t num_m;
304  uint8_t units;
305 
306  /* Sanity checks */
307  PIOS_DEBUG_Assert(nmea_latlon);
308  PIOS_DEBUG_Assert(latlon);
309 
310  if (*nmea_latlon == '\0') { /* empty lat/lon field */
311  return false;
312  }
313 
314  if (!NMEA_parse_real(&num_DDDMM, &num_m, &units, nmea_latlon)) {
315  return false;
316  }
317 
318  /* scale up the mmmm[mm] field apropriately depending on # of digits */
319  /* not using 1eN notation because that forces fixed point and lost precision */
320  switch (units) {
321  case 0:
322  /* no digits, value is zero so no scaling */
323  break;
324  case 1: /* m */
325  num_m *= 1000000; /* m000000 */
326  break;
327  case 2: /* mm */
328  num_m *= 100000; /* mm00000 */
329  break;
330  case 3: /* mmm */
331  num_m *= 10000; /* mmm0000 */
332  break;
333  case 4: /* mmmm */
334  num_m *= 1000; /* mmmm000 */
335  break;
336  case 5: /* mmmmm */
337  num_m *= 100; /* mmmmm00 */
338  break;
339  case 6: /* mmmmmm */
340  num_m *= 10; /* mmmmmm0 */
341  break;
342  default:
343  /* unhandled format */
344  num_m = 0.0f;
345  break;
346  }
347 
348  *latlon = (num_DDDMM / 100) * 10000000; /* scale the whole degrees */
349  *latlon += (num_DDDMM % 100) * 10000000 / 60; /* add in the scaled decimal whole minutes */
350  *latlon += num_m / 60; /* add in the scaled decimal fractional minutes */
351 
352  if (negative)
353  *latlon *= -1;
354 
355  return true;
356 }

365 bool NMEA_update_position(char *nmea_sentence, GPSPositionData *GpsData)
366 {
367  char* p = nmea_sentence;
368  char* params[MAX_NB_PARAMS];
369  uint8_t nbParams;
370 
371 #ifdef DEBUG_MSG_IN
372  DEBUG_MSG("\"%s\"\n", nmea_sentence);
373 #endif
374 
375  // Split the nmea sentence it its parameters, separated by ","
376  // Sample NMEA message: "GPRMC,000131.736,V,,,,,0.00,0.00,060180,,,N*43"
377 
378  // The first parameter starts at the beginning of the message
379  p += 2; // Skip the first two characters (e.g., GP, GN) which is the talker ID
380  params[0] = p;
381  nbParams = 1;
382  while (*p != 0) {
383  if (*p == '*') {
384  // After the * comes the "CRC", we are done,
385  *p = 0; // Zero-terminate this parameter
386  break;
387  } else if (*p == ',') {
388  // This is the end of this parameter
389  *p = 0; // Zero-terminate this parameter
390  // Start new parameter
391  if (nbParams==MAX_NB_PARAMS)
392  break;
393  params[nbParams] = p+1; // For sure there is something at p+1 because at p there is ","
394  nbParams++;
395  }
396  p++;
397  }
398 
399 
400 #ifdef DEBUG_PARAMS
401  int i;
402  for (i=0;i<nbParams; i++) {
403  DEBUG_MSG(" %d \"%s\"\n", i, params[i]);
404  }
405 #endif
406 
407  // The first parameter is the message name, lets see if we find a parser for it
408  const struct nmea_parser *parser;
409  parser = NMEA_find_parser_by_prefix(params[0]);
410  if (!parser) {
411  // No parser found
412  DEBUG_MSG(" NO PARSER (\"%s\")\n", params[0]);
413  return false;
414  }
415 
416  #ifdef DEBUG_MGSID_IN
417  DEBUG_MSG("%s %d ", params[0]);
418  #endif
419  // Send the message to the parser and get it update the GpsData
420  // Information from various different NMEA messages are temporarily
421  // cumulated in the GpsData structure. An actual GPSPosition update
422  // is triggered by GGA messages only. This message type sets the
423  // gpsDataUpdated flag to request this.
424  bool gpsDataUpdated = false;
425 
426  if (!parser->handler(GpsData, &gpsDataUpdated, params, nbParams)) {
427  // Parse failed
428  DEBUG_MSG("PARSE FAILED (\"%s\")\n", params[0]);
429  if (gpsDataUpdated && (GpsData->Status == GPSPOSITION_STATUS_NOFIX)) {
430  GPSPositionSet(GpsData);
431  }
432  return false;
433  }
434 
435 
436  // All is fine :) Update object if data has changed
437  if (gpsDataUpdated) {
438  #ifdef DEBUG_MGSID_IN
439  DEBUG_MSG("U");
440  #endif
441  GPSPositionSet(GpsData);
442  }
443 
444  #ifdef DEBUG_MGSID_IN
445  DEBUG_MSG("\n");
446  #endif
447  return true;
448 }

634 static bool nmeaProcessGPGSV(GPSPositionData * GpsData, bool* gpsDataUpdated, char* param[], uint8_t nbParam)
635 {
636  if (nbParam < 4)
637  return false;
638 
639 #ifdef NMEA_DEBUG_GSV
640  DEBUG_MSG("\n Sentence=%s/%s\n", param[2], param[1]);
641  DEBUG_MSG(" Sats=%s\n", param[3]);
642 #endif
643 
644  uint8_t nbSentences = atoi(param[1]);
645  uint8_t currSentence = atoi(param[2]);
646 
647  *gpsDataUpdated = false;
648 
649  if (nbSentences < 1 || nbSentences > 8 || currSentence < 1 || currSentence > nbSentences)
650  return false;
651 
652  gsv_partial.SatsInView = atoi(param[3]);
653 
654  // Find out if this is the first sentence in the GSV set
655  if (currSentence == 1) {
656  if (gsv_expected_mask != gsv_processed_mask) {
657  // We are starting over when we haven't yet finished our previous GSV group
658  gsv_incomplete_error++;
659  }
660 
661  // First GSV sentence in the sequence, reset our expected_mask
662  gsv_expected_mask = (1 << nbSentences) - 1;
663  }
664 
665  uint8_t current_sentence_id = (1 << (currSentence - 1));
666  if (gsv_processed_mask & current_sentence_id) {
667  /* Duplicate sentence in this GSV set */
668  gsv_duplicate_error++;
669  } else {
670  /* Note that we've seen this sentence */
671  gsv_processed_mask |= current_sentence_id;
672  }
673 
674  uint8_t parIdx = 4;
675 
676 #ifdef NMEA_DEBUG_GSV
677  DEBUG_MSG(" PRN:");
678 #endif
679 
680  /* Make sure this sentence can fit in our GPSSatellites object */
681  if ((currSentence * 4) <= NELEMENTS(gsv_partial.PRN)) {
682  /* Process 4 blocks of satellite info */
683  for (uint8_t i = 0; parIdx+4 <= nbParam && i < 4; i++) {
684  uint8_t sat_index = ((currSentence - 1) * 4) + i;
