109 static void batteryTask(void * parameters)
110 {
111  const float dT = SAMPLE_PERIOD_MS / 1000.0f;
112 
113  battery_settings_updated = true;
114  bool cells_calculated = false;
115  unsigned cells = 1;
116 
117  FlightBatteryStateData flightBatteryData;
118  FlightBatterySettingsData batterySettings;
119 
120  FlightBatteryStateGet(&flightBatteryData);
121 
122  // Main task loop
123  uint32_t lastSysTime;
124  lastSysTime = PIOS_Thread_Systime();
125  while (true) {
126  PIOS_Thread_Sleep_Until(&lastSysTime, SAMPLE_PERIOD_MS);
127  float energyRemaining;
128 
129  if (battery_settings_updated) {
130  battery_settings_updated = false;
131  FlightBatterySettingsGet(&batterySettings);
132 
133  voltageADCPin = batterySettings.VoltagePin;
134  if (voltageADCPin == FLIGHTBATTERYSETTINGS_VOLTAGEPIN_NONE)
135  voltageADCPin = -1;
136 
137  currentADCPin = batterySettings.CurrentPin;
138  if (currentADCPin == FLIGHTBATTERYSETTINGS_CURRENTPIN_NONE)
139  currentADCPin = -1;
140 
141  cells_calculated = false;
142  }
143 
144  bool adc_pin_invalid = false;
145  bool adc_offset_invalid = false;
146 
147  // handle voltage
148  if (voltageADCPin >= 0) {
149  float adc_voltage = (float)PIOS_ADC_GetChannelVolt(voltageADCPin);
150  float scaled_voltage = 0.0f;
151 
152  // A negative result indicates an error (PIOS_ADC_GetChannelVolt returns negative on error)
153  if(adc_voltage < 0.0f)
154  adc_pin_invalid = true;
155  else {
156  // scale to actual voltage
157  scaled_voltage = (adc_voltage * 1000.0f
158  / batterySettings.SensorCalibrationFactor[FLIGHTBATTERYSETTINGS_SENSORCALIBRATIONFACTOR_VOLTAGE])
159  + batterySettings.SensorCalibrationOffset[FLIGHTBATTERYSETTINGS_SENSORCALIBRATIONOFFSET_VOLTAGE]; //in Volts
160 
161  // disallow negative values as these are cast to unsigned integral types
162  // in some telemetry layers
163  if (scaled_voltage < 0.0f) {
164  scaled_voltage = 0.0f;
165  adc_offset_invalid = true;
166  } else if (batterySettings.MaxCellVoltage > 0.0f && scaled_voltage > 2.5f) {
167  if (!cells_calculated) {
168  cells = ((scaled_voltage / batterySettings.MaxCellVoltage) + 0.9f);
169  if (cells > 0) {
170  cells_calculated = true;
171  flightBatteryData.DetectedCellCount = cells;
172  }
173  }
174  } else {
175  cells_calculated = false;
176  }
177 
178  if (!cells_calculated) {
179  cells = batterySettings.NbCells;
180  flightBatteryData.DetectedCellCount = 0;
181  }
182  }
183 
184  flightBatteryData.Voltage = scaled_voltage;
185 
186  // generate alarms and warnings
187  if (flightBatteryData.Voltage < (batterySettings.CellVoltageThresholds[FLIGHTBATTERYSETTINGS_CELLVOLTAGETHRESHOLDS_ALARM] * cells))
188  AlarmsSet(SYSTEMALARMS_ALARM_BATTERY, SYSTEMALARMS_ALARM_CRITICAL);
189  else if (flightBatteryData.Voltage < (batterySettings.CellVoltageThresholds[FLIGHTBATTERYSETTINGS_CELLVOLTAGETHRESHOLDS_WARNING] * cells))
190  AlarmsSet(SYSTEMALARMS_ALARM_BATTERY, SYSTEMALARMS_ALARM_WARNING);
191  else
192  AlarmsClear(SYSTEMALARMS_ALARM_BATTERY);
193  } else {
194  flightBatteryData.Voltage = 0;
195  }
196 
197  // handle current
198  if (currentADCPin >= 0) {
199  float adc_voltage = (float)PIOS_ADC_GetChannelVolt(currentADCPin);
200  float scaled_current = 0.0f;
201 
202  // A negative result indicates an error (PIOS_ADC_GetChannelVolt returns -1 on error)
203  if(adc_voltage < 0.0f)
204  adc_pin_invalid = true;
205  else {
206  // scale to actual current
207  scaled_current = (adc_voltage * 1000.0f
208  / batterySettings.SensorCalibrationFactor[FLIGHTBATTERYSETTINGS_SENSORCALIBRATIONFACTOR_CURRENT])
209  + batterySettings.SensorCalibrationOffset[FLIGHTBATTERYSETTINGS_SENSORCALIBRATIONOFFSET_CURRENT]; //in Amps
210 
211  // disallow negative values as these are cast to unsigned integral types
212  // in some telemetry layers
213  if(scaled_current < 0.0f) {
214  scaled_current = 0.0f;
215  adc_offset_invalid = true;
216  }
217  }
218 
219  flightBatteryData.Current = scaled_current;
220 
221  if (flightBatteryData.Current > flightBatteryData.PeakCurrent)
222  flightBatteryData.PeakCurrent = flightBatteryData.Current; //in Amps
223 
224  flightBatteryData.ConsumedEnergy += (flightBatteryData.Current * dT * 1000.0f / 3600.0f); //in mAh
225 
226  //Apply a 2 second rise time low-pass filter to average the current
227  float alpha = 1.0f - dT / (dT + 2.0f);
228  flightBatteryData.AvgCurrent = alpha * flightBatteryData.AvgCurrent + (1 - alpha) * flightBatteryData.Current; //in Amps
229 
230  // XXX Arguably this should be to 10% capacity or 15%
231  energyRemaining = batterySettings.Capacity - flightBatteryData.ConsumedEnergy; // in mAh
232 
233  // And for time estimation, smooth things much more.
234  // Second order since it incorporates the above
235  // smoothing, 12s time constant for this layer.
236  alpha = 1.0f - dT / (dT + 12.0f);
237 
238  avg_current_lpf_for_time = avg_current_lpf_for_time * alpha + (1 - alpha) * flightBatteryData.AvgCurrent;
239 
240  if (avg_current_lpf_for_time > 0.1f)
241  flightBatteryData.EstimatedFlightTime = (energyRemaining / (avg_current_lpf_for_time * 1000.0f)) * 3600.0f; //in Sec
242  else
243  flightBatteryData.EstimatedFlightTime = 9999;
244 
245  // generate alarms and warnings
246  if ((batterySettings.FlightTimeThresholds[FLIGHTBATTERYSETTINGS_FLIGHTTIMETHRESHOLDS_ALARM] > 0)
247  && (flightBatteryData.EstimatedFlightTime < batterySettings.FlightTimeThresholds[FLIGHTBATTERYSETTINGS_FLIGHTTIMETHRESHOLDS_ALARM]))
248  AlarmsSet(SYSTEMALARMS_ALARM_FLIGHTTIME, SYSTEMALARMS_ALARM_CRITICAL);
249  else if ((batterySettings.FlightTimeThresholds[FLIGHTBATTERYSETTINGS_FLIGHTTIMETHRESHOLDS_WARNING] > 0)
250  && (flightBatteryData.EstimatedFlightTime < batterySettings.FlightTimeThresholds[FLIGHTBATTERYSETTINGS_FLIGHTTIMETHRESHOLDS_WARNING]))
251  AlarmsSet(SYSTEMALARMS_ALARM_FLIGHTTIME, SYSTEMALARMS_ALARM_WARNING);
252  else
253  AlarmsClear(SYSTEMALARMS_ALARM_FLIGHTTIME);
254  } else {
255  flightBatteryData.Current = 0;
256  }
257 
258  if(adc_pin_invalid)
259  AlarmsSet(SYSTEMALARMS_ALARM_ADC, SYSTEMALARMS_ALARM_CRITICAL);
260  else if(adc_offset_invalid)
261  AlarmsSet(SYSTEMALARMS_ALARM_ADC, SYSTEMALARMS_ALARM_WARNING);
262  else if(voltageADCPin >= 0 || currentADCPin >= 0)
263  AlarmsSet(SYSTEMALARMS_ALARM_ADC, SYSTEMALARMS_ALARM_OK);
264  else
265  AlarmsSet(SYSTEMALARMS_ALARM_ADC, SYSTEMALARMS_ALARM_UNINITIALISED);
266 
267  FlightBatteryStateSet(&flightBatteryData);
268  }
269 }
