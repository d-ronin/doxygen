223 static void registerObject(telem_t telem, UAVObjHandle obj)
224 {
225  if (UAVObjIsMetaobject(obj)) {
226  /* Only connect change notifications for meta objects. No periodic updates */
227  UAVObjConnectQueue(obj, telem->queue, EV_MASK_ALL_UPDATES);
228  return;
229  } else {
230  UAVObjMetadata metadata;
231  UAVObjUpdateMode updateMode;
232  UAVObjGetMetadata(obj, &metadata);
233  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
234 
235  // Setup object for periodic updates
236  UAVObjEvent ev = {
237  .obj = obj,
238  .instId = UAVOBJ_ALL_INSTANCES,
239  .event = EV_UPDATED_PERIODIC,
240  };
241 
242  /* Only create a periodic event for objects that are periodic */
243  if (updateMode == UPDATEMODE_PERIODIC) {
244  EventPeriodicQueueCreate(&ev, telem->queue, 0);
245  }
246 
247  // Setup object for telemetry updates
248  updateObject(telem, obj, EV_NONE);
249  }

256 static void updateObject(telem_t telem, UAVObjHandle obj, int32_t eventType)
257 {
258  UAVObjMetadata metadata;
259  UAVObjUpdateMode updateMode;
260  int32_t eventMask;
261 
262  if (UAVObjIsMetaobject(obj)) {
263  /* This function updates the periodic updates for the object.
264  * Meta Objects cannot have periodic updates.
265  */
266  PIOS_Assert(false);
267  return;
268  }
269 
270  // Get metadata
271  UAVObjGetMetadata(obj, &metadata);
272  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
273 
274  // Setup object depending on update mode
275  switch (updateMode) {
276  case UPDATEMODE_PERIODIC:
277  // Set update period
278  setUpdatePeriod(telem, obj, metadata.telemetryUpdatePeriod);
279 
280  // Connect queue
281  eventMask = EV_UPDATED_PERIODIC | EV_UPDATED_MANUAL;
282  UAVObjConnectQueueThrottled(obj, telem->queue, eventMask, 0);
283  break;
284  case UPDATEMODE_ONCHANGE:
285  // Set update period
286  setUpdatePeriod(telem, obj, 0);
287 
288  // Connect queue
289  eventMask = EV_UPDATED | EV_UPDATED_MANUAL;
290  UAVObjConnectQueueThrottled(obj, telem->queue, eventMask, 0);
291  break;
292  case UPDATEMODE_THROTTLED:
293  setUpdatePeriod(telem, obj, 0);
294 
295  eventMask = EV_UPDATED | EV_UPDATED_MANUAL;
296  UAVObjConnectQueueThrottled(obj, telem->queue, eventMask,
297  metadata.telemetryUpdatePeriod);
298  break;
299  case UPDATEMODE_MANUAL:
300  // Set update period
301  setUpdatePeriod(telem, obj, 0);
302 
303  // Connect queue
304  eventMask = EV_UPDATED_MANUAL;
305  UAVObjConnectQueueThrottled(obj, telem->queue, eventMask, 0);
306  break;
307  }
308 }

316 static void ackResendOrTimeout(telem_t telem, int idx)
317 {
318  if (telem->acks[idx].retry_count++ > MAX_RETRIES) {
319  DEBUG_PRINTF(3, "telem: abandoning ack wait for %d/%d\n",
320  UAVObjGetID(telem->acks[idx].obj),
321  telem->acks[idx].inst_id);
322 
323  /* Two different kinds of things in tx_errors-- objects that
324  * failed to transfer, and send errors (shouldn't happen) */
325  telem->tx_errors++;
326 
327  telem->acks[idx].obj = NULL;
328  } else {
329  DEBUG_PRINTF(3, "telem: abandoning ack wait for %d/%d\n",
330  UAVObjGetID(telem->acks[idx].obj),
331  telem->acks[idx].inst_id);
332 
333  int32_t success;
334 
335  telem->acks[idx].timeout = PIOS_Thread_Systime() +
336  ACK_TIMEOUT_MS;
337 
338  /* Must not hold lock while sending an object, because
339  * of lock ordering issues (though the lock should be
340  * severely squashed in uavtalk.c
341  */
342  PIOS_Mutex_Unlock(telem->reqack_mutex);
343  success = UAVTalkSendObject(telem->uavTalkCon,
344  telem->acks[idx].obj,
345  telem->acks[idx].inst_id,
346  true);
347  PIOS_Mutex_Lock(telem->reqack_mutex, PIOS_MUTEX_TIMEOUT_MAX);
348 
349  telem->tx_retries++;
350 
351  if (success == -1) {
352  telem->tx_errors++;
353  }
354  }
355 }

384 static void addAckPending(telem_t telem, UAVObjHandle obj, uint16_t inst_id)
385 {
386  DEBUG_PRINTF(3, "telem: want ack for %d/%d\n", UAVObjGetID(obj),
387  inst_id);
388 
389  while (true) {
390  PIOS_Mutex_Lock(telem->reqack_mutex, PIOS_MUTEX_TIMEOUT_MAX);
391  ackHousekeeping(telem);
392 
393  /* There's a slight race here, if we're --already-- waiting
394  * for an ack on an object, we can end up putting it in here
395  * to wait again. In a perfect storm-- past retries cause two
396  * acks, and then the new object is lost-- we can be fooled.
397  * It seems like a lesser evil than most of the alternatives
398  * that keep good pipelining, though
399  */
400  for (int i = 0; i < MAX_ACKS_PENDING; i++) {
401  if (!telem->acks[i].obj) {
402  telem->acks[i].obj = obj;
403  telem->acks[i].inst_id = inst_id;
404  telem->acks[i].timeout =
405  PIOS_Thread_Systime() +
406  ACK_TIMEOUT_MS;
407 
408  telem->acks[i].retry_count = 0;
409 
410  PIOS_Mutex_Unlock(telem->reqack_mutex);
411  return;
412  }
413  }
414 
415  DEBUG_PRINTF(3, "telem: blocking because acks are full\n");
416 
417  /* Oh no. This is the bad case--- maximum number of things
418  * needing ack in flight.
419  */
420 
421  PIOS_Mutex_Unlock(telem->reqack_mutex);
422 
423  /* TODO: Consider nicer wakeup mechanism here. */
424  PIOS_Thread_Sleep(3);
425  }
426 }

440 static int32_t fileReqCallback(void *ctx, uint8_t *buf,
441  uint32_t file_id, uint32_t offset, uint32_t len)
442 {
443  if (file_id < FLASH_PARTITION_NUM_LABELS) {
444  uintptr_t part_id;
445 
446  if (PIOS_FLASH_find_partition_id(file_id, &part_id)) {
447  return -1;
448  }
449 
450  uint32_t size;
451 
452  if (PIOS_FLASH_get_partition_size(part_id, &size)) {
453  return -2;
454  }
455 
456  if (offset >= size) {
457  return 0;
458  }
459 
460  uint32_t remaining = size - offset;
461 
462  if (len > remaining) {
463  len = remaining;
464  }
465 
466  if (PIOS_FLASH_start_transaction(part_id)) {
467  return -3;
468  }
469 
470  if (PIOS_FLASH_read_data(part_id, offset, buf, len)) {
471  len = -4;
472  }
473 
474  PIOS_FLASH_end_transaction(part_id);
475 
476  return len;
477  }
478 
479  return -1;
480 }

519 static void ackCallback(void *ctx, uint32_t obj_id, uint16_t inst_id)
520 {
521  telem_t telem = ctx;
522 
523  PIOS_Mutex_Lock(telem->reqack_mutex, PIOS_MUTEX_TIMEOUT_MAX);
524 
525  for (int i = 0; i < MAX_ACKS_PENDING; i++) {
526  if (!telem->acks[i].obj) {
527  continue;
528  }
529 
530  if (inst_id != telem->acks[i].inst_id) {
531  continue;
532  }
533 
534  uint32_t other_obj_id = UAVObjGetID(telem->acks[i].obj);
535 
536  if (obj_id != other_obj_id) {
537  continue;
538  }
539 
540  telem->acks[i].obj = NULL;
541 
542  DEBUG_PRINTF(3, "telem: Got ack for %d/%d\n", obj_id, inst_id);
543  PIOS_Mutex_Unlock(telem->reqack_mutex);
544  return;
545  }
546 
547  PIOS_Mutex_Unlock(telem->reqack_mutex);
548 
549  DEBUG_PRINTF(3, "telem: Got UNEXPECTED ack for %d/%d\n", obj_id, inst_id);
550 }

555 static void processObjEvent(telem_t telem, UAVObjEvent * ev)
556 {
557  if (ev->obj == 0) {
558  updateTelemetryStats(telem);
559  } else if (ev->obj == GCSTelemetryStatsHandle()) {
560  gcsTelemetryStatsUpdated(telem);
561  } else {
562  // Act on event
563  if (ev->event == EV_UPDATED || ev->event == EV_UPDATED_MANUAL ||
564  ev->event == EV_UPDATED_PERIODIC) {
565  int32_t success = -1;
566 
567  UAVObjMetadata metadata;
568 
569  bool acked = false;
570 
571  // Get object metadata
572  UAVObjGetMetadata(ev->obj, &metadata);
573 
574  if (UAVObjGetTelemetryAcked(&metadata)) {
575  acked = true;
576 
577  addAckPending(telem, ev->obj, ev->instId);
578  }
579 
580  success = UAVTalkSendObject(telem->uavTalkCon,
581  ev->obj, ev->instId,
582  acked);
583 
584  if (success == -1) {
585  telem->tx_errors++;
586  }
587  }
588 
589  // If this is a metaobject then make necessary
590  // telemetry updates
591  if (UAVObjIsMetaobject(ev->obj)) {
592  // linked object will be the actual object the
593  // metadata are for
594  updateObject(telem, UAVObjGetLinkedObj(ev->obj),
595  EV_NONE);
596  }
597  }
598 }

600 static void sendRequestedObjs(telem_t telem)
601 {
602  // Must be called with the reqack mutex.
603 
604  while (telem->reqs[0].valid) {
605  struct pending_req preq = telem->reqs[0];
606 
607  /* Expensive, but for as little as this comes up,
608  * who cares.
609  */
610  int i;
611 
612  for (i = 1; i < MAX_REQS_PENDING; i++) {
613  if (!telem->reqs[i].valid) {
614  break;
615  }
616  telem->reqs[i-1] = telem->reqs[i];
617  }
618 
619  telem->reqs[i-1].valid = 0;
620 
621  // Unlock, to ensure new requests can come in OK.
622  PIOS_Mutex_Unlock(telem->reqack_mutex);
623 
624  // handle request
625  UAVObjHandle obj = UAVObjGetByID(preq.obj_id);
626 
627  // Send requested object if message is of type OBJ_REQ
628  if (!obj)
629  UAVTalkSendNack(telem->uavTalkCon, preq.obj_id);
630  else
631  UAVTalkSendObject(telem->uavTalkCon,
632  obj, preq.inst_id,
633  false);
634 
635  PIOS_Mutex_Lock(telem->reqack_mutex,
636  PIOS_MUTEX_TIMEOUT_MAX);
637  }
638 }

682 static bool processUsbActivity(bool seen_active)
683 {
684  uint32_t this_systime = PIOS_Thread_Systime();
685  uint32_t fixedup_time = this_systime +
686  USB_ACTIVITY_TIMEOUT_MS;
687 
688  if (seen_active) {
689  if (fixedup_time < this_systime) {
690  // (mostly) handle wrap.
691  usb_timeout_time = UINT32_MAX;
692  } else {
693  usb_timeout_time = fixedup_time;
694  }
695 
696  return true;
697  }
698 
699  uint32_t usb_time = usb_timeout_time;
700 
701  if (this_systime >= usb_time) {
702  usb_timeout_time = 0;
703 
704  return false;
705  }
706 
707  /* If the timeout is too far in the future ...
708  * (because of the above wrap case..) */
709  if (fixedup_time < usb_time) {
710  if (fixedup_time > this_systime) {
711  /* and we're not wrapped, then fixup the
712  * time. */
713  usb_timeout_time = fixedup_time;
714  }
715  }
716 
717  return true;
718 }

816  uint8_t forceUpdate;
817  uint8_t connectionTimeout;
818  uint32_t timeNow;
819 
820  // Get stats
821  UAVTalkGetStats(telem->uavTalkCon, &utalkStats);
822 
823  // Get object data
824  FlightTelemetryStatsGet(&flightStats);
825  GCSTelemetryStatsGet(&gcsStats);
826 
827  // Update stats object
828  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
829  /* This is bogus because updates to GCSTelemetryStats trigger
830  * this call, in addition to just the periodic stuff.
831  */
832  flightStats.RxDataRate = (float)utalkStats.rxBytes / ((float)STATS_UPDATE_PERIOD_MS / 1000.0f);
833  flightStats.TxDataRate = (float)utalkStats.txBytes / ((float)STATS_UPDATE_PERIOD_MS / 1000.0f);
834  flightStats.RxFailures += utalkStats.rxErrors;
835  flightStats.TxFailures += telem->tx_errors;
836  flightStats.TxRetries += telem->tx_retries;
837  telem->tx_errors = 0;
838  telem->tx_retries = 0;
839  } else {
840  flightStats.RxDataRate = 0;
841  flightStats.TxDataRate = 0;
842  flightStats.RxFailures = 0;
843  flightStats.TxFailures = 0;
844  flightStats.TxRetries = 0;
845  telem->tx_errors = 0;
846  telem->tx_retries = 0;
847  }
848 
849  // Check for connection timeout
850  timeNow = PIOS_Thread_Systime();
851  if (utalkStats.rxObjects > 0) {
852  telem->time_of_last_update = timeNow;
853  }
854  if ((timeNow - telem->time_of_last_update) > CONNECTION_TIMEOUT_MS) {
855  connectionTimeout = 1;
856  } else {
857  connectionTimeout = 0;
858  }
859 
860  // Update connection state
861  forceUpdate = 1;
862  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED) {
863  // Wait for connection request
864  if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_HANDSHAKEREQ) {
865  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_HANDSHAKEACK;
866  }
867  } else if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_HANDSHAKEACK) {
868  // Wait for connection
869  if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_CONNECTED) {
870  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_CONNECTED;
871  } else if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_DISCONNECTED) {
872  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
873  }
874  } else if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
875  if (gcsStats.Status != GCSTELEMETRYSTATS_STATUS_CONNECTED || connectionTimeout) {
876  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
877  } else {
878  forceUpdate = 0;
879  }
880  } else {
881  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
882  }
883 
884  // Update the telemetry alarm
885  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
886  AlarmsClear(SYSTEMALARMS_ALARM_TELEMETRY);
887  } else {
888  AlarmsSet(SYSTEMALARMS_ALARM_TELEMETRY, SYSTEMALARMS_ALARM_ERROR);
889  }
890 
891  // Update object
892  FlightTelemetryStatsSet(&flightStats);
893 
894  // Force telemetry update if not connected
895  if (forceUpdate) {
896  FlightTelemetryStatsUpdated();
897  }
898 }
