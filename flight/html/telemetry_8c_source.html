157 static void registerObject(UAVObjHandle obj)
158 {
159  if (UAVObjIsMetaobject(obj)) {
160  /* Only connect change notifications for meta objects. No periodic updates */
161  UAVObjConnectQueue(obj, queue, EV_MASK_ALL_UPDATES);
162  return;
163  } else {
164  UAVObjMetadata metadata;
165  UAVObjUpdateMode updateMode;
166  UAVObjGetMetadata(obj, &metadata);
167  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
168 
169  // Setup object for periodic updates
170  UAVObjEvent ev = {
171  .obj = obj,
172  .instId = UAVOBJ_ALL_INSTANCES,
173  .event = EV_UPDATED_PERIODIC,
174  };
175 
176  /* Only create a periodic event for objects that are periodic */
177  if (updateMode == UPDATEMODE_PERIODIC) {
178  EventPeriodicQueueCreate(&ev, queue, 0);
179  }
180 
181  // Setup object for telemetry updates
182  updateObject(obj, EV_NONE);
183  }

190 static void updateObject(UAVObjHandle obj, int32_t eventType)
191 {
192  UAVObjMetadata metadata;
193  UAVObjUpdateMode updateMode;
194  int32_t eventMask;
195 
196  if (UAVObjIsMetaobject(obj)) {
197  /* This function updates the periodic updates for the object.
198  * Meta Objects cannot have periodic updates.
199  */
200  PIOS_Assert(false);
201  return;
202  }
203 
204  // Get metadata
205  UAVObjGetMetadata(obj, &metadata);
206  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
207 
208  // Setup object depending on update mode
209  switch (updateMode) {
210  case UPDATEMODE_PERIODIC:
211  // Set update period
212  setUpdatePeriod(obj, metadata.telemetryUpdatePeriod);
213 
214  // Connect queue
215  eventMask = EV_UPDATED_PERIODIC | EV_UPDATED_MANUAL;
216  UAVObjConnectQueueThrottled(obj, queue, eventMask, 0);
217  break;
218  case UPDATEMODE_ONCHANGE:
219  // Set update period
220  setUpdatePeriod(obj, 0);
221 
222  // Connect queue
223  eventMask = EV_UPDATED | EV_UPDATED_MANUAL;
224  UAVObjConnectQueueThrottled(obj, queue, eventMask, 0);
225  break;
226  case UPDATEMODE_THROTTLED:
227  setUpdatePeriod(obj, 0);
228 
229  eventMask = EV_UPDATED | EV_UPDATED_MANUAL;
230  UAVObjConnectQueueThrottled(obj, queue, eventMask,
231  metadata.telemetryUpdatePeriod);
232  break;
233  case UPDATEMODE_MANUAL:
234  // Set update period
235  setUpdatePeriod(obj, 0);
236 
237  // Connect queue
238  eventMask = EV_UPDATED_MANUAL;
239  UAVObjConnectQueueThrottled(obj, queue, eventMask, 0);
240  break;
241  }
242 }

247 static void processObjEvent(UAVObjEvent * ev)
248 {
249  UAVObjMetadata metadata;
250  FlightTelemetryStatsData flightStats;
251  int32_t retries;
252  int32_t success;
253 
254  if (ev->obj == 0) {
255  updateTelemetryStats();
256  } else if (ev->obj == GCSTelemetryStatsHandle()) {
257  gcsTelemetryStatsUpdated();
258  } else {
259  FlightTelemetryStatsGet(&flightStats);
260  // Get object metadata
261  UAVObjGetMetadata(ev->obj, &metadata);
262 
263  // Act on event
264  retries = 0;
265  success = -1;
266  if (ev->event == EV_UPDATED || ev->event == EV_UPDATED_MANUAL ||
267  ev->event == EV_UPDATED_PERIODIC) {
268  // Send update to GCS (with retries)
269  while (retries < MAX_RETRIES && success == -1) {
270  success = UAVTalkSendObject(uavTalkCon, ev->obj, ev->instId, UAVObjGetTelemetryAcked(&metadata), REQ_TIMEOUT_MS); // call blocks until ack is received or timeout
271 
272  ++retries;
273  }
274  // Update stats
275  txRetries += (retries - 1);
276  if (success == -1) {
277  ++txErrors;
278  }
279  }
280 
281  // If this is a metaobject then make necessary telemetry updates
282  if (UAVObjIsMetaobject(ev->obj)) {
283  updateObject(UAVObjGetLinkedObj(ev->obj), EV_NONE); // linked object will be the actual object the metadata are for
284  }
285  }
286 }

315 static bool processUsbActivity(bool seen_active)
316 {
317  uint32_t this_systime = PIOS_Thread_Systime();
318  uint32_t fixedup_time = this_systime +
319  USB_ACTIVITY_TIMEOUT_MS;
320 
321  if (seen_active) {
322  if (fixedup_time < this_systime) {
323  // (mostly) handle wrap.
324  usb_timeout_time = UINT32_MAX;
325  } else {
326  usb_timeout_time = fixedup_time;
327  }
328 
329  return true;
330  }
331 
332  uint32_t usb_time = usb_timeout_time;
333 
334  if (this_systime >= usb_time) {
335  usb_timeout_time = 0;
336 
337  return false;
338  }
339 
340  /* If the timeout is too far in the future ...
341  * (because of the above wrap case..) */
342  if (fixedup_time < usb_time) {
343  if (fixedup_time > this_systime) {
344  /* and we're not wrapped, then fixup the
345  * time. */
346  usb_timeout_time = fixedup_time;
347  }
348  }
349 
350  return true;
351 }

357 static void telemetryRxTask(void *parameters)
358 {
359  // Task loop
360  while (1) {
361  uintptr_t inputPort = getComPort();
362 
363  if (inputPort) {
364  // Block until data are available
365  uint8_t serial_data[16];
366  uint16_t bytes_to_process;
367 
368  bytes_to_process = PIOS_COM_ReceiveBuffer(inputPort, serial_data, sizeof(serial_data), 500);
369  if (bytes_to_process > 0) {
370  for (uint8_t i = 0; i < bytes_to_process; i++) {
371  UAVTalkProcessInputStream(uavTalkCon,serial_data[i]);
372  }
373 
374 #if defined(PIOS_INCLUDE_USB)

376  processUsbActivity(true);
377  }
378 #endif
379  }
380  } else {
381  PIOS_Thread_Sleep(3);
382  }
383  }

440 static void updateTelemetryStats()
441 {
442  UAVTalkStats utalkStats;
443  FlightTelemetryStatsData flightStats;
444  GCSTelemetryStatsData gcsStats;
445  uint8_t forceUpdate;
446  uint8_t connectionTimeout;
447  uint32_t timeNow;
448 
449  // Get stats
450  UAVTalkGetStats(uavTalkCon, &utalkStats);
451  UAVTalkResetStats(uavTalkCon);
452 
453  // Get object data
454  FlightTelemetryStatsGet(&flightStats);
455  GCSTelemetryStatsGet(&gcsStats);
456 
457  // Update stats object
458  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
459  flightStats.RxDataRate = (float)utalkStats.rxBytes / ((float)STATS_UPDATE_PERIOD_MS / 1000.0f);
460  flightStats.TxDataRate = (float)utalkStats.txBytes / ((float)STATS_UPDATE_PERIOD_MS / 1000.0f);
461  flightStats.RxFailures += utalkStats.rxErrors;
462  flightStats.TxFailures += txErrors;
463  flightStats.TxRetries += txRetries;
464  txErrors = 0;
465  txRetries = 0;
466  } else {
467  flightStats.RxDataRate = 0;
468  flightStats.TxDataRate = 0;
469  flightStats.RxFailures = 0;
470  flightStats.TxFailures = 0;
471  flightStats.TxRetries = 0;
472  txErrors = 0;
473  txRetries = 0;
474  }
475 
476  // Check for connection timeout
477  timeNow = PIOS_Thread_Systime();
478  if (utalkStats.rxObjects > 0) {
479  timeOfLastObjectUpdate = timeNow;
480  }
481  if ((timeNow - timeOfLastObjectUpdate) > CONNECTION_TIMEOUT_MS) {
482  connectionTimeout = 1;
483  } else {
484  connectionTimeout = 0;
485  }
486 
487  // Update connection state
488  forceUpdate = 1;
489  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED) {
490  // Wait for connection request
491  if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_HANDSHAKEREQ) {
492  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_HANDSHAKEACK;
493  }
494  } else if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_HANDSHAKEACK) {
495  // Wait for connection
496  if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_CONNECTED) {
497  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_CONNECTED;
498  } else if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_DISCONNECTED) {
499  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
500  }
501  } else if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
502  if (gcsStats.Status != GCSTELEMETRYSTATS_STATUS_CONNECTED || connectionTimeout) {
503  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
504  } else {
505  forceUpdate = 0;
506  }
507  } else {
508  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
509  }
510 
511  // Update the telemetry alarm
512  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
513  AlarmsClear(SYSTEMALARMS_ALARM_TELEMETRY);
514  } else {
515  AlarmsSet(SYSTEMALARMS_ALARM_TELEMETRY, SYSTEMALARMS_ALARM_ERROR);
516  }
517 
518  // Update object
519  FlightTelemetryStatsSet(&flightStats);
520 
521  // Force telemetry update if not connected
522  if (forceUpdate) {
523  FlightTelemetryStatsUpdated();
524  }
525 }
