180 static void at_new_actuators(UAVObjEvent * ev, void *ctx, void *obj, int len) {
181  (void) ev; (void) ctx;
182 
183  static bool last_sample_unpushed = false;
184 
185  static bool running = false;
186  static bool first_cycle = false;
187 
188  if (last_sample_unpushed) {
189  /* Last time we were unable to advance the write pointer.
190  * Try again, last chance! */
191  if (circ_queue_advance_write(at_queue)) {
192  at_points_spilled++;
193  }
194  }
195 
196  ActuatorDesiredData actuators;
197  ActuatorDesiredGet(&actuators);
198 
199  GyrosData g;
200  GyrosGet(&g);
201 
202  struct at_queued_data *q_item = circ_queue_write_pos(at_queue,
203  NULL, NULL);
204 
205  if (actuators.SystemIdentCycle == 0xffff) {
206  if (running) {
207  // Signify end of actuation stream.
208  q_item->throttle = THROTTLE_EOF;
209 
210  if (circ_queue_advance_write(at_queue) == 0) {
211  running = false;
212  first_cycle = false;
213  }
214  }
215 
216  return; // No data
217  }
218 
219  /* Simple state machine.
220  * !running !first_cycle -> running first_cycle -> running !first_cycle
221  */
222  if (!running || first_cycle) {
223  if (actuators.SystemIdentCycle == 0x0000) {
224  running = true;
225  first_cycle = ! first_cycle;
226  }
227  }
228 
229 #ifdef AUTOTUNE_AVERAGING_MODE
230  struct at_measurement *avg_point = &at_averages[actuators.SystemIdentCycle / AUTOTUNE_AVERAGING_DECIMATION];
231 
232  if (first_cycle) {
233  *avg_point = (struct at_measurement) { { 0 } };
234  }
235 
236  avg_point->y[0] += g.x;
237  avg_point->y[1] += g.y;
238  avg_point->y[2] += g.z;
239 
240  avg_point->u[0] += actuators.Roll;
241  avg_point->u[1] += actuators.Pitch;
242  avg_point->u[2] += actuators.Yaw;
243 #endif
244 
245  q_item->sample_num = actuators.SystemIdentCycle;
246 
247  q_item->meas.y[0] = g.x;
248  q_item->meas.y[1] = g.y;
249  q_item->meas.y[2] = g.z;
250 
251  q_item->meas.u[0] = actuators.Roll;
252  q_item->meas.u[1] = actuators.Pitch;
253  q_item->meas.u[2] = actuators.Yaw;
254 
255  q_item->throttle = actuators.Thrust;
256 
257  if (circ_queue_advance_write(at_queue) != 0) {
258  last_sample_unpushed = true;
259  } else {
260  last_sample_unpushed = false;
261  }
262 }

342 static void AutotuneTask(void *parameters)
343 {
344  enum AUTOTUNE_STATE state = AT_INIT;
345 
346  float X[AF_NUMX] = {0};
347  float P[AF_NUMP] = {0};
348  float noise[3] = {0};
349 
350  uint16_t last_samp = 0;
351 
352  const uint32_t YIELD_MS = 2;
353 
354 #ifdef AUTOTUNE_AVERAGING_MODE
355  /* Wait for stabilization module to complete startup and tell us
356  * its dT, etc.
357  */
358  while (!ident_wiggle_points) {
359  PIOS_Thread_Sleep(25);
360  }
361 
362  decim_wiggle_points = ident_wiggle_points / AUTOTUNE_AVERAGING_DECIMATION;
363 
364  uint16_t buf_size = sizeof(*at_averages) * decim_wiggle_points;
365  at_averages = PIOS_malloc(buf_size);
366 
367  while (!at_averages) {
368  /* Infinite loop because we couldn't get our buffer */
369  /* Assert alarm XXX? */
370  PIOS_Thread_Sleep(2500);
371  }
372 #endif /* AUTOTUNE_AVERAGING_MODE */
373 
374  float dT_expected = 0.001f;
375  uint16_t sample_rate = PIOS_SENSORS_GetSampleRate(PIOS_SENSOR_GYRO);
376 
377  if (sample_rate) {
378  dT_expected = 1.0f / sample_rate;
379  }
380 
381  ActuatorDesiredConnectCallback(at_new_actuators);
382 
383  bool save_needed = false;
384 
385  while(1) {
386  static uint32_t update_counter = 0;
387 
388  uint8_t armed;
389 
390  FlightStatusArmedGet(&armed);
391 
392  if (save_needed) {
393  if (armed == FLIGHTSTATUS_ARMED_DISARMED) {
394 #ifdef AUTOTUNE_AVERAGING_MODE
395  if (autotune_save_averaging()) {
396  // Try again in a second, I guess.
397  PIOS_Thread_Sleep(1000);
398  continue;
399  }
400 #endif /* AUTOTUNE_AVERAGING_MODE */
401 
402  // Save the settings locally.
403  UAVObjSave(SystemIdentHandle(), 0);
404  state = AT_INIT;
405 
406  save_needed = false;
407  }
408 
409  }
410 
411  // Only allow this module to run when roll axis is in a tune
412  // mode
413  uint8_t stab_modes[STABILIZATIONDESIRED_STABILIZATIONMODE_NUMELEM];
414 
415  StabilizationDesiredStabilizationModeGet(stab_modes);
416 
417  switch (stab_modes[0]) {
418  case STABILIZATIONDESIRED_STABILIZATIONMODE_SYSTEMIDENT:
419  case STABILIZATIONDESIRED_STABILIZATIONMODE_SYSTEMIDENTRATE:
420  break;
421  default:
422  state = AT_INIT;
423  PIOS_Thread_Sleep(30);
424  continue;
425  }
426 
427  switch(state) {
428  case AT_INIT:
429  // Reset save status; only save if this tune
430  // completes.
431  save_needed = false;
432 
433  // Only start when armed and flying
434  if (armed == FLIGHTSTATUS_ARMED_ARMED) {
435 
436  af_init(X,P);
437 
438  state = AT_START;
439 
440  }
441 
442  break;
443 
444  case AT_START:
445  /* Drain the queue of all current data */
446  circ_queue_clear(at_queue);
447 
448  /* And reset the point spill counter */
449 
450  update_counter = 0;
451  at_points_spilled = 0;
452 
453  throttle_accumulator = 0;
454 
455  state = AT_WAITFIRSTPOINT;
456 
457  break;
458 
459  case AT_WAITFIRSTPOINT:
460  {
461  struct at_queued_data *pt;
462 
463  /* Grab an autotune point */
464  pt = circ_queue_read_pos(at_queue, NULL, NULL);
465 
466  if (!pt) {
467  // Still no data showing up
468  break;
469  }
470 
471  if (pt->throttle == THROTTLE_EOF) {
472  // Ignore old EOFs, just in case
473  break;
474  }
475 
476  state = AT_RUN;
477  }
478 
479  case AT_RUN:
480  while (true) {
481  struct at_queued_data *pt;
482 
483  /* Grab an autotune point */
484  pt = circ_queue_read_pos(at_queue, NULL, NULL);
485 
486  if (!pt) {
487  /* We've drained the buffer
488  * fully. Yay! */
489  break;
490  }
491 
492  if (pt->throttle == THROTTLE_EOF) {
493  // End of actuation, clean up
494  float hover_throttle = ((float)(throttle_accumulator/update_counter))/10000.0f;
495  UpdateSystemIdent(X, noise, update_counter, at_points_spilled, hover_throttle, true);
496 
497  save_needed = true;
498  state = AT_WAITING;
499 
500  break;
501  }
502 
503  /* calculate time between successive
504  * points */
505 
506  uint16_t samp_interval = pt->sample_num - last_samp;
507 
508  if (samp_interval > ident_wiggle_points) {
509  /* Handle wrap case. */
510  samp_interval += ident_wiggle_points;
511  }
512 
513  if ((samp_interval == 0) ||
514  (samp_interval > 10)) {
515  samp_interval = 10;
516  }
517 
518  float dT_s = samp_interval * dT_expected;
519 
520  last_samp = pt->sample_num;
521 
522  af_predict(X, P, &pt->meas, dT_s, pt->throttle);
523 
524  for (uint32_t i = 0; i < 3; i++) {
525  const float NOISE_ALPHA = 0.9997f; // 10 second time constant at 300 Hz
526  noise[i] = NOISE_ALPHA * noise[i] + (1-NOISE_ALPHA) * (pt->meas.y[i] - X[i]) * (pt->meas.y[i] - X[i]);
527  }
528 
529  //This will work up to 8kHz with an 89% throttle position before overflow
530  throttle_accumulator += 10000 * pt->throttle;
531 
532  /* Free the buffer containing an AT point */
533  circ_queue_read_completed(at_queue);
534 
535  // Update uavo every 256 cycles to avoid
536  // telemetry spam
537  if (!((++update_counter) & 0xff)) {
538  float hover_throttle = ((float)(throttle_accumulator/update_counter))/10000.0f;
539  UpdateSystemIdent(X, noise, update_counter, at_points_spilled, hover_throttle, false);
540  }
541  }
542 
543  break;
544 
545  case AT_WAITING:
546  default:
547  break;
548  }
549 
550  PIOS_Thread_Sleep(YIELD_MS);
551  }
552 }

562 __attribute__((always_inline)) static inline void af_predict(float X[AF_NUMX], float P[AF_NUMP], const struct at_measurement *measurement, const float dT_s, const float t_in)
563 {
564  const float Ts = dT_s;
565  const float Tsq = Ts * Ts;
566  const float Tsq3 = Tsq * Ts;
567  const float Tsq4 = Tsq * Tsq;
568 
569  // for convenience and clarity code below uses the named versions of
570  // the state variables
571  float w1 = X[0]; // roll rate estimate
572  float w2 = X[1]; // pitch rate estimate
573  float w3 = X[2]; // yaw rate estimate
574  float u1 = X[3]; // scaled roll torque
575  float u2 = X[4]; // scaled pitch torque
576  float u3 = X[5]; // scaled yaw torque
577  const float e_b1 = expapprox(X[6]); // roll torque scale
578  const float b1 = X[6];
579  const float e_b2 = expapprox(X[7]); // pitch torque scale
580  const float b2 = X[7];
581  const float e_b3 = expapprox(X[8]); // yaw torque scale
582  const float b3 = X[8];
583  const float e_tau = expapprox(X[9]); // time response of the motors
584  const float tau = X[9];
585  const float bias1 = X[10]; // bias in the roll torque
586  const float bias2 = X[11]; // bias in the pitch torque
587  const float bias3 = X[12]; // bias in the yaw torque
588 
589  // inputs to the system (roll, pitch, yaw)
590  const float u1_in = 4*t_in*measurement->u[0];
591  const float u2_in = 4*t_in*measurement->u[1];
592  const float u3_in = 4*t_in*measurement->u[2];
593 
594  // measurements from gyro
595  const float gyro_x = measurement->y[0];
596  const float gyro_y = measurement->y[1];
597  const float gyro_z = measurement->y[2];
598 
599  // update named variables because we want to use predicted
600  // values below
601  w1 = X[0] = w1 - Ts*bias1*e_b1 + Ts*u1*e_b1;
602  w2 = X[1] = w2 - Ts*bias2*e_b2 + Ts*u2*e_b2;
603  w3 = X[2] = w3 - Ts*bias3*e_b3 + Ts*u3*e_b3;
604  u1 = X[3] = (Ts*u1_in)/(Ts + e_tau) + (u1*e_tau)/(Ts + e_tau);
605  u2 = X[4] = (Ts*u2_in)/(Ts + e_tau) + (u2*e_tau)/(Ts + e_tau);
606  u3 = X[5] = (Ts*u3_in)/(Ts + e_tau) + (u3*e_tau)/(Ts + e_tau);
607  // X[6] to X[12] unchanged
608 
609  /**** filter parameters ****/
610  const float q_w = 1e-3f;
611  const float q_ud = 1e-3f;
612  const float q_B = 1e-6f;
613  const float q_tau = 1e-6f;
614  const float q_bias = 1e-19f;
615  const float s_a = 150.0f; // expected gyro measurement noise
616 
617  const float Q[AF_NUMX] = {q_w, q_w, q_w, q_ud, q_ud, q_ud, q_B, q_B, q_B, q_tau, q_bias, q_bias, q_bias};
618 
619  float D[AF_NUMP];
620  for (uint32_t i = 0; i < AF_NUMP; i++)
621  D[i] = P[i];
622 
623  const float e_tau2 = e_tau * e_tau;
624  const float e_tau3 = e_tau * e_tau2;
625  const float e_tau4 = e_tau2 * e_tau2;
626  const float Ts_e_tau2 = (Ts + e_tau) * (Ts + e_tau);
627  const float Ts_e_tau4 = Ts_e_tau2 * Ts_e_tau2;
628 
629  // covariance propagation - D is stored copy of covariance
630  P[0] = D[0] + Q[0] + 2*Ts*e_b1*(D[3] - D[28] - D[9]*bias1 + D[9]*u1) + Tsq*(e_b1*e_b1)*(D[4] - 2*D[29] + D[32] - 2*D[10]*bias1 + 2*D[30]*bias1 + 2*D[10]*u1 - 2*D[30]*u1 + D[11]*(bias1*bias1) + D[11]*(u1*u1) - 2*D[11]*bias1*u1);
631  P[1] = D[1] + Q[1] + 2*Ts*e_b2*(D[5] - D[33] - D[12]*bias2 + D[12]*u2) + Tsq*(e_b2*e_b2)*(D[6] - 2*D[34] + D[37] - 2*D[13]*bias2 + 2*D[35]*bias2 + 2*D[13]*u2 - 2*D[35]*u2 + D[14]*(bias2*bias2) + D[14]*(u2*u2) - 2*D[14]*bias2*u2);
632  P[2] = D[2] + Q[2] + 2*Ts*e_b3*(D[7] - D[38] - D[15]*bias3 + D[15]*u3) + Tsq*(e_b3*e_b3)*(D[8] - 2*D[39] + D[42] - 2*D[16]*bias3 + 2*D[40]*bias3 + 2*D[16]*u3 - 2*D[40]*u3 + D[17]*(bias3*bias3) + D[17]*(u3*u3) - 2*D[17]*bias3*u3);
