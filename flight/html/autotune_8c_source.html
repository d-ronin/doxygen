149 static void at_new_actuators(UAVObjEvent * ev, void *ctx, void *obj, int len) {
150  (void) ev; (void) ctx;
151 
152  static bool first_cycle = false;
153 
154  ActuatorDesiredData actuators;
155  ActuatorDesiredGet(&actuators);
156 
157  GyrosData g;
158  GyrosGet(&g);
159 
160  if (actuators.SystemIdentCycle == 0xffff) {
161  if (tune_running) {
162  tune_running = false;
163  first_cycle = false;
164  }
165 
166  return; // No data
167  }
168 
169  /* Simple state machine.
170  * !running !first_cycle -> running first_cycle -> running !first_cycle
171  */
172  if (!tune_running || first_cycle) {
173  if (actuators.SystemIdentCycle == 0x0000) {
174  if (!tune_running) {
175  update_counter = 0;
176  throttle_accumulator = 0;
177  }
178 
179  tune_running = true;
180  first_cycle = ! first_cycle;
181  }
182  }
183 
184  struct at_measurement *avg_point = &at_averages[actuators.SystemIdentCycle / AUTOTUNE_AVERAGING_DECIMATION];
185 
186  if (first_cycle) {
187  *avg_point = (struct at_measurement) { { 0 } };
188  }
189 
190  avg_point->y[0] += g.x;
191  avg_point->y[1] += g.y;
192  avg_point->y[2] += g.z;
193 
194  avg_point->u[0] += actuators.Roll;
195  avg_point->u[1] += actuators.Pitch;
196  avg_point->u[2] += actuators.Yaw;
197 
198  update_counter++;
199  throttle_accumulator += 10000 * actuators.Thrust;
200 }

214 static int autotune_save_averaging() {
215  uintptr_t part_id;
216 
217  if (PIOS_FLASH_find_partition_id(
218  FLASH_PARTITION_LABEL_AUTOTUNE,
219  &part_id)) {
220  return -1;
221  }
222 
223 
224  if (PIOS_FLASH_start_transaction(part_id)) {
225  return -2;
226  }
227 
228  if (PIOS_FLASH_erase_partition(part_id)) {
229  PIOS_FLASH_end_transaction(part_id);
230  return -3;
231  }
232 
233  struct at_flash_header hdr = {
234  .magic = ATFLASH_MAGIC,
235  .wiggle_points = decim_wiggle_points,
236  .aux_data_len = 0,
237  .sample_rate = PIOS_SENSORS_GetSampleRate(PIOS_SENSOR_GYRO) / AUTOTUNE_AVERAGING_DECIMATION,
238  };
239 
240  uint32_t offset = 0;
241 
242  if (PIOS_FLASH_write_data(part_id, 0, (uint8_t *) &hdr,
243  sizeof(hdr))) {
244  PIOS_FLASH_end_transaction(part_id);
245  return -4;
246  }
247 
248  offset += sizeof(hdr);
249 
250  if (PIOS_FLASH_write_data(part_id, offset, (uint8_t *) at_averages,
251  sizeof(*at_averages) * decim_wiggle_points)) {
252  PIOS_FLASH_end_transaction(part_id);
253  return -5;
254  }
255 
256  PIOS_FLASH_end_transaction(part_id);
257  return 0;
258 }

263 static void AutotuneTask(void *parameters)
264 {
265  const uint32_t YIELD_MS = 100;
266 
267  /* Wait for stabilization module to complete startup and tell us
268  * its dT, etc.
269  */
270  while (!ident_wiggle_points) {
271  PIOS_Thread_Sleep(25);
272  }
273 
274  decim_wiggle_points = ident_wiggle_points / AUTOTUNE_AVERAGING_DECIMATION;
275 
276  uint16_t buf_size = sizeof(*at_averages) * decim_wiggle_points;
277  at_averages = PIOS_malloc(buf_size);
278 
279  while (!at_averages) {
280  /* Infinite loop because we couldn't get our buffer */
281  /* Assert alarm XXX? */
282  PIOS_Thread_Sleep(2500);
283  }
284 
285  ActuatorDesiredConnectCallback(at_new_actuators);
286 
287  bool save_needed = false;
288  enum autotune_state state = AT_INIT;
289 
290  while(1) {
291  uint8_t armed;
292 
293  FlightStatusArmedGet(&armed);
294 
295  if (save_needed) {
296  if (armed == FLIGHTSTATUS_ARMED_DISARMED) {
297  if (autotune_save_averaging()) {
298  // Try again in a second, I guess.
299  PIOS_Thread_Sleep(1000);
300  continue;
301  }
302 
303  float hover_throttle = ((float) (throttle_accumulator / update_counter)) / 10000.0f;
304 
305  UpdateSystemIdent(update_counter, hover_throttle,
306  true);
307 
308  // Save the UAVO locally.
309  UAVObjSave(SystemIdentHandle(), 0);
310  state = AT_INIT;
311 
312  save_needed = false;
313  }
314 
315  }
316 
317  switch(state) {
318  default:
319  case AT_INIT:
320  if (tune_running) {
321  // Reset save status; only save if this
322  // tune completes.
323  save_needed = false;
324  state = AT_RUN;
325  }
326 
327  break;
328 
329  case AT_RUN:
330  (void) 0;
331 
332  float hover_throttle = ((float) (throttle_accumulator / update_counter)) / 10000.0f;
333 
334  UpdateSystemIdent(update_counter, hover_throttle,
335  false);
336 
337  if (!tune_running) {
338  /* Threshold: 24 seconds of data @ 500Hz */
339  if (update_counter > 12000) {
340  save_needed = true;
341  }
342 
343  state = AT_INIT;
344  }
345 
346  break;
347  }
348 
349  PIOS_Thread_Sleep(YIELD_MS);
350  }
351 }
