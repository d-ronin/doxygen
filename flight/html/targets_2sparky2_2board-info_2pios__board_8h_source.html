136 #define PIOS_SYSCLK 168000000
137 // Peripherals that belongs to APB1 are:
138 // DAC |PWR |CAN1,2
139 // I2C1,2,3 |UART4,5 |USART3,2
140 // I2S3Ext |SPI3/I2S3 |SPI2/I2S2
141 // I2S2Ext |IWDG |WWDG
142 // RTC/BKP reg
143 // TIM2,3,4,5,6,7,12,13,14
144 
145 // Calculated as SYSCLK / APBPresc * (APBPre == 1 ? 1 : 2)
146 // Default APB1 Prescaler = 4
147 #define PIOS_PERIPHERAL_APB1_CLOCK (PIOS_SYSCLK / 4)
148 
149 // Peripherals belonging to APB2
