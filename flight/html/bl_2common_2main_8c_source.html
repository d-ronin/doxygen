273 {
274  if (context->fsm_timer_enabled) {
275  if (elapsed_us >= context->fsm_timer_remaining_us) {
276  /* Timer has expired */
277  context->fsm_timer_remaining_us = 0;

279  /* Timer is still running, account for the elapsed time */
280  context->fsm_timer_remaining_us -= elapsed_us;

315 static void go_jumping_to_app(struct bl_fsm_context * context)
316 {
317  bl_fsm_timer_cancel(context);
318 
319  PIOS_ANNUNC_On(PIOS_LED_HEARTBEAT);
320 
321  /* Recover a pointer to the bootloader board info blob */
322  const struct pios_board_info * bdinfo = &pios_board_info_blob;
323 
324  /* Load the application's PC and SP from the beginning of the firmware flash bank */
325  uintptr_t initial_sp = *((__IO uint32_t*) bdinfo->fw_base);
326  uintptr_t initial_pc = *((__IO uint32_t*) (bdinfo->fw_base + 4));
327 
328  /* Simple sanity check to make sure that the SP actually points to RAM */
329  if (((initial_sp & 0x2FFE0000) != 0x20000000) && ((initial_sp & 0x1FFE0000) != 0x10000000)) {
330  /* sp is not sane, don't attempt the jump */
331  return;
332  }
333 
334  /* Simple sanity check to make sure that the PC actually points to FLASH */
335  if ((initial_pc & 0x08000000) != 0x08000000) {
336  /* pc is not sane, don't attempt the jump */
337  return;
338  }
339 
340  /*
341  * pc and sp are sane, try the jump to the application
342  */
343 
344  /* Disable USB */
345  PIOS_USBHOOK_Deactivate();
346 
347  /* Re-lock the internal flash */
348  FLASH_Lock();
349 
350  /* Reset all peripherals */
351  RCC_APB2PeriphResetCmd(0xffffffff, ENABLE);
352  RCC_APB1PeriphResetCmd(0xffffffff, ENABLE);
353  RCC_APB2PeriphResetCmd(0xffffffff, DISABLE);
354  RCC_APB1PeriphResetCmd(0xffffffff, DISABLE);
355 
356  /* Initialize user application's stack pointer */
357  __set_MSP(initial_sp);
358 
359  /* Jump to the application entry point */
360  ((void (*)(void))initial_pc)();
361 }

393 static void bl_fsm_process_auto(struct bl_fsm_context *context)
394 {
395  while (bl_transitions[context->curr_state].next_state[BL_EVENT_AUTO]) {
396  context->curr_state = bl_transitions[context->curr_state].next_state[BL_EVENT_AUTO];
397 
398  /* Call the entry function (if any) for the next state. */
399  if (bl_transitions[context->curr_state].entry_fn) {
400  bl_transitions[context->curr_state].entry_fn(context);
401  }
402  }

405 static void bl_fsm_inject_event(struct bl_fsm_context * context, enum bl_events event)
406 {
407  /*
408  * Move to the next state
409  *
410  * This is done prior to calling the new state's entry function to
411  * guarantee that the entry function never depends on the previous
412  * state. This way, it cannot ever know what the previous state was.
413  */
414  context->curr_state = bl_transitions[context->curr_state].next_state[event];
415 
416  /* Call the entry function (if any) for the next state. */
417  if (bl_transitions[context->curr_state].entry_fn) {
418  bl_transitions[context->curr_state].entry_fn(context);
419  }
420 
421  /* Process any AUTO transitions in the FSM */
422  bl_fsm_process_auto(context);
423 }

435 int main(void)
436 {
437  /* Configure and enable system clocks */
438  PIOS_SYS_Init();
439 
440  /* Initialize the board-specific HW configuration */
441  PIOS_Board_Init();
442 
443  /* Initialize the bootloader FSM */
444  struct bl_fsm_context bl_fsm_context;
445  bl_fsm_init(&bl_fsm_context);
446 
447  /* Check if the user has requested that we boot into DFU mode */
448  PIOS_IAP_Init();
449 
450 #ifdef F1_UPGRADER
451  PIOS_IAP_ClearRequest();
452  bl_fsm_inject_event(&bl_fsm_context, BL_EVENT_ENTER_DFU);
453 #else
454  if (PIOS_IAP_CheckRequest() == true) {
455  /* User has requested that we boot into DFU mode */
456  PIOS_IAP_ClearRequest();
457  bl_fsm_inject_event(&bl_fsm_context, BL_EVENT_ENTER_DFU);
458  } else if (PIOS_Boot_CheckRequest() == true) {
459  /* User has requested that we boot into firmware */
460  PIOS_IAP_ClearRequest();
461  PIOS_DELAY_WaitmS(1000);//needed so OS can detect FW USB disconnect
462  bl_fsm_inject_event(&bl_fsm_context, BL_EVENT_FORCE_BOOT);
463  }
464 #endif
465 
466  /* Assume no USB connected */
467  bool usb_connected = false;
468 
469  uint32_t prev_ticks = PIOS_DELAY_GetuS();
470  while (1) {
471  uint32_t elapsed_ticks = PIOS_DELAY_GetuSSince(prev_ticks);
472 
473  /* Run the fsm timer */
474  if (elapsed_ticks) {
475  bl_fsm_timer_add_ticks(&bl_fsm_context, elapsed_ticks);
476  if (bl_fsm_timer_expired_p(&bl_fsm_context) == true) {
477  /* Timer has expired, inject an expiry event */
478  bl_fsm_inject_event(&bl_fsm_context, BL_EVENT_TIMER_EXPIRY);
479  }
480 
481  /* pulse the LEDs */
482  led_pwm_add_ticks(&bl_fsm_context.leds, elapsed_ticks);
483  led_pwm_update_leds(&bl_fsm_context.leds);
484 
485  prev_ticks += elapsed_ticks;
486  }
487 
488  /* check for changes in USB connection state */
489  if (!usb_connected) {
490  if (PIOS_USB_CableConnected(0)) {
491  bl_fsm_inject_event(&bl_fsm_context, BL_EVENT_USB_CONNECTED);
492  usb_connected = true;
493  }
494  } else {
495  if (!PIOS_USB_CableConnected(0)) {
496  bl_fsm_inject_event(&bl_fsm_context, BL_EVENT_USB_DISCONNECTED);
497  usb_connected = false;
498  }
499  }
500 
501  /* Manage any active read transfers */
502  if (bl_fsm_get_state(&bl_fsm_context) == BL_STATE_DFU_READ_IN_PROGRESS) {
503  if (bl_xfer_send_next_read_packet(&bl_fsm_context.xfer)) {
504  /* Sent a packet. Are we finished? */
505  if (bl_xfer_completed_p(&bl_fsm_context.xfer)) {
506  /* Transfer is finished */
507  bl_fsm_inject_event(&bl_fsm_context, BL_EVENT_TRANSFER_DONE);
508  }
509  } else {
510  /* Failed to send packet, fail the transfer */
511  bl_fsm_inject_event(&bl_fsm_context, BL_EVENT_TRANSFER_ERROR);
512  }
513  }
514 
515  /* check for an incoming packet */
516  bool packet_available = PIOS_COM_MSG_Receive(PIOS_COM_TELEM_USB,
517  (uint8_t *)&bl_fsm_context.msg,
518  sizeof(bl_fsm_context.msg));
519  if (packet_available) {
520  process_packet_rx(&bl_fsm_context, &bl_fsm_context.msg);
521  }
522  }
523 }

574 static void process_packet_rx(struct bl_fsm_context * context, const struct bl_messages * msg)
575 {
576  /* Sanity checks */
577  if (!msg)
578  return;
579 
580  /* Split the flags_command field into flags and command */
581  uint8_t flags = msg->flags_command & BL_MSG_FLAGS_MASK;
582  uint8_t command = msg->flags_command & BL_MSG_COMMAND_MASK;
583 
584  struct bl_messages echo;
585  if (flags & BL_MSG_FLAGS_ECHO_REQ) {
586  /* Request should be echoed back to the sender. Make a copy of the request */
587  memcpy (&echo, msg, sizeof(echo));
588  }
589 
590  switch (command) {
591  case BL_MSG_CAP_REQ:
592  bl_send_capabilities(context, msg->v.cap_req.device_number);
593  break;
594  case BL_MSG_ENTER_DFU:
595  if (bl_select_dfu_device(context, msg->v.enter_dfu.device_number)) {
596  bl_fsm_inject_event(context, BL_EVENT_ENTER_DFU);
597  } else {
598  /* Failed to select the requested device */
599  }
600  break;
601  case BL_MSG_JUMP_FW:
602  if (BE16_TO_CPU(msg->v.jump_fw.safe_word) == 0x5afe) {
603  /* Force board into safe mode */
604  PIOS_IAP_WriteBootCount(0xFFFF);
605  }
606  bl_fsm_inject_event(context, BL_EVENT_JUMP_TO_APP);
607  break;
608  case BL_MSG_RESET:
609  PIOS_IRQ_Disable();
610  PIOS_DELAY_WaitmS(1500);
611 
612  PIOS_SYS_Reset();
613  break;
614  case BL_MSG_OP_ABORT:
615  bl_fsm_inject_event(context, BL_EVENT_ABORT_OPERATION);
616  break;
617 
618  case BL_MSG_WRITE_START:
619  if (bl_xfer_write_start(&context->xfer, &(msg->v.xfer_start))) {
620  bl_fsm_inject_event(context, BL_EVENT_WRITE_START);
621  } else {
622  /* Failed to start the write */
623  }
624  break;
625  case BL_MSG_WRITE_CONT:
626  if (bl_fsm_get_state(context) == BL_STATE_DFU_WRITE_IN_PROGRESS) {
627  if (!bl_xfer_write_cont(&context->xfer, &(msg->v.xfer_cont))) {
628  /* Invalid packet, fail the transfer */
629  bl_fsm_inject_event(context, BL_EVENT_TRANSFER_ERROR);
630  }
631  }
632  break;
633 
634  case BL_MSG_OP_END:
635  if (bl_fsm_get_state(context) == BL_STATE_DFU_WRITE_IN_PROGRESS) {
636  if (bl_xfer_completed_p(&context->xfer)) {
637  /* Transfer is finished, check the CRC */
638  if (bl_xfer_crc_ok_p(&context->xfer)) {
639  bl_fsm_inject_event(context, BL_EVENT_TRANSFER_DONE);
640  } else {
641  /* Mismatched CRC */
642  bl_fsm_inject_event(context, BL_EVENT_TRANSFER_ERROR);
643  }
644  }
645  }
646  break;
647 
648  case BL_MSG_READ_START:
649  if (bl_xfer_read_start(&context->xfer, &(msg->v.xfer_start))) {
650  bl_fsm_inject_event(context, BL_EVENT_READ_START);
651  } else {
652  /* Failed to start the read */
653  }
654  break;
655 
656  case BL_MSG_STATUS_REQ:
657  bl_send_status(context);
658  break;
659 
660  case BL_MSG_WIPE_PARTITION:
661  bl_xfer_wipe_partition(&(msg->v.wipe_partition));
662  break;
663 
664  case BL_MSG_CAP_REP:
665  case BL_MSG_STATUS_REP:
666  case BL_MSG_READ_CONT:
667  /* We've received a *reply* packet when we expected a request. */
668  break;
669  case BL_MSG_RESERVED:
670  /* We've received an reserved command. Ignore this packet. */
671  break;
672 
673  default:
674  /* We've received an unknown command. Ignore this packet. */
675  break;
676  }
677 
678  if (flags & BL_MSG_FLAGS_ECHO_REQ) {
679  /* Reflect the original message back to the originator as an echo reply */
680  echo.flags_command &= ~(BL_MSG_FLAGS_MASK);
681  echo.flags_command |= BL_MSG_FLAGS_ECHO_REP;
682 
683  PIOS_COM_MSG_Send(PIOS_COM_TELEM_USB, (uint8_t *)&echo, sizeof(echo));
684  }
685 
686 }
