220  * Note: this is regular crc-16-ccitt CRC

223  * Poly = 0x1021

226  * Xor_Out = 0x0000
227  * ReflectOut = False

229  * @brief Update a CRC with a data buffer

236 {

239  uint8_t c;
240

242  c = *data++;
