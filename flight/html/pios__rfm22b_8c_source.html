437  // Do add'l validation that the device is present so we don't
438  // incorrectly talk to other SPI devices
439  uint8_t version = rfm22_read(rfm22b_dev, RFM22_DEVICE_VERSION);
440 
441  if ((version != RFM22_DEVICE_VERSION_V2) &&
442  (version != RFM22_DEVICE_VERSION_A0) &&
443  (version != RFM22_DEVICE_VERSION_B1)) {
444  return -1;
445  }
446 
447  // Initialize our configuration parameters
448  rfm22b_dev->datarate = RFM22B_DEFAULT_RX_DATARATE;
449  rfm22b_dev->tx_power = RFM22B_DEFAULT_TX_POWER;
450  rfm22b_dev->coordinator = false;
451  rfm22b_dev->coordinatorID = 0;
452 
453  // Initialize the com callbacks.
454  rfm22b_dev->rx_in_cb = NULL;
455  rfm22b_dev->tx_out_cb = NULL;
456 
457  // Initialzie the PPM callback.
458  rfm22b_dev->rfm22b_rcvr_id = 0;
459 
460  // Initialize the stats.
461  rfm22b_dev->stats.rx_good = 0;
462  rfm22b_dev->stats.rx_corrected = 0;
463  rfm22b_dev->stats.rx_error = 0;
464  rfm22b_dev->stats.resets = 0;
465  rfm22b_dev->stats.timeouts = 0;
466  rfm22b_dev->stats.link_quality = 0;
467  rfm22b_dev->stats.rssi = 0;
468

470  PIOS_RFM22B_Config(*rfm22b_id,
471  RFM22B_DEFAULT_RX_DATARATE,
472  RFM22B_DEFAULT_MIN_CHANNEL,
473  RFM22B_DEFAULT_MAX_CHANNEL,
474  0, false, false, false);
475 
476  // Bind the configuration to the device instance
477  rfm22b_dev->cfg = *cfg;
478 
479  // Create our (hopefully) unique 32 bit id from the processor serial number.
480  uint8_t crcs[] = { 0, 0, 0, 0 };
481  {
482  char serial_no_str[33];
483  PIOS_SYS_SerialNumberGet(serial_no_str);
484  // Create a 32 bit value using 4 8 bit CRC values.
485  for (uint8_t i = 0; serial_no_str[i] != 0; ++i) {
486  crcs[i % 4] =
487  PIOS_CRC_updateByte(crcs[i % 4],
488  serial_no_str[i]);
489  }
490  }
491  rfm22b_dev->deviceID = crcs[0] | crcs[1] << 8 | crcs[2] << 16 | crcs[3] << 24;
492  if (rfm22b_dev->deviceID == 0)
493  rfm22b_dev->deviceID = 1;
494  DEBUG_PRINTF(2, "RF device ID: %x\n\r", rfm22b_dev->deviceID);
495 
496  // Initialize the external interrupt.
497  PIOS_EXTI_Init(cfg->exti_cfg);
498 
499  // Register the watchdog timer for the radio driver task
500 #if defined(PIOS_INCLUDE_WDG) && defined(PIOS_WDG_RFM22B)
501  PIOS_WDG_RegisterFlag(PIOS_WDG_RFM22B);
502 #endif /* PIOS_WDG_RFM22B */
503 
504  // Initialize the ECC library.
505  initialize_ecc();
506 
507  // Set the state to initializing.
508  rfm22b_dev->state = RADIO_STATE_UNINITIALIZED;
509 
510  // Initialize the radio device.
511  init_requested = true;
512 
513  // Start the driver task. This task controls the radio state machine and removed all of the IO from the IRQ handler.
514  rfm22b_dev->taskHandle = PIOS_Thread_Create(pios_rfm22_task, "PIOS_RFM22B_Task", STACK_SIZE_BYTES, (void *)rfm22b_dev, TASK_PRIORITY);
515 
516  return 0;
517 }

645 void PIOS_RFM22B_Config(uint32_t rfm22b_id,
646  HwSharedMaxRfSpeedOptions datarate,
647  uint8_t min_chan, uint8_t max_chan,
648  uint32_t coordinator_id,
649  bool oneway, bool ppm_mode,
650  bool ppm_only)
651 {
652  struct pios_rfm22b_dev *rfm22b_dev = (struct pios_rfm22b_dev *)rfm22b_id;
653 
654  if (!PIOS_RFM22B_Validate(rfm22b_dev)) {
655  return;
656  }
657 
658  bool coordinator = coordinator_id == 0;
659 
660  ppm_mode = ppm_mode || ppm_only;
661  rfm22b_dev->coordinator = coordinator;
662  rfm22b_dev->coordinatorID = coordinator_id;
663  rfm22b_dev->ppm_send_mode = ppm_mode && coordinator;
664  rfm22b_dev->ppm_recv_mode = ppm_mode && !coordinator;
665 
666  // If datarate is so slow that we can only do PPM, force this
667  if (ppm_mode && (datarate <= RFM22B_PPM_ONLY_DATARATE)) {
668  ppm_only = true;
669  }
670  rfm22b_dev->ppm_only_mode = ppm_only;
671  if (ppm_only) {
672  rfm22b_dev->one_way_link = true;
673  datarate = RFM22B_PPM_ONLY_DATARATE;
674  rfm22b_dev->datarate = RFM22B_PPM_ONLY_DATARATE;
675  } else {
676  rfm22b_dev->one_way_link = false;
677  rfm22b_dev->datarate = datarate;
678  }
679 
680  rfm22b_dev->packet_time = (ppm_mode ? packet_time_ppm[datarate] : packet_time[datarate]);
681  if (!rfm22b_dev->one_way_link)
682  rfm22b_dev->packet_time *= 2; // double the time to allow a send and receive in each slice
683 
684  // Find the first N channels that meet the min/max criteria out of the random channel list.
685  uint32_t crc = 0;
686  const uint8_t CRC_INC = 0x39;
687  if (coordinator) {
688  crc = PIOS_CRC_updateByte(rfm22b_dev->deviceID, CRC_INC);
689  } else {
690  crc = PIOS_CRC_updateByte(rfm22b_dev->coordinatorID, CRC_INC);
691  }
692 
693  uint8_t num_found = 0;
694  while (num_found < num_channels[datarate]) {
695  crc = PIOS_CRC_updateByte(crc, CRC_INC);
696  uint8_t chan = min_chan + (crc % (max_chan - min_chan));
697 
698  if (chan < RFM22B_NUM_CHANNELS) {
699  // skip any duplicates
700  for (int32_t i = 0; i < num_found; i++) {
701  if (rfm22b_dev->channels[i] == chan)
702  continue;
703  }
704  rfm22b_dev->channels[num_found++] = chan;
705  }
706  }
707 
708  // Calculate the maximum packet length from the datarate.
709  float bytes_per_period =
710  (float)data_rate[datarate] * (float)(rfm22b_dev->packet_time -
711  2) / 9000;
712 
713  rfm22b_dev->max_packet_len =
714  bytes_per_period - TX_PREAMBLE_NIBBLES / 2 - SYNC_BYTES -
715  HEADER_BYTES - LENGTH_BYTES;
716  if (rfm22b_dev->max_packet_len > RFM22B_MAX_PACKET_LEN) {
717  rfm22b_dev->max_packet_len = RFM22B_MAX_PACKET_LEN;
718  }
719 }

784 bool PIOS_RFM22B_ReceivePacket(uint32_t rfm22b_id, uint8_t * p)
785 {
786  struct pios_rfm22b_dev *rfm22b_dev =
787  (struct pios_rfm22b_dev *)rfm22b_id;
788 
789  if (!PIOS_RFM22B_Validate(rfm22b_dev)) {
790  return false;
791  }
792  rfm22b_dev->rx_packet_handle = p;
793 
794  // Claim the SPI bus.
795  rfm22_claimBus(rfm22b_dev);
796 
797  // disable interrupts
798  rfm22_write(rfm22b_dev, RFM22_interrupt_enable1, 0x00);
799  rfm22_write(rfm22b_dev, RFM22_interrupt_enable2, 0x00);
800 
801  // Switch to TUNE mode
802  rfm22_write(rfm22b_dev, RFM22_op_and_func_ctrl1,
803  RFM22_opfc1_pllon);
804 
805 #ifdef PIOS_RFM22B_DEBUG_ON_TELEM
806  D2_LED_OFF;
807 #endif // PIOS_RFM22B_DEBUG_ON_TELEM
808  RX_LED_OFF;
809  TX_LED_OFF;
810 
811  // empty the rx buffer
812  rfm22b_dev->rx_buffer_wr = 0;
813 
814  // Clear the TX buffer.
815  rfm22b_dev->tx_data_rd = rfm22b_dev->tx_data_wr = 0;
816 
817  // clear FIFOs
818  rfm22_write(rfm22b_dev, RFM22_op_and_func_ctrl2,
819  RFM22_opfc2_ffclrrx | RFM22_opfc2_ffclrtx);
820  rfm22_write(rfm22b_dev, RFM22_op_and_func_ctrl2, 0x00);
821 
822  // enable RX interrupts
823  rfm22_write(rfm22b_dev, RFM22_interrupt_enable1,
824  RFM22_ie1_encrcerror | RFM22_ie1_enpkvalid |
825  RFM22_ie1_enrxffafull | RFM22_ie1_enfferr);
826  rfm22_write(rfm22b_dev, RFM22_interrupt_enable2,
827  RFM22_ie2_enpreaval | RFM22_ie2_enswdet);
828 
829  // enable the receiver
830  rfm22_write(rfm22b_dev, RFM22_op_and_func_ctrl1,
831  RFM22_opfc1_pllon | RFM22_opfc1_rxon);
832 
833  // Release the SPI bus.
834  rfm22_releaseBus(rfm22b_dev);
835 
836  // Indicate that we're in RX wait mode.
837  rfm22b_dev->rfm22b_state = RFM22B_STATE_RX_WAIT;
838 
839  return true;
840 }

849 bool PIOS_RFM22B_TransmitPacket(uint32_t rfm22b_id, uint8_t * p,
850  uint8_t len)
851 {
852  struct pios_rfm22b_dev *rfm22b_dev =
853  (struct pios_rfm22b_dev *)rfm22b_id;
854 
855  if (!PIOS_RFM22B_Validate(rfm22b_dev)) {
856  return false;
857  }
858 
859  rfm22b_dev->tx_packet_handle = p;
860  rfm22b_dev->stats.tx_byte_count += len;
861  rfm22b_dev->packet_start_ticks = PIOS_Thread_Systime();
862  if (rfm22b_dev->packet_start_ticks == 0) {
863  rfm22b_dev->packet_start_ticks = 1;
864  }
865  // Claim the SPI bus.
866  rfm22_claimBus(rfm22b_dev);
867 
868  // Disable interrupts
869  rfm22_write(rfm22b_dev, RFM22_interrupt_enable1, 0x00);
870  rfm22_write(rfm22b_dev, RFM22_interrupt_enable2, 0x00);
871 
872  // set the tx power
873  rfm22_write(rfm22b_dev, RFM22_tx_power, RFM22_tx_pwr_lna_sw | rfm22b_dev->tx_power);
874 
875  // TUNE mode
876  rfm22_write(rfm22b_dev, RFM22_op_and_func_ctrl1,
877  RFM22_opfc1_pllon);
878 
879  // Queue the data up for sending
880  rfm22b_dev->tx_data_wr = len;
881 
882  RX_LED_OFF;
883 
884  // Set the destination address in the transmit header.
885  uint32_t id = rfm22_destinationID(rfm22b_dev);
886  rfm22_write(rfm22b_dev, RFM22_transmit_header0, id & 0xff);
887  rfm22_write(rfm22b_dev, RFM22_transmit_header1, (id >> 8) & 0xff);
888  rfm22_write(rfm22b_dev, RFM22_transmit_header2, (id >> 16) & 0xff);
889  rfm22_write(rfm22b_dev, RFM22_transmit_header3, (id >> 24) & 0xff);
890 
891  // FIFO mode, GFSK modulation
892  uint8_t fd_bit = rfm22_read(rfm22b_dev,
893  RFM22_modulation_mode_control2) & RFM22_mmc2_fd;
894  rfm22_write(rfm22b_dev, RFM22_modulation_mode_control2,
895  fd_bit | RFM22_mmc2_dtmod_fifo | RFM22_mmc2_modtyp_gfsk);
896 
897  // Clear the FIFOs.
898  rfm22_write(rfm22b_dev, RFM22_op_and_func_ctrl2,
899  RFM22_opfc2_ffclrrx | RFM22_opfc2_ffclrtx);
900  rfm22_write(rfm22b_dev, RFM22_op_and_func_ctrl2, 0x00);
901 
902  // Set the total number of data bytes we are going to transmit.
903  rfm22_write(rfm22b_dev, RFM22_transmit_packet_length, len);
904 
905  // Add some data to the chips TX FIFO before enabling the transmitter
906  uint8_t *tx_buffer = rfm22b_dev->tx_packet_handle;
907  rfm22_assertCs(rfm22b_dev);
908  PIOS_SPI_TransferByte(rfm22b_dev->spi_id, RFM22_fifo_access | 0x80);
909  int bytes_to_write = (rfm22b_dev->tx_data_wr - rfm22b_dev->tx_data_rd);
910  bytes_to_write = (bytes_to_write > FIFO_SIZE) ? FIFO_SIZE : bytes_to_write;
911  PIOS_SPI_TransferBlock(rfm22b_dev->spi_id, &tx_buffer[rfm22b_dev->tx_data_rd],
912  NULL, bytes_to_write);
913  rfm22b_dev->tx_data_rd += bytes_to_write;
914  rfm22_deassertCs(rfm22b_dev);
915 
916  // Enable TX interrupts.
917  rfm22_write(rfm22b_dev, RFM22_interrupt_enable1,
918  RFM22_ie1_enpksent | RFM22_ie1_entxffaem);
919 
920  // Enable the transmitter.
921  rfm22_write(rfm22b_dev, RFM22_op_and_func_ctrl1,
922  RFM22_opfc1_pllon | RFM22_opfc1_txon);
923 
924  // Release the SPI bus.
925  rfm22_releaseBus(rfm22b_dev);
926 
927  // We're in Tx mode.
928  rfm22b_dev->rfm22b_state = RFM22B_STATE_TX_MODE;
929 
930  TX_LED_ON;
931 
932 #ifdef PIOS_RFM22B_DEBUG_ON_TELEM
933  D1_LED_ON;
934 #endif
935 
936  return true;
937 }

945 pios_rfm22b_int_result PIOS_RFM22B_ProcessTx(uint32_t rfm22b_id)
946 {
947  struct pios_rfm22b_dev *rfm22b_dev =
948  (struct pios_rfm22b_dev *)rfm22b_id;
949 
950  if (!PIOS_RFM22B_Validate(rfm22b_dev)) {
951  return PIOS_RFM22B_INT_FAILURE;
952  }
953  // Read the device status registers
954  if (!pios_rfm22_readStatus(rfm22b_dev)) {
955  return PIOS_RFM22B_INT_FAILURE;
956  }
957  // TX FIFO almost empty, it needs filling up
958  if (rfm22b_dev->status_regs.int_status_1.tx_fifo_almost_empty) {
959  // Add data to the TX FIFO buffer
960  uint8_t *tx_buffer = rfm22b_dev->tx_packet_handle;
961  uint16_t max_bytes = FIFO_SIZE - TX_FIFO_LO_WATERMARK - 1;
962  rfm22_claimBus(rfm22b_dev);
963  rfm22_assertCs(rfm22b_dev);
964  PIOS_SPI_TransferByte(rfm22b_dev->spi_id, RFM22_fifo_access | 0x80);
965  int bytes_to_write = (rfm22b_dev->tx_data_wr - rfm22b_dev->tx_data_rd);
966  bytes_to_write = (bytes_to_write > max_bytes) ? max_bytes : bytes_to_write;
967  PIOS_SPI_TransferBlock(rfm22b_dev->spi_id, &tx_buffer[rfm22b_dev->tx_data_rd],
968  NULL, bytes_to_write);
969  rfm22b_dev->tx_data_rd += bytes_to_write;
970  rfm22_deassertCs(rfm22b_dev);
971  rfm22_releaseBus(rfm22b_dev);
972 
973  return PIOS_RFM22B_INT_SUCCESS;
974  } else if (rfm22b_dev->status_regs.int_status_1.packet_sent_interrupt) {
975  // Transition out of Tx mode.
976  rfm22b_dev->rfm22b_state = RFM22B_STATE_TRANSITION;
977  return PIOS_RFM22B_TX_COMPLETE;
978  }
979 
980  return 0;

989 pios_rfm22b_int_result PIOS_RFM22B_ProcessRx(uint32_t rfm22b_id)
990 {
991  struct pios_rfm22b_dev *rfm22b_dev = (struct pios_rfm22b_dev *)rfm22b_id;
992 
993  if (!PIOS_RFM22B_Validate(rfm22b_dev)) {
994  return PIOS_RFM22B_INT_FAILURE;
995  }
996  uint8_t *rx_buffer = rfm22b_dev->rx_packet_handle;
997  pios_rfm22b_int_result ret = PIOS_RFM22B_INT_SUCCESS;
998 
999  // Read the device status registers
1000  if (!pios_rfm22_readStatus(rfm22b_dev)) {
1001  rfm22_rxFailure(rfm22b_dev);
1002  return PIOS_RFM22B_INT_FAILURE;
1003  }
1004  // FIFO under/over flow error. Restart RX mode.
1005  if (rfm22b_dev->status_regs.int_status_1.fifo_underoverflow_error
1006  || rfm22b_dev->status_regs.int_status_1.crc_error) {
1007  rfm22_rxFailure(rfm22b_dev);
1008  return PIOS_RFM22B_INT_FAILURE;
1009  }
1010  // Valid packet received
1011  if (rfm22b_dev->status_regs.int_status_1.valid_packet_received) {
1012  // Claim the SPI bus.
1013  rfm22_claimBus(rfm22b_dev);
1014 
1015  // read the total length of the packet data
1016  uint32_t len = rfm22_read(rfm22b_dev, RFM22_received_packet_length);
1017 
1018  // The received packet is going to be larger than the receive buffer
1019  if (len > rfm22b_dev->max_packet_len) {
1020  rfm22_releaseBus(rfm22b_dev);
1021  rfm22_rxFailure(rfm22b_dev);
1022  return PIOS_RFM22B_INT_FAILURE;
1023  }
1024  // there must still be data in the RX FIFO we need to get
1025  if (rfm22b_dev->rx_buffer_wr < len) {
1026  int32_t bytes_to_read = len - rfm22b_dev->rx_buffer_wr;
1027  // Fetch the data from the RX FIFO
1028  rfm22_assertCs(rfm22b_dev);
1029  PIOS_SPI_TransferByte(rfm22b_dev->spi_id, RFM22_fifo_access & 0x7F);
1030  rfm22b_dev->rx_buffer_wr +=
1031  (PIOS_SPI_TransferBlock(rfm22b_dev->spi_id, OUT_FF,
1032  (uint8_t *) & rx_buffer[rfm22b_dev->rx_buffer_wr],
1033  bytes_to_read) == 0) ? bytes_to_read : 0;
1034  rfm22_deassertCs(rfm22b_dev);
1035  }
1036  // Read the packet header (destination ID)
1037  rfm22b_dev->rx_destination_id = rfm22_read(rfm22b_dev, RFM22_received_header0);
1038  rfm22b_dev->rx_destination_id |= (rfm22_read(rfm22b_dev, RFM22_received_header1) << 8);
1039  rfm22b_dev->rx_destination_id |= (rfm22_read(rfm22b_dev, RFM22_received_header2) << 16);
1040  rfm22b_dev->rx_destination_id |= (rfm22_read(rfm22b_dev, RFM22_received_header3) << 24);
1041 
1042  // Release the SPI bus.
1043  rfm22_releaseBus(rfm22b_dev);
1044 
1045  // Is there a length error?
1046  if (rfm22b_dev->rx_buffer_wr != len) {
1047  rfm22_rxFailure(rfm22b_dev);
1048  return PIOS_RFM22B_INT_FAILURE;
1049  }
1050  // Increment the total byte received count.
1051  rfm22b_dev->stats.rx_byte_count += rfm22b_dev->rx_buffer_wr;
1052 
1053  // We're finished with Rx mode
1054  rfm22b_dev->rfm22b_state = RFM22B_STATE_TRANSITION;
1055 
1056  ret = PIOS_RFM22B_RX_COMPLETE;
1057  } else if (rfm22b_dev->status_regs.int_status_1.rx_fifo_almost_full) {
1058  // RX FIFO almost full, it needs emptying
1059  // read data from the rf chips FIFO buffer
1060 
1061  // Claim the SPI bus.
1062  rfm22_claimBus(rfm22b_dev);
1063 
1064  // Read the total length of the packet data
1065  uint16_t len = rfm22_read(rfm22b_dev, RFM22_received_packet_length);
1066 
1067  // The received packet is going to be larger than the specified length
1068  if ((rfm22b_dev->rx_buffer_wr + RX_FIFO_HI_WATERMARK) > len) {
1069  rfm22_releaseBus(rfm22b_dev);
1070  rfm22_rxFailure(rfm22b_dev);
1071  return PIOS_RFM22B_INT_FAILURE;
1072  }
1073  // The received packet is going to be larger than the receive buffer
1074  if ((rfm22b_dev->rx_buffer_wr + RX_FIFO_HI_WATERMARK) > rfm22b_dev->max_packet_len) {
1075  rfm22_releaseBus(rfm22b_dev);
1076  rfm22_rxFailure(rfm22b_dev);
1077  return PIOS_RFM22B_INT_FAILURE;
1078  }
1079  // Fetch the data from the RX FIFO
1080  rfm22_assertCs(rfm22b_dev);
1081  PIOS_SPI_TransferByte(rfm22b_dev->spi_id, RFM22_fifo_access & 0x7F);
1082  rfm22b_dev->rx_buffer_wr += (PIOS_SPI_TransferBlock(rfm22b_dev->spi_id, OUT_FF,
1083  (uint8_t *) & rx_buffer[rfm22b_dev->rx_buffer_wr], RX_FIFO_HI_WATERMARK
1084  ) == 0) ? RX_FIFO_HI_WATERMARK : 0;
1085  rfm22_deassertCs(rfm22b_dev);
1086 
1087  // Release the SPI bus.
1088  rfm22_releaseBus(rfm22b_dev);
1089 
1090  // Make sure that we're in RX mode.
1091  rfm22b_dev->rfm22b_state = RFM22B_STATE_RX_MODE;
1092  } else if (rfm22b_dev->status_regs.int_status_2.valid_preamble_detected) {
1093  // Valid preamble detected
1094  RX_LED_ON;
1095 
1096  // Sync word detected
1097 #ifdef PIOS_RFM22B_DEBUG_ON_TELEM
1098  D2_LED_ON;
1099 #endif // PIOS_RFM22B_DEBUG_ON_TELEM
1100  rfm22b_dev->packet_start_ticks = PIOS_Thread_Systime();
1101  if (rfm22b_dev->packet_start_ticks == 0) {
1102  rfm22b_dev->packet_start_ticks = 1;
1103  }
1104  // We detected the preamble, now wait for sync.
1105  rfm22b_dev->rfm22b_state = RFM22B_STATE_RX_WAIT_SYNC;
1106  } else if (rfm22b_dev->status_regs.int_status_2.sync_word_detected) {
1107  // Claim the SPI bus.
1108  rfm22_claimBus(rfm22b_dev);
1109 
1110  // read the 10-bit signed afc correction value
1111  // bits 9 to 2
1112  uint16_t afc_correction = (uint16_t) rfm22_read(rfm22b_dev,
1113  RFM22_afc_correction_read) << 8;
1114  // bits 1 & 0
1115  afc_correction |= (uint16_t) rfm22_read(rfm22b_dev,
1116  RFM22_ook_counter_value1) & 0x00c0;
1117  afc_correction >>= 6;
1118 
1119  // convert the afc value to Hz
1120  int32_t afc_corr = (int32_t) (rfm22b_dev->frequency_step_size *
1121  afc_correction + 0.5f);
1122  rfm22b_dev->afc_correction_Hz = (afc_corr <
1123  -127) ? -127 : ((afc_corr > 127) ? 127 : afc_corr);
1124 
1125  // read rx signal strength .. 45 = -100dBm, 205 = -20dBm
1126  uint8_t rssi = rfm22_read(rfm22b_dev, RFM22_rssi);
1127  // convert to dBm
1128  rfm22b_dev->rssi_dBm = (int8_t) (rssi >> 1) - 122;
1129 
1130  // Release the SPI bus.
1131  rfm22_releaseBus(rfm22b_dev);
1132 
1133  // Indicate that we're in RX mode.
1134  rfm22b_dev->rfm22b_state = RFM22B_STATE_RX_MODE;
1135  } else if ((rfm22b_dev->rfm22b_state == RFM22B_STATE_RX_WAIT_SYNC)
1136  && !rfm22b_dev->status_regs.int_status_2.
1137  valid_preamble_detected) {
1138  // Waiting for the preamble timed out.
1139  rfm22_rxFailure(rfm22b_dev);
1140  return PIOS_RFM22B_INT_FAILURE;
1141  }
1142 
1143  return ret;
1144 }

1200 
1201  if (!PIOS_RFM22B_Validate(rfm22b_dev)) {
1202  return;
1203  }
1204

1206  uint32_t curTime_ms = lastEventTime_ms;
1207 
1208  while (1) {
1209 
1210 #if defined(PIOS_INCLUDE_WDG) && defined(PIOS_WDG_RFM22B)
1211  // Update the watchdog timer
1212  PIOS_WDG_UpdateFlag(PIOS_WDG_RFM22B);
1213 #endif /* PIOS_WDG_RFM22B */
1214

1216  rfm22_process_event(rfm22b_dev, RADIO_EVENT_INITIALIZE);

1226  rfm22_process_event(rfm22b_dev, RADIO_EVENT_INT_RECEIVED);
1227  }
1228  }
1229 
1230  // The main task for the radio must be serviced reliably every ms
1231  curTime_ms = PIOS_Thread_Systime();
1232 
1233  // Throw an error if it has been too long since the last ISR
1234  if (pios_rfm22_time_difference_ms(lastEventTime_ms, curTime_ms) > PIOS_RFM22B_SUPERVISOR_TIMEOUT) {
1235  lastEventTime_ms = PIOS_Thread_Systime();
1236  rfm22_process_event(rfm22b_dev, RADIO_EVENT_ERROR);
1237  }
1238 
1239  // Change channels if necessary.
1240  if (rfm22_changeChannel(rfm22b_dev)) {
1241  rfm22_process_event(rfm22b_dev, RADIO_EVENT_RX_MODE);
1242  }
1243 
1244  // Update the connected status
1245  rfm22_setConnected(rfm22b_dev, rfm22b_dev->sync_pulses_missed < RADIO_SYNC_PULSES_DISCONNECT);
1246 
1247  // Have we been sending / receiving this packet too long?
1248  if ((rfm22b_dev->packet_start_ticks > 0) &&
1249  (pios_rfm22_time_difference_ms(rfm22b_dev->packet_start_ticks, curTime_ms) > (rfm22b_dev->packet_time * 3))) {
1250  rfm22_process_event(rfm22b_dev, RADIO_EVENT_TIMEOUT);
1251  }

1253  bool time_to_send = rfm22_timeToSend(rfm22b_dev);
1254 #ifdef PIOS_RFM22B_DEBUG_ON_TELEM

1256  D4_LED_ON;
1257  } else {
1258  D4_LED_OFF;
1259  }
1260 #endif
1261  if (time_to_send && rfm22_InRxWait(rfm22b_dev)) {
1262  rfm22_process_event(rfm22b_dev, RADIO_EVENT_TX_START);
1263  } else if (time_to_send) {
1264  rfm22b_add_rx_status(rfm22b_dev,RADIO_ERROR_TX_MISSED);
1265  }
1266 
1267 #if defined(PIOS_LED_LINK)
1268  // If not listening for PPM indicate link status with link led
1269  if (!rfm22b_dev->ppm_recv_mode) {
1270  if (rfm22b_dev->stats.link_state == RFM22BSTATUS_LINKSTATE_CONNECTED)
1271  PIOS_ANNUNC_On(PIOS_LED_LINK);
1272  else
1273  PIOS_ANNUNC_Off(PIOS_LED_LINK);
1274  }
1275 #endif /* PIOS_LED_LINK */
1276 
1277  }
1278 }

1280 uint8_t PIOS_RFM22B_RSSI_Get(void)
1281 {
1282  if (!PIOS_RFM22B_Validate(g_rfm22b_dev)) {
1283  return 0;
1284  }
1285 
1286  // Use a similar metric to OpenLRS "Combined" RSSI. If link quality reflects
1287  // missing packets that scale RSSI less than 128
1288  if (g_rfm22b_dev->stats.link_quality != 127)
1289  return g_rfm22b_dev->stats.link_quality;
1290 
1291  // Converting the RSSI to the same scale as with OpenLRS so add back
1292  // the 122 used to convert the raw register in half into dBm, and then
1293  // add 128 because we use (128,255) as the range for good RSSI when LQ
1294  // is perfect. At this point and RSSI of 150 reflects -100 dBm (very bad)
1295  // and 230 reflects -20 dBm (max possible register)
1296  return (uint8_t) ((int32_t) g_rfm22b_dev->rssi_dBm) + 122 + 128;
1297 }

1310 static enum pios_radio_event rfm22_process_state_transition(struct pios_rfm22b_dev *rfm22b_dev,
1311  enum pios_radio_event event)
1312 {
1313  // No event
1314  if (event >= RADIO_EVENT_NUM_EVENTS) {
1315  return RADIO_EVENT_NUM_EVENTS;
1316  }
1317  // Don't transition if there is no transition defined
1318  enum pios_radio_state next_state =
1319  rfm22b_transitions[rfm22b_dev->state].next_state[event];
1320  if (!next_state) {
1321  return RADIO_EVENT_NUM_EVENTS;
1322  }
1323 
1324  /*
1325  * Move to the next state
1326  *
1327  * This is done prior to calling the new state's entry function to
1328  * guarantee that the entry function never depends on the previous
1329  * state. This way, it cannot ever know what the previous state was.
1330  */
1331  rfm22b_dev->state = next_state;
1332 
1333  /* Call the entry function (if any) for the next state. */
1334  if (rfm22b_transitions[rfm22b_dev->state].entry_fn) {
1335  return rfm22b_transitions[rfm22b_dev->state].
1336  entry_fn(rfm22b_dev);
1337  }
1338 
1339  return RADIO_EVENT_NUM_EVENTS;
1340 }

1349 static void rfm22_process_event(struct pios_rfm22b_dev *rfm22b_dev,
1350  enum pios_radio_event event)
1351 {
1352  // Process all state transitions.
1353  while (event != RADIO_EVENT_NUM_EVENTS) {
1354  event = rfm22_process_state_transition(rfm22b_dev, event);
1355  }
1356 }

1543

1546  TX_PREAMBLE_NIBBLES);
1547  // x-nibbles rx preamble detection
1548  rfm22_write(rfm22b_dev, RFM22_preamble_detection_ctrl1,
1549  RX_PREAMBLE_NIBBLES << 3);
1550 
1551  // header control - using a 4 by header with broadcast of 0xffffffff
1552  rfm22_write(rfm22b_dev, RFM22_header_control1,
1553  RFM22_header_cntl1_bcen_0 |
1554  RFM22_header_cntl1_bcen_1 |
1555  RFM22_header_cntl1_bcen_2 |
1556  RFM22_header_cntl1_bcen_3 |

1558  RFM22_header_cntl1_hdch_1 |
1559  RFM22_header_cntl1_hdch_2 | RFM22_header_cntl1_hdch_3);
1560  // Check all bit of all bytes of the header, unless we're an unbound modem.
1561  uint8_t header_mask =
1562  (rfm22_destinationID(rfm22b_dev) == 0xffffffff) ? 0 : 0xff;
1563  rfm22_write(rfm22b_dev, RFM22_header_enable0, header_mask);
1564  rfm22_write(rfm22b_dev, RFM22_header_enable1, header_mask);
1565  rfm22_write(rfm22b_dev, RFM22_header_enable2, header_mask);
1566  rfm22_write(rfm22b_dev, RFM22_header_enable3, header_mask);
1567  // The destination ID and receive ID should be the same.
1568  uint32_t id = rfm22_destinationID(rfm22b_dev);
1569  rfm22_write(rfm22b_dev, RFM22_check_header0, id & 0xff);
1570  rfm22_write(rfm22b_dev, RFM22_check_header1, (id >> 8) & 0xff);
1571  rfm22_write(rfm22b_dev, RFM22_check_header2, (id >> 16) & 0xff);

1701 static void rfm22_setNominalCarrierFrequency(struct pios_rfm22b_dev
1702  *rfm22b_dev,
1703  uint8_t init_chan,
1704  uint32_t frequency_hz)
1705 {
1706  // The step size is 10MHz / 250 channels = 40khz, and the step size is specified in 10khz increments.
1707  uint8_t freq_hop_step_size = 4;
1708 
1709  // holds the hbsel (1 or 2)
1710  uint8_t hbsel;
1711

1713  hbsel = 0;
1714  } else {
1715  hbsel = 1;
1716  }
1717  float freq_mhz = (float)(frequency_hz) / 1000000.0f;
1718  float xtal_freq_khz = 30000.0f;

1720  freq_mhz / (10.0f * (xtal_freq_khz / 30000.0f) * (1 + hbsel));
1721  uint32_t fb = (uint32_t) sfreq - 24 + (64 + 32 * hbsel);
1722  uint32_t fc = (uint32_t) ((sfreq - (uint32_t) sfreq) * 64000.0f);

1727  rfm22_claimBus(rfm22b_dev);

1731  freq_hop_step_size);
1732 
1733  rfm22b_dev->frequency_step_size = 156.25f * hbsel;
1734

1736  rfm22b_dev->channel = init_chan;
1737  rfm22_write(rfm22b_dev, RFM22_frequency_hopping_channel_select,
1738  init_chan);
1739

1741  rfm22_write(rfm22b_dev, RFM22_frequency_offset1, 0);

1836 static enum pios_radio_event radio_txStart(struct pios_rfm22b_dev
1837  *radio_dev)
1838 {
1839  uint8_t *p = radio_dev->tx_packet;
1840  uint8_t len = 0;
1841  uint8_t max_data_len =
1842  radio_dev->max_packet_len -
1843  (radio_dev->ppm_only_mode ? 0 : RS_ECC_NPARITY);
1844 
1845  // Don't send if it's not our turn, or if we're receiving a packet.
1846  if (!rfm22_timeToSend(radio_dev) || !rfm22_InRxWait(radio_dev)) {
1847  return RADIO_EVENT_RX_MODE;
1848  }
1849 
1850  // Don't send anything if we're bound to a coordinator and not yet connected.
1851  if (!rfm22_isCoordinator(radio_dev) && !rfm22_isConnected(radio_dev)) {
1852  return RADIO_EVENT_RX_MODE;
1853  }
1854 
1855  // Should we append PPM data to the packet?
1856  if (radio_dev->ppm_send_mode) {
1857  len = RFM22B_PPM_NUM_CHANNELS + (radio_dev->ppm_only_mode ? 2 : 1);
1858 
1859  // Ensure we can fit the PPM data in the packet.
1860  if (max_data_len < len) {
1861  return RADIO_EVENT_RX_MODE;
1862  }
1863  // The first byte is a bitmask of valid channels.
1864  p[0] = 0;
1865 
1866  // Read the PPM input.
1867  for (uint8_t i = 0; i < RFM22B_PPM_NUM_CHANNELS; ++i) {
1868  int32_t val = radio_dev->ppm[i];
1869  if ((val == PIOS_RCVR_INVALID) || (val == PIOS_RCVR_TIMEOUT)) {
1870  p[i + 1] = 0;
1871  } else {
1872  p[0] |= 1 << i;
1873  p[i + 1] =
1874  (val < 1000) ? 0 : ((val >= 1900) ? 255
1875  : (uint8_t) (256 * (val - 1000) / 900));
1876  }
1877  }
1878 
1879  // The last byte is a CRC.
1880  if (radio_dev->ppm_only_mode) {
1881  uint8_t crc = 0;
1882  for (uint8_t i = 0;
1883  i < RFM22B_PPM_NUM_CHANNELS + 1; ++i) {
1884  crc = PIOS_CRC_updateByte(crc, p[i]);
1885  }
1886  p[RFM22B_PPM_NUM_CHANNELS + 1] = crc;
1887  }
1888  }
1889 
1890  // Append data from the com interface if applicable.
1891  if (!radio_dev->ppm_only_mode && radio_dev->tx_out_cb) {
1892  // Try to get some data to send
1893  bool need_yield = false;
1894  len += (radio_dev->tx_out_cb) (radio_dev->tx_out_context, p + len, max_data_len - len, NULL, &need_yield);
1895  }
1896 
1897  // Always send a packet on the sync channel. So if length is zero (no data)
1898  // and not sync channel, return to listener mode.
1899  if ((len == 0) && (radio_dev->channel_index != 0)) {
1900  return RADIO_EVENT_RX_MODE;
1901  }
1902 
1903  // Add the error correcting code.
1904  if (!radio_dev->ppm_only_mode) {
1905  if (len != 0) {
1906  encode_data((unsigned char *)p, len, (unsigned char *)p);
1907  } else {
1908  for (uint32_t i = 0; i < RS_ECC_NPARITY; i++)
1909  p[i] = EMPTY_PACKET + i;
1910  }
1911  len += RS_ECC_NPARITY;
1912  }
1913  // Transmit the packet.
1914  PIOS_RFM22B_TransmitPacket((uint32_t) radio_dev, p, len);
1915 
1916  return RADIO_EVENT_NUM_EVENTS;
1917 }

1978 static enum pios_radio_event radio_receivePacket(struct pios_rfm22b_dev
1979  *radio_dev, uint8_t * p,
1980  uint16_t rx_len)
1981 {
1982  bool good_packet = false;
1983  bool corrected_packet = false;
1984  bool empty_packet = false;
1985  uint8_t data_len = rx_len;
1986 
1987  if (!radio_dev->ppm_only_mode) {
1988  data_len -= RS_ECC_NPARITY;
1989 
1990  // Attempt to correct any errors in the packet.
1991  if (data_len > 0) {
1992  decode_data((unsigned char *)p, rx_len);
1993  good_packet = check_syndrome() == 0;
1994 
1995  // We have an error. Try to correct it.
1996  if (!good_packet &&
1997  (correct_errors_erasures((unsigned char *)p, rx_len, 0, 0) != 0)) {
1998  // We corrected it
1999  corrected_packet = true;
2000  }
2001  } else {
2002  // Empty packets have specific code for ECC
2003  empty_packet = true;
2004  for (uint32_t i = 0; i < RS_ECC_NPARITY; i++)
2005  empty_packet &= (p[i] == EMPTY_PACKET + i);
2006  }
2007  } else {
2008  // We don't rsencode ppm only packets.
2009  good_packet = true;
2010  }
2011 
2012  uint8_t ppm_len = RFM22B_PPM_NUM_CHANNELS + (radio_dev->ppm_only_mode ? 2 : 1);
2013 
2014  // Parse PPM data from the packet when expecting it
2015  if ((good_packet || corrected_packet) && radio_dev->ppm_recv_mode) {
2016 
2017 #if defined(PIOS_LED_LINK)
2018  // if we have a link LED and are expecting PPM, that is the most
2019  // important thing to know, so use the LED to indicate that.
2020  PIOS_ANNUNC_Toggle(PIOS_LED_LINK);
2021 #endif /* PIOS_LED_LINK */
2022 
2023  // Ensure the packet it long enough
2024  if (data_len < ppm_len) {
2025  good_packet = false;
2026  }
2027 
2028  // Verify the CRC if this is a PPM only packet.
2029  if (good_packet && radio_dev->ppm_only_mode) {
2030  uint8_t crc = 0;
2031  for (uint8_t i = 0; i < RFM22B_PPM_NUM_CHANNELS + 1; ++i) {
2032  crc = PIOS_CRC_updateByte(crc, p[i]);
2033  }
2034  if (p[RFM22B_PPM_NUM_CHANNELS + 1] != crc) {
2035  good_packet = false;
2036  corrected_packet = false;
2037  }
2038  }
2039 
2040  if (good_packet) {
2041  for (uint8_t i = 0; i < RFM22B_PPM_NUM_CHANNELS; ++i) {
2042  // Is this a valid channel?
2043  if (p[0] & (1 << i)) {
2044  uint32_t val = p[i + 1];
2045  radio_dev->ppm[i] = (uint16_t) (1000 + val * 900 / 256);
2046  } else {
2047  radio_dev->ppm[i] = PIOS_RCVR_TIMEOUT;
2048  }
2049  }
2050 
2051  p += RFM22B_PPM_NUM_CHANNELS + 1;
2052  data_len -= RFM22B_PPM_NUM_CHANNELS + 1;
2053 
2054  // Call the PPM received callback if it's available.
2055  if (radio_dev->rfm22b_rcvr_id) {
2056 #if defined(PIOS_INCLUDE_RFM22B_RCVR)
2057  PIOS_RFM22B_Rcvr_UpdateChannels(radio_dev->rfm22b_rcvr_id, radio_dev->ppm);
2058 #endif
2059  }
2060  }
2061  }
2062 
2063  // Set the packet status
2064  if (good_packet) {
2065  rfm22b_add_rx_status(radio_dev, RADIO_GOOD_RX_PACKET);
2066  } else if (corrected_packet) {
2067  // We corrected the error.
2068  rfm22b_add_rx_status(radio_dev, RADIO_CORRECTED_RX_PACKET);
2069  } else if (!empty_packet) {
2070  // There was data but we could not correct it
2071  rfm22b_add_rx_status(radio_dev, RADIO_ERROR_RX_PACKET);
2072  }
2073 
2074  if (good_packet || corrected_packet) {
2075  // Send the data to the com port
2076  bool rx_need_yield;
2077  if (radio_dev->rx_in_cb && (data_len > 0) && !radio_dev->ppm_only_mode) {
2078  (radio_dev->rx_in_cb) (radio_dev->rx_in_context, p, data_len, NULL, &rx_need_yield);
2079  }
2080  }
2081 
2082  // Flag that we have received a packet
2083  if (good_packet || corrected_packet || empty_packet) {
2084 
2085  radio_dev->packet_received_slice = true;
2086 
2087  // We only synchronize the clock on packets from our coordinator on the sync channel.
2088  // These packets are not error checked. This can be improved in the future
2089  if (!rfm22_isCoordinator(radio_dev) &&
2090  radio_dev->rx_destination_id == rfm22_destinationID(radio_dev) &&
2091  radio_dev->channel_index == 0) {
2092 
2093  radio_dev->sync_pulses_missed = 0;
2094  rfm22_setConnected(radio_dev, true);
2095 
2096  rfm22_synchronizeClock(radio_dev);
2097  }
2098  }
2099 
2100  return RADIO_EVENT_RX_COMPLETE;
2101 }
