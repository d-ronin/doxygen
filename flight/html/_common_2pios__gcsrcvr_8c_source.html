95 {

100  if (!gcsrcvr_dev)

102 
103  for (uint8_t i = 0; i < GCSRECEIVER_CHANNEL_NUMELEM; i++) {

106  }

114  }

136 static void PIOS_gcsrcvr_Supervisor(uintptr_t gcsrcvr_id) {
137  /* Recover our device context */
138  struct pios_gcsrcvr_dev * gcsrcvr_dev = (struct pios_gcsrcvr_dev *)gcsrcvr_id;
139 
140  if (!PIOS_gcsrcvr_validate(gcsrcvr_dev)) {
141  /* Invalid device specified */
142  return;
143  }
144 
145  /*
146  * RTC runs at 625Hz.
147  */
148  if(++(gcsrcvr_dev->supv_timer) < (PIOS_GCSRCVR_TIMEOUT_MS * 1000 / 625)) {
149  return;
150  }
151  gcsrcvr_dev->supv_timer = 0;
152 
153  if (!gcsrcvr_dev->Fresh)
154  for (int32_t i = 0; i < GCSRECEIVER_CHANNEL_NUMELEM; i++)
155  gcsreceiverdata.Channel[i] = PIOS_RCVR_TIMEOUT;
156 
157  gcsrcvr_dev->Fresh = false;
158 }
