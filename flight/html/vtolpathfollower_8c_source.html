151 static void vtolPathFollowerTask(void *parameters)
152 {
153  SystemSettingsData systemSettings;
154  FlightStatusData flightStatus;
155 
156  VtolPathFollowerSettingsConnectCallback(vtol_follower_control_settings_updated);
157  AltitudeHoldSettingsConnectCallback(vtol_follower_control_settings_updated);
158  vtol_follower_control_settings_updated(NULL, NULL, NULL, 0);
159 
160  VtolPathFollowerSettingsGet(&guidanceSettings);
161 
162  // Main task loop
163  while (1) {
164 
165  SystemSettingsGet(&systemSettings);
166  if ( (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_VTOL) &&
167  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_QUADP) &&
168  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_QUADX) &&
169  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_HEXA) &&
170  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_HEXAX) &&
171  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_HEXACOAX) &&
172  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_OCTO) &&
173  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_OCTOV) &&
174  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_OCTOCOAXP) &&
175  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_OCTOCOAXX) &&
176  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_TRI) &&
177  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_HELICP) )
178  {
179  // This should be a critical alarm since the system will not attempt to
180  // control in this situation.
181  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_CRITICAL);
182  PIOS_Thread_Sleep(1000);
183  continue;
184  }
185 
186  // Make sure when flight mode toggles, to immediately update the path
187  UAVObjEvent ev;
188  PIOS_Queue_Receive(queue, &ev, guidanceSettings.UpdatePeriod);
189 
190  static uint8_t last_flight_mode;
191  FlightStatusGet(&flightStatus);
192 
193  static bool fsm_running = false;
194 
195  if (flightStatus.FlightMode != last_flight_mode) {
196  // The mode has changed
197 
198  last_flight_mode = flightStatus.FlightMode;
199 
200  switch(flightStatus.FlightMode) {
201  case FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME:
202  vtol_follower_fsm_activate_goal(GOAL_LAND_HOME);
203  fsm_running = true;
204  break;
205  case FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD:
206  vtol_follower_fsm_activate_goal(GOAL_HOLD_POSITION);
207  fsm_running = true;
208  break;
209  case FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER:
210  case FLIGHTSTATUS_FLIGHTMODE_TABLETCONTROL:
211  // TODO: currently when in this mode the follower just
212  // attempts to fly the path segments blindly which means
213  // the FSM cannot be utilized in a meaningful way. It might
214  // be better when flying in path planner mode for the path
215  // planner to specify the goals in PathDesired so things like
216  // RTH can be used. However, for now this isn't critical.
217  vtol_follower_fsm_activate_goal(GOAL_FLY_PATH);
218  fsm_running = true;
219  break;
220  default:
221  vtol_follower_fsm_activate_goal(GOAL_LAND_NONE);
222  fsm_running = false;
223  break;
224  }
225  }
226 
227  if (fsm_running) {
228  vtol_follower_fsm_update();
229  } else {
230  for (uint32_t i = 0; i < VTOL_PID_NUM; i++)
231  pid_zero(&vtol_pids[i]);
232 
233  // Track throttle before engaging this mode. Cheap system ident
234  StabilizationDesiredThrustGet(&vtol_pids[DOWN_VELOCITY].iAccumulator);
235  // Note the negative sign because this is the accumulation for down.
236  vtol_pids[DOWN_VELOCITY].iAccumulator *= -1;
237  }
238 
239  AlarmsClear(SYSTEMALARMS_ALARM_PATHFOLLOWER);
240 
241  }
242 }
