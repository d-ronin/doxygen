357 static void update_accels(struct pios_sensor_accel_data *accels)
358 {
359  // Average and scale the accels before rotation
360  float accels_out[3] = {
361  accels->x * accel_scale[0] - accel_bias[0],
362  accels->y * accel_scale[1] - accel_bias[1],
363  accels->z * accel_scale[2] - accel_bias[2]
364  };
365 
366  lpfilter_run(accel_filter, accels_out);
367 
368  if (rotate) {
369  float accel_rotated[3];
370  rot_mult(Rsb, accels_out, accel_rotated, true);
371  accelsData.x = accel_rotated[0];
372  accelsData.y = accel_rotated[1];
373  accelsData.z = accel_rotated[2];
374  } else {
375  accelsData.x = accels_out[0];
376  accelsData.y = accels_out[1];
377  accelsData.z = accels_out[2];
378  }
379 
380  accelsData.z += z_accel_offset;
381  accelsData.temperature = accels->temperature;
382 
383  AccelsSet(&accelsData);
384 }

390 static void update_gyros(struct pios_sensor_gyro_data *gyros)
391 {
392  // Scale the gyros
393  float gyros_out[3] = {
394  gyros->x * gyro_scale[0],
395  gyros->y * gyro_scale[1],
396  gyros->z * gyro_scale[2]
397  };
398 
399  lpfilter_run(gyro_filter, gyros_out);
400 
401  GyrosData gyrosData;
402  gyrosData.temperature = gyros->temperature;
403 
404  // Update the bias due to the temperature
405  updateTemperatureComp(gyrosData.temperature, gyro_temp_bias);
406 
407  // Apply temperature bias correction before the rotation
408  if (bias_correct_gyro) {
409  gyros_out[0] -= gyro_temp_bias[0];
410  gyros_out[1] -= gyro_temp_bias[1];
411  gyros_out[2] -= gyro_temp_bias[2];
412  }
413 
414  if (rotate) {
415  float gyros[3];
416  rot_mult(Rsb, gyros_out, gyros, true);
417  gyrosData.x = gyros[0];
418  gyrosData.y = gyros[1];
419  gyrosData.z = gyros[2];
420  } else {
421  gyrosData.x = gyros_out[0];
422  gyrosData.y = gyros_out[1];
423  gyrosData.z = gyros_out[2];
424  }
425 
426  if (bias_correct_gyro) {
427  // Apply bias correction to the gyros from the state estimator
428  GyrosBiasData gyrosBias;
429  GyrosBiasGet(&gyrosBias);
430  gyrosData.x -= gyrosBias.x;
431  gyrosData.y -= gyrosBias.y;
432  gyrosData.z -= gyrosBias.z;
433 
434  const float GYRO_BIAS_WARN = 10.0f;
435  if (fabsf(gyrosBias.x) > GYRO_BIAS_WARN ||
436  fabsf(gyrosBias.y) > GYRO_BIAS_WARN ||
437  fabsf(gyrosBias.z) > GYRO_BIAS_WARN) {
438  AlarmsSet(SYSTEMALARMS_ALARM_GYROBIAS, SYSTEMALARMS_ALARM_WARNING);
439  } else {
440  AlarmsClear(SYSTEMALARMS_ALARM_GYROBIAS);
441  }
442  }
443 
444  GyrosSet(&gyrosData);
445 }

561 static void updateTemperatureComp(float temperature, float *temp_bias)
562 {
563  static int temp_counter = -1;
564  static float temp_accum = 0;
565  static const float TEMP_MIN = -10;
566  static const float TEMP_MAX = 60;
567 
568  if (temperature < TEMP_MIN)
569  temperature = TEMP_MIN;
570  if (temperature > TEMP_MAX)
571  temperature = TEMP_MAX;
572 
573  if (temp_counter < 500) {
574  temp_accum += temperature;
575  temp_counter ++;
576  } else {
577  float t = temp_accum / temp_counter;
578  temp_accum = 0;
579  temp_counter = 0;
580 
581  // Compute a third order polynomial for each chanel after each 500 samples
582  temp_bias[0] = gyro_coeff_x[0] + gyro_coeff_x[1] * t +
583  gyro_coeff_x[2] * powf(t,2) + gyro_coeff_x[3] * powf(t,3);
584  temp_bias[1] = gyro_coeff_y[0] + gyro_coeff_y[1] * t +
585  gyro_coeff_y[2] * powf(t,2) + gyro_coeff_y[3] * powf(t,3);
586  temp_bias[2] = gyro_coeff_z[0] + gyro_coeff_z[1] * t +
587  gyro_coeff_z[2] * powf(t,2) + gyro_coeff_z[3] * powf(t,3);
588  }
589 }

596 static void mag_calibration_prelemari(MagnetometerData *mag)
597 {
598  // Constants, to possibly go into a UAVO
599  static const float MIN_NORM_DIFFERENCE = 50;
600 
601  static float B2[3] = {0, 0, 0};
602 
603  MagBiasData magBias;
604  MagBiasGet(&magBias);
605 
606  // Remove the current estimate of the bias
607  mag->x -= magBias.x;
608  mag->y -= magBias.y;
609  mag->z -= magBias.z;
610 
611  // First call
612  if (B2[0] == 0 && B2[1] == 0 && B2[2] == 0) {
613  B2[0] = mag->x;
614  B2[1] = mag->y;
615  B2[2] = mag->z;
616  return;
617  }
618 
619  float B1[3] = {mag->x, mag->y, mag->z};
620  float norm_diff = sqrtf(powf(B2[0] - B1[0],2) + powf(B2[1] - B1[1],2) + powf(B2[2] - B1[2],2));
621  if (norm_diff > MIN_NORM_DIFFERENCE) {
622  float norm_b1 = sqrtf(B1[0]*B1[0] + B1[1]*B1[1] + B1[2]*B1[2]);
623  float norm_b2 = sqrtf(B2[0]*B2[0] + B2[1]*B2[1] + B2[2]*B2[2]);
624  float scale = insSettings.MagBiasNullingRate * (norm_b2 - norm_b1) / norm_diff;
625  float b_error[3] = {(B2[0] - B1[0]) * scale, (B2[1] - B1[1]) * scale, (B2[2] - B1[2]) * scale};
626 
627  magBias.x += b_error[0];
628  magBias.y += b_error[1];
629  magBias.z += b_error[2];
630 
631  MagBiasSet(&magBias);
632 
633  // Store this value to compare against next update
634  B2[0] = B1[0]; B2[1] = B1[1]; B2[2] = B1[2];
635  }
636 }

665  float B_e[3];
666  float xy[2];
667  float delta[3];

671 
672  // Rotate the mag into the NED frame
673  B_e[0] = R[0][0] * mag->x + R[1][0] * mag->y + R[2][0] * mag->z;
674  B_e[1] = R[0][1] * mag->x + R[1][1] * mag->y + R[2][1] * mag->z;
675  B_e[2] = R[0][2] * mag->x + R[1][2] * mag->y + R[2][2] * mag->z;
676 
677  float cy = cosf(attitude.Yaw * DEG2RAD);
678  float sy = sinf(attitude.Yaw * DEG2RAD);
679 
680  xy[0] = cy * B_e[0] + sy * B_e[1];
681  xy[1] = -sy * B_e[0] + cy * B_e[1];
682 
683  float xy_norm = sqrtf(xy[0]*xy[0] + xy[1]*xy[1]);
684 
685  delta[0] = -rate * (xy[0] / xy_norm * Rxy - xy[0]);
686  delta[1] = -rate * (xy[1] / xy_norm * Rxy - xy[1]);
687  delta[2] = -rate * (Rz - B_e[2]);
688 
689  if (delta[0] == delta[0] && delta[1] == delta[1] && delta[2] == delta[2]) {
690  magBias.x += delta[0];
691  magBias.y += delta[1];
692  magBias.z += delta[2];
693  MagBiasSet(&magBias);
694  }
