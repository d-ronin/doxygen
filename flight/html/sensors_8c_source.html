213 bool sensors_step()
214 {
215  static uint32_t good_runs = 0;
216  static uint32_t last_baro_update_time;
217 
218  bool ret = false; /* Are gyros OK this time? */
219 
220  if (settings_updated) {
221  sensors_settings_update();
222  }

224  struct pios_sensor_gyro_data gyros;
225  struct pios_sensor_accel_data accels;
226  struct pios_sensor_mag_data mags;
227  struct pios_sensor_baro_data baro;
228 
229  uint32_t timeval = PIOS_DELAY_GetRaw();
230 
231  // First do everything NON-gyro/accel, so as to minimize gyro->stab
232  // latency.
233  bool test_good_run = good_runs > REQUIRED_GOOD_CYCLES;
234 
235  if (PIOS_SENSORS_GetData(PIOS_SENSOR_MAG, &mags, 0) != false) {
236  update_mags(&mags);
237 #ifdef PIOS_TOLERATE_MISSING_SENSORS
238  } else if (test_good_run) {
239  if (PIOS_SENSORS_GetMissing(PIOS_SENSOR_MAG)) {
240  // Keep alarm asserted
241  test_good_run = false;
242 
243  AlarmsSet(SYSTEMALARMS_ALARM_SENSORS,
244  missing_sensor_severity);
245  }
246 #endif
247  }
248 
249  if (PIOS_SENSORS_GetData(PIOS_SENSOR_BARO, &baro, 0) != false) {
250  // we can use the timeval because it contains the current time stamp (PIOS_DELAY_GetRaw())
251  last_baro_update_time = timeval;
252  update_baro(&baro);
253  AlarmsClear(SYSTEMALARMS_ALARM_TEMPBARO);
254 #ifdef PIOS_TOLERATE_MISSING_SENSORS
255  } else if (PIOS_SENSORS_GetMissing(PIOS_SENSOR_BARO)) {
256  // Keep alarm asserted
257  test_good_run = false;
258 
259  AlarmsSet(SYSTEMALARMS_ALARM_TEMPBARO,
260  missing_sensor_severity);
261 #endif
262  } else if (PIOS_SENSORS_IsRegistered(PIOS_SENSOR_BARO)) {
263  // Check that we got valid sensor datas
264  uint32_t dT_baro_datas = PIOS_DELAY_DiffuS(last_baro_update_time);
265  // if the last valid sensor datas older than 100 ms report an error
266  if (dT_baro_datas > MAX_TIME_BETWEEN_VALID_BARO_DATAS_MS) {
267  AlarmsSet(SYSTEMALARMS_ALARM_TEMPBARO, SYSTEMALARMS_ALARM_ERROR);
268  }
269  }
270 
271 #if defined(PIOS_INCLUDE_OPTICALFLOW)
272  struct pios_sensor_optical_flow_data optical_flow;
273  if (PIOS_SENSORS_GetData(PIOS_SENSOR_OPTICAL_FLOW, &optical_flow, 0) != false) {
274  update_optical_flow(&optical_flow);
275  }
276 #endif /* PIOS_INCLUDE_OPTICALFLOW */
277 
278 #if defined(PIOS_INCLUDE_RANGEFINDER)
279  struct pios_sensor_rangefinder_data rangefinder;
280  if (PIOS_SENSORS_GetData(PIOS_SENSOR_RANGEFINDER, &rangefinder, 0) != false) {
281  update_rangefinder(&rangefinder);
282  }
283 #endif /* PIOS_INCLUDE_RANGEFINDER */
284 
285  //Block on gyro data but nothing else
286  if (PIOS_SENSORS_GetData(PIOS_SENSOR_GYRO, &gyros, SENSOR_PERIOD) == false) {
287  good_runs = 0;
288  test_good_run = false;
289  } else {
290  ret = true;
291  }
292 
293  if (PIOS_SENSORS_GetData(PIOS_SENSOR_ACCEL, &accels, 0) == false) {
294  //If no new accels data is ready, reuse the latest sample
295  AccelsSet(&accelsData);
296  } else {
297  update_accels(&accels);
298  }
299 
300  // Update gyros after the accels since the rest of the code expects
301  // the accels to be available first
302  update_gyros(&gyros);
303 
304  // Check total time to get the sensors wasn't over the limit
305  uint32_t dT_us = PIOS_DELAY_DiffuS(timeval);
306  if (dT_us > (SENSOR_PERIOD * 1000)) {
307  good_runs = 0;
308  }
309 
310  if (good_runs == 0) {
311  AlarmsSet(SYSTEMALARMS_ALARM_SENSORS, SYSTEMALARMS_ALARM_CRITICAL);
312  } else if (test_good_run) {
313  AlarmsClear(SYSTEMALARMS_ALARM_SENSORS);
314  }
315 
316  if (ret && (!test_good_run)) {
317  good_runs++;
318  }
319 
320 
321  return ret;
322 }

328 static void update_accels(struct pios_sensor_accel_data *accels)
329 {
330  // Average and scale the accels before rotation
331  float accels_out[3] = {
332  accels->x * accel_scale[0] - accel_bias[0],
333  accels->y * accel_scale[1] - accel_bias[1],
334  accels->z * accel_scale[2] - accel_bias[2]
335  };
336 
337  lpfilter_run(accel_filter, accels_out);
338 
339  if (rotate) {
340  float accel_rotated[3];
341  rot_mult(Rsb, accels_out, accel_rotated, true);
342  accelsData.x = accel_rotated[0];
343  accelsData.y = accel_rotated[1];
344  accelsData.z = accel_rotated[2];
345  } else {
346  accelsData.x = accels_out[0];
347  accelsData.y = accels_out[1];
348  accelsData.z = accels_out[2];
349  }
350 
351  accelsData.z += z_accel_offset;
352  accelsData.temperature = accels->temperature;
353 
354  AccelsSet(&accelsData);
355 }

361 static void update_gyros(struct pios_sensor_gyro_data *gyros)
362 {
363  // Scale the gyros
364  float gyros_out[3] = {
365  gyros->x * gyro_scale[0],
366  gyros->y * gyro_scale[1],
367  gyros->z * gyro_scale[2]
368  };
369 
370  lpfilter_run(gyro_filter, gyros_out);
371 
372  GyrosData gyrosData;
373  gyrosData.temperature = gyros->temperature;
374 
375  // Update the bias due to the temperature
376  updateTemperatureComp(gyrosData.temperature, gyro_temp_bias);
377 
378  // Apply temperature bias correction before the rotation
379  if (bias_correct_gyro) {
380  gyros_out[0] -= gyro_temp_bias[0];
381  gyros_out[1] -= gyro_temp_bias[1];
382  gyros_out[2] -= gyro_temp_bias[2];
383  }
384 
385  if (rotate) {
386  float gyros[3];
387  rot_mult(Rsb, gyros_out, gyros, true);
388  gyrosData.x = gyros[0];
389  gyrosData.y = gyros[1];
390  gyrosData.z = gyros[2];
391  } else {
392  gyrosData.x = gyros_out[0];
393  gyrosData.y = gyros_out[1];
394  gyrosData.z = gyros_out[2];
395  }
396 
397  if (bias_correct_gyro) {
398  // Apply bias correction to the gyros from the state estimator
399  GyrosBiasData gyrosBias;
400  GyrosBiasGet(&gyrosBias);
401  gyrosData.x -= gyrosBias.x;
402  gyrosData.y -= gyrosBias.y;
403  gyrosData.z -= gyrosBias.z;
404 
405  const float GYRO_BIAS_WARN = 10.0f;
406  if (fabsf(gyrosBias.x) > GYRO_BIAS_WARN ||
407  fabsf(gyrosBias.y) > GYRO_BIAS_WARN ||
408  fabsf(gyrosBias.z) > GYRO_BIAS_WARN) {
409  AlarmsSet(SYSTEMALARMS_ALARM_GYROBIAS, SYSTEMALARMS_ALARM_WARNING);
410  } else {
411  AlarmsClear(SYSTEMALARMS_ALARM_GYROBIAS);
412  }
413  }
414 
415  GyrosSet(&gyrosData);
416 }

422 static void update_mags(struct pios_sensor_mag_data *mag)
423 {
424  float mags[3] = {
425  mag->x * mag_scale[0] - mag_bias[0],
426  mag->y * mag_scale[1] - mag_bias[1],
427  mag->z * mag_scale[2] - mag_bias[2]
428  };
429 
430  MagnetometerData magData;
431  if (rotate) {
432  float mag_out[3];
433  rot_mult(Rsb, mags, mag_out, true);
434  magData.x = mag_out[0];
435  magData.y = mag_out[1];
436  magData.z = mag_out[2];
437  } else {
438  magData.x = mags[0];
439  magData.y = mags[1];
440  magData.z = mags[2];
441  }
442 
443  // Correct for mag bias and update if the rate is non zero
444  if (insSettings.MagBiasNullingRate > 0) {
445  switch (mag_calibration_algo) {
446  case MAG_CALIBRATION_PRELEMARI:
447  mag_calibration_prelemari(&magData);
448  break;
449  case MAG_CALIBRATION_NORMALIZE_LENGTH:
450  mag_calibration_fix_length(&magData);
451  break;
452  default:
453  // No calibration
454  break;
455  }
456  }
457 
458  MagnetometerSet(&magData);
459 }

532 static void updateTemperatureComp(float temperature, float *temp_bias)
533 {
534  static int temp_counter = -1;
535  static float temp_accum = 0;
536  static const float TEMP_MIN = -10;
537  static const float TEMP_MAX = 60;
538 
539  if (temperature < TEMP_MIN)
540  temperature = TEMP_MIN;
541  if (temperature > TEMP_MAX)
542  temperature = TEMP_MAX;
543 
544  if (temp_counter < 500) {
545  temp_accum += temperature;
546  temp_counter ++;
547  } else {
548  float t = temp_accum / temp_counter;
549  temp_accum = 0;
550  temp_counter = 0;
551 
552  // Compute a third order polynomial for each chanel after each 500 samples
553  temp_bias[0] = gyro_coeff_x[0] + gyro_coeff_x[1] * t +
554  gyro_coeff_x[2] * powf(t,2) + gyro_coeff_x[3] * powf(t,3);
555  temp_bias[1] = gyro_coeff_y[0] + gyro_coeff_y[1] * t +
556  gyro_coeff_y[2] * powf(t,2) + gyro_coeff_y[3] * powf(t,3);
557  temp_bias[2] = gyro_coeff_z[0] + gyro_coeff_z[1] * t +
558  gyro_coeff_z[2] * powf(t,2) + gyro_coeff_z[3] * powf(t,3);
559  }
560 }

567 static void mag_calibration_prelemari(MagnetometerData *mag)
568 {
569  // Constants, to possibly go into a UAVO
570  static const float MIN_NORM_DIFFERENCE = 50;
571 
572  static float B2[3] = {0, 0, 0};
573 
574  MagBiasData magBias;
575  MagBiasGet(&magBias);
576 
577  // Remove the current estimate of the bias
578  mag->x -= magBias.x;
579  mag->y -= magBias.y;
580  mag->z -= magBias.z;
581 
582  // First call
583  if (B2[0] == 0 && B2[1] == 0 && B2[2] == 0) {
584  B2[0] = mag->x;
585  B2[1] = mag->y;
586  B2[2] = mag->z;
587  return;
588  }
589 
590  float B1[3] = {mag->x, mag->y, mag->z};
591  float norm_diff = sqrtf(powf(B2[0] - B1[0],2) + powf(B2[1] - B1[1],2) + powf(B2[2] - B1[2],2));
592  if (norm_diff > MIN_NORM_DIFFERENCE) {
593  float norm_b1 = sqrtf(B1[0]*B1[0] + B1[1]*B1[1] + B1[2]*B1[2]);
594  float norm_b2 = sqrtf(B2[0]*B2[0] + B2[1]*B2[1] + B2[2]*B2[2]);
595  float scale = insSettings.MagBiasNullingRate * (norm_b2 - norm_b1) / norm_diff;
596  float b_error[3] = {(B2[0] - B1[0]) * scale, (B2[1] - B1[1]) * scale, (B2[2] - B1[2]) * scale};
597 
598  magBias.x += b_error[0];
599  magBias.y += b_error[1];
600  magBias.z += b_error[2];
601 
602  MagBiasSet(&magBias);
603 
604  // Store this value to compare against next update
605  B2[0] = B1[0]; B2[1] = B1[1]; B2[2] = B1[2];
606  }
607 }

615 static void mag_calibration_fix_length(MagnetometerData *mag)
616 {
617  MagBiasData magBias;
618  MagBiasGet(&magBias);
619 
620  // Remove the current estimate of the bias
621  mag->x -= magBias.x;
622  mag->y -= magBias.y;
623  mag->z -= magBias.z;
624 
625  HomeLocationData homeLocation;
626  HomeLocationGet(&homeLocation);
627 
628  AttitudeActualData attitude;
629  AttitudeActualGet(&attitude);
630 
631  const float Rxy = sqrtf(homeLocation.Be[0]*homeLocation.Be[0] + homeLocation.Be[1]*homeLocation.Be[1]);
632  const float Rz = homeLocation.Be[2];
633 
634  const float rate = insSettings.MagBiasNullingRate;
635  float R[3][3];
636  float B_e[3];
637  float xy[2];
638  float delta[3];
639 
640  // Get the rotation matrix
641  Quaternion2R(&attitude.q1, R);
642 
643  // Rotate the mag into the NED frame
644  B_e[0] = R[0][0] * mag->x + R[1][0] * mag->y + R[2][0] * mag->z;
645  B_e[1] = R[0][1] * mag->x + R[1][1] * mag->y + R[2][1] * mag->z;
646  B_e[2] = R[0][2] * mag->x + R[1][2] * mag->y + R[2][2] * mag->z;
647 
648  float cy = cosf(attitude.Yaw * DEG2RAD);
649  float sy = sinf(attitude.Yaw * DEG2RAD);
650
