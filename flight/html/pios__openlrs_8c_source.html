263 // Sending a x byte packet on bps y takes about (emperical)
264 // usec = (x + 15) * 8200000 / baudrate
265 //
266 // There's 5 bytes of preamble, 4 bytes of header, 1 byte of length,
267 // 2 bytes of sync word, 2 bytes of CRC, yielding 14 bytes.
268 // 15 is used here, for a small margin.
269 //
270 // Similarly, 8 bits/byte, 1000000 microseconds/second + 2.5% for
271 // margin. The values chosen here have interoperability
272 // considerations, so leave them alone :D
273 #define BYTES_AT_BAUD_TO_USEC(bytes, bps, div) ((uint32_t)((bytes) + (div ? 20 : 15)) * 8 * 1025000L / (uint32_t)(bps))

282 static uint32_t get_nominal_packet_interval(struct bind_data *bd,
283  uint32_t *rx_slop)
284 {
285  uint32_t ret = get_control_packet_time(bd);
286 
287  uint32_t pre_slop = ret - 2000;
288 
289  if (bd->flags & TELEMETRY_MASK) {
290  ret += (BYTES_AT_BAUD_TO_USEC(TELEMETRY_PACKETSIZE, modem_params[bd->modem_params].bps, bd->flags&DIVERSITY_ENABLED) + 1000);
291  }
292 
293  // round up to ms
294  ret = ((ret + 999) / 1000) * 1000;
295 
296  if (rx_slop) {
297  /* The "rx slop" is calculated as half the time between the
298  * actual tx frame length, and all the other times-- padding,
299  * rx telemetry length, etc.
300  *
301  * It is when the rx should hop, relative to the end of the
302  * last known packet plus the interval, to maximize margins.
303  */
304  *rx_slop = (ret - pre_slop) / 2;
305  }
306 
307  /* RRRRRRRTTTT--RRRRRRRTTTT--missed!......
308  * ^ we want to hop here
309  */
310 
311  return ret;
312 }

314 static int pack_channels(uint8_t config, int16_t *ppm, uint8_t *buf)
315 {
316  config &= 7;
317 
318  uint8_t *p = buf;
319 
320  /* Pack 4 channels into 5 bytes */
321  for (int i = 0; i <= (config/2); i++) {
322  *(p++) = ppm[0];
323  *(p++) = ppm[1];
324  *(p++) = ppm[2];
325  *(p++) = ppm[3];
326  *(p++) =
327  ((ppm[0] & 0x300) >> 8) |
328  ((ppm[1] & 0x300) >> 6) |
329  ((ppm[2] & 0x300) >> 4) |
330  ((ppm[3] & 0x300) >> 2);
331 
332  ppm += 4;
333  }
334 
335  /* "Switch" channel */
336  if (config & 1) {
337  uint8_t packed_sw = 0;
338 
339  for (int i = 0; i < 4; i++) {
340  packed_sw <<= 2;
341 
342  if (ppm[0] < (166 + 12)) {
343  /* First 1/6 of travel sw pos 0 */
344  packed_sw |= 0;
345  } else if (ppm[0] < (550 + 12)) {
346  /* A little more than the next third is
347  * sw pos 1. Why? So a center value /
348  * 3 pos switch has clear value.
349  */
350  packed_sw |= 1;
351  } else if (ppm[0] < (833 + 12)) {
352  /* Up to 1/6 + 2/3 is sw pos 2 */
353  packed_sw |= 2;
354  } else {
355  /* Leaving the last sixth for sw pos 3 */
356  packed_sw |= 3;
357  }
358  }
359 
360  *(p++) = packed_sw;
361  }
362 
363  return p - buf;
364 }

367 static void txscale_channels(int16_t *chan, int16_t nominal_min, int16_t nominal_max)
368 {
369  if (nominal_min == nominal_max) {
370  /* Prevent divide by zero */
371  nominal_max++;
372  }
373 
374  for (uint32_t i = 0; i < OPENLRS_PPM_NUM_CHANNELS; i++) {
375  int16_t x = chan[i];
376 
377  /*
378  * We have 1024 counts. The middle 1000 should correspond to
379  * 1000..1999, the "normal range". The extra 12 are scaled by
380  * a factor of 16 (from original openlrsng) to make the
381  * overall range from -19% to 119%
382  *
383  * In other words, the "normal resolution" is 0.1%; the
384  * extended resolution is in 1.6% chunks.
385  */
386 
387  float scaled = x;
388  scaled -= nominal_min;
389  scaled /= (nominal_max - nominal_min);
390 
391  const float scale_edge = (12*16) / 1000.0f; /* .192 */
392 
393  if (scaled < -scale_edge) {
394  x = 0;
395  } else if (scaled < 0) {
396  /* [-scale_edge, 0) mapped to 0..12 */
397  x = ((scaled + scale_edge) / scale_edge) * 12;
398  } else if (scaled < 1) {
399  /* [0 .. 1) mapped to 12..1011 */
400  x = scaled * 1000 + 12;
401  } else if (scaled < (1 + scale_edge)) {
402  /* [1, 1 + scale_edge) mapped to 1011..1023 */
403  x = ((scaled - 1) / scale_edge) * 12 + 1011;
404  } else {
405  x = 1023;
406  }
407 
408  chan[i] = x;
409  }
410 }

414 static void unpack_channels(uint8_t config, int16_t *ppm, uint8_t *p)
415 {
416  /* Look at least 3 significant bit of config.
417  * LSB picks whether 4 switch channels are around.
418  * other 2 bits are a number between 1 and 4 for number of
419  * 4 channel groups.
420  *
421  * Therefore there can be up to 20 channels.
422  */
423 
424  config &= 7;
425  /* Ordinary channels are 4ch packed strangely in 5 bytes */
426  for (int i = 0; i<=(config/2); i++) {
427  ppm[0] = ((p[4] << 8) & 0x300) | p[0];
428  ppm[1] = ((p[4] << 6) & 0x300) | p[1];
429  ppm[2] = ((p[4] << 4) & 0x300) | p[2];
430  ppm[3] = ((p[4] << 2) & 0x300) | p[3];
431  p += 5;
432  ppm += 4;
433  }
434 
435  if (config & 1) { // 4ch packed in 1 byte;
436  ppm[0] = ((p[0] >> 6) & 3) * 333 + 12;
437  ppm[1] = ((p[0] >> 4) & 3) * 333 + 12;
438  ppm[2] = ((p[0] >> 2) & 3) * 333 + 12;
439  ppm[3] = ((p[0] >> 0) & 3) * 333 + 12;
440  }
441 }

464 static int pios_openlrs_form_telemetry(pios_openlrs_t openlrs_dev,
465  uint8_t *tx_buf, uint8_t rssi, bool only_serial,
466  int data_len_adjust)
467 {
468  int len = 0;
469 
470  tx_buf[0] &= 0xc0; // Preserve top two sequence bits
471 
472  int max_data_len = TELEMETRY_PACKETSIZE - 1;
473 
474  if (openlrs_dev->bind_data.ext_flags & EXT_FLAG_FASTDATA) {
475  /* Fast data extension: Adjust the amount we're
476  * willing to send from nominal packet lengths.
477  */
478  max_data_len += data_len_adjust;
479 
480  if (max_data_len < 1) {
481  /* Never shrink smaller than a 2b frame */
482  max_data_len = 1;
483  } else if (max_data_len >= MAX_PACKET_SIZE) {
484  max_data_len = MAX_PACKET_SIZE - 1;
485  }
486 
487  if (data_len_adjust < 0) {
488  only_serial = true;
489  }
490  }
491 
492  // Check for data on serial link
493  uint8_t bytes = 0;
494  // Append data from the com interface if applicable.
495  if (openlrs_dev->tx_out_cb) {
496  // Try to get some data to send
497  bool need_yield = false;
498  bytes = (openlrs_dev->tx_out_cb)(openlrs_dev->tx_out_context,
499  &tx_buf[1], max_data_len, NULL, &need_yield);
500  }
501 
502  if (bytes > 0) {
503  if (bytes < (TELEMETRY_PACKETSIZE - 1)) {
504  tx_buf[0] |= (0x37 + bytes);
505  } else {
506  tx_buf[0] |= 0x3f;
507  }
508 
509  if (openlrs_dev->bind_data.ext_flags & EXT_FLAG_FASTDATA) {
510  /* Fast data extension: send only the portion of the
511  * data that we used. Length field is capped at 8.
512  */
513  len = bytes + 1;
514  } else {
515  for (int i = bytes+1; i < TELEMETRY_PACKETSIZE; i++) {
516  tx_buf[i] = 0;
517  }
518 
519  len = TELEMETRY_PACKETSIZE;
520  }
521  } else if (only_serial) {
522  return 0;
523  } else {
524  // tx_buf[0] lowest 6 bits left at 0
525  tx_buf[1] = rssi;
526  if (FlightBatteryStateHandle()) {
527  FlightBatteryStateData bat;
528  FlightBatteryStateGet(&bat);
529  // FrSky protocol normally uses 3.3V at
530  // 255 but divider from display can be
531  // set internally
532  if (bat.DetectedCellCount) {
533  /* Use detected cell count to
534  * maximize useful resolution
535  */
536  tx_buf[2] = (uint8_t) (bat.Voltage / bat.DetectedCellCount / 5.0f * 255);
537  } else {
538  tx_buf[2] = (uint8_t) (bat.Voltage / 25.0f * 255);
539  }
540  tx_buf[3] = (uint8_t) (bat.Current / 60.0f * 255);
541  } else {
542  tx_buf[2] = 0;
543  tx_buf[3] = 0;
544  }
545 
546  /*
547  * 4-5: This used to be AFCC information, which
548  * was never used. Deprecated.
549  */
550  tx_buf[4] = 0;
551  tx_buf[5] = 0;
552  tx_buf[6] = count_set_bits(openlrs_status.LinkQuality & 0x7fff);
553  tx_buf[7] = 0;
554  tx_buf[8] = 0;
555 
556  len = TELEMETRY_PACKETSIZE;
557  }
558 
559  return len;
560 }

954 static bool pios_openlrs_rx_frame(pios_openlrs_t openlrs_dev, uint8_t rssi)
955 {
956  uint8_t *tx_buf = openlrs_dev->tx_buf;
957  uint8_t rx_buf[MAX_PACKET_SIZE];
958 
959  int ctrl_size = get_control_packet_size(&openlrs_dev->bind_data);
960 
961  // Read the packet from RFM22b
962  int size = rfm22_rx_packet_variable(openlrs_dev, rx_buf,
963  MAX_PACKET_SIZE);
964 
965  if (size <= 0) {
966  return false;
967  }
968 
969  pios_openlrs_rx_update_rssi(openlrs_dev, rssi, true);
970 
971 #if defined(PIOS_LED_LINK)
972  PIOS_ANNUNC_Toggle(PIOS_LED_LINK);
973 #endif /* PIOS_LED_LINK */
974 
975  if ((rx_buf[0] & 0x3e) == 0x00) {
976  // This flag indicates receiving control data
977 
978  if (size != ctrl_size) {
979  // These size mismatches should never happen because
980  // we are protected by a CRC, and shared magic hdr.
981  return false;
982  }
983 
984  unpack_channels(openlrs_dev->bind_data.flags,
985  openlrs_dev->ppm, rx_buf + 1);
986  rxscale_channels(openlrs_dev->ppm);
987 
988  // Call the control received callback if it's available.
989  if (openlrs_dev->openlrs_rcvr_id) {
990 #if defined(PIOS_INCLUDE_OPENLRS_RCVR)
991  PIOS_OpenLRS_Rcvr_UpdateChannels(
992  openlrs_dev->openlrs_rcvr_id,
993  openlrs_dev->ppm);
994 #endif
995  }
996  } else if ((rx_buf[0] & 0x3f) == 0x02) {
997  /* This is a "placeholder frame" -- no control data,
998  * no telemetry.
999  */
1000 
1001  /* Still set acknowledge bit to match-- they think we have
1002  * any data, and won't be resending, so be ready to accept
1003  * anything new they send.
1004  */
1005  if (rx_buf[0] & 0x80) {
1006  openlrs_dev->tx_buf[0] |= 0x80;
1007  } else {
1008  openlrs_dev->tx_buf[0] &= ~0x80;
1009  }
1010  } else {
1011  // Not control data. Push into serial RX buffer.
1012  if (((rx_buf[0] & 0x38) == 0x38) &&
1013  ((rx_buf[0] ^ tx_buf[0]) & 0x80)) {
1014  // We got new data... (not retransmission)
1015  bool rx_need_yield;
1016  uint8_t data_len = (rx_buf[0] & 7) + 1;
1017 
1018  if ((data_len + 1) > size) {
1019  return false;
1020  }
1021 
1022  if (data_len < (size - 1)) {
1023  /* Extended / fast-data operation */
1024  data_len = size - 1;
1025  }
1026 
1027  if (openlrs_dev->rx_in_cb) {
1028  (openlrs_dev->rx_in_cb)(openlrs_dev->rx_in_context,
1029  &rx_buf[1], data_len,
1030  NULL, &rx_need_yield);
1031  }
1032  }
1033 
1034  /* Set acknowledge bit to match */
1035  if (rx_buf[0] & 0x80) {
1036  openlrs_dev->tx_buf[0] |= 0x80;
1037  } else {
1038  openlrs_dev->tx_buf[0] &= ~0x80;
1039  }
1040  }
1041 
1042  // When telemetry is enabled we ack packets and send info about FC back
1043  if (openlrs_dev->bind_data.flags & TELEMETRY_MASK) {
1044  if ((tx_buf[0] ^ rx_buf[0]) & 0x40) {
1045  // Last message was lost... resend, unless it
1046  // is just ordinary rx telemetry in which
1047  // case we can update it.
1048 
1049  if ((tx_buf[0] & 0x3F) == 0) {
1050  // Don't swap telem sequence bit though.
1051  openlrs_dev->telem_len =
1052  pios_openlrs_form_telemetry(
1053  openlrs_dev, tx_buf, rssi,
1054  false, ctrl_size - size);
1055  }
1056  } else {
1057  tx_buf[0] ^= 0x40; // Swap telem sequence
1058  openlrs_dev->telem_len = pios_openlrs_form_telemetry(
1059  openlrs_dev, tx_buf, rssi, false,
1060  ctrl_size - size);
1061  }
1062 
1063  if (openlrs_dev->telem_len == 0) {
1064  tx_buf[0] |= 2;
1065  openlrs_dev->telem_len = 1;
1066  }
1067 
1068  // This will block until sent
1069  rfm22_tx_packet(openlrs_dev, tx_buf, openlrs_dev->telem_len);
1070  }
1071 
1072  // Once a packet has been processed, flip back into receiving mode
1073  rfm22_rx_reset(openlrs_dev, false);
1074 
1075  return true;
1076 }

1078 static void pios_openlrs_do_hop(pios_openlrs_t openlrs_dev)
1079 {
1080  openlrs_dev->rf_channel++;
1081 
1082  if (openlrs_dev->rf_channel >= openlrs_dev->hopcount) {
1083  openlrs_dev->rf_channel = 0;
1084  }
1085 
1086  if (openlrs_dev->beacon_armed) {
1087  // Listen for RSSI on beacon channel briefly for
1088  // a sharp rising edge in RSSI to 'trigger' us
1089  uint8_t rssi = beaconGetRSSI(openlrs_dev) << 2;
1090 
1091  if ((openlrs_dev->beacon_rssi_avg + 80) < rssi) {
1092  openlrs_dev->nextBeaconTimeMs =
1093  millis() + 1000L;
1094  }
1095 
1096  openlrs_dev->beacon_rssi_avg =
1097  (openlrs_dev->beacon_rssi_avg * 3 + rssi) >> 2;
1098  }
1099 
1100  rfm22_set_channel(openlrs_dev, openlrs_dev->rf_channel);
1101 }

1103 static void pios_openlrs_rx_after_receive(pios_openlrs_t openlrs_dev,
1104  bool have_frame)
1105 {
1106  uint32_t timeUs, timeMs;
1107 
1108  /* We hop channels here, unless we're hopping slowly and
1109  * it's not time yet.
1110  */
1111  bool willhop = true;
1112 
1113 #if defined(PIOS_INCLUDE_WDG) && defined(PIOS_WDG_RFM22B)
1114  // Update the watchdog timer
1115  PIOS_WDG_UpdateFlag(PIOS_WDG_RFM22B);
1116 #endif /* PIOS_WDG_RFM22B */
1117 
1118  timeUs = PIOS_DELAY_GetuS();
1119  timeMs = millis();
1120 
1121  DEBUG_PRINTF(2,"Time: %d\r\n", timeUs);
1122 
1123  uint32_t interval = get_nominal_packet_interval(
1124  &openlrs_dev->bind_data, NULL);
1125 
1126  // For missing packets to properly trigger a well timed
1127  // channel hop, this method should be called at the
1128  // appropriate time after the packet was expected to trigger
1129  // this path.
1130  if (have_frame) {
1131  // Flag to indicate we ever got a frame
1132  openlrs_dev->link_acquired = true;
1133 
1134  openlrs_dev->beacon_armed = false;
1135  openlrs_status.FailsafeActive = OPENLRSSTATUS_FAILSAFEACTIVE_INACTIVE;
1136  openlrs_dev->numberOfLostPackets = 0;
1137  openlrs_dev->nextBeaconTimeMs = 0;
1138 
1139  /* Could consider splitting difference with interval */
1140  openlrs_dev->lastPacketTimeUs = timeUs;
1141  openlrs_dev->linkLossTimeMs = 0;
1142  } else if (openlrs_dev->numberOfLostPackets < openlrs_dev->hopcount) {
1143  DEBUG_PRINTF(2,"OLRS WARN: Lost packet: %d\r\n",
1144  openlrs_dev->numberOfLostPackets);
1145  /* We lost packet, execute normally timed hop */
1146  openlrs_dev->numberOfLostPackets++;
1147  openlrs_dev->lastPacketTimeUs += interval;
1148  } else if ((openlrs_dev->numberOfLostPackets >= openlrs_dev->hopcount) &&
1149  (PIOS_DELAY_GetuSSince(openlrs_dev->lastPacketTimeUs) >
1150  (interval * (openlrs_dev->hopcount + 1)))) {
1151  DEBUG_PRINTF(2,"OLRS WARN: Trying to sync\r\n");
1152 
1153  // hop slowly to allow resync with TX
1154  openlrs_dev->lastPacketTimeUs = timeUs;
1155 
1156  if (!openlrs_dev->linkLossTimeMs) {
1157  openlrs_dev->linkLossTimeMs = timeMs;
1158  }
1159  } else {
1160  /* Don't have link, but it's not time for slow hop yet */
1161  willhop = false;
1162  }
1163 
1164  if (openlrs_dev->link_acquired && openlrs_dev->numberOfLostPackets) {
1165 #if defined(PIOS_LED_LINK)
1166  PIOS_ANNUNC_Off(PIOS_LED_LINK);
1167 #endif /* PIOS_LED_LINK */
1168 
1169  if (openlrs_dev->failsafeDelay &&
1170  (openlrs_status.FailsafeActive == OPENLRSSTATUS_FAILSAFEACTIVE_INACTIVE) &&
1171  ((timeMs - openlrs_dev->linkLossTimeMs) > ((uint32_t) openlrs_dev->failsafeDelay))) {
1172  DEBUG_PRINTF(2,"Failsafe activated: %d %d\r\n", timeMs, openlrs_dev->linkLossTimeMs);
1173  openlrs_status.FailsafeActive = OPENLRSSTATUS_FAILSAFEACTIVE_ACTIVE;
1174

1361  uint8_t *tx_buf)
1362 {

1366

1386  openlrs_dev->scale_max) / 2;

1393  channels, tx_buf+1);

1399 static bool pios_openlrs_tx_receive_telemetry(pios_openlrs_t openlrs_dev)
1400 {
1401  uint8_t rx_buf[MAX_PACKET_SIZE];
1402 
1403  // Read the packet from RFM22b
1404  int len = rfm22_rx_packet_variable(openlrs_dev, rx_buf,
1405  MAX_PACKET_SIZE);
1406 
1407  if (len <= 0) {
1408  return false;
1409  }
1410 
1411  if (((rx_buf[0] & 0x3f) == 0) && (len == TELEMETRY_PACKETSIZE)) {
1412  /* Generic link telemetry. Grab the data, shove it
1413  * into status.
1414  */
1415 
1416  openlrs_status.LastRSSI = rx_buf[1];
1417  openlrs_status.LinkQuality = (1 << (rx_buf[6])) - 1; /* fudge */
1418 
1419  OpenLRSStatusSet(&openlrs_status);
1420  } else if (((rx_buf[0] & 0x38) == 0x38) &&
1421  ((rx_buf[0] ^ openlrs_dev->tx_buf[0]) & 0x40)) {
1422  bool rx_need_yield;
1423  uint8_t data_len = (rx_buf[0] & 7) + 1;
1424 
1425  if ((data_len + 1) > len) {
1426  return false;
1427  }
1428 
1429  if (data_len < (len - 1)) {
1430  /* Extended / fast-data operation */
1431  data_len = len - 1;
1432  }
1433 
1434  if (openlrs_dev->rx_in_cb) {
1435  (openlrs_dev->rx_in_cb)(openlrs_dev->rx_in_context,
1436  &rx_buf[1], data_len,
1437  NULL, &rx_need_yield);
1438  }
1439  }
1440 
1441 #if defined(PIOS_LED_LINK)
1442  PIOS_ANNUNC_Toggle(PIOS_LED_LINK);
1443 #endif /* PIOS_LED_LINK */
1444 
1445  /* Set acknowledge bit to match */
1446  if (rx_buf[0] & 0x40) {
1447  openlrs_dev->tx_buf[0] |= 0x40;
1448  } else {
1449  openlrs_dev->tx_buf[0] &= ~0x40;
1450  }
1451 
1452  openlrs_dev->tx_prev_rxtelem_hdr = rx_buf[0];
1453 
1454  return true;
1455 }

1471 static void pios_openlrs_tx_frame(pios_openlrs_t openlrs_dev)
1472 {
1473  rfm22_rx_reset(openlrs_dev, false);
1474 
1475  uint32_t rx_slop;
1476  uint32_t interval =
1477  get_nominal_packet_interval(&openlrs_dev->bind_data, &rx_slop);
1478 
1479  pios_openlrs_do_hop(openlrs_dev);
1480 
1481  bool telem_uplink = false;
1482 
1483  openlrs_dev->tx_buf[0] &= 0xc0; // Preserve top two sequence bits
1484 
1485  /* Form a control frame. If we can't, we're allowed to telemeter
1486  * no matter what. Otherwise, we hold it in case telemetry
1487  * decided to not do anything.
1488  */
1489  uint8_t tx_buf[MAX_PACKET_SIZE];
1490 
1491  tx_buf[0] = openlrs_dev->tx_buf[0];
1492 
1493  int len = pios_openlrs_form_control_frame(openlrs_dev, tx_buf);
1494 
1495  if ((!len) || openlrs_dev->tx_ok_to_telemeter) {
1496  openlrs_dev->tx_ok_to_telemeter = false;
1497 
1498  if (openlrs_dev->telem_len &&
1499  ((openlrs_dev->tx_buf[0] ^
1500  openlrs_dev->tx_prev_rxtelem_hdr) & 0x80)) {
1501  /* We're resending the existing buffer. */
1502  telem_uplink = true;
1503  } else {
1504  int ctrl_size =
1505  get_control_packet_size(&openlrs_dev->bind_data);
1506 
1507  /* Try to form a new telemetry lump */
1508  openlrs_dev->telem_len =
1509  pios_openlrs_form_telemetry(openlrs_dev,
1510  openlrs_dev->tx_buf, 0, true,
1511  ctrl_size - TELEMETRY_PACKETSIZE +
1512  TELEMETRY_PACKETSIZE / 2);
1513 
1514  /* If we have 9 byte control frames, allow us to send
1515  * up to 9 + 4 = 13 bytes of data, robbing downlink of
1516  * 4 bytes.
1517  *
1518  * 8 + (9-8+4) = 13
1519  *
1520  * Only FASTDATA obeys this.
1521  */
1522 
1523  if (openlrs_dev->telem_len) {
1524  /* And if we do, flip the send seq */
1525  telem_uplink = true;
1526  openlrs_dev->tx_buf[0] ^= 0x80;
1527  }
1528  }
1529  }
1530 
1531  if (telem_uplink) {
1532  pios_openlrs_tx_wait_and_transmit(openlrs_dev, interval,
1533  openlrs_dev->tx_buf, openlrs_dev->telem_len);
1534  } else if (len > 0) {
1535  /* From the control buf, so we don't touch the telemetry
1536  * buffer-- which may need a retry, after all.
1537  */
1538  pios_openlrs_tx_wait_and_transmit(openlrs_dev, interval,
1539  tx_buf, len);
1540  } else {
1541  /* We have nothing to transmit, but want to keep sync.
1542  * This is a dR protocol extension.
1543  */
1544  tx_buf[0] |= 2;
1545 
1546  pios_openlrs_tx_wait_and_transmit(openlrs_dev, interval,
1547  tx_buf, 1);
1548  }
1549 
1550  if (!(openlrs_dev->bind_data.flags & TELEMETRY_MASK)) {
1551  return;
1552  }
1553 
1554  rfm22_rx_reset(openlrs_dev, false);
1555 
1556  /* We wait until the RX will be listening on the next channel. */
1557  uint32_t expiry = openlrs_dev->lastPacketTimeUs + interval +
1558  rx_slop / 2;
1559 
1560  uint8_t rssi;
1561  bool have_interrupt = wait_interrupt_until(openlrs_dev, expiry,
1562  &rssi);
1563 
1564  bool got_packet = false;
1565 
1566  if (have_interrupt) {
1567  if (pios_openlrs_tx_receive_telemetry(openlrs_dev)) {
1568  if (!telem_uplink) {
1569  /* It's OK to telemeter when we receive
1570  * a downlink telemetry message.. and we
1571  * didn't telemeter this time around.
1572  */
1573  openlrs_dev->tx_ok_to_telemeter = true;
1574  }
1575 
1576  got_packet = true;
1577  }
1578  }
1579 
1580  if (!got_packet) {
1581  /* TODO: link-beep functionality goes here */
1582  if (openlrs_dev->numberOfLostPackets < openlrs_dev->hopcount) {
1583  openlrs_dev->numberOfLostPackets++;
1584  } else {
1585 #if defined(PIOS_LED_LINK)
1586  /* Ensure link light is off */
1587  PIOS_ANNUNC_Off(PIOS_LED_LINK);
1588 #endif
1589  if (!openlrs_dev->linkLossTimeMs) {
1590  openlrs_dev->linkLossTimeMs =
1591  PIOS_Thread_Systime();
1592  }
1593  }
1594  } else {
1595  openlrs_dev->numberOfLostPackets = 0;
1596  openlrs_dev->linkLossTimeMs = 0;
1597  openlrs_dev->link_acquired = true;
1598  }
1599 }

1650 static void pios_openlrs_rx_step(pios_openlrs_t openlrs_dev)
1651 {
1652  rfm22_check_hang(openlrs_dev);
1653 
1654  uint8_t rssi = 0;
1655 
1656  uint32_t interval, slop;
1657 
1658  interval = get_nominal_packet_interval(&openlrs_dev->bind_data,
1659  &slop);
1660 
1661  /*
1662  * It's worth noting that for non-telemetry modes, we need
1663  * to stay very closely in sync because the TX is transmitting
1664  * most of the time. This means that we need to carefully
1665  * split that "dead" non-TX time, arriving to channels slightly
1666  * early, and leaving slightly late if we have not received a
1667  * frame.
1668  *
1669  * For telemetry modes, since "we" have a timeslot the
1670  * tolerances can be much more lax-- we can afford to listen
1671  * into what we believe to be our telemetry timeslot and thus
1672  * cope better if the TX is slower than we expect.
1673  */
1674 
1675  uint32_t expiry = openlrs_dev->lastPacketTimeUs + interval + slop;
1676 
1677  bool have_frame = false;
1678 
1679  if (wait_interrupt_until(openlrs_dev, expiry, &rssi)) {
1680  // Process incoming data
1681  have_frame = pios_openlrs_rx_frame(openlrs_dev, rssi);
1682  } else {
1683  // TODO: We could delay a little bit more if we looked
1684  // at the hardware and see it's midway through
1685  // receiving a valid frame.
1686 
1687  // We timed out because packet was missed
1688  DEBUG_PRINTF(3,
1689  "ISR Timeout. Missed packet: %d %d\r\n",
1690  expiry, interval);
1691  }
1692 
1693  if (!have_frame) {
1694  pios_openlrs_rx_update_rssi(openlrs_dev, rssi,
1695  false);
1696  }
1697 
1698  // Select next channel, update timers, etc.
1699  pios_openlrs_rx_after_receive(openlrs_dev, have_frame);
1700 
1701  DEBUG_PRINTF(3, "Processing time %d\r\n", PIOS_DELAY_GetuSSince(openlrs_dev->lastPacketTimeUs));
1702 }

2210 static void rfm22_set_frequency(pios_openlrs_t openlrs_dev, uint32_t f)
2211 {
2212  /* Protect ourselves from out-of-band frequencies. Ideally we'd latch
2213  * an error here and prevent tx, but this is good enough to protect
2214  * the hardware. */
2215  if ((f < min_freq(openlrs_dev->band)) ||
2216  (f > max_freq(openlrs_dev->band))) {
2217  f = def_carrier_freq(openlrs_dev->band);
2218  }
2219 
2220  DEBUG_PRINTF(3,"rfm22_set_frequency %d\r\n", f);
2221  uint16_t fb, fc, hbsel;
2222  if (f < 480000000) {
2223  hbsel = 0;
2224  fb = f / 10000000 - 24;
2225  fc = (f - (fb + 24) * 10000000) * 4 / 625;
2226  } else {
2227  hbsel = 1;
2228  fb = f / 20000000 - 24;
2229  fc = (f - (fb + 24) * 20000000) * 2 / 625;
2230  }
2231  rfm22_claim_bus(openlrs_dev);
2232  rfm22_write(openlrs_dev, RFM22_frequency_band_select, RFM22_fbs_sbse + (hbsel ? RFM22_fbs_hbsel : 0) + (fb & RFM22_fb_mask));
