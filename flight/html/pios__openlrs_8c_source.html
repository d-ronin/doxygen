167 static uint32_t getInterval(struct bind_data *bd)
168 {
169  uint32_t ret;
170  // Sending a x byte packet on bps y takes about (emperical)
171  // usec = (x + 15) * 8200000 / baudrate
172 #define BYTES_AT_BAUD_TO_USEC(bytes, bps, div) ((uint32_t)((bytes) + (div?20:15)) * 8200000L / (uint32_t)(bps))
173 
174  ret = (BYTES_AT_BAUD_TO_USEC(getPacketSize(bd), modem_params[bd->modem_params].bps, bd->flags&DIVERSITY_ENABLED) + 2000);
175 
176  if (bd->flags & TELEMETRY_MASK) {
177  ret += (BYTES_AT_BAUD_TO_USEC(TELEMETRY_PACKETSIZE, modem_params[bd->modem_params].bps, bd->flags&DIVERSITY_ENABLED) + 1000);
178  }
179 
180  // round up to ms
181  ret = ((ret + 999) / 1000) * 1000;
182 
183  // enable following to limit packet rate to 50Hz at most
184 #ifdef LIMIT_RATE_TO_50HZ
185  if (ret < 20000) {
186  ret = 20000;
187  }
188 #endif
189 
190  return ret;
191 }

193 static void unpackChannels(uint8_t config, int16_t PPM[], uint8_t *p)
194 {
195  uint8_t i;
196  for (i=0; i<=(config/2); i++) { // 4ch packed in 5 bytes
197  PPM[0] = (((uint16_t)p[4] & 0x03) << 8) + p[0];
198  PPM[1] = (((uint16_t)p[4] & 0x0c) << 6) + p[1];
199  PPM[2] = (((uint16_t)p[4] & 0x30) << 4) + p[2];
200  PPM[3] = (((uint16_t)p[4] & 0xc0) << 2) + p[3];
201  p+=5;
202  PPM+=4;
203  }
204  if (config & 1) { // 4ch packed in 1 byte;
205  PPM[0] = (((uint16_t)p[0] >> 6) & 3) * 333 + 12;
206  PPM[1] = (((uint16_t)p[0] >> 4) & 3) * 333 + 12;
207  PPM[2] = (((uint16_t)p[0] >> 2) & 3) * 333 + 12;
208  PPM[3] = (((uint16_t)p[0] >> 0) & 3) * 333 + 12;
209  }

314 static void rfmSetCarrierFrequency(struct pios_openlrs_dev *openlrs_dev, uint32_t f)
315 {
316  /* Protect ourselves from out-of-band frequencies. Ideally we'd latch
317  * an error here and prevent tx, but this is good enough to protect
318  * the hardware. */
319  if ((f < minFreq(openlrs_dev->band)) ||
320  (f > maxFreq(openlrs_dev->band))) {
321  f = defCarrierFreq(openlrs_dev->band);
322  }
323 
324  DEBUG_PRINTF(3,"rfmSetCarrierFrequency %d\r\n", f);
325  uint16_t fb, fc, hbsel;
326  if (f < 480000000) {
327  hbsel = 0;
328  fb = f / 10000000 - 24;
329  fc = (f - (fb + 24) * 10000000) * 4 / 625;
330  } else {
331  hbsel = 1;
332  fb = f / 20000000 - 24;
333  fc = (f - (fb + 24) * 20000000) * 2 / 625;
334  }
335  rfm22_claimBus(openlrs_dev);
336  rfm22_write(openlrs_dev, RFM22_frequency_band_select, RFM22_fbs_sbse + (hbsel ? RFM22_fbs_hbsel : 0) + (fb & RFM22_fb_mask));
337  rfm22_write(openlrs_dev, RFM22_nominal_carrier_frequency1, (fc >> 8));
338  rfm22_write(openlrs_dev, RFM22_nominal_carrier_frequency0, (fc & 0xff));
339  rfm22_releaseBus(openlrs_dev);
340 }

342 static void init_rfm(struct pios_openlrs_dev *openlrs_dev, uint8_t isbind)
343 {
344  DEBUG_PRINTF(2,"init_rfm %d\r\n", isbind);
345 
346  if (!isbind) {
347  DEBUG_PRINTF(2, "Binding settings:\r\n");
348  PIOS_Thread_Sleep(10);
349  DEBUG_PRINTF(2, " version: %d\r\n", openlrs_dev->bind_data.version);
350  PIOS_Thread_Sleep(10);
351  DEBUG_PRINTF(2, " serial_baudrate: %d\r\n", openlrs_dev->bind_data.serial_baudrate);
352  PIOS_Thread_Sleep(10);
353  DEBUG_PRINTF(2, " rf_frequency: %d\r\n", openlrs_dev->bind_data.rf_frequency);
354  PIOS_Thread_Sleep(10);
355  DEBUG_PRINTF(2, " rf_power: %d\r\n", openlrs_dev->bind_data.rf_power);
356  PIOS_Thread_Sleep(10);
357  DEBUG_PRINTF(2, " rf_channel_spacing: %d\r\n", openlrs_dev->bind_data.rf_channel_spacing);
358  PIOS_Thread_Sleep(10);
359  DEBUG_PRINTF(2, " modem_params: %d\r\n", openlrs_dev->bind_data.modem_params);
360  PIOS_Thread_Sleep(10);
361  DEBUG_PRINTF(2, " flags: %d\r\n", openlrs_dev->bind_data.flags);
362  PIOS_Thread_Sleep(10);
363  }
364 
365  rfm22_claimBus(openlrs_dev);
366  openlrs_dev->it_status1 = rfm22_read(openlrs_dev, RFM22_interrupt_status1); // read status, clear interrupt
367  openlrs_dev->it_status2 = rfm22_read(openlrs_dev, RFM22_interrupt_status2);
368  rfm22_write(openlrs_dev, RFM22_interrupt_enable2, 0x00); // disable interrupts
369  rfm22_write(openlrs_dev, RFM22_op_and_func_ctrl1, RF22B_PWRSTATE_READY); // disable lbd, wakeup timer, use internal 32768,xton = 1; in ready mode
370  rfm22_write(openlrs_dev, RFM22_xtal_osc_load_cap, 0x7f); // c = 12.5p
371  rfm22_write(openlrs_dev, RFM22_cpu_output_clk, 0x05);
372  switch (openlrs_dev->cfg.gpio_direction) {
373  case GPIO0_TX_GPIO1_RX:
374  rfm22_write(openlrs_dev, RFM22_gpio0_config, RFM22_gpio0_config_txstate); // gpio0 TX State
375  rfm22_write(openlrs_dev, RFM22_gpio1_config, RFM22_gpio1_config_rxstate); // gpio1 RX State
376  break;
377  case GPIO0_RX_GPIO1_TX:
378  rfm22_write(openlrs_dev, RFM22_gpio0_config, RFM22_gpio0_config_rxstate); // gpio0 RX State
379  rfm22_write(openlrs_dev, RFM22_gpio1_config, RFM22_gpio1_config_txstate); // gpio1 TX State
380  break;
381  }
382  rfm22_write(openlrs_dev, RFM22_gpio2_config, 0xfd); // gpio 2 VDD
383  rfm22_write(openlrs_dev, RFM22_io_port_config, RFM22_io_port_default); // gpio 0, 1,2 NO OTHER FUNCTION.
384  rfm22_releaseBus(openlrs_dev);
385 
386  if (isbind) {
387  setModemRegs(openlrs_dev, &bind_params);
388  } else {
389  setModemRegs(openlrs_dev, &modem_params[openlrs_dev->bind_data.modem_params]);
390  }
391 
392  // Packet settings
393  rfm22_claimBus(openlrs_dev);
394  rfm22_write(openlrs_dev, RFM22_data_access_control, 0x8c); // enable packet handler, msb first, enable crc,
395  rfm22_write(openlrs_dev, RFM22_header_control1, 0x0f); // no broadcast, check header bytes 3,2,1,0
396  rfm22_write(openlrs_dev, RFM22_header_control2, 0x42); // 4 byte header, 2 byte synch, variable pkt size
397  rfm22_write(openlrs_dev, RFM22_preamble_length, (openlrs_dev->bind_data.flags & DIVERSITY_ENABLED)?0x14:0x0a); // 40 bit preamble, 80 with diversity
398  rfm22_write(openlrs_dev, RFM22_preamble_detection_ctrl1, 0x2a); // preath = 5 (20bits), rssioff = 2
399  rfm22_write(openlrs_dev, RFM22_sync_word3, 0x2d); // synchronize word 3
400  rfm22_write(openlrs_dev, RFM22_sync_word2, 0xd4); // synchronize word 2
401  rfm22_write(openlrs_dev, RFM22_sync_word1, 0x00); // synch word 1 (not used)
402  rfm22_write(openlrs_dev, RFM22_sync_word0, 0x00); // synch word 0 (not used)
403 
404  uint32_t magic = isbind ? BIND_MAGIC : openlrs_dev->bind_data.rf_magic;
405  for (uint8_t i = 0; i < 4; i++) {
406  rfm22_write(openlrs_dev, RFM22_transmit_header3 + i, (magic >> 24) & 0xff); // tx header
407  rfm22_write(openlrs_dev, RFM22_check_header3 + i, (magic >> 24) & 0xff); // rx header
408  magic = magic << 8; // advance to next byte
409  }
410 
411  rfm22_write(openlrs_dev, RFM22_header_enable3, 0xff); // all the bit to be checked
412  rfm22_write(openlrs_dev, RFM22_header_enable2, 0xff); // all the bit to be checked
413  rfm22_write(openlrs_dev, RFM22_header_enable1, 0xff); // all the bit to be checked
414  rfm22_write(openlrs_dev, RFM22_header_enable0, 0xff); // all the bit to be checked
415 
416  if (isbind) {
417  rfm22_write(openlrs_dev, RFM22_tx_power, BINDING_POWER);
418  } else {
419  rfm22_write(openlrs_dev, RFM22_tx_power, openlrs_dev->bind_data.rf_power);
420  }
421 
422  rfm22_write(openlrs_dev, RFM22_frequency_hopping_channel_select, 0);
423  rfm22_write(openlrs_dev, RFM22_frequency_hopping_step_size, openlrs_dev->bind_data.rf_channel_spacing); // channel spacing
424 
425  rfm22_write(openlrs_dev, RFM22_frequency_offset1, 0x00);
426  rfm22_write(openlrs_dev, RFM22_frequency_offset2, 0x00); // no offset
427 
428  rfm22_releaseBus(openlrs_dev);
429 
430  rfmSetCarrierFrequency(openlrs_dev, isbind ? bindingFreq(openlrs_dev->band) : openlrs_dev->bind_data.rf_frequency);
431 }

832 static void pios_openlrs_rx_loop(struct pios_openlrs_dev *openlrs_dev)
833 {
834  uint32_t timeUs, timeMs;
835 
836 #if defined(PIOS_INCLUDE_WDG) && defined(PIOS_WDG_RFM22B)
837  // Update the watchdog timer
838  PIOS_WDG_UpdateFlag(PIOS_WDG_RFM22B);
839 #endif /* PIOS_WDG_RFM22B */
840 
841  if (rfm22_read_claim(openlrs_dev, 0x0C) == 0) { // detect the locked module and reboot
842  DEBUG_PRINTF(2,"OLRS ERR: RX hang\r\n");
843  init_rfm(openlrs_dev, 0);
844  to_rx_mode(openlrs_dev);
845  }
846 
847  timeUs = micros();
848  timeMs = millis();
849 
850  DEBUG_PRINTF(2,"Time: %d\r\n", timeUs);
851 
852  uint8_t *tx_buf = openlrs_dev->tx_buf; // convenient variable
853 
854  if (openlrs_dev->rf_mode == Received) {
855 
856  DEBUG_PRINTF(2,"Packet Received. Dt=%d\r\n", timeUs-openlrs_dev->lastPacketTimeUs);
857 
858  // Read the packet from RFM22b
859  rfm22_claimBus(openlrs_dev);
860  rfm22_assertCs(openlrs_dev);
861  PIOS_SPI_TransferByte(openlrs_dev->spi_id, 0x7F);
862  uint32_t packet_size = getPacketSize(&openlrs_dev->bind_data);
863  PIOS_SPI_TransferBlock(openlrs_dev->spi_id, OUT_FF,
864  openlrs_dev->rx_buf, packet_size);
865  rfm22_deassertCs(openlrs_dev);
866  rfm22_releaseBus(openlrs_dev);
867 
868  openlrs_dev->lastAFCCvalue = rfmGetAFCC(openlrs_dev);
869 
870 #if defined(PIOS_LED_LINK)
871  PIOS_ANNUNC_Toggle(PIOS_LED_LINK);
872 #endif /* PIOS_LED_LINK */
873 
874  openlrs_dev->lastPacketTimeUs = timeUs;
875  openlrs_dev->numberOfLostPackets = 0;
876  openlrs_status.LinkQuality <<= 1;
877  openlrs_status.LinkQuality |= 1;
878 
879  if ((openlrs_dev->rx_buf[0] & 0x3e) == 0x00) {
880  // This flag indicates receiving PPM data
881 
882  unpackChannels(openlrs_dev->bind_data.flags & 7, openlrs_dev->ppm, openlrs_dev->rx_buf + 1);
883  rescaleChannels(openlrs_dev->ppm);
884 
885  // Call the PPM received callback if it's available.
886  if (openlrs_dev->openlrs_rcvr_id) {
887 #if defined(PIOS_INCLUDE_OPENLRS_RCVR)
888  PIOS_OpenLRS_Rcvr_UpdateChannels(openlrs_dev->openlrs_rcvr_id, openlrs_dev->ppm);
889 #endif
890  }
891  }
892  else {
893  // Not PPM data. Push into serial RX buffer.
894  if ((openlrs_dev->rx_buf[0] & 0x38) == 0x38) {
895  if ((openlrs_dev->rx_buf[0] ^ tx_buf[0]) & 0x80) {
896  // We got new data... (not retransmission)
897  tx_buf[0] ^= 0x80; // signal that we got it
898  bool rx_need_yield;
899  uint8_t data_len = openlrs_dev->rx_buf[0] & 7;
900  if (openlrs_dev->rx_in_cb && (data_len > 0)) {
901  (openlrs_dev->rx_in_cb) (openlrs_dev->rx_in_context, &openlrs_dev->rx_buf[1], data_len, NULL, &rx_need_yield);
902  }
903  }
904  }
905  }
906 
907  // Flag to indicate ever got a link
908  openlrs_dev->link_acquired |= true;
909  openlrs_status.FailsafeActive = OPENLRSSTATUS_FAILSAFEACTIVE_INACTIVE;
910  openlrs_dev->beacon_armed = false; // when receiving packets make sure beacon cannot emit
911 
912  // When telemetry is enabled we ack packets and send info about FC back
913  if (openlrs_dev->bind_data.flags & TELEMETRY_MASK) {
914  if ((tx_buf[0] ^ openlrs_dev->rx_buf[0]) & 0x40) {
915  // resend last message
916  } else {
917  tx_buf[0] &= 0xc0;
918  tx_buf[0] ^= 0x40; // swap sequence as we have new data
919 
920  // Check for data on serial link
921  uint8_t bytes = 0;
922  // Append data from the com interface if applicable.
923  if (openlrs_dev->tx_out_cb) {
924  // Try to get some data to send
925  bool need_yield = false;
926  bytes = (openlrs_dev->tx_out_cb) (openlrs_dev->tx_out_context, &tx_buf[1], 8, NULL, &need_yield);
927  }
928 
929  if (bytes > 0) {
930  tx_buf[0] |= (0x37 + bytes);
931  } else {
932  // tx_buf[0] lowest 6 bits left at 0
933  tx_buf[1] = openlrs_status.LastRSSI;
934  if (FlightBatteryStateHandle()) {
935  FlightBatteryStateData bat;
936  FlightBatteryStateGet(&bat);
937  // FrSky protocol normally uses 3.3V at 255 but
938  // divider from display can be set internally
939  tx_buf[2] = (uint8_t) (bat.Voltage / 25.0f * 255);
940  tx_buf[3] = (uint8_t) (bat.Current / 60.0f * 255);
941  } else {
942  tx_buf[2] = 0; // these bytes carry analog info. package
943  tx_buf[3] = 0; // battery here
944  }

1237  /* This block of code determines the timing of when to call the loop method. It reaches a bit into
1238  * the internal state of that method to get the optimal timings. This is to keep the loop method as
1239  * similar as possible to the openLRSng implementation (for easier maintenance of compatibility)
1240  * while minimizing overhead spinning in a while loop.
1241  *
1242  * There are three reasons to go into loop:
1243  * 1. the ISR was triggered (packet was received)
1244  * 2. a little before the expected packet (to sample the RSSI while receiving packet)
1245  * 3. a little after expected packet (to channel hop when a packet was missing)

1255  uint32_t time_till_measure_rssi_us = (getInterval(&openlrs_dev->bind_data) - packet_advance_time_us) - time_since_packet_us;
1256  delay_ms = (time_till_measure_rssi_us + 999) / 1000;
1257  DEBUG_PRINTF(3, "T1: %d\r\n", delay_ms);
1258  } else {

1260  uint32_t time_till_timeout_us = (getInterval(&openlrs_dev->bind_data) + packet_timeout_us) - time_since_packet_us;
1261  delay_ms = (time_till_timeout_us + 999) / 1000;
1262  DEBUG_PRINTF(3, "T2: %d %d\r\n", time_till_timeout_us,delay_ms, time_since_packet_us);
1263  }
1264 
1265  // Maximum delay based on packet time
1266  const uint32_t max_delay = (getInterval(&openlrs_dev->bind_data) + packet_timeout_us) / 1000;
1267  if (delay_ms > max_delay) delay_ms = max_delay;
1268

1270  if (!rssi_sampled) {

1272  if (openlrs_dev->numberOfLostPackets < 2) {
1273 
1274  openlrs_dev->lastRSSITimeUs = openlrs_dev->lastPacketTimeUs;

1276

1278  }

1282  pios_openlrs_rx_loop(openlrs_dev);

1293  rssi_sampled = false;
