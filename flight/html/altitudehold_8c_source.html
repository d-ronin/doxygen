145 static void altitudeHoldTask(void *parameters)
146 {
147  bool engaged = false;
148 
149  AltitudeHoldDesiredData altitudeHoldDesired;
150  StabilizationDesiredData stabilizationDesired;
151  AltitudeHoldSettingsData altitudeHoldSettings;
152 
153  UAVObjEvent ev;
154  struct pid velocity_pid;
155 
156  pid_zero(&velocity_pid);
157 
158  // Listen for object updates.
159  AltitudeHoldSettingsConnectQueue(queue);
160  FlightStatusConnectQueue(queue);
161 
162  AltitudeHoldSettingsGet(&altitudeHoldSettings);
163  // Main task loop
164  const uint32_t dt_ms = 20;
165 
166  pid_configure(&velocity_pid, altitudeHoldSettings.VelocityKp,
167  altitudeHoldSettings.VelocityKi, 0.0f, 1.0f,
168  dt_ms / 1000.0f);
169 
170  AlarmsSet(SYSTEMALARMS_ALARM_ALTITUDEHOLD, SYSTEMALARMS_ALARM_OK);
171 
172  uint32_t timeout = dt_ms;
173 
174  while (1) {
175  if (PIOS_Queue_Receive(queue, &ev, timeout) != true) {
176 
177  } else if (ev.obj == FlightStatusHandle()) {
178 
179  uint8_t flight_mode;
180  FlightStatusFlightModeGet(&flight_mode);
181 
182  if (flight_mode == FLIGHTSTATUS_FLIGHTMODE_ALTITUDEHOLD && !engaged) {
183  // Copy the current throttle as a starting point for integral
184  StabilizationDesiredThrustGet(&velocity_pid.iAccumulator);
185  engaged = true;
186  } else if (flight_mode != FLIGHTSTATUS_FLIGHTMODE_ALTITUDEHOLD)
187  engaged = false;
188 
189  // Run loop at 20 Hz when engaged otherwise just slowly wait for it to be engaged
190  timeout = engaged ? dt_ms : 100;
191 
192  continue;
193 
194  } else if (ev.obj == AltitudeHoldSettingsHandle()) {
195  AltitudeHoldSettingsGet(&altitudeHoldSettings);
196 
197  pid_configure(&velocity_pid,
198  altitudeHoldSettings.VelocityKp,
199  altitudeHoldSettings.VelocityKi,
200  0.0f, 1.0f, dt_ms / 1000.0f);
201  continue;
202  }
203 
204  // When engaged compute altitude controller output
205  if (engaged) {
206  float position_z, velocity_z, altitude_error;
207 
208  PositionActualDownGet(&position_z);
209  VelocityActualDownGet(&velocity_z);
210  position_z = -position_z; // Use positive up convention
211  velocity_z = -velocity_z; // Use positive up convention
212 
213  // Compute the altitude error
214  AltitudeHoldDesiredGet(&altitudeHoldDesired);
215  altitude_error = altitudeHoldDesired.Altitude - position_z;
216 
217  bool landing = altitudeHoldDesired.Land == ALTITUDEHOLDDESIRED_LAND_TRUE;
218 
219  // For landing mode allow throttle to go negative to allow the integrals
220  // to stop winding up
221  const float min_throttle = landing ? -0.1f : 0.0f;
222 
223  float velocity_desired = altitude_error * altitudeHoldSettings.PositionKp + altitudeHoldDesired.ClimbRate;
224  float throttle_desired = pid_apply_antiwindup(&velocity_pid,
225  velocity_desired - velocity_z,
226  min_throttle, 1.0f // positive limits since this is throttle
227  );
228 
229  AltitudeHoldStateData altitudeHoldState;
230  altitudeHoldState.VelocityDesired = velocity_desired;
231  altitudeHoldState.Integral = velocity_pid.iAccumulator;
232  altitudeHoldState.AngleGain = 1.0f;
233 
234  if (altitudeHoldSettings.AttitudeComp > 0) {
235  // Thrust desired is at this point the mount desired in the up direction, we can
236  // account for the attitude if desired
237  AttitudeActualData attitudeActual;
238  AttitudeActualGet(&attitudeActual);
239 
240  // Project a unit vector pointing up into the body frame and
241  // get the z component
242  float fraction = attitudeActual.q1 * attitudeActual.q1 -
243  attitudeActual.q2 * attitudeActual.q2 -
244  attitudeActual.q3 * attitudeActual.q3 +
245  attitudeActual.q4 * attitudeActual.q4;
246 
247  // Add ability to scale up the amount of compensation to achieve
248  // level forward flight
249  fraction = powf(fraction, (float) altitudeHoldSettings.AttitudeComp / 100.0f);
250 
251  // Dividing by the fraction remaining in the vertical projection will
252  // attempt to compensate for tilt. This acts like the thrust is linear
253  // with the output which isn't really true. If the fraction is starting
254  // to go negative we are inverted and should shut off throttle
255  throttle_desired = (fraction > 0.1f) ? (throttle_desired / fraction) : 0.0f;
256 
257  altitudeHoldState.AngleGain = 1.0f / fraction;
258  }
259 
260  altitudeHoldState.Thrust = throttle_desired;
261  AltitudeHoldStateSet(&altitudeHoldState);
262 
263  stabilizationDesired.Thrust = bound_min_max(throttle_desired, min_throttle, 1.0f);
264 
265  stabilizationDesired.ReprojectionMode = STABILIZATIONDESIRED_REPROJECTIONMODE_NONE;
266 
267  if (landing) {
268  stabilizationDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
269  stabilizationDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
270  stabilizationDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK;
271  stabilizationDesired.Roll = 0;
272  stabilizationDesired.Pitch = 0;
273  stabilizationDesired.Yaw = 0;
274  } else {
275  stabilizationDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
276  stabilizationDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
277  stabilizationDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK;
278  stabilizationDesired.Roll = altitudeHoldDesired.Roll;
279  stabilizationDesired.Pitch = altitudeHoldDesired.Pitch;
280  stabilizationDesired.Yaw = altitudeHoldDesired.Yaw;
281  }
282  StabilizationDesiredSet(&stabilizationDesired);
283  }
284 
285  }
286 }
