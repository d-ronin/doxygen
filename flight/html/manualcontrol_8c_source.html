112 static void manualControlTask(void *parameters)
113 {
114  /* Make sure disarmed on power up */
115  FlightStatusData flightStatus;
116  FlightStatusGet(&flightStatus);
117  flightStatus.Armed = FLIGHTSTATUS_ARMED_DISARMED;
118  FlightStatusSet(&flightStatus);
119 
120  // Select failsafe before run
121  failsafe_control_select(true);
122 
123  uint32_t arm_time = 1000;
124 
125  enum arm_state {
126  ARM_STATE_SAFETY,
127  ARM_STATE_DISARMED,
128  ARM_STATE_ARMING,
129  ARM_STATE_HOLDING,
130  ARM_STATE_ARMED,
131  } arm_state = ARM_STATE_SAFETY;
132 
133  uint32_t arm_state_time = 0;
134 
135  while (1) {
136 
137  // Process periodic data for each of the controllers, including reading
138  // all available inputs
139  failsafe_control_update();
140  transmitter_control_update();
141  tablet_control_update();
142  geofence_control_update();
143 
144  // Initialize to invalid value to ensure first update sets FlightStatus
145  static FlightStatusControlSourceOptions last_control_selection = -1;
146 
147  // Control logic to select the valid controller
148  FlightStatusControlSourceOptions control_selection =
149  control_source_select();
150  bool reset_controller = control_selection != last_control_selection;
151 
152  int ret = -1;
153 
154  enum control_status status = transmitter_control_get_status();
155 
156  // This logic would be better collapsed into control_source_select but
157  // in the case of the tablet we need to be able to abort and fall back
158  // to failsafe for now
159  switch(control_selection) {
160  case FLIGHTSTATUS_CONTROLSOURCE_TRANSMITTER:
161  ret = transmitter_control_select(reset_controller);
162  break;
163  case FLIGHTSTATUS_CONTROLSOURCE_TABLET:
164  ret = tablet_control_select(reset_controller);
165  break;
166  case FLIGHTSTATUS_CONTROLSOURCE_GEOFENCE:
167  ret = geofence_control_select(reset_controller);
168  break;
169  default:
170  /* Fall through into failsafe */
171  break;
172  }
173 
174  if (ret) {
175  failsafe_control_select(last_control_selection !=
176  FLIGHTSTATUS_CONTROLSOURCE_FAILSAFE);
177  control_selection = FLIGHTSTATUS_CONTROLSOURCE_FAILSAFE;
178  }
179 
180  if (control_selection != last_control_selection) {
181  FlightStatusControlSourceSet(&control_selection);
182  last_control_selection = control_selection;
183  }
184 
185  uint32_t now = PIOS_Thread_Systime();
186 
187 #define GO_STATE(x) \
188  do { \
189  arm_state = (x); \
190  arm_state_time = (now); \
191  } while (0)
192 
193  /* Global state transitions: weird stuff means disarmed +
194  * safety hold-down.
195  */
196  if ((status == STATUS_ERROR) ||
197  (status == STATUS_SAFETYTIMEOUT)) {
198  GO_STATE(ARM_STATE_SAFETY);
199  }
200 
201  /* If there's an alarm that prevents arming, enter a 0.2s
202  * no-arming-holddown. This also stretches transient alarms
203  * and makes them more visible.
204  */
205  if (!ok_to_arm() && (arm_state != ARM_STATE_ARMED)) {
206  GO_STATE(ARM_STATE_SAFETY);
207  }
208 
209  switch (arm_state) {
210  case ARM_STATE_SAFETY:
211  /* state_time > 0.2s + "NONE" or "DISARM" -> DISARMED
212  *
213  * !"NONE" and !"DISARM" -> SAFETY, reset state_time
214  */
215  if ((status == STATUS_NORMAL) ||
216  (status == STATUS_DISARM)) {
217  if (PIOS_Thread_Period_Elapsed(arm_state_time, 200)) {
218  GO_STATE(ARM_STATE_DISARMED);
219  }
220  } else {
221  GO_STATE(ARM_STATE_SAFETY);
222  }
223 
224  break;
225  case ARM_STATE_DISARMED:
226  /* "ARM_*" -> ARMING
227  * "DISCONNECTED" -> SAFETY
228  * "INVALID_FOR_DISARMED" -> SAFETY
229  */
230 
231  if ((status == STATUS_ARM_INVALID) ||
232  (status == STATUS_ARM_VALID)) {
233  GO_STATE(ARM_STATE_ARMING);
234  } else if ((status == STATUS_DISCONNECTED) ||
235  (status == STATUS_INVALID_FOR_DISARMED)) {
236  GO_STATE(ARM_STATE_SAFETY);
237  }
238 
239  ManualControlSettingsArmTimeOptions arm_enum;
240  ManualControlSettingsArmTimeGet(&arm_enum);
241 
242  arm_time = (arm_enum == MANUALCONTROLSETTINGS_ARMTIME_250) ? 250 :
243  (arm_enum == MANUALCONTROLSETTINGS_ARMTIME_500) ? 500 :
244  (arm_enum == MANUALCONTROLSETTINGS_ARMTIME_1000) ? 1000 :
245  (arm_enum == MANUALCONTROLSETTINGS_ARMTIME_2000) ? 2000 : 1000;
246 
247  break;
248  case ARM_STATE_ARMING:
249  /* Anything !"ARM_*" -> SAFETY
250  *
251  * state_time > ArmTime + ARM_INVALID -> HOLDING
252  * state_time > ArmTime + ARM_VALID -> ARMED
253  */
254 
255  if ((status != STATUS_ARM_INVALID) &&
256  (status != STATUS_ARM_VALID)) {
257  GO_STATE(ARM_STATE_SAFETY);
258  } else if (PIOS_Thread_Period_Elapsed(arm_state_time, arm_time)) {
259  if (status == STATUS_ARM_VALID) {
260  GO_STATE(ARM_STATE_ARMED);
261  } else {
262  /* Must be STATUS_ARM_INVALID */
263  GO_STATE(ARM_STATE_HOLDING);
264  }
265  }
266 
267  break;
268  case ARM_STATE_HOLDING:
269  /* ARM_VALID -> ARMED
270  * NONE -> ARMED
271  *
272  * ARM_INVALID -> Stay here
273  *
274  * Anything else -> SAFETY
275  */
276  if ((status == STATUS_ARM_VALID) ||
277  (status == STATUS_NORMAL)) {
278  GO_STATE(ARM_STATE_ARMED);
279  } else if (status == STATUS_ARM_INVALID) {
280  /* TODO: could consider having a maximum
281  * time before we go to safety */
282  } else {
283  GO_STATE(ARM_STATE_SAFETY);
284  }
285 
286  break;
287  case ARM_STATE_ARMED:
288  /* "DISARM" -> SAFETY (lower layer's job to check
289  * DisarmTime)
290  */
291  if (status == STATUS_DISARM) {
292  GO_STATE(ARM_STATE_SAFETY);
293  }
294 
295  break;
296  }
297 
298  FlightStatusArmedOptions armed, prev_armed;
299 
300  FlightStatusArmedGet(&prev_armed);
301 
302  switch (arm_state) {
303  default:
304  case ARM_STATE_SAFETY:
305  case ARM_STATE_DISARMED:
306  armed = FLIGHTSTATUS_ARMED_DISARMED;
307  vehicle_is_armed = false;
308  break;
309  case ARM_STATE_ARMING:
310  armed = FLIGHTSTATUS_ARMED_ARMING;
311  break;
312  case ARM_STATE_HOLDING:
313  /* For now consider "HOLDING" an armed state,
314  * like old code does. (Necessary to get the
315  * "initial spin while armed" that causes user
316  * to release arming position).
317  *
318  * TODO: do something different because
319  * control position invalid.
320  */
321  case ARM_STATE_ARMED:
322  armed = FLIGHTSTATUS_ARMED_ARMED;
323  vehicle_is_armed = true;
324  break;
325  }
326 
327  if (armed != prev_armed) {
328  FlightStatusArmedSet(&armed);
329  }
330 
331  // Wait until next update
332  PIOS_RCVR_WaitActivity(UPDATE_PERIOD_MS);
333  PIOS_WDG_UpdateFlag(PIOS_WDG_MANUAL);
334  }
335 }

348 static FlightStatusControlSourceOptions control_source_select()
349 {
350  // If the geofence controller is triggered, it takes precendence
351  // over even transmitter or failsafe. This means this must be
352  // EXTREMELY robust. The current behavior of geofence is simply
353  // to shut off the motors.
354  if (geofence_control_activate()) {
355  return FLIGHTSTATUS_CONTROLSOURCE_GEOFENCE;
356  }
357 
358  ManualControlCommandData cmd;
359  ManualControlCommandGet(&cmd);
360  if (cmd.Connected != MANUALCONTROLCOMMAND_CONNECTED_TRUE) {
361  return FLIGHTSTATUS_CONTROLSOURCE_FAILSAFE;
362  } else if (transmitter_control_get_flight_mode() ==
363  MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_TABLETCONTROL) {
364  return FLIGHTSTATUS_CONTROLSOURCE_TABLET;
365  } else {
366  return FLIGHTSTATUS_CONTROLSOURCE_TRANSMITTER;
367  }
368 
369 }
374 bool ok_to_arm(void)
375 {
376  // read alarms
377  SystemAlarmsData alarms;
378  SystemAlarmsGet(&alarms);
379 
380  // Check each alarm
381  for (int i = 0; i < SYSTEMALARMS_ALARM_NUMELEM; i++)
382  {
383  if (alarms.Alarm[i] >= SYSTEMALARMS_ALARM_ERROR &&
384  i != SYSTEMALARMS_ALARM_GPS &&
385  i != SYSTEMALARMS_ALARM_TELEMETRY)
386  {
387  return false;
388  }
389  }
390 
391  uint8_t flight_mode;
392  FlightStatusFlightModeGet(&flight_mode);
393 
394  if (flight_mode == FLIGHTSTATUS_FLIGHTMODE_FAILSAFE) {
395  /* Separately mask FAILSAFE arming here. */
396  return false;
397  }
398 
399  return true;
400 }
