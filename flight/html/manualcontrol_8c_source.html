115 static void manualControlTask(void *parameters)
116 {
117  /* Make sure disarmed on power up */
118  FlightStatusData flightStatus;
119  FlightStatusGet(&flightStatus);
120  flightStatus.Armed = FLIGHTSTATUS_ARMED_DISARMED;
121  FlightStatusSet(&flightStatus);
122 
123  // Select failsafe before run
124  failsafe_control_select(true);
125 
126  while (1) {
127 
128  // Process periodic data for each of the controllers, including reading
129  // all available inputs
130  failsafe_control_update();
131  transmitter_control_update();
132  tablet_control_update();
133  geofence_control_update();
134 
135  // Initialize to invalid value to ensure first update sets FlightStatus
136  static FlightStatusControlSourceOptions last_control_selection = -1;
137  enum control_events control_events = CONTROL_EVENTS_NONE;
138 
139  // Control logic to select the valid controller
140  FlightStatusControlSourceOptions control_selection = control_source_select();
141  bool reset_controller = control_selection != last_control_selection;
142 
143  // This logic would be better collapsed into control_source_select but
144  // in the case of the tablet we need to be able to abort and fall back
145  // to failsafe for now
146  switch(control_selection) {
147  case FLIGHTSTATUS_CONTROLSOURCE_TRANSMITTER:
148  transmitter_control_select(reset_controller);
149  control_events = transmitter_control_get_events();
150  break;
151  case FLIGHTSTATUS_CONTROLSOURCE_TABLET:
152  {
153  static bool tablet_previously_succeeded = false;
154  if (tablet_control_select(reset_controller) == 0) {
155  control_events = tablet_control_get_events();
156  tablet_previously_succeeded = true;
157  } else {
158  // Failure in tablet control. This would be better if done
159  // at the selection stage before the tablet is even used.
160  failsafe_control_select(reset_controller || tablet_previously_succeeded);
161  control_events = failsafe_control_get_events();
162  tablet_previously_succeeded = false;
163  }
164  break;
165  }
166  case FLIGHTSTATUS_CONTROLSOURCE_GEOFENCE:
167  geofence_control_select(reset_controller);
168  control_events = geofence_control_get_events();
169  break;
170  case FLIGHTSTATUS_CONTROLSOURCE_FAILSAFE:
171  default:
172  failsafe_control_select(reset_controller);
173  control_events = failsafe_control_get_events();
174  break;
175  }
176  if (control_selection != last_control_selection) {
177  FlightStatusControlSourceSet(&control_selection);
178  last_control_selection = control_selection;
179  }
180 
181  // TODO: This can evolve into a full FSM like I2C possibly
182  switch(control_events) {
183  case CONTROL_EVENTS_NONE:
184  break;
185  case CONTROL_EVENTS_ARM:
186  control_event_arm();
187  break;
188  case CONTROL_EVENTS_ARMING:
189  control_event_arming();
190  break;
191  case CONTROL_EVENTS_DISARM:
192  control_event_disarm();
193  break;
194  }
195 
196  // Wait until next update
197  PIOS_RCVR_WaitActivity(UPDATE_PERIOD_MS);
198  PIOS_WDG_UpdateFlag(PIOS_WDG_MANUAL);
199  }
200 }

250 static FlightStatusControlSourceOptions control_source_select()
251 {
252  // If the geofence controller is triggered, it takes precendence
253  // over even transmitter or failsafe. This means this must be
254  // EXTREMELY robust. The current behavior of geofence is simply
255  // to shut off the motors.
256  if (geofence_control_activate()) {
257  return FLIGHTSTATUS_CONTROLSOURCE_GEOFENCE;
258  }
259 
260  ManualControlCommandData cmd;
261  ManualControlCommandGet(&cmd);
262  if (cmd.Connected != MANUALCONTROLCOMMAND_CONNECTED_TRUE) {
263  return FLIGHTSTATUS_CONTROLSOURCE_FAILSAFE;
264  } else if (transmitter_control_get_flight_mode() ==
265  MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_TABLETCONTROL) {
266  return FLIGHTSTATUS_CONTROLSOURCE_TABLET;
267  } else {
268  return FLIGHTSTATUS_CONTROLSOURCE_TRANSMITTER;
269  }
270 
271 }
276 bool ok_to_arm(void)
277 {
278  // read alarms
279  SystemAlarmsData alarms;
280  SystemAlarmsGet(&alarms);
281 
282  // Check each alarm
283  for (int i = 0; i < SYSTEMALARMS_ALARM_NUMELEM; i++)
284  {
285  if (alarms.Alarm[i] >= SYSTEMALARMS_ALARM_ERROR &&
286  i != SYSTEMALARMS_ALARM_GPS &&
287  i != SYSTEMALARMS_ALARM_TELEMETRY)
288  {
289  return false;
290  }
291  }
292 
293  uint8_t flight_mode;
294  FlightStatusFlightModeGet(&flight_mode);
295 
296  if (flight_mode == FLIGHTSTATUS_FLIGHTMODE_FAILSAFE) {
297  /* Separately mask FAILSAFE arming here. */
298  return false;
299  }
300 
301  return true;
302 }
