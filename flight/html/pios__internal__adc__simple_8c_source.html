86 static void PIOS_INTERNAL_ADC_PinConfig(pios_internal_adc_t adc_dev)
87 {
88  if (!PIOS_INTERNAL_ADC_validate(adc_dev)) {
89  return;
90  }
91  /* Setup analog pins */
92  GPIO_InitTypeDef GPIO_InitStructure;
93  GPIO_StructInit(&GPIO_InitStructure);
94  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz;
95  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AN;
96 
97  for (int32_t i = 0; i < adc_dev->cfg->adc_pin_count; i++) {
98  adc_dev->pin_values[i] = 0xffff;
99 
100  if (adc_dev->cfg->adc_pins[i].port == NULL )
101  continue;
102  GPIO_InitStructure.GPIO_Pin = adc_dev->cfg->adc_pins[i].pin;
103  GPIO_Init(adc_dev->cfg->adc_pins[i].port, (GPIO_InitTypeDef*) &GPIO_InitStructure);
104  }

111 static void PIOS_INTERNAL_ADC_Converter_Config(pios_internal_adc_t adc_dev)
112 {
113  ADC_DeInit(adc_dev->cfg->adc_dev);
114 
115  /* Perform ADC calibration */
116  ADC1->CR |= ADC_CR_ADCAL;
117  while ((ADC1->CR & ADC_CR_ADCAL) != 0);
118 
119  ADC_InitTypeDef ADC_InitStructure;
120  ADC_StructInit(&ADC_InitStructure);
121 
122  ADC_InitStructure.ADC_Resolution = ADC_Resolution_12b;
123  ADC_InitStructure.ADC_ContinuousConvMode = DISABLE;
124  ADC_InitStructure.ADC_ExternalTrigConvEdge = ADC_ExternalTrigConvEdge_None;
125  ADC_InitStructure.ADC_DataAlign = ADC_DataAlign_Left;
126 
127  ADC_Init(adc_dev->cfg->adc_dev, &ADC_InitStructure);
128 
129  ADC_ClockModeConfig(adc_dev->cfg->adc_dev, ADC_ClockMode_AsynClk);
130  ADC_ContinuousModeCmd(adc_dev->cfg->adc_dev, DISABLE);
131  ADC_DiscModeCmd(adc_dev->cfg->adc_dev, ENABLE);
132 
133  /* Do common ADC init */
134  ADC_TempSensorCmd(ENABLE);
135  ADC_VrefintCmd(ENABLE);
136 
137  ADC_Cmd(adc_dev->cfg->adc_dev, ENABLE);
138 
139  while (!ADC_GetFlagStatus(adc_dev->cfg->adc_dev, ADC_FLAG_ADRDY));
140 }

167 void PIOS_INTERNAL_ADC_DoStep(pios_internal_adc_t adc_dev)
168 {
169  if (!adc_dev->conversion_started) {
170  /* Read the DR to ensure that the ADC_FLAG_EOC flag is not set */
171  ADC_GetConversionValue(adc_dev->cfg->adc_dev);
172 
173 
174  adc_dev->cfg->adc_dev->CHSELR = adc_dev->cfg->adc_pins[adc_dev->current_pin].adc_channel;
175  adc_dev->cfg->adc_dev->SMPR = ADC_SampleTime_41_5Cycles;
176  /* 41.5 cycles is a very long integration time, but we're in
177  * no rush */
178 
179  ADC_StartOfConversion(adc_dev->cfg->adc_dev);
180 
181  adc_dev->conversion_started = 1;
182  } else {
183  if (ADC_GetFlagStatus(adc_dev->cfg->adc_dev,
184  ADC_FLAG_ADSTART) == RESET) {
185  adc_dev->pin_values[adc_dev->current_pin] =
186  ADC_GetConversionValue(adc_dev->cfg->adc_dev);
187 
188  adc_dev->current_pin++;
189 
190  if (adc_dev->current_pin > adc_dev->cfg->adc_pin_count) {
191  adc_dev->current_pin = 0;
192  }
193 
194  adc_dev->conversion_started = 0;
195  }
196  }
197 }
