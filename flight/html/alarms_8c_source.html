51 int32_t AlarmsInitialize(void)
52 {

54  lock = PIOS_Mutex_Create();
55  PIOS_Assert(lock != NULL);

57  uint8_t reboot_reason = SYSTEMALARMS_REBOOTCAUSE_UNDEFINED;
58 
59  switch (PIOS_RESET_GetResetReason()) {
60  case PIOS_RESET_FLAG_BROWNOUT: // Brownouts are not detected on STM32F1 or STM32F3
61  reboot_reason = SYSTEMALARMS_REBOOTCAUSE_BROWNOUT;
62  break;
63  case PIOS_RESET_FLAG_PIN:
64  reboot_reason = SYSTEMALARMS_REBOOTCAUSE_PINRESET;
65  break;
66  case PIOS_RESET_FLAG_POWERON:

68  break;
69  case PIOS_RESET_FLAG_SOFTWARE:
70  reboot_reason = SYSTEMALARMS_REBOOTCAUSE_SOFTWARERESET;
71  break;
72  case PIOS_RESET_FLAG_INDEPENDENT_WATCHDOG:
73  reboot_reason = SYSTEMALARMS_REBOOTCAUSE_INDEPENDENTWATCHDOG;
74  break;
75  case PIOS_RESET_FLAG_WINDOW_WATCHDOG:
76  reboot_reason = SYSTEMALARMS_REBOOTCAUSE_WINDOWWATCHDOG;
77  break;
78  case PIOS_RESET_FLAG_LOW_POWER:

97 int32_t AlarmsSet(SystemAlarmsAlarmElem alarm, SystemAlarmsAlarmOptions severity)
98 {
99  SystemAlarmsData alarms;
100 
101  // Check that this is a valid alarm
102  if (alarm >= SYSTEMALARMS_ALARM_NUMELEM)
103  {
104  return -1;
105  }
106 
107  // Lock
108  PIOS_Mutex_Lock(lock, PIOS_MUTEX_TIMEOUT_MAX);
109 
110  // Read alarm and update its severity only if it was changed
111  SystemAlarmsGet(&alarms);
112  if ( alarms.Alarm[alarm] != severity )
113  {
114  alarms.Alarm[alarm] = severity;
115  SystemAlarmsSet(&alarms);
116  }
117 
118  // Release lock
119  PIOS_Mutex_Unlock(lock);
120  return 0;
121 
122 }

219 static int32_t hasSeverity(SystemAlarmsAlarmOptions severity)
220 {
221  SystemAlarmsData alarms;
222  uint32_t n;
223 
224  // Lock
225  PIOS_Mutex_Lock(lock, PIOS_MUTEX_TIMEOUT_MAX);
226 
227  // Read alarms
228  SystemAlarmsGet(&alarms);
229 
230  // Go through alarms and check if any are of the given severity or higher
231  for (n = 0; n < SYSTEMALARMS_ALARM_NUMELEM; ++n)
232  {
233  if ( alarms.Alarm[n] >= severity)
234  {
235  PIOS_Mutex_Unlock(lock);
236  return 1;
237  }
238  }
239 
240  // If this point is reached then no alarms found
241  PIOS_Mutex_Unlock(lock);
242  return 0;
243 }

329 int32_t AlarmString(SystemAlarmsData *alarm, char *buf, size_t buflen, bool blink, uint8_t *state) {
330  *state = SYSTEMALARMS_ALARM_OK;
331  buf[0] = '\0';
332  int pos = 0;
333 
334  // TODO(dustin): sort the alarms by severity. Alarm messages
335  // will get truncated. We want the most urgent stuff to show
336  // up first, having warnings show up only if there's space.
337  for (int i = 0; i < SYSTEMALARMS_ALARM_NUMELEM; i++) {
338  if (((alarm->Alarm[i] == SYSTEMALARMS_ALARM_WARNING) ||
339  (alarm->Alarm[i] == SYSTEMALARMS_ALARM_ERROR) ||
340  (alarm->Alarm[i] == SYSTEMALARMS_ALARM_CRITICAL))) {
341 
342  if (!alarm_names[i][0]) {
343  // zero-length alarm names indicate the alarm is
344  // explicitly ignored
345  continue;
346  }
347 
348  // Returned state is the worst state.
349  if (alarm->Alarm[i] > *state) {
350  *state = alarm->Alarm[i];
351  }
352 
353  char current_msg[LONGEST_MESSAGE+1] = {0};
354  switch (i) {
355  case SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION:
356  strncpy(current_msg,
357  (const char*)config_error_names[alarm->ConfigError],
358  sizeof(*config_error_names));
359  current_msg[sizeof(*config_error_names)] = '\0';
360  break;
361  case SYSTEMALARMS_ALARM_MANUALCONTROL:
362  strncpy(current_msg,
363  (const char*)manual_control_names[alarm->ManualControl],
364  sizeof(*manual_control_names));
365  current_msg[sizeof(*manual_control_names)] = '\0';
366  break;
367  default:
368  strncpy(current_msg, (const char*)alarm_names[i], sizeof(*alarm_names));
369  current_msg[sizeof(*alarm_names)] = '\0';
370  }
371 
372  int this_len = strlen(current_msg);
373 
374  if (pos + this_len + 2 >= buflen) {
375  break;
376  }
377 
378  if ((alarm->Alarm[i] != SYSTEMALARMS_ALARM_WARNING) && !blink) {
379  this_len += 1;
380  while (this_len > 0) {
381  buf[pos++] = ' ';
382  this_len--;
383  }
384  continue;
385  }
386 
387  memcpy(&buf[pos], current_msg, this_len);
388  pos += this_len;
389  buf[pos++] = ' ';
390 
391  }
392  }
393 
394  buf[pos] = '\0';
395  return pos;
396 }
