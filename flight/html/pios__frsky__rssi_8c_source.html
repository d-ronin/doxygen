45 int32_t PIOS_FrSkyRssi_Init(const struct pios_frsky_rssi_cfg * cfg_in)
46 {
47  PIOS_Assert(cfg_in);
48  cfg = cfg_in;
49 
50  // this only works with the advanced timers
51  if ((cfg->channels[0].timer != TIM1) && (cfg->channels[0].timer != TIM8))
52  PIOS_Assert(0);
53 
54  // both channels need to be on the same timer
55  if (cfg->channels[0].timer != cfg->channels[1].timer)
56  PIOS_Assert(0);
57 
58  // we can only use channels 1 and 2
59  if ((cfg->channels[0].timer_chan != TIM_Channel_1) && (cfg->channels[0].timer_chan != TIM_Channel_2))
60  PIOS_Assert(0);
61 
62  if ((cfg->channels[1].timer_chan != TIM_Channel_1) && (cfg->channels[1].timer_chan != TIM_Channel_2))
63  PIOS_Assert(0);
64 
65  // Configure timer clock
66  PIOS_TIM_InitClock(&cfg->clock_cfg);
67 
68  // Setup channels
69  uintptr_t tim_id;
70  if (PIOS_TIM_InitChannels(&tim_id, cfg->channels, 2, NULL, (uintptr_t)cfg)) {
71  return -1;
72  }
73 
74  // Configure the input capture channels
75  TIM_ICInitTypeDef TIM_ICInitStructure = cfg->ic1;
76  TIM_ICInitStructure.TIM_Channel = cfg->channels[0].timer_chan;
77  TIM_ICInit(cfg->channels[0].timer, &TIM_ICInitStructure);
78 
79  TIM_ICInitStructure = cfg->ic2;
80  TIM_ICInitStructure.TIM_Channel = cfg->channels[1].timer_chan;
81  TIM_ICInit(cfg->channels[1].timer, &TIM_ICInitStructure);
82 
83  // slave mode and trigger configuration
84  TIM_SelectSlaveMode(cfg->channels[0].timer, TIM_SlaveMode_Reset);
85 
86  switch(cfg->channels[0].timer_chan) {
87  case TIM_Channel_1:
88  TIM_SelectInputTrigger(cfg->channels[0].timer, TIM_TS_TI1FP1);
89  break;
90  case TIM_Channel_2:
91  TIM_SelectInputTrigger(cfg->channels[0].timer, TIM_TS_TI2FP2);
92  break;
93  }
94 
95  // Enable CC channels
96  TIM_CCxCmd(cfg->channels[0].timer, cfg->channels[0].timer_chan, TIM_CCx_Enable);
97  TIM_CCxCmd(cfg->channels[1].timer, cfg->channels[1].timer_chan, TIM_CCx_Enable);
98 
99  return 0;
100 }
