169 static int PIOS_FLYINGPIO_SendCmd(struct flyingpi_msg *msg) {
170  int len = 0;
171 
172  int ret = 0;
173 
174  uint32_t span = PIOS_DELAY_DiffuS(fpio_dev->last_msg_time);
175 
176  if (span < MIN_QUIET_TIME) {
177  /* Busy wait. No sense in other stuff stacking up.
178  * This has happened because we missed timing last time
179  * around on the actuator side. Ensure we get this actuator
180  * message out ASAP and queues don't fill in the meantime.
181  */
182  PIOS_DELAY_WaituS(MIN_QUIET_TIME - span + 20);
183  }
184 
185  flyingpi_calc_crc(msg, true, &len);
186 
187  if (PIOS_SPI_ClaimBus(fpio_dev->spi_id) != 0) {
188  return -1; /* No bueno. */
189  }
190 
191  PIOS_SPI_SetClockSpeed(fpio_dev->spi_id, 15000000);
192 
193  PIOS_SPI_RC_PinSet(fpio_dev->spi_id, fpio_dev->spi_slave,
194  false);
195 
196  struct flyingpi_msg resp;
197  if (PIOS_SPI_TransferBlock(fpio_dev->spi_id, (uint8_t *)msg,
198  (uint8_t *)&resp, len)) {
199  ret = -1;
200  }
201 
202  PIOS_SPI_RC_PinSet(fpio_dev->spi_id, fpio_dev->spi_slave,
203  true);
204  PIOS_SPI_ReleaseBus(fpio_dev->spi_id);
205 
206  fpio_dev->last_msg_time = PIOS_DELAY_GetRaw();
207 
208  /* Handle response data-- after releasing SPI */
209  if (!ret) {
210  if (!flyingpi_calc_crc(&resp, false, NULL)) {
211  printf("fpio: VERYBAD: Bad CRC on response data %d\n\t",
212  fpio_dev->msg_num);
213 
214  uint8_t *msg_data = (void *) &resp;
215 
216  for (int i = 0; i < 16; i++) {
217  printf("%02x ", msg_data[i]);
218  }
219 
220  printf("\n");
221 
222  ret = -1;
223 
224  } else if (resp.id != FLYINGPIRESP_IO) {
225  printf("fpio: VERYBAD: Unexpected response message type\n");
226  ret = -1;
227  } else {
228  struct flyingpiresp_io_10 *data = &resp.body.io_10;
229  if (data->valid_messages_recvd != fpio_dev->msg_num) {
230  printf("fpio: BAD: sequence number mismatch %d vs %d\n", data->valid_messages_recvd, fpio_dev->msg_num);
231 
232  // Signal that caller should reconfig.
233  ret = -1;
234 
235  fpio_dev->msg_num = data->valid_messages_recvd;
236  } else {
237  for (int i = 0; i < FPPROTO_MAX_RCCHANS; i++) {
238  fpio_dev->rcvr_value[i] = data->chan_data[i];
239  }
240 
241  for (int i = 0; i < FPPROTO_MAX_ADCCHANS; i++) {
242  fpio_dev->adc_value[i] = data->adc_data[i];
243 #if 0
244  if (!(fpio_dev->msg_num & 1023)) {
245  printf("%d : %d\n", i, data->adc_data[i]);
246  }
247 #endif
248  }
249  }
250  }
251  fpio_dev->msg_num++;
252 
253  }
254 
255  if (!ret) {
256  fpio_dev->err_cnt = 0;
257  } else {
258  fpio_dev->err_cnt++;
259 
260  PIOS_Assert(fpio_dev->err_cnt < MAX_CONSEC_ERRS);
261  }
262 
263  return ret;
264 
265 }

267 static void PIOS_FLYINGPIO_ActuatorUpdate()
268 {
269  /* Make sure we've already been configured */
270  PIOS_Assert(actuator_cfg.id == FLYINGPICMD_CFG);
271 
272  static int upd_num = 0;
273 
274  upd_num++;
275 
276  PIOS_Assert(!PIOS_FLYINGPIO_Validate(fpio_dev));
277 
278  struct flyingpi_msg tx_buf;
279 
280  tx_buf.id = FLYINGPICMD_ACTUATOR;
281 
282  struct flyingpicmd_actuator_fc *cmd = &tx_buf.body.actuator_fc;
283  struct flyingpicmd_cfg_fa *cfg = &actuator_cfg.body.cfg_fa;
284 
285  for (int i = 0; i < FPPROTO_MAX_SERVOS; i++) {
286  uint16_t span = cfg->actuators[i].max - cfg->actuators[i].min;
287  if (span == 0) {
288  // Prevent div by 0, this is a simple case.
289  cmd->values[i] = cfg->actuators[i].max;
290  } else {
291  float fraction = channel_values[i] -
292  cfg->actuators[i].min;
293 
294  fraction /= span;
295  fraction *= 65535.0f;
296  fraction += 0.5f;
297 
298  if (fraction > 65535) {
299  fraction = 65535;
300  } else if (fraction < 0) {
301  fraction = 0;
302  }
303 
304  cmd->values[i] = fraction;
305 
306 #if 0
307  if (!(upd_num & 1023)) {
308  printf("fpio: actuator: %d %f\n", i, fraction);
309  }
310 #endif
311  }
312  }
313 
314  cmd->led_status = PIOS_ANNUNC_GetStatus(PIOS_LED_HEARTBEAT);
315 
316  if (PIOS_FLYINGPIO_SendCmd(&tx_buf)) {
317  // If there's a sequence mismatch, reconfig.
318  PIOS_FLYINGPIO_SendCmd(&actuator_cfg);
319  PIOS_DELAY_WaituS(3 * MIN_QUIET_TIME);
320  }
321 }

323 static int PIOS_FLYINGPIO_ActuatorSetMode(const uint16_t *out_rate,
324  const int banks, const uint16_t *channel_max,
325  const uint16_t *channel_min) {
326  actuator_cfg.id = FLYINGPICMD_CFG;
327 
328  struct flyingpicmd_cfg_fa *cmd = &actuator_cfg.body.cfg_fa;
329 
330  PIOS_Assert(banks >= FPPROTO_MAX_BANKS);
331 
332  for (int i = 0; i < FPPROTO_MAX_BANKS; i++) {
333  cmd->rate[i] = out_rate[i];
334  }
335 
336  for (int i = 0; i < ACTUATORCOMMAND_CHANNEL_NUMELEM; i++) {
337  uint16_t true_min, true_max;
338 
339  if (channel_min[i] > channel_max[i]) {
340  true_min = channel_max[i];
341  true_max = channel_min[i];
342  } else {
343  true_min = channel_min[i];
344  true_max = channel_max[i];
345  }
346 
347  cmd->actuators[i].min = true_min;
348  cmd->actuators[i].max = true_max;
349  }
350 
351  /* XXX needs to be specified / configured somewhere... */
352  cmd->receiver_protocol = HWSHARED_PORTTYPES_SBUS;
353 
354  while (PIOS_FLYINGPIO_SendCmd(&actuator_cfg)) { // retry until OK
355  PIOS_Thread_Sleep(5);
356  }
357 
358  PIOS_Thread_Sleep(4);
359 
360  return 0;
361 }
