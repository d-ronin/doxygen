170 static void AttitudeTask(void *parameters)
171 {
172  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
173 
174  // Force settings update to make sure rotation loaded
175  settingsUpdatedCb(NULL, NULL, NULL, 0);
176 
177  enum complimentary_filter_status complimentary_filter_status;
178  complimentary_filter_status = CF_POWERON;
179 
180  uint32_t arming_count = 0;
181 
182  // Main task loop
183  while (1) {
184 
185  FlightStatusData flightStatus;
186  FlightStatusGet(&flightStatus);
187 
188  if (complimentary_filter_status == CF_POWERON) {
189 
190  complimentary_filter_status = (PIOS_Thread_Systime() > 1000) ?
191  CF_INITIALIZING : CF_POWERON;
192 
193  } else if(complimentary_filter_status == CF_INITIALIZING &&
194  (PIOS_Thread_Systime() < 7000) &&
195  (PIOS_Thread_Systime() > 1000)) {
196 
197  // For first 7 seconds use accels to get gyro bias
198  accelKp = 20 + 20 * (PIOS_Thread_Systime() < 4000);
199  accelKi = 1;
200  accel_filter_enabled = false;
201 
202  } else if (zero_during_arming &&
203  (flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMING)) {
204 
205  // Use a rapidly decrease accelKp to force the attitude to snap back
206  // to level and then converge more smoothly
207  if (arming_count < 20)
208  accelKp = 50;
209  else if (accelKp > 20)
210  accelKp -= 1;
211  arming_count++;
212 
213  accelKi = 1;
214  accel_filter_enabled = false;
215 
216  // Indicate arming so that after arming it reloads
217  // the normal settings
218  if (complimentary_filter_status != CF_ARMING) {
219  accumulate_gyro_zero();
220  complimentary_filter_status = CF_ARMING;
221  accumulating_gyro = true;
222  }
223 
224  } else if (complimentary_filter_status == CF_ARMING ||
225  complimentary_filter_status == CF_INITIALIZING) {
226 
227  // Reload settings (all the rates)
228  AttitudeSettingsAccKiGet(&accelKi);
229  AttitudeSettingsAccKpGet(&accelKp);
230  if(accel_alpha > 0.0f)
231  accel_filter_enabled = true;
232 
233  // If arming that means we were accumulating gyro
234  // samples. Compute new bias.
235  if (complimentary_filter_status == CF_ARMING) {
236  accumulate_gyro_compute();
237  accumulating_gyro = false;
238  arming_count = 0;
239  }
240 
241  // Indicate normal mode to prevent rerunning this code
242  complimentary_filter_status = CF_NORMAL;
243  }
244 
245  PIOS_WDG_UpdateFlag(PIOS_WDG_ATTITUDE);
246 
247  AccelsData accels;
248  GyrosData gyros;
249  int32_t retval = 0;
250 
251  retval = updateSensorsDigital(&accels, &gyros);
252 
253  // During power on set to angle from accel
254  if (complimentary_filter_status == CF_POWERON) {
255  float RPY[3];
256  float theta = atan2f(accels.x, -accels.z);
257  RPY[1] = theta * RAD2DEG;
258  RPY[0] = atan2f(-accels.y, -accels.z / cosf(theta)) * RAD2DEG;
259  RPY[2] = 0;
260  RPY2Quaternion(RPY, q);
261  }
262 
263  // Only update attitude when sensor data is good
264  if (retval != 0) {
265  AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE, SYSTEMALARMS_ALARM_ERROR);
266  PIOS_Thread_Sleep(SENSOR_PERIOD);
267  continue;
268  }
269 
270  // Do not update attitude data in simulation mode
271  if (!AttitudeActualReadOnly())
272  updateAttitude(&accels, &gyros);
273 
274  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
275  }
276 }

341 static void update_gyros(struct pios_sensor_gyro_data *gyros, GyrosData * gyrosData)
342 {
343  static float gyro_temp_bias[3] = {0,0,0};
344 
345  // Scale the gyros
346  float gyros_out[3] = {gyros->x * sensorSettings.GyroScale[0],
347  gyros->y * sensorSettings.GyroScale[1],
348  gyros->z * sensorSettings.GyroScale[2]};
349 
350  // Update the bias due to the temperature
351  updateTemperatureComp(gyrosData->temperature, gyro_temp_bias);
352 
353  // Apply temperature bias correction before the rotation
354  if (bias_correct_gyro) {
355  gyros_out[0] -= gyro_temp_bias[0];
356  gyros_out[1] -= gyro_temp_bias[1];
357  gyros_out[2] -= gyro_temp_bias[2];
358  }
359 
360  // When computing the bias accumulate samples
361  accumulate_gyro(gyros_out);
362 
363 
364  if (rotate) {
365  float gyros[3];
366  rot_mult(Rsb, gyros_out, gyros, true);
367  gyrosData->x = gyros[0];
368  gyrosData->y = gyros[1];
369  gyrosData->z = gyros[2];
370  } else {
371  gyrosData->x = gyros_out[0];
372  gyrosData->y = gyros_out[1];
373  gyrosData->z = gyros_out[2];
374  }
375 
376  if(bias_correct_gyro) {
377  // Applying integral component here so it can be seen on the gyros and correct bias
378  gyrosData->x -= gyro_correct_int[0];
379  gyrosData->y -= gyro_correct_int[1];
380  gyrosData->z -= gyro_correct_int[2];
381  }
382 
383  gyrosData->temperature = gyros->temperature;
384 }

446 static void updateAttitude(AccelsData * accelsData, GyrosData * gyrosData)
447 {
448  float dT;
449  uint32_t thisSysTime = PIOS_Thread_Systime();
450  static uint32_t lastSysTime = 0;
451  static float accels_filtered[3] = {0,0,0};
452  static float grot_filtered[3] = {0,0,0};
453 
454  dT = (thisSysTime == lastSysTime) ? 0.001f : (PIOS_THREAD_TIMEOUT_MAX & (thisSysTime - lastSysTime)) / 1000.0f;
455  lastSysTime = thisSysTime;
456 
457  // Bad practice to assume structure order, but saves memory
458  float * gyros = &gyrosData->x;
459  float * accels = &accelsData->x;
460 
461  float grot[3];
462  float accel_err[3];
463 
464  // Apply smoothing to accel values, to reduce vibration noise before main calculations.
465  apply_accel_filter(accels,accels_filtered);
466 
467  // Rotate gravity to body frame, filter and cross with accels
468  grot[0] = -(2 * (q[1] * q[3] - q[0] * q[2]));
469  grot[1] = -(2 * (q[2] * q[3] + q[0] * q[1]));
470  grot[2] = -(q[0] * q[0] - q[1]*q[1] - q[2]*q[2] + q[3]*q[3]);
471 
472  // Apply same filtering to the rotated attitude to match delays
473  apply_accel_filter(grot,grot_filtered);
474 
475  // Compute the error between the predicted direction of gravity and smoothed acceleration
476  CrossProduct((const float *) accels_filtered, (const float *) grot_filtered, accel_err);
477 
478  // Account for accel magnitude
479  float accel_mag = sqrtf(accels_filtered[0]*accels_filtered[0] + accels_filtered[1]*accels_filtered[1] + accels_filtered[2]*accels_filtered[2]);
480 
481  // Account for filtered gravity vector magnitude
482  float grot_mag;
483 
484  if (accel_filter_enabled)
485  grot_mag = sqrtf(grot_filtered[0]*grot_filtered[0] + grot_filtered[1]*grot_filtered[1] + grot_filtered[2]*grot_filtered[2]);
486  else
487  grot_mag = 1.0f;
488 
489  if (grot_mag > 1.0e-3f && accel_mag > 1.0e-3f) {
490  accel_err[0] /= (accel_mag*grot_mag);
491  accel_err[1] /= (accel_mag*grot_mag);
492  accel_err[2] /= (accel_mag*grot_mag);
493 
494  // Accumulate integral of error. Scale here so that units are (deg/s) but Ki has units of s
495  gyro_correct_int[0] -= accel_err[0] * accelKi * dT;
496  gyro_correct_int[1] -= accel_err[1] * accelKi * dT;
497 
498  // Correct rates based on error, integral component dealt with in updateSensors
499  gyros[0] += accel_err[0] * accelKp;
500  gyros[1] += accel_err[1] * accelKp;
501  gyros[2] += accel_err[2] * accelKp;
502  }
503 
504  { // scoping variables to save memory
505  // Work out time derivative from INSAlgo writeup
506  // Also accounts for the fact that gyros are in deg/s
507  float qdot[4];
508  qdot[0] = (-q[1] * gyros[0] - q[2] * gyros[1] - q[3] * gyros[2]) * dT * DEG2RAD / 2;
509  qdot[1] = (q[0] * gyros[0] - q[3] * gyros[1] + q[2] * gyros[2]) * dT * DEG2RAD / 2;
510  qdot[2] = (q[3] * gyros[0] + q[0] * gyros[1] - q[1] * gyros[2]) * dT * DEG2RAD / 2;
511  qdot[3] = (-q[2] * gyros[0] + q[1] * gyros[1] + q[0] * gyros[2]) * dT * DEG2RAD / 2;
512 
513  // Take a time step
514  q[0] = q[0] + qdot[0];
515  q[1] = q[1] + qdot[1];
516  q[2] = q[2] + qdot[2];
517  q[3] = q[3] + qdot[3];
518 
519  if(q[0] < 0) {
520  q[0] = -q[0];
521  q[1] = -q[1];
522  q[2] = -q[2];
523  q[3] = -q[3];
524  }
525  }
526 
527  // Renomalize
528  float qmag = sqrtf(q[0]*q[0] + q[1]*q[1] + q[2]*q[2] + q[3]*q[3]);
529  q[0] = q[0] / qmag;
530  q[1] = q[1] / qmag;
531  q[2] = q[2] / qmag;
532  q[3] = q[3] / qmag;
533 
534  // If quaternion has become inappropriately short or is nan reinit.
535  // THIS SHOULD NEVER ACTUALLY HAPPEN
536  if((fabsf(qmag) < 1e-3f) || IS_NOT_FINITE(qmag)) {
537  q[0] = 1;
538  q[1] = 0;
539  q[2] = 0;
540  q[3] = 0;
541  }
542 
543  AttitudeActualData attitudeActual = {
544  .q1 = q[0],
545  .q2 = q[1],
546  .q3 = q[2],
547  .q4 = q[3],
548  };
549 
550  // Convert into eueler degrees (makes assumptions about RPY order)
551  Quaternion2RPY(&attitudeActual.q1,&attitudeActual.Roll);
552 
553  AttitudeActualSet(&attitudeActual);
554 }

560 static void updateTemperatureComp(float temperature, float *temp_bias)
561 {
562  static int temp_counter = 0;
563  static float temp_accum = 0;
564 
565  static const float TEMP_MIN = -10;
566  static const float TEMP_MAX = 60;
567 
568  if (temperature < TEMP_MIN)
569  temperature = TEMP_MIN;
570  if (temperature > TEMP_MAX)
571  temperature = TEMP_MAX;
572 
573  if (temp_counter < 500) {
574  temp_accum += temperature;
575  temp_counter ++;
576  } else {
577  float t = temp_accum / temp_counter;
578  temp_accum = 0;
579  temp_counter = 0;
580 
581  // Compute a third order polynomial for each channel after each 500 samples
582  // a + t * (b + t * (c + t * d)) = a + b*t + c*t^2 + d*t^3
583  temp_bias[0] = sensorSettings.XGyroTempCoeff[0] +
584  t * (sensorSettings.XGyroTempCoeff[1] +
585  t * (sensorSettings.XGyroTempCoeff[2] +
586  t * sensorSettings.XGyroTempCoeff[3]));
587  temp_bias[1] = sensorSettings.YGyroTempCoeff[0] +
588  t * (sensorSettings.YGyroTempCoeff[1] +
589  t * (sensorSettings.YGyroTempCoeff[2] +
590  t * sensorSettings.YGyroTempCoeff[3]));
591  temp_bias[2] = sensorSettings.ZGyroTempCoeff[0] +
592  t * (sensorSettings.ZGyroTempCoeff[1] +
593  t * (sensorSettings.ZGyroTempCoeff[2] +
594  t * sensorSettings.ZGyroTempCoeff[3]));
595  }
596 }
