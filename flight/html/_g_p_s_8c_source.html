115 int32_t GPSInitialize(void)
116 {
117  gpsPort = PIOS_COM_GPS;
118  uint8_t gpsProtocol;
119 
120 #ifdef MODULE_GPS_BUILTIN
121  module_enabled = true;
122 #else
123  module_enabled = PIOS_Modules_IsEnabled(PIOS_MODULE_GPS);
124 #endif
125 
126  // These things are only conditional on small F1 targets.
127  // Expected to be always present otherwise.
128  if (GPSPositionInitialize() == -1 \
129  || GPSVelocityInitialize() == -1) {
130  module_enabled = false;
131  return -1;
132  }
133  if (GPSTimeInitialize() == -1 \
134  || GPSSatellitesInitialize() == -1 \
135  || HomeLocationInitialize() == -1 \
136  || UBloxInfoInitialize() == -1) {
137  module_enabled = false;
138  return -1;
139  }
140 #if defined(PIOS_GPS_PROVIDES_AIRSPEED)
141  if (AirspeedActualInitialize() == -1) {
142  module_enabled = false;
143  return -1;
144  }
145 #endif
146  updateSettings();
147 
148  if (gpsPort && module_enabled) {
149  ModuleSettingsGPSDataProtocolGet(&gpsProtocol);
150  switch (gpsProtocol) {
151  case MODULESETTINGS_GPSDATAPROTOCOL_NMEA:
152  gps_rx_buffer = PIOS_malloc(NMEA_MAX_PACKET_LENGTH);
153  if (gps_rx_buffer == NULL) {
154  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_ERROR);
155  return -1;
156  }
157  break;
158  case MODULESETTINGS_GPSDATAPROTOCOL_UBX:
159  gps_rx_buffer = PIOS_malloc(sizeof(struct UBXPacket));
160  if (gps_rx_buffer == NULL) {
161  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_ERROR);
162  return -1;
163  }
164  break;
165  default:
166  gps_rx_buffer = NULL;
167  }
168  PIOS_Assert(gps_rx_buffer);
169 
170  return 0;
171  }
172 
173  return -1;
174 }

183  ModuleSettingsGPSAutoConfigureGet(&gpsAutoConfigure);

189  switch (gpsProtocol) {

206  ModuleSettingsGPSSBASConstellationGet(&sbas_const);
207  ModuleSettingsGPSDynamicsModeGet(&dyn_mode);

212

225 static void gpsTask(void *parameters)
226 {
227  GPSPositionData gpsposition;
228 
229  uint32_t timeOfLastUpdateMs = 0;
230  uint32_t timeOfConfigAttemptMs = 0;
231 
232  uint8_t gpsProtocol;
233 
234 #ifdef PIOS_GPS_PROVIDES_AIRSPEED
235  gps_airspeed_initialize();
236 #endif
237 
238  GPSPositionGet(&gpsposition);
239 
240  // Wait for power to stabilize before talking to external devices
241  PIOS_Thread_Sleep(1000);
242 
243  // Loop forever
244  while (1) {
245  uint32_t xDelay = GPS_COM_TIMEOUT_MS;
246 
247  uint32_t loopTimeMs = PIOS_Thread_Systime();
248 
249  // XXX TODO: also on modulesettings change..
250  if (!timeOfConfigAttemptMs) {
251  ModuleSettingsGPSDataProtocolGet(&gpsProtocol);
252 
253  gpsConfigure(gpsProtocol);
254  timeOfConfigAttemptMs = PIOS_Thread_Systime();
255 
256  continue;
257  }
258 
259  uint8_t c;
260 
261  // This blocks the task until there is something on the buffer
262  while (PIOS_COM_ReceiveBuffer(gpsPort, &c, 1, xDelay) > 0)
263  {
264  int res;
265  switch (gpsProtocol) {
266 #if defined(PIOS_INCLUDE_GPS_NMEA_PARSER)
267  case MODULESETTINGS_GPSDATAPROTOCOL_NMEA:
268  res = parse_nmea_stream (c,gps_rx_buffer, &gpsposition, &gpsRxStats);
269  break;
270 #endif
271 #if defined(PIOS_INCLUDE_GPS_UBX_PARSER)
272  case MODULESETTINGS_GPSDATAPROTOCOL_UBX:
273  res = parse_ubx_stream (c,gps_rx_buffer, &gpsposition, &gpsRxStats);
274  break;
275 #endif
276  default:
277  res = NO_PARSER; // this should not happen
278  break;
279  }
280 
281  if (res == PARSER_COMPLETE) {
282  timeOfLastUpdateMs = loopTimeMs;
283  }
284 
285  xDelay = 0; // For now on, don't block / wait,
286  // but consume what we can from the fifo
287  }
288 
289  // Check for GPS timeout
290  if ((loopTimeMs - timeOfLastUpdateMs) >= GPS_TIMEOUT_MS) {
291  // we have not received any valid GPS sentences for a while.
292  // either the GPS is not plugged in or a hardware problem or the GPS has locked up.
293  uint8_t status = GPSPOSITION_STATUS_NOGPS;
294  GPSPositionStatusSet(&status);
295  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_ERROR);
296  /* Don't reinitialize too often. */
297  if ((loopTimeMs - timeOfConfigAttemptMs) >= GPS_TIMEOUT_MS) {
298  timeOfConfigAttemptMs = 0; // reinit next loop
299  }
300  } else {
301  // we appear to be receiving GPS sentences OK, we've had an update
302  //criteria for GPS-OK taken from this post
303  if (gpsposition.PDOP < 3.5f &&
304  gpsposition.Satellites >= 7 &&
305  (gpsposition.Status == GPSPOSITION_STATUS_FIX3D ||
306  gpsposition.Status == GPSPOSITION_STATUS_DIFF3D)) {
307  AlarmsClear(SYSTEMALARMS_ALARM_GPS);
308  } else if (gpsposition.Status == GPSPOSITION_STATUS_FIX3D ||
309  gpsposition.Status == GPSPOSITION_STATUS_DIFF3D)
310  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_WARNING);
311  else
312  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_CRITICAL);
313  }
314 
315  }
