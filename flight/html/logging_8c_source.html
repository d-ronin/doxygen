216 static void loggingTask(void *parameters)
217 {
218  bool armed = false;
219  uint32_t now = PIOS_Thread_Systime();
220 
221 #ifdef PIOS_INCLUDE_LOG_TO_FLASH
222  bool write_open = false;
223  bool read_open = false;
224  int32_t read_sector = 0;
225  uint8_t read_data[LOGGINGSTATS_FILESECTOR_NUMELEM];
226 #endif
227 
228  // Get settings automatically for now on
229  LoggingSettingsConnectCopy(&settings);
230 
231  LoggingStatsGet(&loggingData);
232  loggingData.BytesLogged = 0;
233 
234 #ifdef PIOS_INCLUDE_LOG_TO_FLASH
235  if (destination_onboard_flash) {
236  loggingData.MinFileId = PIOS_STREAMFS_MinFileId(logging_com_id);
237  loggingData.MaxFileId = PIOS_STREAMFS_MaxFileId(logging_com_id);
238  }
239 #endif
240 
241  if (!destination_onboard_flash) {
242  updateSettings();
243  }
244 
245  if (settings.LogBehavior == LOGGINGSETTINGS_LOGBEHAVIOR_LOGONSTART) {
246  loggingData.Operation = LOGGINGSTATS_OPERATION_INITIALIZING;
247  } else {
248  loggingData.Operation = LOGGINGSTATS_OPERATION_IDLE;
249  }
250 
251  LoggingStatsSet(&loggingData);
252 
253  // Loop forever
254  while (1)
255  {
256  LoggingStatsGet(&loggingData);
257 
258  // Check for change in armed state if logging on armed
259 
260  if (settings.LogBehavior == LOGGINGSETTINGS_LOGBEHAVIOR_LOGONARM) {
261  FlightStatusData flightStatus;
262  FlightStatusGet(&flightStatus);
263 
264  if (flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMED && !armed) {
265  // Start logging because just armed
266  loggingData.Operation = LOGGINGSTATS_OPERATION_INITIALIZING;
267  armed = true;
268  LoggingStatsSet(&loggingData);
269  } else if (flightStatus.Armed == FLIGHTSTATUS_ARMED_DISARMED && armed) {
270  loggingData.Operation = LOGGINGSTATS_OPERATION_IDLE;
271  armed = false;
272  LoggingStatsSet(&loggingData);
273  }
274  }
275 
276  switch (loggingData.Operation) {
277  case LOGGINGSTATS_OPERATION_FORMAT:
278  // Format the file system
279 #ifdef PIOS_INCLUDE_LOG_TO_FLASH
280  if (destination_onboard_flash){
281  if (read_open || write_open) {
282  PIOS_STREAMFS_Close(logging_com_id);
283  read_open = false;
284  write_open = false;
285  }
286 
287  PIOS_STREAMFS_Format(logging_com_id);
288  loggingData.MinFileId = PIOS_STREAMFS_MinFileId(logging_com_id);
289  loggingData.MaxFileId = PIOS_STREAMFS_MaxFileId(logging_com_id);
290  }
291 #endif /* PIOS_INCLUDE_LOG_TO_FLASH */
292  loggingData.Operation = LOGGINGSTATS_OPERATION_IDLE;
293  LoggingStatsSet(&loggingData);
294  break;
295  case LOGGINGSTATS_OPERATION_INITIALIZING:
296  // Unregister all objects
297  UAVObjIterate(&unregister_object);
298 #ifdef PIOS_INCLUDE_LOG_TO_FLASH
299  if (destination_onboard_flash){
300  // Close the file if it is open for reading
301  if (read_open) {
302  PIOS_STREAMFS_Close(logging_com_id);
303  read_open = false;
304  }
305 
306  // Open the file if it is not open for writing
307  if (!write_open) {
308  if (PIOS_STREAMFS_OpenWrite(logging_com_id) != 0) {
309  loggingData.Operation = LOGGINGSTATS_OPERATION_ERROR;
310  continue;
311  } else {
312  write_open = true;
313  }
314  loggingData.MinFileId = PIOS_STREAMFS_MinFileId(logging_com_id);
315  loggingData.MaxFileId = PIOS_STREAMFS_MaxFileId(logging_com_id);
316  LoggingStatsSet(&loggingData);
317  }
318  }
319  else {
320  read_open = false;
321  write_open = true;
322  }
323 #endif /* PIOS_INCLUDE_LOG_TO_FLASH */
324 
325  // Write information at start of the log file
326  writeHeader();
327 
328  // Log settings
329  if (settings.InitiallyLog == LOGGINGSETTINGS_INITIALLYLOG_ALLOBJECTS) {
330  UAVObjIterate(&logAll);
331  } else if (settings.InitiallyLog == LOGGINGSETTINGS_INITIALLYLOG_SETTINGSOBJECTS) {
332  UAVObjIterate(&logSettings);
333  }
334 
335  // Register objects to be logged
336  switch (settings.Profile) {
337  case LOGGINGSETTINGS_PROFILE_BASIC:
338  register_default_profile();
339  break;
340  case LOGGINGSETTINGS_PROFILE_CUSTOM:
341  case LOGGINGSETTINGS_PROFILE_FULLBORE:
342  UAVObjIterate(&register_object);
343  break;
344  }
345 
346  // Empty the queue
347  LoggingStatsBytesLoggedSet(&written_bytes);
348  loggingData.Operation = LOGGINGSTATS_OPERATION_LOGGING;
349  LoggingStatsSet(&loggingData);
350  break;
351  case LOGGINGSTATS_OPERATION_LOGGING:
352  {
353  // Sleep between updating stats.
354  PIOS_Thread_Sleep_Until(&now, LOGGING_PERIOD_MS);
355 
356  LoggingStatsBytesLoggedSet(&written_bytes);
357 
358  now = PIOS_Thread_Systime();
359  }
360  break;
361  case LOGGINGSTATS_OPERATION_DOWNLOAD:
362 #ifdef PIOS_INCLUDE_LOG_TO_FLASH
363  if (destination_onboard_flash) {
364  if (!read_open) {
365  // Start reading
366  if (PIOS_STREAMFS_OpenRead(logging_com_id, loggingData.FileRequest) != 0) {
367  loggingData.Operation = LOGGINGSTATS_OPERATION_ERROR;
368  } else {
369  read_open = true;
370  read_sector = -1;
371  }
372  }
373  if (read_open && read_sector == loggingData.FileSectorNum) {
374  // Request received for same sector. Reupdate.
375  memcpy(loggingData.FileSector, read_data, LOGGINGSTATS_FILESECTOR_NUMELEM);
376  loggingData.Operation = LOGGINGSTATS_OPERATION_IDLE;
377 
378  } else if (read_open && (read_sector + 1) == loggingData.FileSectorNum) {
379  int32_t bytes_read = PIOS_STREAMFS_Read(logging_com_id, loggingData.FileSector, LOGGINGSTATS_FILESECTOR_NUMELEM);
380  if (bytes_read < 0) {
381  // close on error
382  loggingData.Operation = LOGGINGSTATS_OPERATION_ERROR;
383  loggingData.FileSectorNum = 0xffff;
384  PIOS_STREAMFS_Close(logging_com_id);
385  read_open = false;
386  } else {
387  // Indicate sent
388  loggingData.Operation = LOGGINGSTATS_OPERATION_IDLE;
389 
390  if (bytes_read < LOGGINGSTATS_FILESECTOR_NUMELEM) {
391  // indicate end of file
392  loggingData.Operation = LOGGINGSTATS_OPERATION_COMPLETE;
393  PIOS_STREAMFS_Close(logging_com_id);
394  read_open = false;
395  }
396 
397  }
398  }
399  LoggingStatsSet(&loggingData);
400 
401  // Store the data in case it's needed again /
402  // lost over telemetry link
403  memcpy(read_data, loggingData.FileSector, LOGGINGSTATS_FILESECTOR_NUMELEM);
404  read_sector = loggingData.FileSectorNum;
405  }
406 #endif /* PIOS_INCLUDE_LOG_TO_FLASH */
407 
408  // fall-through to default case
409  default:
410  // Makes sure that we are not hogging the processor
411  PIOS_Thread_Sleep(10);
412 #ifdef PIOS_INCLUDE_LOG_TO_FLASH
413  if (destination_onboard_flash) {
414  // Close the file if necessary
415  if (write_open) {
416  PIOS_STREAMFS_Close(logging_com_id);
417  loggingData.MinFileId = PIOS_STREAMFS_MinFileId(logging_com_id);
418  loggingData.MaxFileId = PIOS_STREAMFS_MaxFileId(logging_com_id);
419  LoggingStatsSet(&loggingData);
420  write_open = false;
421  }
422  }
423 #endif /* PIOS_INCLUDE_LOG_TO_FLASH */
424  }
425  }
426 }

550 static void register_object(UAVObjHandle obj)
551 {
552  uint16_t period;
553 
554  if (settings.Profile == LOGGINGSETTINGS_PROFILE_FULLBORE) {
555  if (UAVObjIsSettings(obj)) {
556  return;
557  }
558 
559  period = 1;
560  } else {
561  UAVObjMetadata meta_data;
562  if (UAVObjGetMetadata(obj, &meta_data) < 0){
563  return;
564  }
565 
566  if (meta_data.loggingUpdatePeriod == 0){
567  return;
568  }
569 
570  period = meta_data.loggingUpdatePeriod;
571  }
572 
573  period = MAX(period, get_minimum_logging_period());
574 
575  if (period == 1) {
576  // log every update
577  UAVObjConnectCallback(obj, obj_updated_callback, NULL, EV_UPDATED | EV_UNPACKED);
578  } else {
579  // log updates throttled
580  UAVObjConnectCallbackThrottled(obj, obj_updated_callback, NULL, EV_UPDATED | EV_UNPACKED, period);
581  }
582 }

654 static void writeHeader()
655 {
656  uint8_t fw_blob[100];
657  PIOS_BL_HELPER_FLASH_Read_Description((uint8_t *) &fw_blob,
658  sizeof(fw_blob));
659 
660  int pos;
661 #define STR_BUF_LEN 45
662  char tmp_str[STR_BUF_LEN];
663  char *info_str;
664  char this_char;
665  DateTimeT date_time;
666 
667  // Header
668  #define LOG_HEADER "dRonin git hash:\n"
669  send_data((uint8_t *)LOG_HEADER, strlen(LOG_HEADER));
670 
671  // Commit tag name
672  // XXX all of thse should use the fw_version_info structure instead of
673  // doing offset math, but we need to factor out the fw_version_info
674  // struct from the templates first.
675  info_str = (char*)(fw_blob + 14);
676  send_data((uint8_t*)info_str, strnlen(info_str, 26));
677 
678  // Git commit hash
679  pos = 0;
680  tmp_str[pos++] = ':';
681  for (int i = 0; i < 4; i++){
682  this_char = *(char*)(fw_blob + 7 - i);
683  tmp_str[pos++] = DIGITS[(this_char & 0xF0) >> 4];
684  tmp_str[pos++] = DIGITS[(this_char & 0x0F)];
685  }
686  send_data((uint8_t*)tmp_str, pos);
687 
688  // Date
689  date_from_timestamp(*(uint32_t *)(fw_blob + 8), &date_time);
690  uint8_t len = snprintf(tmp_str, STR_BUF_LEN, " %d%02d%02d\n", 1900 + date_time.year, date_time.mon + 1, date_time.mday);
691  send_data((uint8_t*)tmp_str, len);
692 
693  // UAVO SHA1
694  pos = 0;
695  for (int i = 0; i < 20; i++){
696  this_char = *(char*)(fw_blob + 60 + i);
697  tmp_str[pos++] = DIGITS[(this_char & 0xF0) >> 4];
698  tmp_str[pos++] = DIGITS[(this_char & 0x0F)];
699  }
700  tmp_str[pos++] = '\n';
701  send_data((uint8_t*)tmp_str, pos);
702 }
