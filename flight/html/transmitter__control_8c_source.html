199 int32_t transmitter_control_update()
200 {
201  lastSysTime = PIOS_Thread_Systime();
202

204  bool validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_NUMELEM];
205 
206  if (settings_updated) {
207  settings_updated = false;
208  ManualControlSettingsGet(&settings);
209  }
210 
211  /* Update channel activity monitor */
212  uint8_t arm_status;
213  FlightStatusArmedGet(&arm_status);
214 
215  if (arm_status == FLIGHTSTATUS_ARMED_DISARMED) {
216  if (updateRcvrActivity(&activity_fsm)) {
217  /* Reset the aging timer because activity was detected */
218  lastActivityTime = lastSysTime;
219  }
220  }
221 
222  if (timeDifferenceMs(lastActivityTime, lastSysTime) > 5000) {
223  resetRcvrActivity(&activity_fsm);
224  lastActivityTime = lastSysTime;
225  }
226 
227  if (ManualControlCommandReadOnly()) {
228  FlightTelemetryStatsData flightTelemStats;
229  FlightTelemetryStatsGet(&flightTelemStats);
230  if(flightTelemStats.Status != FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
231  /* trying to fly via GCS and lost connection. fall back to transmitter */
232  UAVObjMetadata metadata;
233  ManualControlCommandGetMetadata(&metadata);
234  UAVObjSetAccess(&metadata, ACCESS_READWRITE);
235  ManualControlCommandSetMetadata(&metadata);
236  }
237 
238  // Don't process anything else when GCS is overriding the objects
239  return 0;
240  }
241 
242  if (settings.RssiType != MANUALCONTROLSETTINGS_RSSITYPE_NONE) {
243  int32_t value = 0;
244 
245  switch (settings.RssiType) {
246  case MANUALCONTROLSETTINGS_RSSITYPE_ADC:
247 #if defined(PIOS_INCLUDE_ADC)
248  if (settings.RssiChannelNumber > 0) {
249  value = PIOS_ADC_GetChannelRaw(settings.RssiChannelNumber - 1);
250  } else {
251  value = 0;
252  }
253 #endif
254  break;
255  case MANUALCONTROLSETTINGS_RSSITYPE_OPENLRS:
256 #if defined(PIOS_INCLUDE_OPENLRS_RCVR)
257  value = PIOS_OpenLRS_RSSI_Get();
258 #endif /* PIOS_INCLUDE_OPENLRS_RCVR */
259  break;
260  case MANUALCONTROLSETTINGS_RSSITYPE_FRSKYPWM:
261 #if defined(PIOS_INCLUDE_FRSKY_RSSI)
262  value = PIOS_FrSkyRssi_Get();
263 #endif /* PIOS_INCLUDE_FRSKY_RSSI */
264  break;
265  case MANUALCONTROLSETTINGS_RSSITYPE_RFM22B:
266 #if defined(PIOS_INCLUDE_RFM22B)
267  value = PIOS_RFM22B_RSSI_Get();
268 #endif /* PIOS_INCLUDE_RFM22B */
269  break;
270  default:
271  (void) 0 ;
272  int mapped = rssitype_to_channelgroup();
273 
274  if (mapped >= 0) {
275  value = PIOS_RCVR_Read(
276  pios_rcvr_group_map[mapped],
277  settings.RssiChannelNumber);
278  }
279 
280  break;
281  }
282 
283  if (value < settings.RssiMin) {
284  cmd.Rssi = 0;
285  } else if (settings.RssiMax == settings.RssiMin) {
286  cmd.Rssi = 0;
287  } else if (value > settings.RssiMax) {
288  if (value > (settings.RssiMax + settings.RssiMax/4)) {
289  cmd.Rssi = 0;
290  } else {
291  cmd.Rssi = 100;
292  }
293  } else {
294  cmd.Rssi = ((float)(value - settings.RssiMin)/((float)settings.RssiMax-settings.RssiMin)) * 100;
295  }
296 
297  cmd.RawRssi = value;
298  }
299 
300  bool valid_input_detected = true;
301 
302  // Read channel values in us
303  for (uint8_t n = 0;
304  n < MANUALCONTROLSETTINGS_CHANNELGROUPS_NUMELEM && n < MANUALCONTROLCOMMAND_CHANNEL_NUMELEM;
305  ++n) {
306 
307  if (settings.ChannelGroups[n] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
308  cmd.Channel[n] = PIOS_RCVR_INVALID;
309  validChannel[n] = false;
310  } else {
311  cmd.Channel[n] = PIOS_RCVR_Read(pios_rcvr_group_map[settings.ChannelGroups[n]],
312  settings.ChannelNumber[n]);
313  }
314 
315  // If a channel has timed out this is not valid data and we shouldn't update anything
316  // until we decide to go to failsafe
317  if(cmd.Channel[n] == (uint16_t) PIOS_RCVR_TIMEOUT) {
318  valid_input_detected = false;
319  validChannel[n] = false;
320  } else {
321  scaledChannel[n] = scaleChannel(n, cmd.Channel[n]);
322  validChannel[n] = validInputRange(n, cmd.Channel[n], CONNECTION_OFFSET);
323  }
324  }
325 
326  // Check settings, if error raise alarm
327  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
328  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
329  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
330  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
331  // Check all channel mappings are valid
332  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] == (uint16_t) PIOS_RCVR_INVALID ||
333  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] == (uint16_t) PIOS_RCVR_INVALID ||
334  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] == (uint16_t) PIOS_RCVR_INVALID ||
335  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] == (uint16_t) PIOS_RCVR_INVALID ||
336  // Check the driver exists
337  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] == (uint16_t) PIOS_RCVR_NODRIVER ||
338  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] == (uint16_t) PIOS_RCVR_NODRIVER ||
339  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] == (uint16_t) PIOS_RCVR_NODRIVER ||
340  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] == (uint16_t) PIOS_RCVR_NODRIVER ||
341  // Check the FlightModeNumber is valid
342  settings.FlightModeNumber < 1 || settings.FlightModeNumber > MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_NUMELEM ||
343  // If we've got more than one possible valid FlightMode, we require a configured FlightMode channel
344  ((settings.FlightModeNumber > 1) && (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE)) ||
345  // Whenever FlightMode channel is configured, it needs to be valid regardless of FlightModeNumber settings
346  ((settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] < MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) && (
347  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] == (uint16_t) PIOS_RCVR_INVALID ||
348  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] == (uint16_t) PIOS_RCVR_NODRIVER ))) {
349 
350  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_SETTINGS);
351 
352  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_FALSE;
353  ManualControlCommandSet(&cmd);
354 
355  // Need to do this here since we don't process armed status. Since this shouldn't happen in flight (changed config)
356  // immediately disarm
357  arm_state = ARM_STATE_DISARMED;
358  set_armed_if_changed(arm_state);
359 
360  return -1;
361  }
362 
363  // the block above validates the input configuration. this simply checks that the range is valid if flight mode is configured.
364  bool flightmode_valid_input = settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
365  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE];
366 
367  // because arming is optional we must determine if it is needed before checking it is valid
368  bool arming_valid_input = (settings.Arming < MANUALCONTROLSETTINGS_ARMING_SWITCH) ||
369  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ARMING];
370 
371  // decide if we have valid manual input or not
372  valid_input_detected &= validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] &&
373  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] &&
374  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] &&
375  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] &&
376  flightmode_valid_input &&
377  arming_valid_input;
378 
379  // Implement hysteresis loop on connection status
380  if (valid_input_detected && (++connected_count > 10)) {
381  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_TRUE;
382  connected_count = 0;
383  disconnected_count = 0;
384  } else if (!valid_input_detected && (++disconnected_count > 10)) {
385  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_FALSE;
386  connected_count = 0;
387  disconnected_count = 0;
388  }
389 
390  if (cmd.Connected == MANUALCONTROLCOMMAND_CONNECTED_FALSE) {
391  // These values are not used but just put ManualControlCommand in a sane state. When
392  // Connected is false, then the failsafe submodule will be in control.
393 
394  cmd.Throttle = -1;
395  cmd.Roll = 0;
396  cmd.Yaw = 0;
397  cmd.Pitch = 0;
398  cmd.Collective = 0;
399 
400  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_NORX);
401 
402  } else if (valid_input_detected) {
403  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_NONE);
404 
405  // Scale channels to -1 -> +1 range
406  cmd.Roll = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL];
407  cmd.Pitch = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH];
408  cmd.Yaw = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW];
409  cmd.Throttle = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE];
410  cmd.ArmSwitch = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ARMING] > 0 ?
411  MANUALCONTROLCOMMAND_ARMSWITCH_ARMED : MANUALCONTROLCOMMAND_ARMSWITCH_DISARMED;
412  flight_mode_value = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE];
413 
414  // Apply deadband for Roll/Pitch/Yaw stick inputs
415  if (settings.Deadband) {
416  applyDeadband(&cmd.Roll, settings.Deadband);
417  applyDeadband(&cmd.Pitch, settings.Deadband);
418  applyDeadband(&cmd.Yaw, settings.Deadband);
419  }
420 
421  if(cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_INVALID &&
422  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_NODRIVER &&
423  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_TIMEOUT) {
424  cmd.Collective = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE];
425  }
426 
427  // Set Accessory 0
428  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY0] !=
429  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
430  cmd.Accessory[0] = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY0];
431  }
432 
433  // Set Accessory 1
434  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY1] !=
435  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
436  cmd.Accessory[1] = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY1];
437  }
438  // Set Accessory 2
439  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY2] !=
440  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
441  cmd.Accessory[2] = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY1];
442  }
443  }
444 
445  // Process arming outside conditional so system will disarm when disconnected. Notice this
446  // is processed in the _update method instead of _select method so the state system is always
447  // evalulated, even if not detected.
448  process_transmitter_events(&cmd, &settings, valid_input_detected);
449 
450  // Update cmd object
451  ManualControlCommandSet(&cmd);
452 
453  return 0;
454 }

461 {

463  set_flight_mode();

474 
475  switch(flightMode) {
476  case FLIGHTSTATUS_FLIGHTMODE_MANUAL:
477  case FLIGHTSTATUS_FLIGHTMODE_ACRO:
478  case FLIGHTSTATUS_FLIGHTMODE_ACROPLUS:
479  case FLIGHTSTATUS_FLIGHTMODE_ACRODYNE:
480  case FLIGHTSTATUS_FLIGHTMODE_LEVELING:
481  case FLIGHTSTATUS_FLIGHTMODE_VIRTUALBAR:
482  case FLIGHTSTATUS_FLIGHTMODE_HORIZON:
483  case FLIGHTSTATUS_FLIGHTMODE_AXISLOCK:
484  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED1:
485  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED2:
486  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED3:
487  case FLIGHTSTATUS_FLIGHTMODE_FAILSAFE:
488  case FLIGHTSTATUS_FLIGHTMODE_AUTOTUNE:
489  update_stabilization_desired(&cmd, &settings, &airframe_type);
490  break;
491  case FLIGHTSTATUS_FLIGHTMODE_ALTITUDEHOLD:
492  altitude_hold_desired(&cmd, lastFlightMode != flightMode, &airframe_type);

501  break;
502  default:
503  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_UNDEFINED);
504  break;
505  }
506  lastFlightMode = flightMode;
507 
508  return 0;

553 static bool arming_position(ManualControlCommandData * cmd, ManualControlSettingsData * settings) {
554 
555  // If system is not appropriate to arm, do not even attempt
556  if (!ok_to_arm())
557  return false;
558 
559  bool lowThrottle = cmd->Throttle <= 0;
560 
561  switch(settings->Arming) {
562  case MANUALCONTROLSETTINGS_ARMING_ALWAYSDISARMED:
563  return false;
564  case MANUALCONTROLSETTINGS_ARMING_ALWAYSARMED:
565  return true;
566  case MANUALCONTROLSETTINGS_ARMING_ROLLLEFTTHROTTLE:
567  return lowThrottle && cmd->Roll < -ARMED_THRESHOLD;
568  case MANUALCONTROLSETTINGS_ARMING_ROLLRIGHTTHROTTLE:
569  return lowThrottle && cmd->Roll > ARMED_THRESHOLD;
570  case MANUALCONTROLSETTINGS_ARMING_YAWLEFTTHROTTLE:
571  return lowThrottle && cmd->Yaw < -ARMED_THRESHOLD;
572  case MANUALCONTROLSETTINGS_ARMING_YAWRIGHTTHROTTLE:
573  return lowThrottle && cmd->Yaw > ARMED_THRESHOLD;

632 static void process_transmitter_events(ManualControlCommandData * cmd, ManualControlSettingsData * settings, bool valid)
633 {
634 
635  /* State machine to determine arming of disarming:
636  DISARMED: if invalid input, remain.
637  look for the arm signal to go true.
638  if a switch go straight to ARMED
639  else store time and go to ARMING
640  ARMING: if arm signal ends or invalid go to IDLE
641  if time expires go to ARMED_STILL_HOLDING
642  ARMED_STILL_HOLDING: if arm signal ends go to ARMED
643  ARMED: if invalid look for failsafe
644  look for disarm signal.
645  if switch go to DISARMED
646  else store time and go to DISARMING
647  DISARMING: if arm signal ends return to ARMED
648  if time expires to go DISARMED_STILL_HOLDING
649  DISARMED_STILL_HOLDING: wait for release
650  */
651 
652  static uint32_t armedDisarmStart;
653 
654  valid &= cmd->Connected == MANUALCONTROLCOMMAND_CONNECTED_TRUE;
655 
656  switch(arm_state) {
657  case ARM_STATE_DISARMED:
658  {
659  set_armed_if_changed(FLIGHTSTATUS_ARMED_DISARMED);
660 
661  // last_arm used for detecting a rising edge to trigger switch arming
662  static bool last_arm = false;
663  bool arm = arming_position(cmd, settings) && valid;
664 
665  if (arm && (settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCH ||
666  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLE)) {
667  if (!last_arm) {
668  armedDisarmStart = lastSysTime;
669  arm_state = ARM_STATE_ARMED;
670  }
671  } else if (arm && (settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLEDELAY ||
672  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHDELAY)) {
673  if (!last_arm) {
674  armedDisarmStart = lastSysTime;
675  arm_state = ARM_STATE_ARMING;
676  }
677  } else if (arm) {
678  armedDisarmStart = lastSysTime;
679  arm_state = ARM_STATE_ARMING;
680  }
681 
682  last_arm = arm;
683  }
684  break;
685 
686  case ARM_STATE_ARMING:
687  {
688  set_armed_if_changed(FLIGHTSTATUS_ARMED_ARMING);
689 
690  bool arm = arming_position(cmd, settings) && valid;
691  uint16_t arm_time = (settings->ArmTime == MANUALCONTROLSETTINGS_ARMTIME_250) ? 250 : \
692  (settings->ArmTime == MANUALCONTROLSETTINGS_ARMTIME_500) ? 500 : \
693  (settings->ArmTime == MANUALCONTROLSETTINGS_ARMTIME_1000) ? 1000 : \
694  (settings->ArmTime == MANUALCONTROLSETTINGS_ARMTIME_2000) ? 2000 : 1000;
695  if (arm && timeDifferenceMs(armedDisarmStart, lastSysTime) > arm_time) {
696  if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLEDELAY ||
697  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHDELAY) {
698  arm_state = ARM_STATE_ARMED;
699  } else {
700  arm_state = ARM_STATE_ARMED_STILL_HOLDING;
701  }
702  } else if (!arm) {
703  arm_state = ARM_STATE_DISARMED;
704  }
705  }
706  break;
707 
708  case ARM_STATE_ARMED_STILL_HOLDING:
709  {
710  set_armed_if_changed(FLIGHTSTATUS_ARMED_ARMED);
711 
712  bool arm = arming_position(cmd, settings) && valid;
713  if (!arm) {
714  arm_state = ARM_STATE_ARMED;
715  }
716  }
717  break;
718 
719  case ARM_STATE_ARMED:
720  {
721  set_armed_if_changed(FLIGHTSTATUS_ARMED_ARMED);
722 
723  // Determine whether to disarm when throttle is low
724  uint8_t flight_mode;
725  FlightStatusFlightModeGet(&flight_mode);
726  bool autonomous_mode =
727  flight_mode == FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD ||
728  flight_mode == FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME ||
729  flight_mode == FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER ||
730  flight_mode == FLIGHTSTATUS_FLIGHTMODE_ALTITUDEHOLD ||
731  flight_mode == FLIGHTSTATUS_FLIGHTMODE_TABLETCONTROL;
732  bool check_throttle = settings->ArmTimeoutAutonomous == MANUALCONTROLSETTINGS_ARMTIMEOUTAUTONOMOUS_ENABLED ||
733  !autonomous_mode;
734  bool low_throttle = cmd->Throttle < 0;
735 
736  // Check for arming timeout if transmitter invalid or throttle is low and checked
737  if (!valid) {
738  /* This is a separate armed timeout for invalid input only.
739  * It defaults to 0 for now (in which case we still check the below
740  * low-throttle-or-invalid-input timeout */
741  if ((settings->InvalidRXArmedTimeout != 0) && (timeDifferenceMs(armedDisarmStart, lastSysTime) > settings->InvalidRXArmedTimeout)) {
742  arm_state = ARM_STATE_DISARMED;
743  break;
744  }
745  }
746 
747  if (!valid || (low_throttle && check_throttle)) {
748  if ((settings->ArmedTimeout != 0) && (timeDifferenceMs(armedDisarmStart, lastSysTime) > settings->ArmedTimeout)) {
749  arm_state = ARM_STATE_DISARMED;
750  break;
751  }
752  } else {
753  armedDisarmStart = lastSysTime;
754  }
755 
756  bool disarm = disarming_position(cmd, settings) && valid;
757  if (disarm) {
758  armedDisarmStart = lastSysTime;
759  arm_state = ARM_STATE_DISARMING;
760  }
761  }
762  break;
763 
764  case ARM_STATE_DISARMING:
765  {
766  bool disarm = disarming_position(cmd, settings) && valid;
767  uint16_t disarm_time = (settings->DisarmTime == MANUALCONTROLSETTINGS_DISARMTIME_250) ? 250 : \
768  (settings->DisarmTime == MANUALCONTROLSETTINGS_DISARMTIME_500) ? 500 : \
769  (settings->DisarmTime == MANUALCONTROLSETTINGS_DISARMTIME_1000) ? 1000 : \
770  (settings->DisarmTime == MANUALCONTROLSETTINGS_DISARMTIME_2000) ? 2000 : 1000;
771 
772  if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCH ||
773  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHDELAY ||
774  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLE ||
775  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLEDELAY) {
776  disarm_time = 100;
777  }
778 
779  if (disarm && timeDifferenceMs(armedDisarmStart, lastSysTime) > disarm_time) {
780  if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCH ||
781  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHDELAY ||
782  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLE ||
783  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLEDELAY) {
784  arm_state = ARM_STATE_DISARMED;
785  } else {
786  arm_state = ARM_STATE_DISARMED_STILL_HOLDING;
787  }
788  } else if (!disarm) {
789  arm_state = ARM_STATE_ARMED;
790  }
791  }
792  break;
793 
794  case ARM_STATE_DISARMED_STILL_HOLDING:
795  {
796  set_armed_if_changed(FLIGHTSTATUS_ARMED_DISARMED);
797 
798  bool disarm = disarming_position(cmd, settings) && valid;
799  if (!disarm) {
800  arm_state = ARM_STATE_DISARMED;
801  }
802  }
803  break;
804  }
805 
806 }

853 static bool updateRcvrActivityCompare(uintptr_t rcvr_id, struct rcvr_activity_fsm * fsm)
854 {
855  uint8_t active_channel = 0;
856 
857  int threshold = RCVR_ACTIVITY_MONITOR_MIN_RANGE;
858 
859  int rssi_channel = 0;
860 
861  /* Start off figuring out if there's a receive-rssi-injection channel */
862  int rssi_group = rssitype_to_channelgroup();
863 
864  /* If so, see if it's what we're processing */
865  if ((rssi_group >= 0) && (pios_rcvr_group_map[rssi_group] == rcvr_id)) {
866  // Yup. So let's skip the configured RSSI channel
867  rssi_channel = settings.RssiChannelNumber;
868  }
869 
870  /* Compare the current value to the previous sampled value */
871  for (uint8_t channel = 1;
872  channel <= RCVR_ACTIVITY_MONITOR_CHANNELS_PER_GROUP;
873  channel++) {
874  if (rssi_channel == channel) {
875  // Don't spot activity on the configured RSSI channel
876  continue;
877  }
878 
879  uint16_t delta;
880  uint16_t prev = fsm->prev[channel - 1]; // Subtract 1 because channels are 1 indexed
881  uint16_t curr = PIOS_RCVR_Read(rcvr_id, channel);
882  if (curr > prev) {
883  delta = curr - prev;
884  } else {
885  delta = prev - curr;
886  }
887 
888  if (delta > threshold) {
889  active_channel = channel;
890 
891  // This is the best channel so far. Only look for
892  // "more active" channels now.
893  threshold = delta;
894  }
895  }
896 
897  if (active_channel) {
898  /* Mark this channel as active */
899  ReceiverActivityActiveGroupSet((uint8_t *)&fsm->group);
900  ReceiverActivityActiveChannelSet(&active_channel);
901 
902  return true;
903  }
904 
905  return false;
906 }

908 static bool updateRcvrActivity(struct rcvr_activity_fsm * fsm)
909 {
910  bool activity_updated = false;
911 
912  if (fsm->group >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
913  /* We're out of range, reset things */
914  resetRcvrActivity(fsm);
915  }
916 
917  if (fsm->check_count && (pios_rcvr_group_map[fsm->group])) {
918  /* Compare with previous sample */
919  activity_updated = updateRcvrActivityCompare(pios_rcvr_group_map[fsm->group], fsm);
920  }
921 
922  if (!activity_updated && fsm->check_count < 3) {
923  // First time through this is 1, so basically be willing
924  // to check twice.
925  fsm->check_count++;
926 
927  return false;
928  }
929 
930  /* Reset the sample counter */
931  fsm->check_count = 0;
932 
933  /* Find the next active group, but limit search so we can't loop forever here */
934  for (uint8_t i = 0; i <= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE; i++) {
935  /* Move to the next group */
936  fsm->group++;
937  if (fsm->group >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
938  /* Wrap back to the first group */
939  fsm->group = 0;
940  }
941  if (pios_rcvr_group_map[fsm->group]) {
942  /*
943  * Found an active group, take a sample here to avoid an
944  * extra 20ms delay in the main thread so we can speed up
945  * this algorithm.
946  */
947  updateRcvrActivitySample(pios_rcvr_group_map[fsm->group],
948  fsm->prev,
949  NELEMENTS(fsm->prev));
950 
951  fsm->check_count = 1;
952 
953  break;
954  }
955  }
956 
957  return (activity_updated);
958 }
