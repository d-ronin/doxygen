165 static float get_thrust_source(ManualControlCommandData *manual_control_command,
166  bool always_fullrange)
167 {
168  float thrust = collective_is_thrust ?
169  manual_control_command->Collective :
170  manual_control_command->Throttle;
171 
172  if (thrust_is_bidir) {
173  /* Thrust is bidirectional; apply a deadband and call it good */
174  apply_channel_deadband(&thrust, settings.ThrustDeadband);
175 
176  return thrust;
177  }
178 
179  if (thrust < 0) {
180  /* Thrust is not intended to be bidirectional, squish the
181  * bottom half of the range.
182  */
183  thrust = 0;
184  }
185 
186  if (!always_fullrange) {
187  /* We have a 0..1 value, and they are OK with a 0..1 value */
188  return thrust;
189  }
190 
191  /* We want a -1..1 value, but thrust is squished into a 0..1 range. */
192  thrust = thrust * 2.0f - 1.0f;
193 
194  apply_channel_deadband(&thrust, settings.ThrustDeadband);
195 
196  return thrust;
197 }

266 int32_t transmitter_control_update()
267 {
268  lastSysTime = PIOS_Thread_Systime();
269 
270  float scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_NUMELEM];
271  bool validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_NUMELEM];
272 
273  static float filter_rssi;
274  bool settings_updated_this_cycle = false;
275 
276  if (settings_updated) {
277  perform_tc_settings_update();
278 
279  /* TODO: Factor more of the valid-configuration checks up
280  * here, so they don't need to be performed each time.
281  * Also do additonal validation .. e.g. a mostly-negative
282  * throttle range doesn't make sense!
283  */
284 
285  settings_updated_this_cycle = true;
286  }
287 
288  /* Update channel activity monitor */
289  uint8_t arm_status;
290  FlightStatusArmedGet(&arm_status);
291 
292  if (arm_status == FLIGHTSTATUS_ARMED_DISARMED) {
293  if (updateRcvrActivity(&activity_fsm)) {
294  /* Reset the aging timer because activity was detected */
295  lastActivityTime = lastSysTime;
296  }
297  }
298 
299  if (timeDifferenceMs(lastActivityTime, lastSysTime) > 5000) {
300  resetRcvrActivity(&activity_fsm);
301  lastActivityTime = lastSysTime;
302  }
303 
304  if (settings.RssiType != MANUALCONTROLSETTINGS_RSSITYPE_NONE) {
305  int32_t value = 0;
306 
307  switch (settings.RssiType) {
308  case MANUALCONTROLSETTINGS_RSSITYPE_ADC:
309 #if defined(PIOS_INCLUDE_ADC)
310  if (settings.RssiChannelNumber > 0) {
311  value = PIOS_ADC_GetChannelRaw(settings.RssiChannelNumber - 1) >> 4;
312  /*
313  * Shift right 4 bits because most targets
314  * return a 16 bit value now but existing configs
315  * expect a 12 bit value.
316  */
317  } else {
318  value = 0;
319  }
320 #endif
321  break;
322  case MANUALCONTROLSETTINGS_RSSITYPE_OPENLRS:
323 #if defined(PIOS_INCLUDE_OPENLRS_RCVR)
324  value = PIOS_OpenLRS_RSSI_Get();
325 #endif /* PIOS_INCLUDE_OPENLRS_RCVR */
326  break;
327  default:
328  (void) 0 ;
329  int mapped = rssitype_to_channelgroup();
330 
331  if (mapped >= 0) {
332  value = PIOS_RCVR_Read(
333  pios_rcvr_group_map[mapped],
334  settings.RssiChannelNumber);
335  }
336 
337  break;
338  }
339 
340  float rssi;
341 
342  if (value < settings.RssiMin) {
343  rssi = 0;
344  } else if (settings.RssiMax == settings.RssiMin) {
345  rssi = 0;
346  } else if (value > settings.RssiMax) {
347  if (value > (settings.RssiMax + settings.RssiMax/4)) {
348  rssi = 0;
349  } else {
350  rssi = 1;
351  }
352  } else {
353  rssi = ((float)(value - settings.RssiMin)/((float)settings.RssiMax-settings.RssiMin));
354  }
355 
356  /* Runs at least every 20ms. Takes 4 updates to go halfway
357  * to value, so makes RSSI on OSD a fair bit less frenetic.
358  * Still gets 90% there in 300ms with slow updates, and
359  * 99% there in 600ms.
360  */
361  filter_rssi = ((filter_rssi * 6) + rssi) / 7;
362 
363  cmd.Rssi = filter_rssi * 100;
364 
365  cmd.RawRssi = value;
366  }
367 
368  bool valid_input_detected = true;
369 
370  // Read channel values in us
371  for (uint8_t n = 0;
372  n < MANUALCONTROLSETTINGS_CHANNELGROUPS_NUMELEM && n < MANUALCONTROLCOMMAND_CHANNEL_NUMELEM;
373  ++n) {
374 
375  if (settings.ChannelGroups[n] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
376  cmd.Channel[n] = PIOS_RCVR_INVALID;
377  validChannel[n] = false;
378  } else {
379  cmd.Channel[n] = PIOS_RCVR_Read(pios_rcvr_group_map[settings.ChannelGroups[n]],
380  settings.ChannelNumber[n]);
381  }
382 
383  // If a channel has timed out this is not valid data and we shouldn't update anything
384  // until we decide to go to failsafe
385  if(cmd.Channel[n] == (uint16_t) PIOS_RCVR_TIMEOUT) {
386  valid_input_detected = false;
387  validChannel[n] = false;
388  } else {
389  scaledChannel[n] = scaleChannel(n, cmd.Channel[n]);
390  validChannel[n] = validInputRange(n, cmd.Channel[n], CONNECTION_OFFSET);
391  }
392  }
393 
394  // Check settings, if error raise alarm
395  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
396  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
397  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
398  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
399  // Check all channel mappings are valid
400  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] == (uint16_t) PIOS_RCVR_INVALID ||
401  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] == (uint16_t) PIOS_RCVR_INVALID ||
402  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] == (uint16_t) PIOS_RCVR_INVALID ||
403  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] == (uint16_t) PIOS_RCVR_INVALID ||
404  // Check the driver exists
405  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] == (uint16_t) PIOS_RCVR_NODRIVER ||
406  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] == (uint16_t) PIOS_RCVR_NODRIVER ||
407  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] == (uint16_t) PIOS_RCVR_NODRIVER ||
408  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] == (uint16_t) PIOS_RCVR_NODRIVER ||
409  // Check the FlightModeNumber is valid
410  settings.FlightModeNumber < 1 || settings.FlightModeNumber > MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_NUMELEM ||
411  // If we've got more than one possible valid FlightMode, we require a configured FlightMode channel
412  ((settings.FlightModeNumber > 1) && (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE)) ||
413  // Whenever FlightMode channel is configured, it needs to be valid regardless of FlightModeNumber settings
414  ((settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] < MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) && (
415  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] == (uint16_t) PIOS_RCVR_INVALID ||
416  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] == (uint16_t) PIOS_RCVR_NODRIVER ))) {
417 
418  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_SETTINGS);
419 
420  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_FALSE;
421  ManualControlCommandSet(&cmd);
422 
423  control_status = STATUS_ERROR;
424 
425  return -1;
426  }
427 
428  // the block above validates the input configuration. this simply checks that the range is valid if flight mode is configured.
429  bool flightmode_valid_input = settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
430  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE];
431 
432  // because arming is optional we must determine if it is needed before checking it is valid
433  bool arming_valid_input = (settings.Arming < MANUALCONTROLSETTINGS_ARMING_SWITCH) ||
434  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ARMING];
435 
436  // decide if we have valid manual input or not
437  valid_input_detected &= validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] &&
438  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] &&
439  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] &&
440  validChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] &&
441  flightmode_valid_input &&
442  arming_valid_input;
443 
444  // Implement hysteresis loop on connection status
445  if (valid_input_detected && (++connected_count > 10)) {
446  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_TRUE;
447  connected_count = 0;
448  disconnected_count = 0;
449  } else if (!valid_input_detected && (++disconnected_count > 10)) {
450  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_FALSE;
451  connected_count = 0;
452  disconnected_count = 0;
453  }
454 
455  if (cmd.Connected == MANUALCONTROLCOMMAND_CONNECTED_FALSE) {
456  // These values are not used but just put ManualControlCommand in a sane state. When
457  // Connected is false, then the failsafe submodule will be in control.
458 
459  cmd.Throttle = 0;
460  cmd.Roll = 0;
461  cmd.Yaw = 0;
462  cmd.Pitch = 0;
463  cmd.Collective = 0;
464 
465  /* If all relevant channel groups aren't in error mode, then the receiver seems to be working fine.
466  So if we still reach this point, the channel min/max/neutral values on the input page are faulty. */
467  bool rx_signal_detected = cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] < (uint16_t)PIOS_RCVR_NODRIVER &&
468  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] < (uint16_t)PIOS_RCVR_NODRIVER &&
469  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] < (uint16_t)PIOS_RCVR_NODRIVER &&
470  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] < (uint16_t)PIOS_RCVR_NODRIVER &&
471  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ARMING] < (uint16_t)PIOS_RCVR_NODRIVER &&
472  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] < (uint16_t)PIOS_RCVR_NODRIVER;
473 
474  if (rx_signal_detected)
475  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_CHANNELCONFIGURATION);
476  else
477  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_NORX);
478 
479  } else if (valid_input_detected) {
480  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_NONE);
481 
482  // Scale channels to -1 -> +1 range
483  cmd.Roll = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL];
484  cmd.Pitch = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH];
485  cmd.Yaw = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW];
486  cmd.Throttle = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE];
487  cmd.ArmSwitch = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ARMING] > 0 ?
488  MANUALCONTROLCOMMAND_ARMSWITCH_ARMED : MANUALCONTROLCOMMAND_ARMSWITCH_DISARMED;
489  flight_mode_value = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE];
490 
491  // Apply deadband for Roll/Pitch/Yaw stick inputs
492  if (settings.Deadband) {
493  apply_channel_deadband(&cmd.Roll, settings.Deadband);
494  apply_channel_deadband(&cmd.Pitch, settings.Deadband);
495  apply_channel_deadband(&cmd.Yaw, settings.Deadband);
496  }
497 
498  if(cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_INVALID &&
499  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_NODRIVER &&
500  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_TIMEOUT) {
501  cmd.Collective = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE];
502  }
503 
504  // Set Accessory 0
505  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY0] !=
506  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
507  cmd.Accessory[0] = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY0];
508  }
509 
510  // Set Accessory 1
511  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY1] !=
512  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
513  cmd.Accessory[1] = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY1];
514  }
515  // Set Accessory 2
516  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY2] !=
517  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
518  cmd.Accessory[2] = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY2];
519  }
520  }
521 
522  // Process arming outside conditional so system will disarm when disconnected. Notice this
523  // is processed in the _update method instead of _select method so the state system is always
524  // evalulated, even if not detected.
525  process_transmitter_events(&cmd, &settings, valid_input_detected,
526  settings_updated_this_cycle);
527 
528  // Update cmd object
529  ManualControlCommandSet(&cmd);
530 
531  return 0;
532 }

743 static void process_transmitter_events(ManualControlCommandData * cmd, ManualControlSettingsData * settings, bool valid, bool settings_updated)
744 {
745  if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_ALWAYSDISARMED) {
746  /*
747  * Disconnected is used to force a "disarmed" position being
748  * reached before arming. e.g. on config change.
749  */
750  control_status = STATUS_DISCONNECTED;
751  }
752 
753  valid &= cmd->Connected == MANUALCONTROLCOMMAND_CONNECTED_TRUE;
754 
755  if (!valid || (cmd->Connected != MANUALCONTROLCOMMAND_CONNECTED_TRUE)) {
756  /* disarm timeout check-- use least favorable timeout.
757  * Note that if things were low throttle, and are now
758  * disconnected, we keep the old timer running.
759  *
760  * (The same is true for vice-versa).
761  */
762  uint32_t timeout = 0;
763 
764  if (settings->InvalidRXArmedTimeout && settings->ArmedTimeout) {
765  timeout = MIN(settings->InvalidRXArmedTimeout,
766  settings->ArmedTimeout);
767  } else if (settings->InvalidRXArmedTimeout) {
768  timeout = settings->InvalidRXArmedTimeout;
769  } else {
770  timeout = settings->ArmedTimeout;
771  }
772 
773  if (check_receiver_timer(timeout)) {
774  control_status = STATUS_SAFETYTIMEOUT;
775  } else {
776  control_status = STATUS_DISCONNECTED;
777  }
778 
779  return;
780  }
781 
782  bool low_throt = is_low_throttle_for_arming(cmd);
783 
784  if (low_throt) {
785  /* Determine whether to disarm when throttle is low */
786  uint8_t flt_mode;
787  FlightStatusFlightModeGet(&flt_mode);
788 
789  if (flt_mode == FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD ||
790  flt_mode == FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME ||
791  flt_mode == FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER ||
792  flt_mode == FLIGHTSTATUS_FLIGHTMODE_ALTITUDEHOLD ||
793  flt_mode == FLIGHTSTATUS_FLIGHTMODE_TABLETCONTROL) {
794  low_throt = false;
795  }
796  }
797 
798  if (low_throt) {
799  if (check_receiver_timer(settings->ArmedTimeout)) {
800  control_status = STATUS_DISARM;
801  return;
802  }
803  } else {
804  /* Not low throt, not disconnected. */
805  reset_receiver_timer();
806  }
807 
808  if (settings_updated) {
809  /* Make sure that we don't arm as a result of a settings
810  * change.
811  */
812  control_status = STATUS_INVALID_FOR_DISARMED;
813  return;
814  }
815 
816  if (disarm_commanded(cmd, settings, low_throt)) {
817  /* Separate timer from the above */
818 
819  control_status = STATUS_DISARM;
820  return;
821  } else if (arming_position(cmd, settings, low_throt)) {
822  /* Not disarming pos, not low throt timeout, not disconnected. */
823 
824  if (control_status == STATUS_DISARM) {
825  /* Edge detector says it's time to reset the receiver
826  * timer
827  */
828  reset_receiver_timer();
829  }
830 
831  if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCH ||
832  settings->Arming == MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLE) {
833  control_status = STATUS_ARM_VALID;
834  } else if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_CORNERSTHROTTLE) {
835  control_status = STATUS_TOGGLE_ARM;
836  } else {
837  control_status = STATUS_ARM_INVALID;
838  }
839 
840  return;
841  } else {
842  /* If the arming switch is flipped on, but we don't otherwise
843  * qualify for arming, let the upper layer know.
844  */
845  if (settings->Arming ==
846  MANUALCONTROLSETTINGS_ARMING_SWITCHTHROTTLE) {
847  if (cmd->ArmSwitch == MANUALCONTROLCOMMAND_ARMSWITCH_ARMED) {
848  control_status = STATUS_INVALID_FOR_DISARMED;
849  return;
850  }
851  }
852  }
853 
854  control_status = STATUS_NORMAL;
855 
856  return;
857 
858 }

905 static bool updateRcvrActivityCompare(uintptr_t rcvr_id, struct rcvr_activity_fsm * fsm)
906 {
907  uint8_t active_channel = 0;
908 
909  int threshold = RCVR_ACTIVITY_MONITOR_MIN_RANGE;
910 
911  int rssi_channel = 0;
912 
913  /* Start off figuring out if there's a receive-rssi-injection channel */
914  int rssi_group = rssitype_to_channelgroup();
915 
916  /* If so, see if it's what we're processing */
917  if ((rssi_group >= 0) && (pios_rcvr_group_map[rssi_group] == rcvr_id)) {
918  // Yup. So let's skip the configured RSSI channel
919  rssi_channel = settings.RssiChannelNumber;
920  }
921 
922  /* Compare the current value to the previous sampled value */
923  for (uint8_t channel = 1;
924  channel <= RCVR_ACTIVITY_MONITOR_CHANNELS_PER_GROUP;
925  channel++) {
926  if (rssi_channel == channel) {
927  // Don't spot activity on the configured RSSI channel
928  continue;
929  }
930 
931  uint16_t delta;
932  uint16_t prev = fsm->prev[channel - 1]; // Subtract 1 because channels are 1 indexed
933  uint16_t curr = PIOS_RCVR_Read(rcvr_id, channel);
934  if (curr > prev) {
935  delta = curr - prev;
936  } else {
937  delta = prev - curr;
938  }
939 
940  if (delta > threshold) {
941  active_channel = channel;
942 
943  // This is the best channel so far. Only look for
944  // "more active" channels now.
945  threshold = delta;
946  }
947  }
948 
949  if (active_channel) {
950  /* Mark this channel as active */
951  ReceiverActivityActiveGroupSet((uint8_t *)&fsm->group);
952  ReceiverActivityActiveChannelSet(&active_channel);
953 
954  return true;
955  }
956 
957  return false;
958 }

960 static bool updateRcvrActivity(struct rcvr_activity_fsm * fsm)
961 {
962  bool activity_updated = false;
963 
964  if (fsm->group >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
965  /* We're out of range, reset things */
966  resetRcvrActivity(fsm);
967  }
968 
969  if (fsm->check_count && (pios_rcvr_group_map[fsm->group])) {
970  /* Compare with previous sample */
971  activity_updated = updateRcvrActivityCompare(pios_rcvr_group_map[fsm->group], fsm);
972  }
973 
974  if (!activity_updated && fsm->check_count < 3) {
975  // First time through this is 1, so basically be willing
976  // to check twice.
977  fsm->check_count++;
978 
979  return false;
980  }
981 
982  /* Reset the sample counter */
983  fsm->check_count = 0;
984 
985  /* Find the next active group, but limit search so we can't loop forever here */
986  for (uint8_t i = 0; i <= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE; i++) {
987  /* Move to the next group */
988  fsm->group++;
989  if (fsm->group >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
990  /* Wrap back to the first group */
991  fsm->group = 0;
992  }
993  if (pios_rcvr_group_map[fsm->group]) {
994  /*
995  * Found an active group, take a sample here to avoid an
996  * extra 20ms delay in the main thread so we can speed up
997  * this algorithm.
998  */
999  updateRcvrActivitySample(pios_rcvr_group_map[fsm->group],
1000  fsm->prev,
1001  NELEMENTS(fsm->prev));
1002 
1003  fsm->check_count = 1;
1004 
1005  break;
1006  }
1007  }
1008 
1009  return (activity_updated);
1010 }
