79 static void draw_rect(charosd_state_t state, uint8_t l, uint8_t t, uint8_t w, uint8_t h, bool filled, uint8_t attr)
80 {
81  const uint8_t _rect_thin [] = {0xd0, 0xd1, 0xd2, 0xd3, 0xd4, 0xd5, 0xd6, 0xd7};
82  const uint8_t _rect_fill [] = {0xd8, 0xd9, 0xda, 0xdb, 0xdc, 0xdd, 0xde, 0xdf};
83 
84  if (w < 2 || h < 2) return;
85 
86  uint8_t r = w - 1;
87  uint8_t b = h - 1;
88 
89  const uint8_t *rect = filled ? _rect_fill : _rect_thin;
90 
91  char buffer [w + 1];
92 
93  for (uint8_t i = 0; i < h; i ++) {
94  char spacer;
95  if (i == 0) {
96  buffer [0] = rect [0];
97  buffer [r] = rect [2];
98  spacer = rect [1];
99  } else if (i == b) {
100  buffer [0] = rect [4];
101  buffer [r] = rect [6];
102  spacer = rect [5];
103  } else {
104  buffer [0] = rect [3];
105  buffer [r] = rect [7];
106  spacer = ' ';
107  }
108 
109  if (w > 2) {
110  memset(buffer + 1, spacer, w - 2);
111  }
112 
113  buffer [w] = 0;
114 
115  PIOS_MAX7456_puts(state->dev, l, t + i, buffer, attr);
116  }
117 }

300 void HORIZON_update(charosd_state_t state, uint8_t x, uint8_t y)
301 {
302  char buffer [PANEL_HORIZON_HEIGHT][PANEL_HORIZON_WIDTH + 1];
303 
304  for (uint8_t i = 0; i < PANEL_HORIZON_HEIGHT; i ++) {
305  memcpy(buffer[i], i == PANEL_HORIZON_HEIGHT / 2 ? _horz_center : _horz_line, PANEL_HORIZON_WIDTH);
306  buffer[i][PANEL_HORIZON_WIDTH] = 0;
307  }
308 
309  // code below from minoposd
310  int16_t pitch_line = tanf(DEG2RAD * (-state->telemetry.attitude_actual.Pitch)) * _PAN_HORZ_LINES;
311  float roll = tanf(DEG2RAD * state->telemetry.attitude_actual.Roll);
312  for (uint8_t col = 1; col <= _PAN_HORZ_INT_WIDTH; col ++) {
313  // center X point at middle of each column
314  float middle = col * _PAN_HORZ_INT_WIDTH - (_PAN_HORZ_INT_WIDTH * _PAN_HORZ_INT_WIDTH / 2) - _PAN_HORZ_INT_WIDTH / 2;
315  // calculating hit point on Y plus offset to eliminate negative values
316  int16_t hit = roll * middle + pitch_line + _PAN_HORZ_LINES;
317  if (hit > 0 && hit < _PAN_HORZ_TOTAL_LINES) {
318  int8_t row = PANEL_HORIZON_HEIGHT - ((hit - 1) / _PAN_HORZ_CHAR_LINES);
319  int8_t subval = (hit - (_PAN_HORZ_TOTAL_LINES - row * _PAN_HORZ_CHAR_LINES + 1)) * _PAN_HORZ_VRES / _PAN_HORZ_CHAR_LINES;
320  if (subval == _PAN_HORZ_VRES - 1)
321  buffer [row - 2][col] = _PAN_HORZ_TOP;
322  buffer[row - 1][col] = _PAN_HORZ_LINE + subval;
323  }
324  }
325 
326  for (uint8_t i = 0; i < PANEL_HORIZON_HEIGHT; i ++) {
327  PIOS_MAX7456_puts(state->dev, x, y + i, buffer[i], 0);
328  }
329 }

419  char buffer[12];

421  0xc2, 0xc0, 0xc1, 0xc0,
