282 static inline uint8_t indicate_error(const char **sequence)
283 {
284 #ifdef PIPXTREME
285  *sequence="t"; /* Dashes */
286  return SYSTEMALARMS_ALARM_WARNING;
287 #else
288 
289  SystemAlarmsData alarms;
290  SystemAlarmsGet(&alarms);
291 
292  *sequence = NULL;
293 
294  uint8_t worst_sev = 0;
295 
296  bool generic = false;
297 
298  for (uint32_t i = 0; i < SYSTEMALARMS_ALARM_NUMELEM; i++) {
299  // If this is less severe than the current alarm, continue.
300  if (alarms.Alarm[i] < worst_sev) {
301  continue;
302  }
303 
304  // If there's a tie and the previous answer is not the
305  // 'generic' alarm, continue.
306  if ((!generic) && (alarms.Alarm[i] == worst_sev)) {
307  continue;
308  }
309 
310  uint8_t thresh = SYSTEMALARMS_ALARM_WARNING;
311 
312  if (i == SYSTEMALARMS_ALARM_TELEMETRY) {
313  // Suppress most alarms from telemetry.
314  // The user can identify if present from GCS.
315  thresh = SYSTEMALARMS_ALARM_CRITICAL;
316  }
317 
318  if (i == SYSTEMALARMS_ALARM_EVENTSYSTEM) {
319  // Skip event system alarms-- nuisance
320  continue;
321  }
322 
323  if (alarms.Alarm[i] < thresh) {
324  continue;
325  }
326 
327  // OK, we have a candidate alarm.
328 
329  worst_sev = alarms.Alarm[i];
330  generic = false;
331 
332  switch (i) {
333  case SYSTEMALARMS_ALARM_BATTERY:
334  // -... b for battery
335  *sequence = "b ";
336  break;
337  case SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION:
338  // -.-. c for configuration
339  *sequence = "c ";
340  break;
341  case SYSTEMALARMS_ALARM_GPS:
342  // --. g for GPS
343  *sequence = "g ";
344  break;
345  case SYSTEMALARMS_ALARM_MANUALCONTROL:
346  // .-. r for radio
347  *sequence = BLINK_STRING_RADIO;
348  break;
349  default:
350  // .- a for alarm
351  *sequence = "a ";
352 
353  generic = true;
354  break;
355  }
356  }
357 
358  return worst_sev;
359 #endif /* PIPXTREME */
360 }

415 static void systemPeriodicCb(UAVObjEvent *ev, void *ctx, void *obj_data, int len) {
416  (void) ev; (void) ctx; (void) obj_data; (void) len;
417 
418  static unsigned int counter=0;
419 
420  counter++;
421 
422 #ifndef NO_SENSORS
423  if (config_check_needed) {
424  configuration_check();
425  config_check_needed = false;
426  }
427 #endif
428 
429 #ifdef SYSTEM_RAPID_UPDATES
430  bool fourth = (counter & 3) == 0;
431 #else
432  bool fourth = true; // All of this stuff, do each time
433 #endif
434 
435  if (fourth) {
436  // Update the system statistics
437  updateStats();
438 
439 #ifndef PIPXTREME
440  // Update the system alarms
441  updateSystemAlarms();
442 #endif /* PIPXTREME */
443 
444 #if defined(WDG_STATS_DIAGNOSTICS)
445  updateWDGstats();
446 #endif
447

454 #if defined(PIOS_INCLUDE_ANNUNC)
455  // Figure out what we should be doing.
456 
457  static const char *blink_string = NULL;
458  static uint32_t blink_state = 0;
459  static uint8_t blink_prio = 0;
460  static bool ever_armed = false;
461  static uint8_t armed_status = FLIGHTSTATUS_ARMED_DISARMED;
462  static bool is_manual_control = false;
463  static int morse;
464 
465 #ifdef SYSTEMMOD_RGBLED_SUPPORT
466  static bool led_override = false;
467 #endif
468 
469  // Evaluate all our possible annunciator sources.
470  // The most important: indicate_error / alarms
471 
472  if (fourth) {
473  const char *candidate_blink;
474  uint8_t candidate_prio;
475 
476  if (custom_blink_string) {
477  candidate_prio = ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_HAIRONFIRE;
478 
479  candidate_blink = custom_blink_string;
480  } else {
481  candidate_prio = indicate_error(&candidate_blink);
482  }
483 
484  if (candidate_prio > blink_prio) {
485  // Preempt!
486  blink_state = 0;
487  blink_string = candidate_blink;
488  blink_prio = candidate_prio;
489 
490  if (blink_prio == ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_HAIRONFIRE) {
491  custom_blink_string = NULL;
492  }
493 
494  if (blink_string &&
495  !strcmp(blink_string, BLINK_STRING_RADIO)) {
496  /* XXX do this in a more robust way */
497  is_manual_control = true;
498  } else {
499  is_manual_control = false;
500  }
501  }
502 
503  FlightStatusArmedGet(&armed_status);
504 
505  if (armed_status == FLIGHTSTATUS_ARMED_ARMED) {
506  ever_armed = true;
507  } else {
508  if ((counter & 15) == 0) {
509  /* Every 16 cycles through here, kick off
510  * a config check next cycle when disarmed.
511  * Works around #1776 where tasks race
512  * through startup... but it also just
513  * seems prudent to check this stuff
514  * every half second or so while disarmed.
515  * (for lost events, etc.)
516  */
517  config_check_needed = true;
518  }
519  }
520 
521  if ((blink_prio == 0) && (blink_state == 0)) {
522  // Nothing else to do-- show armed status
523  if (armed_status == FLIGHTSTATUS_ARMED_ARMED) {
524  blink_string = "I"; // .. pairs of blinks.
525  } else {
526  blink_string = "T"; // - single long blinks
527  }
528 
529  blink_prio = SHAREDDEFS_ALARMLEVELS_OK;
530  }
531 
532  morse = morse_send(&blink_string, &blink_state);
533 
534  AnnunciatorSettingsData annunciatorSettings;
535  AnnunciatorSettingsGet(&annunciatorSettings);
536 
537 #ifdef PIOS_LED_HEARTBEAT
538  consider_annunc(&annunciatorSettings, morse > 0, ever_armed,
539  is_manual_control,
540  blink_prio, PIOS_LED_HEARTBEAT,
541  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_LED_HEARTBEAT);
542 #endif
543 
544 #ifdef PIOS_LED_ALARM
545  consider_annunc(&annunciatorSettings, morse > 0, ever_armed,
546  is_manual_control,
547  blink_prio, PIOS_LED_ALARM,
548  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_LED_ALARM);
549 #endif
550 
551  uint8_t buzzer_prio = blink_prio;
552 
553 #ifndef PIPXTREME
554  if (annunciatorSettings.ManualBuzzer !=
555  ANNUNCIATORSETTINGS_MANUALBUZZER_DISABLED) {
556  float acc[MANUALCONTROLCOMMAND_ACCESSORY_NUMELEM];
557  ManualControlCommandAccessoryGet(acc);
558  /* DONT_BUILD_IF to protect this at end of file */
559  if (acc[annunciatorSettings.ManualBuzzer - 1] > 0.0f)
560  buzzer_prio = ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_HAIRONFIRE;
561  }
562 #endif
563 
564  (void) buzzer_prio;
565 
566 #ifdef PIOS_ANNUNCIATOR_BUZZER
567  consider_annunc(&annunciatorSettings, morse > 0, ever_armed,
568  is_manual_control,
569  buzzer_prio, PIOS_ANNUNCIATOR_BUZZER,
570  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_BUZZER);
571 #endif
572 
573 #ifdef PIOS_INCLUDE_DAC_ANNUNCIATOR
574  if (pios_dac_annunciator_id) {
575  if (should_annunc(&annunciatorSettings, ever_armed,
576  is_manual_control, buzzer_prio,
577  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_DAC)) {
578  PIOS_ANNUNCDAC_SetValue(pios_dac_annunciator_id,
579  true, morse > 0);
580  } else {
581  PIOS_ANNUNCDAC_SetValue(pios_dac_annunciator_id,
582  false, false);
583  }
584  }
585 #endif
586 
587 #ifdef PIOS_INCLUDE_ACTUATOR_ANNUNCIATOR
588  if (should_annunc(&annunciatorSettings, ever_armed,
589  is_manual_control, buzzer_prio,
590  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_ACTUATORS)) {
591  actuator_send_dshot_command(DSHOT_CMD_BEACON3,
592  1, 0xffff);
593  }
594 #endif
595 
596 #ifdef SYSTEMMOD_RGBLED_SUPPORT
597  led_override = should_annunc(&annunciatorSettings,
598  ever_armed, is_manual_control, blink_prio,
599  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_RGB_LEDS);
600 #endif
601 
602  if (morse < 0) {
603  // This means we were told "completed"
604  blink_string = NULL;
605  blink_prio = 0;
606 
607  is_manual_control = false;
608  }
609  }
610 
611 #ifdef SYSTEMMOD_RGBLED_SUPPORT
612  systemmod_process_rgb_leds(led_override, morse > 0, blink_prio,
613  FLIGHTSTATUS_ARMED_DISARMED != armed_status,
614  (FLIGHTSTATUS_ARMED_ARMING == armed_status) &&
615  ((counter & 3) < 2));
616 #endif /* SYSTEMMOD_RGBLED_SUPPORT */
617 
618 #endif /* PIOS_INCLUDE_ANNUNC */
619 }

625 static void objectUpdatedCb(UAVObjEvent * ev, void *ctx, void *obj_data, int len)
626 {
627  (void) ctx; (void) obj_data; (void) len;
628 
629  ObjectPersistenceData objper;
630  UAVObjHandle obj;
631 
632  // If the object updated was the ObjectPersistence execute requested action
633  if (ev->obj == ObjectPersistenceHandle()) {
634  // Get object data
635  ObjectPersistenceGet(&objper);
636 
637  int retval = 1;
638 
639  // When this is called because of this method don't do anything
640  if (objper.Operation == OBJECTPERSISTENCE_OPERATION_ERROR ||
641  objper.Operation == OBJECTPERSISTENCE_OPERATION_COMPLETED) {
642  return;
643  }
644 
645  if (objper.Operation == OBJECTPERSISTENCE_OPERATION_LOAD) {
646  // Get selected object
647  obj = UAVObjGetByID(objper.ObjectID);
648  if (obj == 0) {
649  return;
650  }
651  // Load selected instance
652  retval = UAVObjLoad(obj, objper.InstanceID);
653  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_SAVE) {
654  // Get selected object
655  obj = UAVObjGetByID(objper.ObjectID);
656  if (obj == 0) {
657  return;
658  }
659 
660  // Save selected instance
661  retval = UAVObjSave(obj, objper.InstanceID);
662  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_DELETE) {
663  // Delete selected instance
664  retval = UAVObjDeleteById(objper.ObjectID, objper.InstanceID);
665  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_FULLERASE) {
666  retval = -1;
667 #if defined(PIOS_INCLUDE_LOGFS_SETTINGS)
668  extern uintptr_t pios_uavo_settings_fs_id;
669  retval = PIOS_FLASHFS_Format(pios_uavo_settings_fs_id);
670 #endif
671  }
672 
673  // Yield when saving, so if there's a ton of updates we don't
674  // prevent other threads from updatin'.
675  PIOS_Thread_Sleep(25);
676 
677  switch(retval) {
678  case 0:
679  objper.Operation = OBJECTPERSISTENCE_OPERATION_COMPLETED;
680  ObjectPersistenceSet(&objper);
681  break;
682  case -1:
683  objper.Operation = OBJECTPERSISTENCE_OPERATION_ERROR;
684  ObjectPersistenceSet(&objper);
685  break;
686  default:
687  break;
688  }
689  }
690 }

929 {

933  // Check that the object is not already connected

954  objEntry->evInfo.cb = cb;

1009 static uint32_t processPeriodicUpdates()
1010 {
1011  PeriodicObjectList* objEntry;
1012  int32_t timeNow;
1013  int32_t offset;
1014 
1015  // Get lock
1016  PIOS_Recursive_Mutex_Lock(mutex, PIOS_MUTEX_TIMEOUT_MAX);
1017 
1018  // Iterate through each object and update its timer, if zero then transmit object.
1019  // Also calculate smallest delay to next update.
1020  uint32_t now = PIOS_Thread_Systime();
1021  uint32_t timeToNextUpdate = PIOS_Thread_Systime() + MAX_UPDATE_PERIOD_MS;
1022  LL_FOREACH(objList, objEntry)
1023  {
1024  // If object is configured for periodic updates
1025  if (objEntry->updatePeriodMs > 0)
1026  {
1027  // Check if time for the next update
1028  timeNow = PIOS_Thread_Systime();
1029  if (objEntry->timeToNextUpdateMs <= timeNow)
1030  {
1031  // Reset timer
1032  offset = ( timeNow - objEntry->timeToNextUpdateMs ) % objEntry->updatePeriodMs;
1033  objEntry->timeToNextUpdateMs = timeNow + objEntry->updatePeriodMs - offset;
1034  // Invoke callback, if one
1035  if ( objEntry->evInfo.cb != 0)
1036  {
1037  objEntry->evInfo.cb(&objEntry->evInfo.ev, NULL, NULL, 0); // the function is expected to copy the event information
1038  }
1039  // Push event to queue, if one
1040  if ( objEntry->evInfo.queue != 0)
1041  {
1042  if (PIOS_Queue_Send(objEntry->evInfo.queue, &objEntry->evInfo.ev, 0) != true ) // do not block if queue is full
1043  {
1044  if (objEntry->evInfo.ev.obj != NULL)
1045  stats.lastErrorID = UAVObjGetID(objEntry->evInfo.ev.obj);
1046  ++stats.eventErrors;
1047  }
1048  }
1049  }
1050  // Update minimum delay
1051  if (objEntry->timeToNextUpdateMs < timeToNextUpdate)
1052  {
1053  timeToNextUpdate = objEntry->timeToNextUpdateMs;
1054  }
1055  }
1056  }
1057 
1058  // Done
1059  PIOS_Recursive_Mutex_Unlock(mutex);
1060  return timeToNextUpdate - now;
1061 }
