282 static inline uint8_t indicate_error(const char **sequence)
283 {
284  SystemAlarmsData alarms;
285  SystemAlarmsGet(&alarms);
286 
287  *sequence = NULL;
288 
289  uint8_t worst_sev = 0;
290 
291  bool generic = false;
292 
293  for (uint32_t i = 0; i < SYSTEMALARMS_ALARM_NUMELEM; i++) {
294  // If this is less severe than the current alarm, continue.
295  if (alarms.Alarm[i] < worst_sev) {
296  continue;
297  }
298 
299  // If there's a tie and the previous answer is not the
300  // 'generic' alarm, continue.
301  if ((!generic) && (alarms.Alarm[i] == worst_sev)) {
302  continue;
303  }
304 
305  uint8_t thresh = SYSTEMALARMS_ALARM_WARNING;
306 
307  if (i == SYSTEMALARMS_ALARM_TELEMETRY) {
308  // Suppress most alarms from telemetry.
309  // The user can identify if present from GCS.
310  thresh = SYSTEMALARMS_ALARM_CRITICAL;
311  }
312 
313  if (i == SYSTEMALARMS_ALARM_EVENTSYSTEM) {
314  // Skip event system alarms-- nuisance
315  continue;
316  }
317 
318  if (alarms.Alarm[i] < thresh) {
319  continue;
320  }
321 
322  // OK, we have a candidate alarm.
323 
324  worst_sev = alarms.Alarm[i];
325  generic = false;
326 
327  switch (i) {
328  case SYSTEMALARMS_ALARM_BATTERY:
329  // -... b for battery
330  *sequence = "b ";
331  break;
332  case SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION:
333  // -.-. c for configuration
334  *sequence = "c ";
335  break;
336  case SYSTEMALARMS_ALARM_GPS:
337  // --. g for GPS
338  *sequence = "g ";
339  break;
340  case SYSTEMALARMS_ALARM_MANUALCONTROL:
341  // .-. r for radio
342  *sequence = BLINK_STRING_RADIO;
343  break;
344  default:
345  // .- a for alarm
346  *sequence = "a ";
347 
348  generic = true;
349  break;
350  }
351  }
352 
353  return worst_sev;
354 }

452  static uint8_t blink_prio = 0;
453  static bool ever_armed = false;
454  static uint8_t armed_status = FLIGHTSTATUS_ARMED_DISARMED;

457 
458 #ifdef SYSTEMMOD_RGBLED_SUPPORT
459  static bool led_override = false;
460 #endif
461 
462  // Evaluate all our possible annunciator sources.
463 
464  // The most important: indicate_error / alarms
465 
466  if (fourth) {
467  const char *candidate_blink;
468  uint8_t candidate_prio;
469 
470  candidate_prio = indicate_error(&candidate_blink);
471 
472  if (candidate_prio > blink_prio) {
473  // Preempt!
474  blink_state = 0;
475  blink_string = candidate_blink;
476  blink_prio = candidate_prio;
477 
478  if (blink_string &&
479  !strcmp(blink_string, BLINK_STRING_RADIO)) {
480  // XXX do this in a more robust way */

486

488

490  ever_armed = true;
491  }
492 
493  if ((blink_prio == 0) && (blink_state == 0)) {
494  // Nothing else to do-- show armed status
495  if (armed_status == FLIGHTSTATUS_ARMED_ARMED) {
496  blink_string = "I"; // .. pairs of blinks.
497  } else {
498  blink_string = "T"; // - single long blinks

507  AnnunciatorSettingsGet(&annunciatorSettings);
508 
509 #ifdef PIOS_LED_HEARTBEAT
510  consider_annunc(&annunciatorSettings, morse > 0, ever_armed,

512  blink_prio, PIOS_LED_HEARTBEAT,
513  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_LED_HEARTBEAT);
514 #endif

516 #ifdef PIOS_LED_ALARM
517  consider_annunc(&annunciatorSettings, morse > 0, ever_armed,
518  is_manual_control,
519  blink_prio, PIOS_LED_ALARM,
520  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_LED_ALARM);
521 #endif
522 
523 #ifdef PIOS_ANNUNCIATOR_BUZZER
524  consider_annunc(&annunciatorSettings, morse > 0, ever_armed,
525  is_manual_control,
526  blink_prio, PIOS_ANNUNCIATOR_BUZZER,
527  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_BUZZER);
528 #endif
529 
530 #ifdef PIOS_INCLUDE_DAC_ANNUNCIATOR
531  if (pios_dac_annunciator_id) {
532  if (should_annunc(&annunciatorSettings, ever_armed,
533  is_manual_control, blink_prio,
534  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_DAC)) {
535  PIOS_ANNUNCDAC_SetValue(pios_dac_annunciator_id,
536  true, morse > 0);
537  } else {
538  PIOS_ANNUNCDAC_SetValue(pios_dac_annunciator_id,
539  false, false);
540  }
541  }
542 #endif
543 
544 #ifdef SYSTEMMOD_RGBLED_SUPPORT
545  led_override = should_annunc(&annunciatorSettings,
546  ever_armed, is_manual_control, blink_prio,
547  ANNUNCIATORSETTINGS_ANNUNCIATEANYTIME_RGB_LEDS);
548 #endif
549 
550  if (morse < 0) {
551  // This means we were told "completed"
552  blink_string = NULL;
553  blink_prio = 0;
554 
555  is_manual_control = false;
556  }
557  }
558 
559 #ifdef SYSTEMMOD_RGBLED_SUPPORT
560  systemmod_process_rgb_leds(led_override, morse > 0, blink_prio,

565 
566 #endif /* !PIPXTREME && PIOS_INCLUDE_ANNUNC */
567 }

573 static void objectUpdatedCb(UAVObjEvent * ev, void *ctx, void *obj_data, int len)
574 {
575  (void) ctx; (void) obj_data; (void) len;
576 
577  /* Handled in RadioComBridge on pipxtreme. */
578 #ifndef PIPXTREME
579  ObjectPersistenceData objper;
580  UAVObjHandle obj;
581 
582  // If the object updated was the ObjectPersistence execute requested action
583  if (ev->obj == ObjectPersistenceHandle()) {
584  // Get object data
585  ObjectPersistenceGet(&objper);
586 
587  int retval = 1;
588 
589  // When this is called because of this method don't do anything
590  if (objper.Operation == OBJECTPERSISTENCE_OPERATION_ERROR ||
591  objper.Operation == OBJECTPERSISTENCE_OPERATION_COMPLETED) {
592  return;
593  }
594 
595  if (objper.Operation == OBJECTPERSISTENCE_OPERATION_LOAD) {
596  // Get selected object
597  obj = UAVObjGetByID(objper.ObjectID);
598  if (obj == 0) {
599  return;
600  }
601  // Load selected instance
602  retval = UAVObjLoad(obj, objper.InstanceID);
603  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_SAVE) {
604  // Get selected object
605  obj = UAVObjGetByID(objper.ObjectID);
606  if (obj == 0) {
607  return;
608  }
609 
610  // Save selected instance
611  retval = UAVObjSave(obj, objper.InstanceID);
612  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_DELETE) {
613  // Delete selected instance
614  retval = UAVObjDeleteById(objper.ObjectID, objper.InstanceID);
615  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_FULLERASE) {
616  retval = -1;
617 #if defined(PIOS_INCLUDE_LOGFS_SETTINGS)
618  extern uintptr_t pios_uavo_settings_fs_id;
619  retval = PIOS_FLASHFS_Format(pios_uavo_settings_fs_id);
620 #endif
621  }
622 
623  // Yield when saving, so if there's a ton of updates we don't
624  // prevent other threads from updatin'.
625  PIOS_Thread_Sleep(25);
626 
627  switch(retval) {
628  case 0:
629  objper.Operation = OBJECTPERSISTENCE_OPERATION_COMPLETED;
630  ObjectPersistenceSet(&objper);
631  break;
632  case -1:
633  objper.Operation = OBJECTPERSISTENCE_OPERATION_ERROR;
634  ObjectPersistenceSet(&objper);
635  break;
636  default:
637  break;
638  }
639  }
640 #endif
641 }

678 static void updateRfm22bStats() {
679 #if defined(PIOS_INCLUDE_RFM22B)
680  if (pios_rfm22b_id) {
681 
682  // Update the RFM22BStatus UAVO
683  RFM22BStatusData rfm22bStatus;
684  RFM22BStatusInstGet(RFM22BSTATUSINST, &rfm22bStatus);
685 
686  // Get the stats from the radio device
687  struct rfm22b_stats radio_stats;
688  PIOS_RFM22B_GetStats(pios_rfm22b_id, &radio_stats);
689 
690  // Update the LInk status
691  static bool first_time = true;
692  static uint16_t prev_tx_count = 0;
693  static uint16_t prev_rx_count = 0;
694  rfm22bStatus.HeapRemaining = PIOS_heap_get_free_size();
695  rfm22bStatus.RxGood = radio_stats.rx_good;
696  rfm22bStatus.RxCorrected = radio_stats.rx_corrected;
697  rfm22bStatus.RxErrors = radio_stats.rx_error;
698  rfm22bStatus.RxSyncMissed = radio_stats.rx_sync_missed;
699  rfm22bStatus.TxMissed = radio_stats.tx_missed;
700  rfm22bStatus.RxFailure = radio_stats.rx_failure;
701  rfm22bStatus.Resets = radio_stats.resets;
702  rfm22bStatus.Timeouts = radio_stats.timeouts;
703  rfm22bStatus.RSSI = radio_stats.rssi;
704  rfm22bStatus.LinkQuality = radio_stats.link_quality;
705  if (first_time) {
706  first_time = false;
707  } else {
708  uint16_t tx_count = radio_stats.tx_byte_count;
709  uint16_t rx_count = radio_stats.rx_byte_count;
710  uint16_t tx_bytes = (tx_count < prev_tx_count) ? (0xffff - prev_tx_count + tx_count) : (tx_count - prev_tx_count);
711  uint16_t rx_bytes = (rx_count < prev_rx_count) ? (0xffff - prev_rx_count + rx_count) : (rx_count - prev_rx_count);

787  idleCounterClear = 1;
788 
789 #if defined(PIOS_INCLUDE_ADC) && defined(PIOS_ADC_USE_TEMP_SENSOR)
790  float temp_voltage = 3.3f * PIOS_ADC_DevicePinGet(PIOS_INTERNAL_ADC, 0) / ((1 << 12) - 1);

957 static int32_t eventPeriodicCreate(UAVObjEvent* ev, UAVObjEventCallback cb, struct pios_queue *queue, uint16_t periodMs)
958 {
959  PeriodicObjectList* objEntry;
960  // Get lock
961  PIOS_Recursive_Mutex_Lock(mutex, PIOS_MUTEX_TIMEOUT_MAX);
962  // Check that the object is not already connected
963  LL_FOREACH(objList, objEntry)
964  {
965  if (objEntry->evInfo.cb == cb &&
966  objEntry->evInfo.queue == queue &&
967  objEntry->evInfo.ev.obj == ev->obj &&
968  objEntry->evInfo.ev.instId == ev->instId &&
969  objEntry->evInfo.ev.event == ev->event)
970  {
971  // Already registered, do nothing
972  PIOS_Recursive_Mutex_Unlock(mutex);
973  return -1;
974  }
975  }
976  // Create handle
977  objEntry = (PeriodicObjectList*)PIOS_malloc_no_dma(sizeof(PeriodicObjectList));
978  if (objEntry == NULL) return -1;
979  objEntry->evInfo.ev.obj = ev->obj;
980  objEntry->evInfo.ev.instId = ev->instId;
981  objEntry->evInfo.ev.event = ev->event;
982  objEntry->evInfo.cb = cb;
983  objEntry->evInfo.queue = queue;
984  objEntry->updatePeriodMs = periodMs;
985  objEntry->timeToNextUpdateMs = randomize_int(periodMs); // avoid bunching of updates
986  // Add to list
987  LL_APPEND(objList, objEntry);
988  // Release lock
989  PIOS_Recursive_Mutex_Unlock(mutex);

1037 static uint32_t processPeriodicUpdates()
1038 {
1039  PeriodicObjectList* objEntry;
1040  int32_t timeNow;
1041  int32_t offset;
1042 
1043  // Get lock
1044  PIOS_Recursive_Mutex_Lock(mutex, PIOS_MUTEX_TIMEOUT_MAX);
1045 
1046  // Iterate through each object and update its timer, if zero then transmit object.
1047  // Also calculate smallest delay to next update.
1048  uint32_t now = PIOS_Thread_Systime();
1049  uint32_t timeToNextUpdate = PIOS_Thread_Systime() + MAX_UPDATE_PERIOD_MS;
1050  LL_FOREACH(objList, objEntry)
1051  {
1052  // If object is configured for periodic updates
1053  if (objEntry->updatePeriodMs > 0)
1054  {
1055  // Check if time for the next update
1056  timeNow = PIOS_Thread_Systime();
1057  if (objEntry->timeToNextUpdateMs <= timeNow)
1058  {
1059  // Reset timer
1060  offset = ( timeNow - objEntry->timeToNextUpdateMs ) % objEntry->updatePeriodMs;
1061  objEntry->timeToNextUpdateMs = timeNow + objEntry->updatePeriodMs - offset;
1062  // Invoke callback, if one
1063  if ( objEntry->evInfo.cb != 0)
1064  {
1065  objEntry->evInfo.cb(&objEntry->evInfo.ev, NULL, NULL, 0); // the function is expected to copy the event information
1066  }
1067  // Push event to queue, if one
1068  if ( objEntry->evInfo.queue != 0)
1069  {
1070  if (PIOS_Queue_Send(objEntry->evInfo.queue, &objEntry->evInfo.ev, 0) != true ) // do not block if queue is full
1071  {
1072  if (objEntry->evInfo.ev.obj != NULL)
1073  stats.lastErrorID = UAVObjGetID(objEntry->evInfo.ev.obj);
1074  ++stats.eventErrors;
1075  }
1076  }
1077  }
1078  // Update minimum delay
1079  if (objEntry->timeToNextUpdateMs < timeToNextUpdate)
1080  {
1081  timeToNextUpdate = objEntry->timeToNextUpdateMs;
1082  }
1083  }
1084  }
1085 
1086  // Done
1087  PIOS_Recursive_Mutex_Unlock(mutex);
1088  return timeToNextUpdate - now;
1089 }
