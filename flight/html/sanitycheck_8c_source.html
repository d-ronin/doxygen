68 int32_t configuration_check()
69 {
70  SystemAlarmsConfigErrorOptions error_code = SYSTEMALARMS_CONFIGERROR_NONE;
71 
72  // For when modules are not running we should explicitly check the objects are
73  // valid
74  if (ManualControlSettingsHandle() == NULL ||
75  SystemSettingsHandle() == NULL) {
76  AlarmsSet(SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION, SYSTEMALARMS_ALARM_CRITICAL);
77  return 0;
78  }
79 
80  // Classify airframe type
81  bool multirotor = true;
82  uint8_t airframe_type;
83  SystemSettingsAirframeTypeGet(&airframe_type);
84  switch(airframe_type) {
85  case SYSTEMSETTINGS_AIRFRAMETYPE_QUADX:
86  case SYSTEMSETTINGS_AIRFRAMETYPE_QUADP:
87  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXA:
88  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTO:
89  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXAX:
90  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTOV:
91  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTOCOAXP:
92  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXACOAX:
93  case SYSTEMSETTINGS_AIRFRAMETYPE_TRI:
94  multirotor = true;
95  break;
96  default:
97  multirotor = false;
98  }
99 
100  // For each available flight mode position sanity check the available
101  // modes
102  uint8_t num_modes;
103  uint8_t modes[MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_NUMELEM];
104  ManualControlSettingsFlightModeNumberGet(&num_modes);
105  ManualControlSettingsFlightModePositionGet(modes);
106 
107  for(uint32_t i = 0; i < num_modes; i++) {
108  switch(modes[i]) {
109  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_MANUAL:
110  if (multirotor) {
111  error_code = SYSTEMALARMS_CONFIGERROR_STABILIZATION;
112  }
113  break;
114  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_ACRO:
115  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_ACROPLUS:
116  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_ACRODYNE:
117  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_LEVELING:
118  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_HORIZON:
119  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_AXISLOCK:
120  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_VIRTUALBAR:
121  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_FAILSAFE:
122  // always ok
123  break;
124  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED1:
125  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(1, multirotor) : error_code;
126  break;
127  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED2:
128  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(2, multirotor) : error_code;
129  break;
130  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED3:
131  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(3, multirotor) : error_code;
132  break;
133  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_AUTOTUNE:
134  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_AUTOTUNE))
135  error_code = SYSTEMALARMS_CONFIGERROR_AUTOTUNE;
136  break;
137  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_ALTITUDEHOLD:
138  if ( !TaskMonitorQueryRunning(TASKINFO_RUNNING_ALTITUDEHOLD) )
139  error_code = SYSTEMALARMS_CONFIGERROR_ALTITUDEHOLD;
140  break;
141  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_POSITIONHOLD:
142  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_RETURNTOHOME:
143  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER)) {
144  error_code = SYSTEMALARMS_CONFIGERROR_PATHPLANNER;
145  } else {
146  error_code = check_safe_autonomous();
147  }
148  break;
149  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_PATHPLANNER:
150  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_TABLETCONTROL:
151  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER) ||
152  !TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHPLANNER)) {
153  error_code = SYSTEMALARMS_CONFIGERROR_PATHPLANNER;
154  } else {
155  error_code = check_safe_autonomous();
156  }
157  break;
158  default:
159  // Uncovered modes are automatically an error
160  error_code = SYSTEMALARMS_CONFIGERROR_UNDEFINED;
161  }
162  }
163 
164  // Check the stabilization rates are within what the sensors can track
165  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_rates() : error_code;
166 
167  // Only check safe to arm if no other errors exist
168  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_safe_to_arm() : error_code;
169 
170  set_config_error(error_code);
171 
172  return 0;
173 }

182 static int32_t check_stabilization_settings(int index, bool multirotor)
183 {
184  // Make sure the modes have identical sizes
185  if (MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM != MANUALCONTROLSETTINGS_STABILIZATION2SETTINGS_NUMELEM ||
186  MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM != MANUALCONTROLSETTINGS_STABILIZATION3SETTINGS_NUMELEM)
187  return SYSTEMALARMS_CONFIGERROR_MULTIROTOR;
188 
189  uint8_t modes[MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM];
190 
191  // Get the different axis modes for this switch position
192  switch(index) {
193  case 1:
194  ManualControlSettingsStabilization1SettingsGet(modes);
195  break;
196  case 2:
197  ManualControlSettingsStabilization2SettingsGet(modes);
198  break;
199  case 3:
200  ManualControlSettingsStabilization3SettingsGet(modes);
201  break;
202  default:
203  return SYSTEMALARMS_CONFIGERROR_NONE;
204  }
205 
206  // For multirotors verify that nothing is set to "disabled" or "manual"
207  if (multirotor) {
208  for(uint32_t i = 0; i < NELEMENTS(modes); i++) {
209  if (modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_DISABLED || modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_MANUAL)
210  return SYSTEMALARMS_CONFIGERROR_MULTIROTOR;
211 
212  // If this axis allows enabling an autotune behavior without the module
213  // running then set an alarm now that aututune module initializes the
214  // appropriate objects
215  if ((modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_SYSTEMIDENT) &&
216  (!TaskMonitorQueryRunning(TASKINFO_RUNNING_AUTOTUNE)))
217  return SYSTEMALARMS_CONFIGERROR_AUTOTUNE;
218  }
219  }
220 
221  // POI mode is only valid for YAW in the case it is enabled and camera stab is running
222  if (modes[MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_ROLL] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_POI ||
223  modes[MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_PITCH] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_POI)
224  return SYSTEMALARMS_CONFIGERROR_STABILIZATION;
225  if (modes[MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_YAW] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_POI) {
226 #if !defined(CAMERASTAB_POI_MODE)
227  return SYSTEMALARMS_CONFIGERROR_STABILIZATION;
228 #endif
229  // TODO: Need to check camera stab is actually running
230  }
231 
232 
233  // Warning: This assumes that certain conditions in the XML file are met. That
234  // MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_DISABLED has the same numeric value for each channel
235  // and is the same for STABILIZATIONDESIRED_STABILIZATIONMODE_DISABLED
236 
237  return SYSTEMALARMS_CONFIGERROR_NONE;
238 }

249 static int32_t check_safe_to_arm()
250 {
251  FlightStatusData flightStatus;
252  FlightStatusGet(&flightStatus);
253 
254  // Only arm in traditional modes where pilot has control
255  if (flightStatus.Armed != FLIGHTSTATUS_ARMED_ARMED) {
256  switch (flightStatus.FlightMode) {
257  case FLIGHTSTATUS_FLIGHTMODE_MANUAL:
258  case FLIGHTSTATUS_FLIGHTMODE_ACRO:
259  case FLIGHTSTATUS_FLIGHTMODE_ACROPLUS:
260  case FLIGHTSTATUS_FLIGHTMODE_ACRODYNE:
261  case FLIGHTSTATUS_FLIGHTMODE_LEVELING:
262  case FLIGHTSTATUS_FLIGHTMODE_HORIZON:
263  case FLIGHTSTATUS_FLIGHTMODE_AXISLOCK:
264  case FLIGHTSTATUS_FLIGHTMODE_VIRTUALBAR:
265  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED1:
266  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED2:
267  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED3:
268  case FLIGHTSTATUS_FLIGHTMODE_ALTITUDEHOLD:
269  break;
270 
271  case FLIGHTSTATUS_FLIGHTMODE_FAILSAFE:
272  /* for failsafe, we don't want to prevent
273  * arming here because it makes an ugly looking
274  * GCS config error.
275  */
276  break;
277  default:
278  // Any mode not specifically allowed prevents arming
279  return SYSTEMALARMS_CONFIGERROR_UNSAFETOARM;
280  }
281  }
282 
283  return SYSTEMALARMS_CONFIGERROR_NONE;
284 }

290 static int32_t check_safe_autonomous()
291 {
292  // The current filter combinations are safe for navigation
293  // Attitude | Navigation
294  // Comp | Raw (not recommended)
295  // Comp | INS (recommmended)
296  // Anything | None (unsafe)
297  // INSOutdoor | INS
298  // INSIndoor | INS (unsafe)
299 
300  StateEstimationData stateEstimation;
301  StateEstimationGet(&stateEstimation);
302 
303  if (stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_INSINDOOR)
304  return SYSTEMALARMS_CONFIGERROR_NAVFILTER;
305 
306  // Anything not allowed is invalid, safe default
307  if (stateEstimation.NavigationFilter != STATEESTIMATION_NAVIGATIONFILTER_INS &&
308  stateEstimation.NavigationFilter != STATEESTIMATION_NAVIGATIONFILTER_RAW)
309  return SYSTEMALARMS_CONFIGERROR_NAVFILTER;
310 
311  return SYSTEMALARMS_CONFIGERROR_NONE;
312 }

319 static int32_t check_stabilization_rates()
320 {
321  const float MAXIMUM_SAFE_FRACTIONAL_RATE = 0.85;
322  int32_t max_safe_rate = PIOS_SENSORS_GetMaxGyro() * MAXIMUM_SAFE_FRACTIONAL_RATE;
323  float rates[3];
324 
325  StabilizationSettingsManualRateGet(rates);
326  if (rates[0] > max_safe_rate || rates[1] > max_safe_rate || rates[2] > max_safe_rate)
327  return SYSTEMALARMS_CONFIGERROR_STABILIZATION;
328 
329  StabilizationSettingsMaximumRateGet(rates);
330  if (rates[0] > max_safe_rate || rates[1] > max_safe_rate || rates[2] > max_safe_rate)
331  return SYSTEMALARMS_CONFIGERROR_STABILIZATION;
332 
333  return SYSTEMALARMS_CONFIGERROR_NONE;
334 }

340 void set_config_error(SystemAlarmsConfigErrorOptions error_code)
341 {
342  // Get the severity of the alarm given the error code
343  SystemAlarmsAlarmOptions severity;
344 
345  static bool sticky = false;
346 
347  /* Once a sticky error occurs, never change the error code */
348  if (sticky) return;
349 
350  switch (error_code) {
351  case SYSTEMALARMS_CONFIGERROR_NONE:
352  severity = SYSTEMALARMS_ALARM_OK;
353  break;
354  default:
355  error_code = SYSTEMALARMS_CONFIGERROR_UNDEFINED;
356  /* and fall through */
357 
358  case SYSTEMALARMS_CONFIGERROR_DUPLICATEPORTCFG:
359  sticky = true;
360  /* and fall through */
361  case SYSTEMALARMS_CONFIGERROR_AUTOTUNE:
362  case SYSTEMALARMS_CONFIGERROR_ALTITUDEHOLD:
363  case SYSTEMALARMS_CONFIGERROR_MULTIROTOR:
364  case SYSTEMALARMS_CONFIGERROR_NAVFILTER:
365  case SYSTEMALARMS_CONFIGERROR_PATHPLANNER:
366  case SYSTEMALARMS_CONFIGERROR_POSITIONHOLD:
367  case SYSTEMALARMS_CONFIGERROR_STABILIZATION:
368  case SYSTEMALARMS_CONFIGERROR_UNDEFINED:
369  case SYSTEMALARMS_CONFIGERROR_UNSAFETOARM:
370  severity = SYSTEMALARMS_ALARM_ERROR;
371  break;
372  }
373 
374  // Make sure not to set the error code if it didn't change
375  SystemAlarmsConfigErrorOptions current_error_code;
376  SystemAlarmsConfigErrorGet((uint8_t *) &current_error_code);
377  if (current_error_code != error_code) {
378  SystemAlarmsConfigErrorSet((uint8_t *) &error_code);
379  }
380 
381  // AlarmSet checks only updates on toggle
382  AlarmsSet(SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION, (uint8_t) severity);
383 }
