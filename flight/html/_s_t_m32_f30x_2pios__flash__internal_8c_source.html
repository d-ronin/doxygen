91 static int32_t PIOS_Flash_Internal_StartTransaction(uintptr_t chip_id)
92 {
93  struct pios_internal_flash_dev *flash_dev = (struct pios_internal_flash_dev *)chip_id;
94 
95  if (!PIOS_Flash_Internal_Validate(flash_dev))
96  return -1;
97 
98  if (PIOS_Semaphore_Take(flash_dev->transaction_lock, PIOS_SEMAPHORE_TIMEOUT_MAX) != true)
99  return -2;
100 
101  /* Unlock the internal flash so we can write to it */
102  FLASH_Unlock();
103 
104  return 0;

156 static int32_t PIOS_Flash_Internal_WriteData(uintptr_t chip_id, uint32_t chip_offset, const uint8_t *data, uint16_t len)
157 {
158  PIOS_Assert(data);
159  PIOS_Assert((chip_offset & 0x0001) == 0)
160 
161  struct pios_internal_flash_dev *flash_dev = (struct pios_internal_flash_dev *)chip_id;
162 
163  if (!PIOS_Flash_Internal_Validate(flash_dev))
164  return -1;
165 
166  /* Write the data */
167  for (uint16_t i = 0; i < (len & ~1); i += 2) {
168  /* Check if content has been changed prior to write.
169  * This should enhance performance and make this compatible with the F3 chip.
170  */
171  uint8_t temp[2];
172  if (PIOS_Flash_Internal_ReadData(chip_id, chip_offset + i, temp, sizeof(temp)) != 0)
173  return -4;
174 
175  if (temp[0] == data[i] && temp[1] == data[i + 1])
176  continue;
177 
178  FLASH_Status status;
179  status = FLASH_ProgramHalfWord(FLASH_BASE + chip_offset + i, data[i + 1] << 8 | data[i]);
180  PIOS_Assert(status == FLASH_COMPLETE);
181  }
182  /* Handle uneven writes by filling up with 0xff*/
183  if ((len & 1) != 0) {
184  FLASH_Status status;
185  status = FLASH_ProgramHalfWord(FLASH_BASE + chip_offset + len - 1, 0xff << 8 | data[len - 1]);
186  PIOS_Assert(status == FLASH_COMPLETE);
187  }
188 
189  return 0;
190 }
