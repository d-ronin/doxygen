163  struct pios_adc_dev * adc_dev = sub_device_list.sub_device_pointers[x];
164  if (!PIOS_ADC_validate(adc_dev)) {

170  return (adc_dev->driver->get_pin)(adc_dev->lower_id, channel - offset);
