52 void PIOS_FLASH_fixup_partitions_for_capacity(
53  struct pios_flash_partition *partition_table,
54  uint8_t partition_table_len,
55  const struct pios_flash_chip *descriptor,
56  struct pios_flash_sector_range *sectors,
57  uint32_t num_bytes)
58 {
59  PIOS_Assert(descriptor->sector_blocks == sectors);
60 
61  uint32_t basic_capacity = 0;
62 
63  for (int i = 0; i < descriptor->num_blocks; i++) {
64  basic_capacity +=
65  (sectors[i].last_sector - sectors[i].base_sector + 1) *
66  sectors[i].sector_size;
67  }
68 
69  PIOS_Assert(num_bytes >= basic_capacity);
70 
71  if (num_bytes == basic_capacity) {
72  return; // nothing to see here.
73  }
74 
75  /* Select the last group of sectors */
76  sectors += descriptor->num_blocks - 1;
77 
78  uint32_t old_last_sector = sectors->last_sector;
79 
80  uint32_t additional_bytes = num_bytes - basic_capacity;
81  uint32_t additional_sectors = additional_bytes / sectors->sector_size;
82 
83  PIOS_Assert(additional_bytes ==
84  (additional_sectors * sectors->sector_size));
85 
86  sectors->last_sector += additional_sectors;
87 
88  // iterate partitions.. where descriptor and old last sector
89  // matches, increase the size.
90  for (int i = 0; i < partition_table_len; i++) {
91  if (partition_table[i].last_sector == old_last_sector) {
92  if (partition_table[i].chip_desc == descriptor) {
93  partition_table[i].last_sector =
94  sectors->last_sector;
95  partition_table[i].size += additional_bytes;
96  }
97  }
98  }
99 }

183 static bool pios_flash_get_partition_first_sector(const struct pios_flash_partition *partition, struct pios_flash_sector_desc *curr)
184 {
185  /* Find the beginning of the partition */
186  uint32_t chip_offset = 0;
187  for (uint16_t block_id = 0; block_id < partition->chip_desc->num_blocks; block_id++) {
188  const struct pios_flash_sector_range *block = &partition->chip_desc->sector_blocks[block_id];
189 
190  if ((partition->first_sector >= block->base_sector) && (partition->first_sector <= block->last_sector)) {
191  /* Sector is in this block. Compute offset within this block */
192  chip_offset += (partition->first_sector - block->base_sector) *block->sector_size;
193 
194  curr->block_id = block_id;
195  curr->sector = partition->first_sector;
196 
197  curr->chip_offset = chip_offset;
198  curr->sector_size = block->sector_size;
199  curr->partition_offset = 0;
200 
201  return true;
202  } else {
203  /* Not this block. Skip forward to the next block. */
204  uint32_t num_sectors_in_range = (block->last_sector - block->base_sector + 1);
205  chip_offset += num_sectors_in_range *block->sector_size;
206  }
207  }
208 
209  return false;
210 }

212 static bool pios_flash_get_partition_next_sector(const struct pios_flash_partition *partition, struct pios_flash_sector_desc *curr)
213 {
214  /* Are we past the end of the device? */
215  if (curr->block_id >= partition->chip_desc->num_blocks)
216  return false;
217 
218  const struct pios_flash_sector_range *block = &partition->chip_desc->sector_blocks[curr->block_id];
219 
220  /* Is the current sector within the current block? */
221  if ((curr->sector < block->base_sector) || (curr->sector > block->last_sector))
222  return false;
223 
224  /* Is the current sector within the current partition? */
225  if ((curr->sector < partition->first_sector) || (curr->sector > partition->last_sector))
226  return false;
227 
228  /*
229  * We've been given a valid context, find the next sector in the partition
230  */
231 
232  /* Accumulate the size of the current sector into the offsets */
233  curr->chip_offset += block->sector_size;
234  curr->partition_offset += block->sector_size;
235 
236  curr->sector++;
237 
238  /* Are we still in the partition? */
239  if (curr->sector > partition->last_sector)
240  return false;
241 
242  /* Are we now beyond the end of the current block? */
243  if (curr->sector > block->last_sector) {
244  /* Move to the next block */
245  curr->block_id++;
246 
247  /* Are we still within the chip boundaries? */
248  if (curr->block_id >= partition->chip_desc->num_blocks)
249  return false;
250 
251  block = &partition->chip_desc->sector_blocks[curr->block_id];
252  }
253 
254  curr->sector_size = block->sector_size;
255 
256  return true;
257 }

259 static bool pios_flash_partition_get_chip_extents(const struct pios_flash_partition *partition, uint32_t *chip_start_offset, uint32_t *chip_end_offset)
260 {
261  struct pios_flash_sector_desc sector_desc;
262  if (!pios_flash_get_partition_first_sector(partition, &sector_desc))
263  return false;
264 
265  if (chip_start_offset)
266  *chip_start_offset = sector_desc.chip_offset;
267 
268  /* Traverse the current partition to find the end of it */
269  do {
270  ;
271  } while (pios_flash_get_partition_next_sector(partition, &sector_desc));
272 
273  if (chip_end_offset)
274  *chip_end_offset = sector_desc.chip_offset - 1;
275 
276  return true;
277 }

358 int32_t PIOS_FLASH_erase_range(uintptr_t partition_id, uint32_t start_offset, uint32_t size)
359 {
360  struct pios_flash_partition *partition = (struct pios_flash_partition *)partition_id;
361 
362  if (!PIOS_FLASH_validate_partition(partition))
363  return -20;
364 
365  if (!partition->chip_desc->driver->erase_sector)
366  return -21;
367 
368  struct pios_flash_sector_desc sector_desc;
369  if (!pios_flash_get_partition_first_sector(partition, &sector_desc))
370  return -22;
371 
372  /* Traverse the current partition and erase sectors within the requested range */
373  uint32_t erase_offset = start_offset;
374  do {
375  /* Have we finished erasing? */
376  if (size == 0)
377  break;
378 
379  /* Is our start offset in the current sector? */
380  if ((erase_offset >= sector_desc.partition_offset) &&
381  (erase_offset < sector_desc.partition_offset + sector_desc.sector_size)) {
382  /* Make sure we're erasing on a sector boundary */
383  if (erase_offset != sector_desc.partition_offset)
384  return -23;
385 
386  /* Make sure we're intending to erase the entire sector */
387  if (size < sector_desc.sector_size)
388  return -24;
389 
390  /* Erase the sector */
391  if (partition->chip_desc->driver->erase_sector(*partition->chip_desc->chip_id,
392  sector_desc.sector,
393  sector_desc.chip_offset) != 0) {
394  return -25;
395  }
396 
397  /* Update our accounting */
398  erase_offset += sector_desc.sector_size;
399  size -= sector_desc.sector_size;
400  }
401  } while (pios_flash_get_partition_next_sector(partition, &sector_desc));
402 
403  return 0;
404 }

457 int32_t PIOS_FLASH_write_data(uintptr_t partition_id, uint32_t partition_offset, const uint8_t *data, uint16_t len)
458 {
459  struct pios_flash_partition *partition = (struct pios_flash_partition *)partition_id;
460 
461  if (!PIOS_FLASH_validate_partition(partition))
462  return -20;
463 
464  if (!partition->chip_desc->driver->write_data)
465  return -21;
466 
467  /* Are we writing past the end of the partition? */
468  if ((partition_offset + len) > partition->size)
469  return -22;
470 
471  /* Fragment the writes to chip's page size */
472  uint32_t page_size = partition->chip_desc->page_size;
473  while (len > 0) {
474  /* Individual writes must fit entirely within a single page buffer. */
475  uint32_t page_remaining = page_size - (partition_offset % page_size);
476  uint16_t write_size = MIN(len, page_remaining);
477 
478  int32_t rc = partition->chip_desc->driver->write_data(*partition->chip_desc->chip_id,
479  partition->chip_offset + partition_offset,
480  data,
481  write_size);
482  if (rc != 0) {
483  /* Failed to write the data to the underlying flash */
484  return rc;
485  }
486 
487  /* Update our accounting */
488  data += write_size;
489  partition_offset += write_size;
490  len -= write_size;
491  }
492 
493  return 0;
494 }
