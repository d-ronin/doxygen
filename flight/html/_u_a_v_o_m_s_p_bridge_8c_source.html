177 static void msp_send(struct msp_bridge *m, uint8_t cmd, const uint8_t *data, size_t len)
178 {
179  uint8_t buf[5];
180  uint8_t cs = (uint8_t)(len) ^ cmd;
181 
182  buf[0] = '$';
183  buf[1] = 'M';
184  buf[2] = '>';
185  buf[3] = (uint8_t)(len);
186  buf[4] = cmd;
187 
188  PIOS_COM_SendBuffer(m->com, buf, sizeof(buf));
189  PIOS_COM_SendBuffer(m->com, data, len);
190 
191  for (int i = 0; i < len; i++) {
192  cs ^= data[i];
193  }
194  cs ^= 0;
195 
196  buf[0] = cs;
197  PIOS_COM_SendBuffer(m->com, buf, 1);
198 }

207 static msp_state msp_state_cmd(struct msp_bridge *m, uint8_t b)
208 {
209  m->cmd_i = 0;
210  m->cmd_id = b;
211  m->checksum ^= m->cmd_id;
212 
213  if (m->cmd_size > sizeof(m->cmd_data)) {
214  // Too large a body. Let's ignore it.
215  return MSP_DISCARD;
216  }
217 
218  return m->cmd_size == 0 ? MSP_CHECKSUM : MSP_FILLBUF;

388 static void msp_send_comp_gps(struct msp_bridge *m)
389 {
390  union {
391  uint8_t buf[0];
392  struct {
393  uint16_t distance_to_home; // meter
394  int16_t direction_to_home; // degree [-180:180]
395  uint8_t home_position_valid; // 0 = Invalid, Dronin MSP specific
396  } __attribute__((packed)) comp_gps;
397  } data;
398 
399  GPSPositionData gps_data = {};
400  HomeLocationData home_data = {};
401 
402  if ((GPSPositionHandle() == NULL) || (HomeLocationHandle() == NULL))
403  {
404  data.comp_gps.distance_to_home = 0;
405  data.comp_gps.direction_to_home = 0;
406  data.comp_gps.home_position_valid = 0; // Home distance and direction will not display on OSD
407  }
408  else
409  {
410  GPSPositionGet(&gps_data);
411  HomeLocationGet(&home_data);
412 
413  if((gps_data.Status < GPSPOSITION_STATUS_FIX2D) || (home_data.Set == HOMELOCATION_SET_FALSE))
414  {
415  data.comp_gps.distance_to_home = 0;
416  data.comp_gps.direction_to_home = 0;
417  data.comp_gps.home_position_valid = 0; // Home distance and direction will not display on OSD
418  }
419  else
420  {
421  data.comp_gps.home_position_valid = 1; // Home distance and direction will display on OSD
422 
423  int32_t delta_lon = (home_data.Longitude - gps_data.Longitude); // degrees * 1e7
424  int32_t delta_lat = (home_data.Latitude - gps_data.Latitude ); // degrees * 1e7
425 
426  float delta_y = (float)delta_lon * WGS84_RADIUS_EARTH_KM * DEG2RAD; // KM * 1e7
427  float delta_x = (float)delta_lat * WGS84_RADIUS_EARTH_KM * DEG2RAD; // KM * 1e7
428 
429  delta_y *= cosf((float)home_data.Latitude * 1e-7f * (float)DEG2RAD); // Latitude compression correction
430 
431  data.comp_gps.distance_to_home = (uint16_t)(sqrtf(delta_x * delta_x + delta_y * delta_y) * 1e-4f); // meters
432 
433  if ((delta_lon == 0) && (delta_lat == 0))
434  data.comp_gps.direction_to_home = 0;
435  else
436  data.comp_gps.direction_to_home = (int16_t)(atan2f(delta_y, delta_x) * RAD2DEG); // degrees;
437  }
438  }
439 
440  msp_send(m, MSP_COMP_GPS, data.buf, sizeof(data));
441 }

573 static msp_state msp_state_checksum(struct msp_bridge *m, uint8_t b)
574 {
575  if ((m->checksum ^ b) != 0) {
576  return MSP_IDLE;
577  }
578 
579  // Respond to interesting things.
580  switch (m->cmd_id) {

582  msp_send_ident(m);
583  break;
584  case MSP_RAW_GPS:
585  msp_send_raw_gps(m);
586  break;
587  case MSP_COMP_GPS:

591  msp_send_altitude(m);
592  break;
593  case MSP_ATTITUDE:
594  msp_send_attitude(m);
595  break;
596  case MSP_STATUS:
597  msp_send_status(m);
598  break;
599  case MSP_ANALOG:
600  msp_send_analog(m);
601  break;
602  case MSP_RC:
603  msp_send_channels(m);
604  break;
605  case MSP_BOXIDS:
606  msp_send_boxids(m);
607  break;
608  case MSP_ALARMS:
609  msp_send_alarms(m);
610  break;
611  }
612  return MSP_IDLE;
613 }

631  m->state = MSP_MAYBE_UAVTALK_SLOW2;
632  break;
633  case '<': // uavtalk matching with 0x3c 0x2x 0xxx 0x0x
634  m->state = MSP_MAYBE_UAVTALK2;
635  break;
636  case '$':
637  m->state = MSP_HEADER_START;
638  break;
639  default:
640  m->state = MSP_IDLE;
641  }
642  break;
643  case MSP_HEADER_START:
644  m->state = b == 'M' ? MSP_HEADER_M : MSP_IDLE;
645  break;
646  case MSP_HEADER_M:
647  m->state = b == '<' ? MSP_HEADER_SIZE : MSP_IDLE;
648  break;
649  case MSP_HEADER_SIZE:
650  m->state = msp_state_size(m, b);
651  break;
652  case MSP_HEADER_CMD:
653  m->state = msp_state_cmd(m, b);
654  break;
655  case MSP_FILLBUF:
656  m->state = msp_state_fill_buf(m, b);
657  break;
658  case MSP_CHECKSUM:
659  m->state = msp_state_checksum(m, b);
660  break;
661  case MSP_DISCARD:
662  m->state = msp_state_discard(m, b);
663  break;
664  case MSP_MAYBE_UAVTALK2:
665  // e.g. 3c 20 1d 00
666  // second possible uavtalk byte
667  m->state = (b&0xf0) == 0x20 ? MSP_MAYBE_UAVTALK3 : MSP_IDLE;
668  break;
669  case MSP_MAYBE_UAVTALK3:
670  // third possible uavtalk byte can be anything
671  m->state = MSP_MAYBE_UAVTALK4;
672  break;
673  case MSP_MAYBE_UAVTALK4:
674  m->state = MSP_IDLE;
675  // If this looks like the fourth possible uavtalk byte, we're done
676  if ((b & 0xf0) == 0) {
677  PIOS_COM_TELEM_RF = m->com;
678  return false;
679  }
680  break;
681  case MSP_MAYBE_UAVTALK_SLOW2:
682  m->state = b == 0x18 ? MSP_MAYBE_UAVTALK_SLOW3 : MSP_IDLE;
683  break;
684  case MSP_MAYBE_UAVTALK_SLOW3:
685  m->state = b == 0x98 ? MSP_MAYBE_UAVTALK_SLOW4 : MSP_IDLE;
686  break;
687  case MSP_MAYBE_UAVTALK_SLOW4:
688  m->state = b == 0x7e ? MSP_MAYBE_UAVTALK_SLOW5 : MSP_IDLE;
689  break;
690  case MSP_MAYBE_UAVTALK_SLOW5:
691  m->state = b == 0x00 ? MSP_MAYBE_UAVTALK_SLOW6 : MSP_IDLE;
692  break;
693  case MSP_MAYBE_UAVTALK_SLOW6:
694  m->state = MSP_IDLE;
695  // If this looks like the sixth possible 57600 baud uavtalk byte, we're done
696  if(b == 0x60) {
697  PIOS_COM_ChangeBaud(m->com, 57600);
698  PIOS_COM_TELEM_RF = m->com;
699  return false;
700  }
701  break;
702  }
703 
704  return true;
705 }
