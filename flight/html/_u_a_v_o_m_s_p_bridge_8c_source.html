246 static void msp_send(struct msp_bridge *m, uint8_t cmd, const uint8_t *data, size_t len)
247 {
248  uint8_t buf[5];
249  uint8_t cs = (uint8_t)(len) ^ cmd;
250 
251  buf[0] = '$';
252  buf[1] = 'M';
253  buf[2] = '>';
254  buf[3] = (uint8_t)(len);
255  buf[4] = cmd;
256 
257  PIOS_COM_SendBuffer(m->com, buf, sizeof(buf));
258  PIOS_COM_SendBuffer(m->com, data, len);
259 
260  for (int i = 0; i < len; i++) {
261  cs ^= data[i];
262  }
263  cs ^= 0;
264 
265  buf[0] = cs;
266  PIOS_COM_SendBuffer(m->com, buf, 1);
267 }

276 static msp_state msp_state_cmd(struct msp_bridge *m, uint8_t b)
277 {
278  m->cmd_i = 0;
279  m->cmd_id = b;
280  m->checksum ^= m->cmd_id;
281 
282  if (m->cmd_size > sizeof(m->cmd_data)) {
283  // Too large a body. Let's ignore it.
284  return MSP_DISCARD;
285  }
286 
287  return m->cmd_size == 0 ? MSP_CHECKSUM : MSP_FILLBUF;
288 }

850 static void msp_send_comp_gps(struct msp_bridge *m)
851 {
852  union {
853  uint8_t buf[0];
854  struct {
855  uint16_t distance_to_home; // meter
856  int16_t direction_to_home; // degree [-180:180]
857  uint8_t home_position_valid; // 0 = Invalid, Dronin MSP specific
858  } __attribute__((packed)) comp_gps;
859  } data;
860 
861  GPSPositionData gps_data = {};
862  HomeLocationData home_data = {};
863 
864  if ((GPSPositionHandle() == NULL) || (HomeLocationHandle() == NULL))
865  {
866  data.comp_gps.distance_to_home = 0;
867  data.comp_gps.direction_to_home = 0;
868  data.comp_gps.home_position_valid = 0; // Home distance and direction will not display on OSD
869  }
870  else
871  {
872  GPSPositionGet(&gps_data);
873  HomeLocationGet(&home_data);
874 
875  if((gps_data.Status < GPSPOSITION_STATUS_FIX2D) || (home_data.Set == HOMELOCATION_SET_FALSE))
876  {
877  data.comp_gps.distance_to_home = 0;
878  data.comp_gps.direction_to_home = 0;
879  data.comp_gps.home_position_valid = 0; // Home distance and direction will not display on OSD
880  }
881  else
882  {
883  data.comp_gps.home_position_valid = 1; // Home distance and direction will display on OSD
884 
885  int32_t delta_lon = (home_data.Longitude - gps_data.Longitude); // degrees * 1e7
886  int32_t delta_lat = (home_data.Latitude - gps_data.Latitude ); // degrees * 1e7
887 
888  float delta_y = (float)delta_lon * WGS84_RADIUS_EARTH_KM * DEG2RAD; // KM * 1e7
889  float delta_x = (float)delta_lat * WGS84_RADIUS_EARTH_KM * DEG2RAD; // KM * 1e7
890 
891  delta_y *= cosf((float)home_data.Latitude * 1e-7f * (float)DEG2RAD); // Latitude compression correction
892 
893  data.comp_gps.distance_to_home = (uint16_t)(sqrtf(delta_x * delta_x + delta_y * delta_y) * 1e-4f); // meters
894 
895  if ((delta_lon == 0) && (delta_lat == 0))
896  data.comp_gps.direction_to_home = 0;
897  else
898  data.comp_gps.direction_to_home = (int16_t)(atan2f(delta_y, delta_x) * RAD2DEG); // degrees;
899  }
900  }
901 
902  msp_send(m, MSP_COMP_GPS, data.buf, sizeof(data));
903 }

1171 static void msp_receive_byte(struct msp_bridge *m, uint8_t b)
1172 {
1173  switch (m->state) {
1174  case MSP_IDLE:
1175  switch (b) {
1176  case '<': // uavtalk matching with 0x3c 0x2x 0xxx 0x0x
1177  m->state = MSP_MAYBE_UAVTALK2;
1178  break;
1179  case '$':
1180  m->state = MSP_HEADER_START;
1181  break;
1182  default:
1183  m->state = MSP_IDLE;
1184  }
1185  break;
1186  case MSP_HEADER_START:
1187  m->state = b == 'M' ? MSP_HEADER_M : MSP_IDLE;
1188  break;
1189  case MSP_HEADER_M:
1190  m->state = b == '<' ? MSP_HEADER_SIZE : MSP_IDLE;
1191  break;
1192  case MSP_HEADER_SIZE:
1193  m->state = msp_state_size(m, b);
1194  break;
1195  case MSP_HEADER_CMD:
1196  m->state = msp_state_cmd(m, b);
1197  break;
1198  case MSP_FILLBUF:
1199  m->state = msp_state_fill_buf(m, b);
1200  break;
1201  case MSP_CHECKSUM:
1202  m->state = msp_state_checksum(m, b);
1203  break;
1204  case MSP_DISCARD:
1205  m->state = msp_state_discard(m, b);
1206  break;
1207  case MSP_MAYBE_UAVTALK2:
1208  // e.g. 3c 20 1d 00
1209  // second possible uavtalk byte
1210  m->state = (b&0xf0) == 0x20 ? MSP_MAYBE_UAVTALK3 : MSP_IDLE;
1211  break;
1212  case MSP_MAYBE_UAVTALK3:
1213  // third possible uavtalk byte can be anything
1214  m->state = MSP_MAYBE_UAVTALK4;
1215  break;
1216  case MSP_MAYBE_UAVTALK4:
1217  m->state = MSP_IDLE;
1218  // If this looks like the fourth possible uavtalk byte, we're done
1219  if ((b & 0xf0) == 0) {
1220  PIOS_COM_TELEM_RF = m->com;
1221 
1222  m->handler = MSP_HANDLER_IDLE;
1223  }
1224  break;
1225  }
1226 }
