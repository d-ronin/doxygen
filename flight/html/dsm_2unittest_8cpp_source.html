71 void DsmTest::pack_channels_10bit(uint16_t channels[DSM_CHANNELS_PER_FRAME], struct pios_dsm_state *state, bool frame)
72 {
73  for (int i = 0; i < DSM_CHANNELS_PER_FRAME; i++) {
74  uint16_t j = idx10[i + DSM_CHANNELS_PER_FRAME*frame];
75  uint16_t val = channels[j];
76  uint16_t word = ((frame & (i == 0)) ? 0x8000 : 0) | ((j & 0x000F) << 10) | (val & 0x03FF);
77  state->received_data[2 + i * 2 + 1] = word & 0x00FF;
78  state->received_data[2 + i * 2] = (word >> 8) & 0x00FF;
79  }
80 }
