99 bool PIOS_SENSORS_GetData(enum pios_sensor_type type, void *buf, int ms_to_wait)
100 {
101  if (type >= PIOS_SENSOR_NUM) {
102  return false;
103  }
104 
105  struct PIOS_Sensor *sensor = &sensors[type];
106 
107  if (!sensor->getdata_cb) {
108  return false;
109  }
110 
111  if (sensor->next_time) {
112  uint32_t now = PIOS_Thread_Systime();
113  int32_t time_until = sensor->next_time - now;
114 
115  if (time_until > ms_to_wait) {
116  return false;
117  }
118 
119  if (time_until > 0) {
120  PIOS_Thread_Sleep(time_until);
121  }
122 
123  /* TODO: could use elapsed time to properly adjust */
124  ms_to_wait -= time_until;
125  }
126 
127  int next_time;
128 
129  bool ret = sensor->getdata_cb(sensor->getdata_ctx, buf, ms_to_wait,
130  &next_time);
131 
132  /* Keep track of next time it *could* have data. Ensure we
133  * don't call before then.
134  */
135  if (next_time) {
136  uint32_t now = PIOS_Thread_Systime();
137 
138  sensor->next_time = now + next_time;
139  } else {
140  sensor->next_time = 0;
141  }
142 
143  return ret;
144 }
