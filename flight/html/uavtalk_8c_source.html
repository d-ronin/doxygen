179 UAVTalkRxState UAVTalkProcessInputStreamQuiet(UAVTalkConnection connectionHandle, uint8_t rxbyte)
180 {
181  UAVTalkConnectionData *connection;
182  CHECKCONHANDLE(connectionHandle,connection,return -1);
183 
184  UAVTalkInputProcessor *iproc = &connection->iproc;
185  ++connection->stats.rxBytes;
186 
187  if (iproc->state == UAVTALK_STATE_ERROR) {
188  connection->stats.rxErrors++;
189  iproc->state = UAVTALK_STATE_SYNC;
190  } else if (iproc->state == UAVTALK_STATE_COMPLETE) {
191  iproc->state = UAVTALK_STATE_SYNC;
192  }
193 
194  if (iproc->rxPacketLength < 0xffff)
195  iproc->rxPacketLength++; // update packet byte count
196 
197  // Receive state machine
198  switch (iproc->state)
199  {
200  case UAVTALK_STATE_SYNC:
201  if (rxbyte != UAVTALK_SYNC_VAL)
202  break;
203 
204  // Initialize and update the CRC
205  iproc->cs = PIOS_CRC_updateByte(0, rxbyte);
206 
207  iproc->rxPacketLength = 1;
208 
209  iproc->state = UAVTALK_STATE_TYPE;
210  break;
211 
212  case UAVTALK_STATE_TYPE:
213 
214  // update the CRC
215  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
216 
217  if ((rxbyte & UAVTALK_TYPE_MASK) != UAVTALK_TYPE_VER) {
218  iproc->state = UAVTALK_STATE_ERROR;
219  break;
220  }
221 
222  iproc->type = rxbyte;
223 
224  iproc->packet_size = 0;
225 
226  iproc->state = UAVTALK_STATE_SIZE;
227  iproc->rxCount = 0;
228  break;
229 
230  case UAVTALK_STATE_SIZE:
231 
232  // update the CRC
233  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
234 
235  if (iproc->rxCount == 0) {
236  iproc->packet_size += rxbyte;
237  iproc->rxCount++;
238  break;
239  }
240 
241  iproc->packet_size += rxbyte << 8;
242 
243  if (iproc->packet_size < UAVTALK_MIN_HEADER_LENGTH ||
244  iproc->packet_size > UAVTALK_MAX_HEADER_LENGTH + UAVTALK_MAX_PAYLOAD_LENGTH) { // incorrect packet size
245  iproc->state = UAVTALK_STATE_ERROR;
246  break;
247  }
248 
249  iproc->rxCount = 0;
250  iproc->objId = 0;
251  iproc->state = UAVTALK_STATE_OBJID;
252  break;
253 
254  case UAVTALK_STATE_OBJID:
255 
256  // update the CRC
257  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
258 
259  iproc->objId += rxbyte << (8*(iproc->rxCount++));
260 
261  if (iproc->rxCount < 4)
262  break;
263 
264  if (iproc->type == UAVTALK_TYPE_FILEREQ) {
265  /* Slightly overloaded from "normal" case. Consume
266  * 4 bytes of offset and 2 bytes of flags.
267  */
268 
269  iproc->instanceLength = 0;
270  iproc->rxCount = 0;
271  iproc->length = 6;
272 
273  if ((iproc->packet_size - iproc->rxPacketLength) !=
274  iproc->length) {
275  iproc->state = UAVTALK_STATE_ERROR;
276  break;
277  }
278 
279  iproc->state = UAVTALK_STATE_DATA;
280 
281  break;
282  }
283 
284  // Search for object.
285  iproc->obj = UAVObjGetByID(iproc->objId);
286 
287  // Determine data length
288  if (iproc->type == UAVTALK_TYPE_OBJ_REQ || iproc->type == UAVTALK_TYPE_ACK || iproc->type == UAVTALK_TYPE_NACK) {
289  iproc->length = 0;
290  iproc->instanceLength = 0;
291 
292  /* Length is always pretty much expected to be 0
293  * here, but it can be 2 if it's a multiple inst
294  * obj requested. Don't peer into metadata to
295  * figure this out-- use the packet length
296  * [so we can properly NAK objects we don't know]
297  */
298  if ((iproc->packet_size - iproc->rxPacketLength) == 2) {
299  iproc->instanceLength = 2;
300  } else if (iproc->length > 0) {
301  iproc->state = UAVTALK_STATE_ERROR;
302  break;
303  }
304  } else {
305  if (iproc->obj) {
306  iproc->length = UAVObjGetNumBytes(iproc->obj);
307  iproc->instanceLength = (UAVObjIsSingleInstance(iproc->obj) ? 0 : 2);
308  } else {
309  // We don't know if it's a multi-instance object, so just assume it's 0.
310  iproc->instanceLength = 0;
311  iproc->length = iproc->packet_size - iproc->rxPacketLength;
312  }
313  }
314 
315  // Check length and determine next state
316  if (iproc->length >= UAVTALK_MAX_PAYLOAD_LENGTH) {
317  iproc->state = UAVTALK_STATE_ERROR;
318  break;
319  }
320 
321  // Check the lengths match
322  if ((iproc->rxPacketLength + iproc->instanceLength + iproc->length) != iproc->packet_size) { // packet error - mismatched packet size
323  iproc->state = UAVTALK_STATE_ERROR;
324  break;
325  }
326 
327  iproc->instId = 0;
328  if (iproc->type == UAVTALK_TYPE_NACK) {
329  // If this is a NACK, we skip to Checksum
330  iproc->state = UAVTALK_STATE_CS;
331  }
332  // Check if this is a single instance object (i.e. if the instance ID field is coming next)
333  else if (iproc->instanceLength) {
334  iproc->state = UAVTALK_STATE_INSTID;
335  } else {
336  // If there is a payload get it, otherwise receive checksum
337  if (iproc->length > 0)
338  iproc->state = UAVTALK_STATE_DATA;
339  else
340  iproc->state = UAVTALK_STATE_CS;
341  }
342  iproc->rxCount = 0;
343 
344  break;
345 
346  case UAVTALK_STATE_INSTID:
347 
348  // update the CRC
349  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
350 
351  iproc->instId += rxbyte << (8*(iproc->rxCount++));
352 
353  if (iproc->rxCount < 2)
354  break;
355 
356  iproc->rxCount = 0;
357 
358  // If there is a payload get it, otherwise receive checksum
359  if (iproc->length > 0)
360  iproc->state = UAVTALK_STATE_DATA;
361  else
362  iproc->state = UAVTALK_STATE_CS;
363 
364  break;
365 
366  case UAVTALK_STATE_DATA:
367 
368  // update the CRC
369  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
370 
371  connection->rxBuffer[iproc->rxCount++] = rxbyte;
372  if (iproc->rxCount < iproc->length)
373  break;
374 
375  iproc->state = UAVTALK_STATE_CS;
376  iproc->rxCount = 0;
377  break;
378 
379  case UAVTALK_STATE_CS:
380 
381  // the CRC byte
382  if (rxbyte != iproc->cs) { // packet error - faulty CRC
383  iproc->state = UAVTALK_STATE_ERROR;
384  break;
385  }
386 
387  if (iproc->rxPacketLength != (iproc->packet_size + 1)) { // packet error - mismatched packet size
388  iproc->state = UAVTALK_STATE_ERROR;
389  break;
390  }
391 
392  connection->stats.rxObjectBytes += iproc->length;
393  connection->stats.rxObjects++;
394 
395  iproc->state = UAVTALK_STATE_COMPLETE;
396  break;
397 
398  default:
399  connection->stats.rxErrors++;
400  iproc->state = UAVTALK_STATE_ERROR;
401  }
402 
403  // Done
404  return iproc->state;
405 }

441 int32_t UAVTalkRelayPacket(UAVTalkConnection inConnectionHandle, UAVTalkConnection outConnectionHandle)
442 {
443  UAVTalkConnectionData *inConnection;
444 
445  CHECKCONHANDLE(inConnectionHandle, inConnection, return -1);
446  UAVTalkInputProcessor *inIproc = &inConnection->iproc;
447 
448  // The input packet must be completely parsed.
449  if (inIproc->state != UAVTALK_STATE_COMPLETE) {
450  inConnection->stats.rxErrors++;
451 
452  return -1;
453  }
454 
455  UAVTalkConnectionData *outConnection;
456  CHECKCONHANDLE(outConnectionHandle, outConnection, return -1);
457 
458  if (!outConnection->outCb) {
459  outConnection->stats.txErrors++;
460 
461  return -1;
462  }
463 
464  // Lock
465  PIOS_Recursive_Mutex_Lock(outConnection->lock, PIOS_MUTEX_TIMEOUT_MAX);
466 
467  outConnection->txBuffer[0] = UAVTALK_SYNC_VAL;
468  // Setup type
469  outConnection->txBuffer[1] = inIproc->type;
470  // next 2 bytes are reserved for data length (inserted here later)
471  // Setup object ID
472  outConnection->txBuffer[4] = (uint8_t)(inIproc->objId & 0xFF);
473  outConnection->txBuffer[5] = (uint8_t)((inIproc->objId >> 8) & 0xFF);
474  outConnection->txBuffer[6] = (uint8_t)((inIproc->objId >> 16) & 0xFF);
475  outConnection->txBuffer[7] = (uint8_t)((inIproc->objId >> 24) & 0xFF);
476  int32_t headerLength = 8;
477 
478  if (inIproc->obj) {
479  if (!UAVObjIsSingleInstance(inIproc->obj)) {
480  // Setup instance ID
481  outConnection->txBuffer[8] = (uint8_t)(inIproc->instId & 0xFF);
482  outConnection->txBuffer[9] = (uint8_t)((inIproc->instId >> 8) & 0xFF);
483  headerLength = 10;
484  }
485  }
486 
487  // Copy data (if any)
488  if (inIproc->length > 0) {
489  memcpy(&outConnection->txBuffer[headerLength], inConnection->rxBuffer, inIproc->length);
490  }
491 
492  // Store the packet length
493  outConnection->txBuffer[2] = (uint8_t)((headerLength + inIproc->length) & 0xFF);
494  outConnection->txBuffer[3] = (uint8_t)(((headerLength + inIproc->length) >> 8) & 0xFF);
495 
496  // Copy the checksum
497  outConnection->txBuffer[headerLength + inIproc->length] = inIproc->cs;
498 
499  // Send the buffer.
500  int32_t rc = (*outConnection->outCb)(outConnection->cbCtx, outConnection->txBuffer, headerLength + inIproc->length + UAVTALK_CHECKSUM_LENGTH);
501 
502  // Update stats
503  outConnection->stats.txBytes += (rc > 0) ? rc : 0;
504 
505  // evaluate return value before releasing the lock
506  int32_t ret = 0;
507  if (rc != (int32_t)(headerLength + inIproc->length + UAVTALK_CHECKSUM_LENGTH)) {
508  outConnection->stats.txErrors++;
509  ret = -1;
510  }
511 
512  // Release lock
513  PIOS_Recursive_Mutex_Unlock(outConnection->lock);
514 
515  // Done
516  return ret;
517 }

659 static void handleFileReq(UAVTalkConnectionData *connection)
660 {
661  UAVTalkInputProcessor *iproc = &connection->iproc;
662  uint32_t file_id = iproc->objId;
663 
664  struct filereq_data *req = (struct filereq_data *) connection->rxBuffer;
665 
666  /* printf("Got filereq for file_id=%08x offs=%d\n", file_id, req->offset); */
667 
668  /* Need txbuffer, and need to make sure file response msgs
669  * are contiguous on link.
670  */
671  PIOS_Recursive_Mutex_Lock(connection->lock, PIOS_MUTEX_TIMEOUT_MAX);
672 
673  connection->txBuffer[0] = UAVTALK_SYNC_VAL; // sync byte
674  connection->txBuffer[1] = UAVTALK_TYPE_FILEDATA;
675  // data length inserted here below
676  connection->txBuffer[4] = (uint8_t)(file_id & 0xFF);
677  connection->txBuffer[5] = (uint8_t)((file_id >> 8) & 0xFF);
678  connection->txBuffer[6] = (uint8_t)((file_id >> 16) & 0xFF);
679  connection->txBuffer[7] = (uint8_t)((file_id >> 24) & 0xFF);
680 
681  uint8_t data_offs = 8;
682 
683  struct fileresp_data *resp =
684  (struct fileresp_data *) (connection->txBuffer + data_offs);
685 
686  data_offs += sizeof(*resp);
687 
688  uint32_t file_offset = req->offset;
689 
690  for (int i = 0; ; i++) {
691  resp->offset = file_offset;
692  resp->flags = 0;
693 
694  int32_t cb_numbytes = -1;
695 
696  if (connection->fileCb) {
697  cb_numbytes = connection->fileCb(connection->cbCtx,
698  connection->txBuffer + data_offs,
699  file_id, file_offset, 100);
700  }
701 
702  uint8_t total_len = data_offs;
703 
704  if (cb_numbytes > 0) {
705  total_len += cb_numbytes;
706 
707  file_offset += cb_numbytes;
708 
709  if (i == 5) { /* 6 messages per chunk */
710  resp->flags = UAVTALK_FILEDATA_LAST;
711  } else {
712  resp->flags = 0;
713  }
714  } else {
715  /* End of file, last chunk in sequence */
716  resp->flags = UAVTALK_FILEDATA_LAST |
717  UAVTALK_FILEDATA_EOF;
718  }
719 
720  // Store the packet length
721  connection->txBuffer[2] = (uint8_t)((total_len) & 0xFF);
722  connection->txBuffer[3] = (uint8_t)(((total_len) >> 8) & 0xFF);
723 
724  // Calculate checksum
725  connection->txBuffer[total_len] = PIOS_CRC_updateCRC(0,
726  connection->txBuffer, total_len);
727 
728  int32_t rc = (*connection->outCb)(connection->cbCtx,
729  connection->txBuffer, total_len + 1);
730 
731  if (rc == total_len) {
732  // Update stats
733  connection->stats.txBytes += total_len;
734  }
735 
736  if (resp->flags & UAVTALK_FILEDATA_LAST) {
737  break;
738  }
739  }
740 
741  PIOS_Recursive_Mutex_Unlock(connection->lock);
742 }

755 static int32_t receiveObject(UAVTalkConnectionData *connection)
756 {
757  int32_t ret = 0;
758 
759  UAVTalkInputProcessor *iproc = &connection->iproc;
760 
761  uint8_t type = iproc->type;
762  uint32_t objId = iproc->objId;
763  uint16_t instId = iproc->instId;
764 
765  UAVObjHandle obj = iproc->obj;
766 
767  /* Handle ACK/NACK --- don't bother to look up IDs etc for these
768  * because we don't need it.
769  *
770  * Treat NACKs and ACKs identically -- ground has done all it wants
771  * with them. This incurs a small penalty from calling the callback
772  * when GCS is a mismatched version and likes NACKing us, but ensures
773  * that we don't start blocking the telemetry session for an ACK that
774  * will never come.
775  */
776  if ((type == UAVTALK_TYPE_NACK) || (type == UAVTALK_TYPE_ACK)) {
777  if (connection->ackCb) {
778  connection->ackCb(connection->cbCtx, objId, instId);
779  }
780 
781  return 0;
782  }
783 
784  /* File request data is a special case. */
785  if (type == UAVTALK_TYPE_FILEREQ) {
786  handleFileReq(connection);
787 
788  return 0;
789  }
790 
791  PIOS_Recursive_Mutex_Lock(connection->lock, PIOS_MUTEX_TIMEOUT_MAX);
792 
793  // Process message type
794  switch (type) {
795  case UAVTALK_TYPE_OBJ:
796  // All instances, not allowed for OBJ messages
797  if (obj && (instId != UAVOBJ_ALL_INSTANCES)) {
798  // Unpack object, if the instance does not exist it will be created!
799  UAVObjUnpack(obj, instId, connection->rxBuffer);
800  } else {
801  ret = -1;
802  }
803  break;
804  case UAVTALK_TYPE_OBJ_ACK:
805  // All instances, not allowed for OBJ_ACK messages
806  if (obj && (instId != UAVOBJ_ALL_INSTANCES)) {
807  // Unpack object, if the instance does not exist it will be created!
808  if (UAVObjUnpack(obj, instId, connection->rxBuffer) == 0) {
809  // Transmit ACK
810  sendObject(connection, obj, instId, UAVTALK_TYPE_ACK);
811  } else {
812  ret = -1;
813  }
814  } else {
815  // We don't know this object, and we complain about it
816  sendNack(connection, objId);
817  ret = -1;
818  }
819  break;
820  case UAVTALK_TYPE_OBJ_REQ:
821  // Send requested object if message is of type OBJ_REQ
822  if (obj == 0)
823  sendNack(connection, objId);
824  else
825  sendObject(connection, obj, instId, UAVTALK_TYPE_OBJ);
826  break;
827  default:
828  ret = -1;
829  }
830 
831  PIOS_Recursive_Mutex_Unlock(connection->lock);
832 
833  // Done
834  return ret;
835 }

846 static int32_t sendObject(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type)
847 {
848  uint32_t numInst;
849  uint32_t n;
850 
851  // If all instances are requested and this is a single instance object, force instance ID to zero
852  if (instId == UAVOBJ_ALL_INSTANCES && UAVObjIsSingleInstance(obj) ) {
853  instId = 0;
854  }
855 
856  // Process message type
857  // XXX some of these combinations don't make sense, like obj with ack
858  // requested for all instances [because if we implemented it properly
859  // it would be -exceptionally- costly]
860  if (type == UAVTALK_TYPE_OBJ || type == UAVTALK_TYPE_OBJ_TS ||
861  type == UAVTALK_TYPE_OBJ_ACK) {
862  if (instId == UAVOBJ_ALL_INSTANCES) {
863  // Get number of instances
864  numInst = UAVObjGetNumInstances(obj);
865  // Send all instances
866  for (n = 0; n < numInst; ++n) {
867  sendSingleObject(connection, obj, n, type);
868  }
869  return 0;
870  } else {
871  return sendSingleObject(connection, obj, instId, type);
872  }
873  } else if (type == UAVTALK_TYPE_OBJ_REQ) {
874  return sendSingleObject(connection, obj, instId, UAVTALK_TYPE_OBJ_REQ);
875  } else if (type == UAVTALK_TYPE_ACK) {
876  if (instId != UAVOBJ_ALL_INSTANCES) {
877  return sendSingleObject(connection, obj, instId, UAVTALK_TYPE_ACK);
878  } else {
879  return -1;
880  }
881  } else {
882  return -1;
883  }
884 }

895 static int32_t sendSingleObject(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type)
896 {
897  int32_t length;
898  int32_t dataOffset;
899  uint32_t objId;
900 
901  if (!connection->outCb) return -1;
902 
903  // Determine data length
904  if (type == UAVTALK_TYPE_OBJ_REQ || type == UAVTALK_TYPE_ACK) {
905  length = 0;
906  } else {
907  length = UAVObjGetNumBytes(obj);
908  }
909 
910  // Check length
911  if (length >= UAVTALK_MAX_PAYLOAD_LENGTH) {
912  return -1;
913  }
914 
915  // Setup type and object id fields
916  objId = UAVObjGetID(obj);
917 
918  PIOS_Recursive_Mutex_Lock(connection->lock, PIOS_MUTEX_TIMEOUT_MAX);
919 
920  connection->txBuffer[0] = UAVTALK_SYNC_VAL; // sync byte
921  connection->txBuffer[1] = type;
922  // data length inserted here below
923  connection->txBuffer[4] = (uint8_t)(objId & 0xFF);
924  connection->txBuffer[5] = (uint8_t)((objId >> 8) & 0xFF);
925  connection->txBuffer[6] = (uint8_t)((objId >> 16) & 0xFF);
926  connection->txBuffer[7] = (uint8_t)((objId >> 24) & 0xFF);
927 
928  // Setup instance ID if one is required
929  if (UAVObjIsSingleInstance(obj)) {
930  dataOffset = 8;
931  } else {
932  connection->txBuffer[8] = (uint8_t)(instId & 0xFF);
933  connection->txBuffer[9] = (uint8_t)((instId >> 8) & 0xFF);
934  dataOffset = 10;
935  }
936 
937  // Add timestamp when the transaction type is appropriate
938  if (type & UAVTALK_TIMESTAMPED) {
939  uint32_t time = PIOS_Thread_Systime();
940  connection->txBuffer[dataOffset] = (uint8_t)(time & 0xFF);
941  connection->txBuffer[dataOffset + 1] = (uint8_t)((time >> 8) & 0xFF);
942  dataOffset += 2;
943  }
944 
945  // Copy data (if any)
946  if (length > 0) {
947  if (UAVObjPack(obj, instId, &connection->txBuffer[dataOffset]) < 0) {
948  PIOS_Recursive_Mutex_Unlock(connection->lock);
949  return -1;
950  }
951  }
952 
953  // Store the packet length
954  connection->txBuffer[2] = (uint8_t)((dataOffset+length) & 0xFF);
955  connection->txBuffer[3] = (uint8_t)(((dataOffset+length) >> 8) & 0xFF);
956 
957  // Calculate checksum
958  connection->txBuffer[dataOffset+length] = PIOS_CRC_updateCRC(0, connection->txBuffer, dataOffset+length);
959 
960  uint16_t tx_msg_len = dataOffset+length+UAVTALK_CHECKSUM_LENGTH;
961  int32_t rc = (*connection->outCb)(connection->cbCtx, connection->txBuffer,
962  tx_msg_len);
963 
964  if (rc == tx_msg_len) {
965  // Update stats
966  ++connection->stats.txObjects;
967  connection->stats.txBytes += tx_msg_len;
968  connection->stats.txObjectBytes += length;
969  }
970 
971  // Done
972  PIOS_Recursive_Mutex_Unlock(connection->lock);
973  return 0;
974 }
