176 UAVTalkRxState UAVTalkProcessInputStreamQuiet(UAVTalkConnection connectionHandle, uint8_t rxbyte)
177 {
178  UAVTalkConnectionData *connection;
179  CHECKCONHANDLE(connectionHandle,connection,return -1);
180 
181  UAVTalkInputProcessor *iproc = &connection->iproc;
182  ++connection->stats.rxBytes;
183 
184  if (iproc->state == UAVTALK_STATE_ERROR || iproc->state == UAVTALK_STATE_COMPLETE)
185  iproc->state = UAVTALK_STATE_SYNC;
186 
187  if (iproc->rxPacketLength < 0xffff)
188  iproc->rxPacketLength++; // update packet byte count
189 
190  // Receive state machine
191  switch (iproc->state)
192  {
193  case UAVTALK_STATE_SYNC:
194  if (rxbyte != UAVTALK_SYNC_VAL)
195  break;
196 
197  // Initialize and update the CRC
198  iproc->cs = PIOS_CRC_updateByte(0, rxbyte);
199 
200  iproc->rxPacketLength = 1;
201 
202  iproc->state = UAVTALK_STATE_TYPE;
203  break;
204 
205  case UAVTALK_STATE_TYPE:
206 
207  // update the CRC
208  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
209 
210  if ((rxbyte & UAVTALK_TYPE_MASK) != UAVTALK_TYPE_VER) {
211  iproc->state = UAVTALK_STATE_ERROR;
212  break;
213  }
214 
215  iproc->type = rxbyte;
216 
217  iproc->packet_size = 0;
218 
219  iproc->state = UAVTALK_STATE_SIZE;
220  iproc->rxCount = 0;
221  break;
222 
223  case UAVTALK_STATE_SIZE:
224 
225  // update the CRC
226  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
227 
228  if (iproc->rxCount == 0) {
229  iproc->packet_size += rxbyte;
230  iproc->rxCount++;
231  break;
232  }
233 
234  iproc->packet_size += rxbyte << 8;
235 
236  if (iproc->packet_size < UAVTALK_MIN_HEADER_LENGTH ||
237  iproc->packet_size > UAVTALK_MAX_HEADER_LENGTH + UAVTALK_MAX_PAYLOAD_LENGTH) { // incorrect packet size
238  iproc->state = UAVTALK_STATE_ERROR;
239  break;
240  }
241 
242  iproc->rxCount = 0;
243  iproc->objId = 0;
244  iproc->state = UAVTALK_STATE_OBJID;
245  break;
246 
247  case UAVTALK_STATE_OBJID:
248 
249  // update the CRC
250  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
251 
252  iproc->objId += rxbyte << (8*(iproc->rxCount++));
253 
254  if (iproc->rxCount < 4)
255  break;
256 
257  // Search for object.
258  iproc->obj = UAVObjGetByID(iproc->objId);
259 
260  // Determine data length
261  if (iproc->type == UAVTALK_TYPE_OBJ_REQ || iproc->type == UAVTALK_TYPE_ACK || iproc->type == UAVTALK_TYPE_NACK) {
262  iproc->length = 0;
263  iproc->instanceLength = 0;
264  } else {
265  if (iproc->obj) {
266  iproc->length = UAVObjGetNumBytes(iproc->obj);
267  iproc->instanceLength = (UAVObjIsSingleInstance(iproc->obj) ? 0 : 2);
268  } else {
269  // We don't know if it's a multi-instance object, so just assume it's 0.
270  iproc->instanceLength = 0;
271  iproc->length = iproc->packet_size - iproc->rxPacketLength;
272  }
273  }
274 
275  // Check length and determine next state
276  if (iproc->length >= UAVTALK_MAX_PAYLOAD_LENGTH) {
277  connection->stats.rxErrors++;
278  iproc->state = UAVTALK_STATE_ERROR;
279  break;
280  }
281 
282  // Check the lengths match
283  if ((iproc->rxPacketLength + iproc->instanceLength + iproc->length) != iproc->packet_size) { // packet error - mismatched packet size
284  connection->stats.rxErrors++;
285  iproc->state = UAVTALK_STATE_ERROR;
286  break;
287  }
288 
289  iproc->instId = 0;
290  if (iproc->type == UAVTALK_TYPE_NACK) {
291  // If this is a NACK, we skip to Checksum
292  iproc->state = UAVTALK_STATE_CS;
293  }
294  // Check if this is a single instance object (i.e. if the instance ID field is coming next)
295  else if ((iproc->obj != 0) && !UAVObjIsSingleInstance(iproc->obj)) {
296  iproc->state = UAVTALK_STATE_INSTID;
297  } else {
298  // If there is a payload get it, otherwise receive checksum
299  if (iproc->length > 0)
300  iproc->state = UAVTALK_STATE_DATA;
301  else
302  iproc->state = UAVTALK_STATE_CS;
303  }
304  iproc->rxCount = 0;
305 
306  break;
307 
308  case UAVTALK_STATE_INSTID:
309 
310  // update the CRC
311  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
312 
313  iproc->instId += rxbyte << (8*(iproc->rxCount++));
314 
315  if (iproc->rxCount < 2)
316  break;
317 
318  iproc->rxCount = 0;
319 
320  // If there is a payload get it, otherwise receive checksum
321  if (iproc->length > 0)
322  iproc->state = UAVTALK_STATE_DATA;
323  else
324  iproc->state = UAVTALK_STATE_CS;
325 
326  break;
327 
328  case UAVTALK_STATE_DATA:
329 
330  // update the CRC
331  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
332 
333  connection->rxBuffer[iproc->rxCount++] = rxbyte;
334  if (iproc->rxCount < iproc->length)
335  break;
336 
337  iproc->state = UAVTALK_STATE_CS;
338  iproc->rxCount = 0;
339  break;
340 
341  case UAVTALK_STATE_CS:
342 
343  // the CRC byte
344  if (rxbyte != iproc->cs) { // packet error - faulty CRC
345  connection->stats.rxErrors++;
346  iproc->state = UAVTALK_STATE_ERROR;
347  break;
348  }
349 
350  if (iproc->rxPacketLength != (iproc->packet_size + 1)) { // packet error - mismatched packet size
351  connection->stats.rxErrors++;
352  iproc->state = UAVTALK_STATE_ERROR;
353  break;
354  }
355 
356  connection->stats.rxObjectBytes += iproc->length;
357  connection->stats.rxObjects++;
358 
359  iproc->state = UAVTALK_STATE_COMPLETE;
360  break;
361 
362  default:
363  connection->stats.rxErrors++;
364  iproc->state = UAVTALK_STATE_ERROR;
365  }
366 
367  // Done
368  return iproc->state;
369 }

405 int32_t UAVTalkRelayPacket(UAVTalkConnection inConnectionHandle, UAVTalkConnection outConnectionHandle)
406 {
407  UAVTalkConnectionData *inConnection;
408 
409  CHECKCONHANDLE(inConnectionHandle, inConnection, return -1);
410  UAVTalkInputProcessor *inIproc = &inConnection->iproc;
411 
412  // The input packet must be completely parsed.
413  if (inIproc->state != UAVTALK_STATE_COMPLETE) {
414  inConnection->stats.rxErrors++;
415 
416  return -1;
417  }
418 
419  UAVTalkConnectionData *outConnection;
420  CHECKCONHANDLE(outConnectionHandle, outConnection, return -1);
421 
422  if (!outConnection->outCb) {
423  outConnection->stats.txErrors++;
424 
425  return -1;
426  }
427 
428  // Lock
429  PIOS_Recursive_Mutex_Lock(outConnection->lock, PIOS_MUTEX_TIMEOUT_MAX);
430 
431  outConnection->txBuffer[0] = UAVTALK_SYNC_VAL;
432  // Setup type
433  outConnection->txBuffer[1] = inIproc->type;
434  // next 2 bytes are reserved for data length (inserted here later)
435  // Setup object ID
436  outConnection->txBuffer[4] = (uint8_t)(inIproc->objId & 0xFF);
437  outConnection->txBuffer[5] = (uint8_t)((inIproc->objId >> 8) & 0xFF);
438  outConnection->txBuffer[6] = (uint8_t)((inIproc->objId >> 16) & 0xFF);
439  outConnection->txBuffer[7] = (uint8_t)((inIproc->objId >> 24) & 0xFF);
440  int32_t headerLength = 8;
441 
442  if (inIproc->obj) {
443  if (!UAVObjIsSingleInstance(inIproc->obj)) {
444  // Setup instance ID
445  outConnection->txBuffer[8] = (uint8_t)(inIproc->instId & 0xFF);
446  outConnection->txBuffer[9] = (uint8_t)((inIproc->instId >> 8) & 0xFF);
447  headerLength = 10;
448  }
449  }
450 
451  // Copy data (if any)
452  if (inIproc->length > 0) {
453  memcpy(&outConnection->txBuffer[headerLength], inConnection->rxBuffer, inIproc->length);
454  }
455 
456  // Store the packet length
457  outConnection->txBuffer[2] = (uint8_t)((headerLength + inIproc->length) & 0xFF);
458  outConnection->txBuffer[3] = (uint8_t)(((headerLength + inIproc->length) >> 8) & 0xFF);
459 
460  // Copy the checksum
461  outConnection->txBuffer[headerLength + inIproc->length] = inIproc->cs;
462 
463  // Send the buffer.
464  int32_t rc = (*outConnection->outCb)(outConnection->cbCtx, outConnection->txBuffer, headerLength + inIproc->length + UAVTALK_CHECKSUM_LENGTH);
465 
466  // Update stats
467  outConnection->stats.txBytes += (rc > 0) ? rc : 0;
468 
469  // evaluate return value before releasing the lock
470  int32_t ret = 0;
471  if (rc != (int32_t)(headerLength + inIproc->length + UAVTALK_CHECKSUM_LENGTH)) {
472  outConnection->stats.txErrors++;
473  ret = -1;
474  }
475 
476  // Release lock
477  PIOS_Recursive_Mutex_Unlock(outConnection->lock);
478 
479  // Done
480  return ret;
481 }

630 static int32_t receiveObject(UAVTalkConnectionData *connection)
631 {
632  int32_t ret = 0;
633 
634  UAVTalkInputProcessor *iproc = &connection->iproc;
635 
636  uint8_t type = iproc->type;
637  uint32_t objId = iproc->objId;
638  uint16_t instId = iproc->instId;
639 
640  UAVObjHandle obj = iproc->obj;
641 
642  /* Handle ACK/NACK first --- don't bother to look up IDs etc for these
643  * because we don't need it.
644  *
645  * Treat NACKs and ACKs identically -- ground has done all it wants
646  * with them. This incurs a small penalty from calling the callback
647  * when GCS is a mismatched version and likes NACKing us, but ensures
648  * that we don't start blocking the telemetry session for an ACK that
649  * will never come.
650  */
651  if ((type == UAVTALK_TYPE_NACK) || (type == UAVTALK_TYPE_ACK)) {
652  if (connection->ackCb) {
653  connection->ackCb(connection->cbCtx, objId, instId);
654  }
655 
656  return 0;
657  }
658 
659  PIOS_Recursive_Mutex_Lock(connection->lock, PIOS_MUTEX_TIMEOUT_MAX);
660 
661  // Process message type
662  switch (type) {
663  case UAVTALK_TYPE_OBJ:
664  // All instances, not allowed for OBJ messages
665  if (obj && (instId != UAVOBJ_ALL_INSTANCES)) {
666  // Unpack object, if the instance does not exist it will be created!
667  UAVObjUnpack(obj, instId, connection->rxBuffer);
668  } else {
669  ret = -1;
670  }
671  break;
672  case UAVTALK_TYPE_OBJ_ACK:
673  // All instances, not allowed for OBJ_ACK messages
674  if (obj && (instId != UAVOBJ_ALL_INSTANCES)) {
675  // Unpack object, if the instance does not exist it will be created!
676  if (UAVObjUnpack(obj, instId, connection->rxBuffer) == 0) {
677  // Transmit ACK
678  sendObject(connection, obj, instId, UAVTALK_TYPE_ACK);
679  } else {
680  ret = -1;
681  }
682  } else {
683  // We don't know this object, and we complain about it
684  sendNack(connection, objId);
685  ret = -1;
686  }
687  break;
688  case UAVTALK_TYPE_OBJ_REQ:
689  // Send requested object if message is of type OBJ_REQ
690  if (obj == 0)
691  sendNack(connection, objId);
692  else
693  sendObject(connection, obj, instId, UAVTALK_TYPE_OBJ);
694  break;
695  default:
696  ret = -1;
697  }
698 
699  PIOS_Recursive_Mutex_Unlock(connection->lock);
700 
701  // Done
702  return ret;
703 }

714 static int32_t sendObject(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type)
715 {
716  uint32_t numInst;
717  uint32_t n;
718 
719  // If all instances are requested and this is a single instance object, force instance ID to zero
720  if (instId == UAVOBJ_ALL_INSTANCES && UAVObjIsSingleInstance(obj) ) {
721  instId = 0;
722  }
723 
724  // Process message type
725  // XXX some of these combinations don't make sense, like obj with ack
726  // requested for all instances [because if we implemented it properly
727  // it would be -exceptionally- costly]
728  if (type == UAVTALK_TYPE_OBJ || type == UAVTALK_TYPE_OBJ_TS ||
729  type == UAVTALK_TYPE_OBJ_ACK) {
730  if (instId == UAVOBJ_ALL_INSTANCES) {
731  // Get number of instances
732  numInst = UAVObjGetNumInstances(obj);
733  // Send all instances
734  for (n = 0; n < numInst; ++n) {
735  sendSingleObject(connection, obj, n, type);
736  }
737  return 0;
738  } else {
739  return sendSingleObject(connection, obj, instId, type);
740  }
741  } else if (type == UAVTALK_TYPE_OBJ_REQ) {
742  return sendSingleObject(connection, obj, instId, UAVTALK_TYPE_OBJ_REQ);
743  } else if (type == UAVTALK_TYPE_ACK) {
744  if (instId != UAVOBJ_ALL_INSTANCES) {
745  return sendSingleObject(connection, obj, instId, UAVTALK_TYPE_ACK);
746  } else {
747  return -1;
748  }
749  } else {
750  return -1;
751  }
752 }

763 static int32_t sendSingleObject(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type)
764 {
765  int32_t length;
766  int32_t dataOffset;
767  uint32_t objId;
768 
769  if (!connection->outCb) return -1;
770 
771  // Determine data length
772  if (type == UAVTALK_TYPE_OBJ_REQ || type == UAVTALK_TYPE_ACK) {
773  length = 0;
774  } else {
775  length = UAVObjGetNumBytes(obj);
776  }
777 
778  // Check length
779  if (length >= UAVTALK_MAX_PAYLOAD_LENGTH) {
780  return -1;
781  }
782 
783  // Setup type and object id fields
784  objId = UAVObjGetID(obj);
785 
786  PIOS_Recursive_Mutex_Lock(connection->lock, PIOS_MUTEX_TIMEOUT_MAX);
787 
788  connection->txBuffer[0] = UAVTALK_SYNC_VAL; // sync byte
789  connection->txBuffer[1] = type;
790  // data length inserted here below
791  connection->txBuffer[4] = (uint8_t)(objId & 0xFF);
792  connection->txBuffer[5] = (uint8_t)((objId >> 8) & 0xFF);
793  connection->txBuffer[6] = (uint8_t)((objId >> 16) & 0xFF);
794  connection->txBuffer[7] = (uint8_t)((objId >> 24) & 0xFF);
795 
796  // Setup instance ID if one is required
797  if (UAVObjIsSingleInstance(obj)) {
798  dataOffset = 8;
799  } else {
