241 static int32_t objectTransaction(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type, int32_t timeoutMs)
242 {
243  bool respReceived;
244 
245  // Send object depending on if a response is needed
246  if (type == UAVTALK_TYPE_OBJ_ACK || type == UAVTALK_TYPE_OBJ_ACK_TS || type == UAVTALK_TYPE_OBJ_REQ) {
247  // Get transaction lock (will block if a transaction is pending)
248  PIOS_Recursive_Mutex_Lock(connection->transLock, PIOS_MUTEX_TIMEOUT_MAX);
249  // Send object
250  PIOS_Recursive_Mutex_Lock(connection->lock, PIOS_MUTEX_TIMEOUT_MAX);
251  connection->respObj = obj;
252  connection->respInstId = instId;
253  sendObject(connection, obj, instId, type);
254  PIOS_Recursive_Mutex_Unlock(connection->lock);
255  // Wait for response (or timeout)
256  respReceived = PIOS_Semaphore_Take(connection->respSema, timeoutMs);
257  // Check if a response was received
258  if (respReceived == false) {
259  // Cancel transaction
260  PIOS_Recursive_Mutex_Lock(connection->lock, PIOS_MUTEX_TIMEOUT_MAX);
261  PIOS_Semaphore_Take(connection->respSema, 0); // non blocking call to make sure the value is reset to zero (binary sema)
262  connection->respObj = 0;
263  PIOS_Recursive_Mutex_Unlock(connection->lock);
264  PIOS_Recursive_Mutex_Unlock(connection->transLock);
265  return -1;
266  } else {
267  PIOS_Recursive_Mutex_Unlock(connection->transLock);
268  return 0;
269  }
270  } else if (type == UAVTALK_TYPE_OBJ || type == UAVTALK_TYPE_OBJ_TS) {
271  PIOS_Recursive_Mutex_Lock(connection->lock, PIOS_MUTEX_TIMEOUT_MAX);
272  sendObject(connection, obj, instId, type);
273  PIOS_Recursive_Mutex_Unlock(connection->lock);
274  return 0;
275  } else {
276  return -1;
277  }
278 }

286 UAVTalkRxState UAVTalkProcessInputStreamQuiet(UAVTalkConnection connectionHandle, uint8_t rxbyte)
287 {
288  UAVTalkConnectionData *connection;
289  CHECKCONHANDLE(connectionHandle,connection,return -1);
290 
291  UAVTalkInputProcessor *iproc = &connection->iproc;
292  ++connection->stats.rxBytes;
293 
294  if (iproc->state == UAVTALK_STATE_ERROR || iproc->state == UAVTALK_STATE_COMPLETE)
295  iproc->state = UAVTALK_STATE_SYNC;
296 
297  if (iproc->rxPacketLength < 0xffff)
298  iproc->rxPacketLength++; // update packet byte count
299 
300  // Receive state machine
301  switch (iproc->state)
302  {
303  case UAVTALK_STATE_SYNC:
304  if (rxbyte != UAVTALK_SYNC_VAL)
305  break;
306 
307  // Initialize and update the CRC
308  iproc->cs = PIOS_CRC_updateByte(0, rxbyte);
309 
310  iproc->rxPacketLength = 1;
311 
312  iproc->state = UAVTALK_STATE_TYPE;
313  break;
314 
315  case UAVTALK_STATE_TYPE:
316 
317  // update the CRC
318  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
319 
320  if ((rxbyte & UAVTALK_TYPE_MASK) != UAVTALK_TYPE_VER) {
321  iproc->state = UAVTALK_STATE_ERROR;
322  break;
323  }
324 
325  iproc->type = rxbyte;
326 
327  iproc->packet_size = 0;
328 
329  iproc->state = UAVTALK_STATE_SIZE;
330  iproc->rxCount = 0;
331  break;
332 
333  case UAVTALK_STATE_SIZE:
334 
335  // update the CRC
336  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
337 
338  if (iproc->rxCount == 0) {
339  iproc->packet_size += rxbyte;
340  iproc->rxCount++;
341  break;
342  }
343 
344  iproc->packet_size += rxbyte << 8;
345 
346  if (iproc->packet_size < UAVTALK_MIN_HEADER_LENGTH ||
347  iproc->packet_size > UAVTALK_MAX_HEADER_LENGTH + UAVTALK_MAX_PAYLOAD_LENGTH) { // incorrect packet size
348  iproc->state = UAVTALK_STATE_ERROR;
349  break;
350  }
351 
352  iproc->rxCount = 0;
353  iproc->objId = 0;
354  iproc->state = UAVTALK_STATE_OBJID;
355  break;
356 
357  case UAVTALK_STATE_OBJID:
358 
359  // update the CRC
360  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
361 
362  iproc->objId += rxbyte << (8*(iproc->rxCount++));
363 
364  if (iproc->rxCount < 4)
365  break;
366 
367  // Search for object.
368  iproc->obj = UAVObjGetByID(iproc->objId);
369 
370  // Determine data length
371  if (iproc->type == UAVTALK_TYPE_OBJ_REQ || iproc->type == UAVTALK_TYPE_ACK || iproc->type == UAVTALK_TYPE_NACK) {
372  iproc->length = 0;
373  iproc->instanceLength = 0;
374  } else {
375  if (iproc->obj) {
376  iproc->length = UAVObjGetNumBytes(iproc->obj);
377  iproc->instanceLength = (UAVObjIsSingleInstance(iproc->obj) ? 0 : 2);
378  iproc->timestampLength = (iproc->type & UAVTALK_TIMESTAMPED) ? 2 : 0;
379  } else {
380  // We don't know if it's a multi-instance object, so just assume it's 0.
381  iproc->instanceLength = 0;
382  iproc->length = iproc->packet_size - iproc->rxPacketLength;
383  }
384  }
385 
386  // Check length and determine next state
387  if (iproc->length >= UAVTALK_MAX_PAYLOAD_LENGTH) {
388  connection->stats.rxErrors++;
389  iproc->state = UAVTALK_STATE_ERROR;
390  break;
391  }
392 
393  // Check the lengths match
394  if ((iproc->rxPacketLength + iproc->instanceLength + iproc->timestampLength + iproc->length) != iproc->packet_size) { // packet error - mismatched packet size
395  connection->stats.rxErrors++;
396  iproc->state = UAVTALK_STATE_ERROR;
397  break;
398  }
399 
400  iproc->instId = 0;
401  if (iproc->type == UAVTALK_TYPE_NACK) {
402  // If this is a NACK, we skip to Checksum
403  iproc->state = UAVTALK_STATE_CS;
404  }
405  // Check if this is a single instance object (i.e. if the instance ID field is coming next)
406  else if ((iproc->obj != 0) && !UAVObjIsSingleInstance(iproc->obj)) {
407  iproc->state = UAVTALK_STATE_INSTID;
408  }
409  // Check if this is a single instance and has a timestamp in it
410  else if ((iproc->obj != 0) && (iproc->type & UAVTALK_TIMESTAMPED)) {
411  iproc->timestamp = 0;
412  iproc->state = UAVTALK_STATE_TIMESTAMP;
413  } else {
414  // If there is a payload get it, otherwise receive checksum
415  if (iproc->length > 0)
416  iproc->state = UAVTALK_STATE_DATA;
417  else
418  iproc->state = UAVTALK_STATE_CS;
419  }
420  iproc->rxCount = 0;
421 
422  break;
423 
424  case UAVTALK_STATE_INSTID:
425 
426  // update the CRC
427  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
428 
429  iproc->instId += rxbyte << (8*(iproc->rxCount++));
430 
431  if (iproc->rxCount < 2)
432  break;
433 
434  iproc->rxCount = 0;
435 
436  // If there is a timestamp, get it
437  if ((iproc->length > 0) && (iproc->type & UAVTALK_TIMESTAMPED)) {
438  iproc->timestamp = 0;
439  iproc->state = UAVTALK_STATE_TIMESTAMP;
440  }
441  // If there is a payload get it, otherwise receive checksum
442  else if (iproc->length > 0)
443  iproc->state = UAVTALK_STATE_DATA;
444  else
445  iproc->state = UAVTALK_STATE_CS;
446 
447  break;
448 
449  case UAVTALK_STATE_TIMESTAMP:
450  // update the CRC
451  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
452 
453  iproc->timestamp += rxbyte << (8*(iproc->rxCount++));
454 
455  if (iproc->rxCount < 2)
456  break;
457 
458  iproc->rxCount = 0;
459 
460  // If there is a payload get it, otherwise receive checksum
461  if (iproc->length > 0)
462  iproc->state = UAVTALK_STATE_DATA;
463  else
464  iproc->state = UAVTALK_STATE_CS;
465  break;
466 
467  case UAVTALK_STATE_DATA:
468 
469  // update the CRC
470  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
471 
472  connection->rxBuffer[iproc->rxCount++] = rxbyte;
473  if (iproc->rxCount < iproc->length)
474  break;
475 
476  iproc->state = UAVTALK_STATE_CS;
477  iproc->rxCount = 0;
478  break;
479 
480  case UAVTALK_STATE_CS:
481 
482  // the CRC byte
483  if (rxbyte != iproc->cs) { // packet error - faulty CRC
484  connection->stats.rxErrors++;
485  iproc->state = UAVTALK_STATE_ERROR;
486  break;
487  }
488 
489  if (iproc->rxPacketLength != (iproc->packet_size + 1)) { // packet error - mismatched packet size
490  connection->stats.rxErrors++;
491  iproc->state = UAVTALK_STATE_ERROR;
492  break;
493  }
494 
495  connection->stats.rxObjectBytes += iproc->length;
496  connection->stats.rxObjects++;
497 
498  iproc->state = UAVTALK_STATE_COMPLETE;
499  break;
500 
501  default:
502  connection->stats.rxErrors++;
503  iproc->state = UAVTALK_STATE_ERROR;
504  }
505 
506  // Done
507  return iproc->state;
508 }

543 int32_t UAVTalkRelayPacket(UAVTalkConnection inConnectionHandle, UAVTalkConnection outConnectionHandle)
544 {
545  UAVTalkConnectionData *inConnection;
546 
547  CHECKCONHANDLE(inConnectionHandle, inConnection, return -1);
548  UAVTalkInputProcessor *inIproc = &inConnection->iproc;
549 
550  // The input packet must be completely parsed.
551  if (inIproc->state != UAVTALK_STATE_COMPLETE) {
552  inConnection->stats.rxErrors++;
553 
554  return -1;
555  }
556 
557  UAVTalkConnectionData *outConnection;
558  CHECKCONHANDLE(outConnectionHandle, outConnection, return -1);
559 
560  if (!outConnection->outStream) {
561  outConnection->stats.txErrors++;
562 
563  return -1;
564  }
565 
566  // Lock
567  PIOS_Recursive_Mutex_Lock(outConnection->lock, PIOS_MUTEX_TIMEOUT_MAX);
568 
569  outConnection->txBuffer[0] = UAVTALK_SYNC_VAL;
570  // Setup type
571  outConnection->txBuffer[1] = inIproc->type;
572  // next 2 bytes are reserved for data length (inserted here later)
573  // Setup object ID
574  outConnection->txBuffer[4] = (uint8_t)(inIproc->objId & 0xFF);
575  outConnection->txBuffer[5] = (uint8_t)((inIproc->objId >> 8) & 0xFF);
576  outConnection->txBuffer[6] = (uint8_t)((inIproc->objId >> 16) & 0xFF);
577  outConnection->txBuffer[7] = (uint8_t)((inIproc->objId >> 24) & 0xFF);
578  int32_t headerLength = 8;
579 
580  if (inIproc->obj) {
581  if (!UAVObjIsSingleInstance(inIproc->obj)) {
582  // Setup instance ID
583  outConnection->txBuffer[8] = (uint8_t)(inIproc->instId & 0xFF);
584  outConnection->txBuffer[9] = (uint8_t)((inIproc->instId >> 8) & 0xFF);
585  headerLength = 10;
586  }
587  }
588 
589  // Add timestamp when the transaction type is appropriate
590  if (inIproc->type & UAVTALK_TIMESTAMPED) {
591  uint32_t time = PIOS_Thread_Systime();
592  outConnection->txBuffer[10] = (uint8_t)(time & 0xFF);
593  outConnection->txBuffer[11] = (uint8_t)((time >> 8) & 0xFF);
594  headerLength += 2;
595  }
596 
597  // Copy data (if any)
598  if (inIproc->length > 0) {
599  memcpy(&outConnection->txBuffer[headerLength], inConnection->rxBuffer, inIproc->length);
600  }
601 
602  // Store the packet length
603  outConnection->txBuffer[2] = (uint8_t)((headerLength + inIproc->length) & 0xFF);
604  outConnection->txBuffer[3] = (uint8_t)(((headerLength + inIproc->length) >> 8) & 0xFF);
605 
606  // Copy the checksum
607  outConnection->txBuffer[headerLength + inIproc->length] = inIproc->cs;
608 
609  // Send the buffer.
610  int32_t rc = (*outConnection->outStream)(outConnection->txBuffer, headerLength + inIproc->length + UAVTALK_CHECKSUM_LENGTH);
611 
612  // Update stats
613  outConnection->stats.txBytes += (rc > 0) ? rc : 0;
614 
615  // evaluate return value before releasing the lock
616  int32_t ret = 0;
617  if (rc != (int32_t)(headerLength + inIproc->length + UAVTALK_CHECKSUM_LENGTH)) {
618  outConnection->stats.txErrors++;
619  ret = -1;
620  }
621 
622  // Release lock
623  PIOS_Recursive_Mutex_Unlock(outConnection->lock);
624 
625  // Done
626  return ret;
627 }

830 static int32_t receiveObject(UAVTalkConnectionData *connection, uint8_t type, uint32_t objId, uint16_t instId, uint8_t* data, int32_t length)
831 {
832  UAVObjHandle obj;
833  int32_t ret = 0;
834 
835  // Get the handle to the Object. Will be zero
836  // if object does not exist.
837  obj = UAVObjGetByID(objId);
838 
839  // Process message type
840  switch (type) {
841  case UAVTALK_TYPE_OBJ:
842  case UAVTALK_TYPE_OBJ_TS:
843  // All instances, not allowed for OBJ messages
844  if (obj && (instId != UAVOBJ_ALL_INSTANCES)) {
845  // Unpack object, if the instance does not exist it will be created!
846  UAVObjUnpack(obj, instId, data);
847  // Check if an ack is pending
848  updateAck(connection, obj, instId);
849  } else {
850  ret = -1;
851  }
852  break;
853  case UAVTALK_TYPE_OBJ_ACK:
854  case UAVTALK_TYPE_OBJ_ACK_TS:
855  // All instances, not allowed for OBJ_ACK messages
856  if (obj && (instId != UAVOBJ_ALL_INSTANCES)) {
857  // Unpack object, if the instance does not exist it will be created!
858  if (UAVObjUnpack(obj, instId, data) == 0) {
859  // Transmit ACK
860  sendObject(connection, obj, instId, UAVTALK_TYPE_ACK);
861  } else {
862  ret = -1;
863  }
864  } else {
865 
866  // We don't know this object, and we complain about it
867  sendNack(connection, objId);
868  ret = -1;
869  }
870  break;
871  case UAVTALK_TYPE_OBJ_REQ:
872  // Send requested object if message is of type OBJ_REQ
873  if (obj == 0)
874  sendNack(connection, objId);
875  else
876  sendObject(connection, obj, instId, UAVTALK_TYPE_OBJ);
877  break;
878  case UAVTALK_TYPE_NACK:
879  // Do nothing on flight side, let it time out.
880  break;
881  case UAVTALK_TYPE_ACK:
882  // All instances, not allowed for ACK messages
883  if (obj && (instId != UAVOBJ_ALL_INSTANCES)) {
884  // Check if an ack is pending
885  updateAck(connection, obj, instId);
886  } else {
887  ret = -1;
888  }
889  break;
890  default:
891  ret = -1;
892  }
893  // Done
894  return ret;
895 }

965 static int32_t sendSingleObject(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type)
966 {
967  int32_t length;
968  int32_t dataOffset;
969  uint32_t objId;
970 
971  if (!connection->outStream) return -1;
972 
973  // Setup type and object id fields
974  objId = UAVObjGetID(obj);
975  connection->txBuffer[0] = UAVTALK_SYNC_VAL; // sync byte
976  connection->txBuffer[1] = type;
977  // data length inserted here below
978  connection->txBuffer[4] = (uint8_t)(objId & 0xFF);
979  connection->txBuffer[5] = (uint8_t)((objId >> 8) & 0xFF);
980  connection->txBuffer[6] = (uint8_t)((objId >> 16) & 0xFF);
981  connection->txBuffer[7] = (uint8_t)((objId >> 24) & 0xFF);
982 
983  // Setup instance ID if one is required
984  if (UAVObjIsSingleInstance(obj)) {
985  dataOffset = 8;
986  } else {
987  connection->txBuffer[8] = (uint8_t)(instId & 0xFF);
988  connection->txBuffer[9] = (uint8_t)((instId >> 8) & 0xFF);
989  dataOffset = 10;
990  }
991 
992  // Add timestamp when the transaction type is appropriate
993  if (type & UAVTALK_TIMESTAMPED) {
994  uint32_t time = PIOS_Thread_Systime();
995  connection->txBuffer[dataOffset] = (uint8_t)(time & 0xFF);
996  connection->txBuffer[dataOffset + 1] = (uint8_t)((time >> 8) & 0xFF);
997  dataOffset += 2;
998  }
999 
1000  // Determine data length
1001  if (type == UAVTALK_TYPE_OBJ_REQ || type == UAVTALK_TYPE_ACK) {
1002  length = 0;
1003  } else {
1004  length = UAVObjGetNumBytes(obj);
1005  }
1006 
1007  // Check length
1008  if (length >= UAVTALK_MAX_PAYLOAD_LENGTH) {
1009  return -1;
1010  }
1011 
1012  // Copy data (if any)
1013  if (length > 0) {
1014  if (UAVObjPack(obj, instId, &connection->txBuffer[dataOffset]) < 0) {
1015  return -1;
1016  }
1017  }
1018 
1019  // Store the packet length
1020  connection->txBuffer[2] = (uint8_t)((dataOffset+length) & 0xFF);
1021  connection->txBuffer[3] = (uint8_t)(((dataOffset+length) >> 8) & 0xFF);
1022 
1023  // Calculate checksum
1024  connection->txBuffer[dataOffset+length] = PIOS_CRC_updateCRC(0, connection->txBuffer, dataOffset+length);
1025 
1026  uint16_t tx_msg_len = dataOffset+length+UAVTALK_CHECKSUM_LENGTH;
1027  int32_t rc = (*connection->outStream)(connection->txBuffer, tx_msg_len);
1028 
1029  if (rc == tx_msg_len) {
1030  // Update stats
1031  ++connection->stats.txObjects;
1032  connection->stats.txBytes += tx_msg_len;
1033  connection->stats.txObjectBytes += length;
1034  }
1035 
1036  // Done
1037  return 0;
1038 }
