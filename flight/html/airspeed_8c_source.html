203 static void airspeedTask(void *parameters)
204 {
205  BaroAirspeedData airspeedData;
206  AirspeedActualData airspeedActualData;
207 
208 #ifdef BARO_AIRSPEED_PRESENT
209  uint32_t lastGPSTime = PIOS_Thread_Systime(); //Time since last GPS-derived airspeed calculation
210  uint32_t lastLoopTime= PIOS_Thread_Systime(); //Time since last loop
211 
212  float airspeedErrInt=0;
213 #endif
214 
215  //GPS airspeed calculation variables
216 #ifdef GPS_AIRSPEED_PRESENT
217  GPSVelocityConnectCallbackCtx(UAVObjCbSetFlag, &gpsNew);
218  if (gps_airspeedInitialize() == -1) {
219  module_enabled = false;
220  return;
221  }
222 #endif
223 
224  AirspeedSettingsConnectCallbackCtx(UAVObjCbSetFlag, &settingsUpdated);
225 
226  // Main task loop
227  uint32_t lastSysTime = PIOS_Thread_Systime();
228 
229  // Get initial values of airspeed object
230  BaroAirspeedGet(&airspeedData);
231 
232  airspeedData.BaroConnected = BAROAIRSPEED_BAROCONNECTED_FALSE;
233 
234  while (1)
235  {
236  if (settingsUpdated) {
237  settingsUpdated = false;
238 
239  doSettingsUpdate();
240  }
241 
242 #ifdef BARO_AIRSPEED_PRESENT
243  float airspeed_tas_baro=0;
244 
245  if(airspeedSensorType != AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_GPSONLY) {
246  //Fetch calibrated airspeed from sensors
247  baro_airspeedGet(&airspeedData, &lastSysTime, airspeedSensorType, airspeedADCPin);
248 
249  //Calculate true airspeed, not taking into account compressibility effects
250  int16_t groundTemperature_10;
251  float groundTemperature;
252  float positionActual_Down;
253 
254  PositionActualDownGet(&positionActual_Down);
255  HomeLocationGroundTemperatureGet(&groundTemperature_10); // Gets tenths of degrees C
256  groundTemperature = groundTemperature_10/10; // Convert into degrees C
257  groundTemperature -= BARO_TEMPERATURE_OFFSET; //Do this just because we suspect that the board heats up relative to its surroundings. THIS IS BAD(tm)
258 
259  struct AirParameters air_STP = initialize_air_structure();
260  air_STP.air_temperature_at_surface = groundTemperature + CELSIUS2KELVIN;
261 
262  #ifdef GPS_AIRSPEED_PRESENT
263  //GPS present, so use baro sensor to filter TAS
264  airspeed_tas_baro = cas2tas(airspeedData.CalibratedAirspeed, -positionActual_Down, &air_STP) + airspeedErrInt * GPS_AIRSPEED_BIAS_KI;
265  #else
266  //No GPS, so TAS comes only from baro sensor
267  airspeedData.TrueAirspeed = cas2tas(airspeedData.CalibratedAirspeed, -positionActual_Down, &air_STP) + airspeedErrInt * GPS_AIRSPEED_BIAS_KI;
268  #endif
269  }
270  else
271 #endif
272  { //Have to catch the fallthrough, or else this loop will monopolize the processor!
273  airspeedData.BaroConnected=BAROAIRSPEED_BAROCONNECTED_FALSE;
274  airspeedData.SensorValue=12345;
275 
276  //Likely, we have a GPS, so let's configure the fallthrough at close to GPS refresh rates
277  PIOS_Thread_Sleep_Until(&lastSysTime, SAMPLING_DELAY_MS_FALLTHROUGH);
278  }
279 
280 #ifdef GPS_AIRSPEED_PRESENT
281  float v_air_GPS=-1.0f;
282 
283  //Check if sufficient time has passed. This will depend on whether we have a pitot tube
284  //sensor or not. In the case we do, shoot for about once per second. Otherwise, consume GPS
285  //as quickly as possible.
286  #ifdef BARO_AIRSPEED_PRESENT
287  float delT = (lastSysTime - lastLoopTime) / 1000.0f;
288  lastLoopTime=lastSysTime;
289  if ( ((lastSysTime - lastGPSTime) > 1000 || airspeedSensorType==AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_GPSONLY)
290  && gpsNew) {
291  lastGPSTime=lastSysTime;
292  #else
293  if (gpsNew) {
294  #endif
295  gpsNew=false; //Do this first
296 
297  //Calculate airspeed as a function of GPS groundspeed and vehicle attitude. From "IMU Wind Estimation (Theory)", by William Premerlani
298  gps_airspeedGet(&v_air_GPS);
299  }
300 
301 
302  //Use the GPS error to correct the airspeed estimate.
303  if (v_air_GPS > 0) //We have valid GPS estimate...
304  {
305  airspeedData.GPSAirspeed=v_air_GPS;
306 
307  #ifdef BARO_AIRSPEED_PRESENT
308  if(airspeedData.BaroConnected==BAROAIRSPEED_BAROCONNECTED_TRUE){ //Check if there is an airspeed sensors present...
309  //Calculate error and error integral
310  float airspeedErr=v_air_GPS - airspeed_tas_baro;
311  airspeedErrInt+=airspeedErr * delT;
312 
313  //Saturate integral component at 5 m/s
314  airspeedErrInt = airspeedErrInt > (5.0f / GPS_AIRSPEED_BIAS_KI) ? (5.0f / GPS_AIRSPEED_BIAS_KI) : airspeedErrInt;
315  airspeedErrInt = airspeedErrInt < -(5.0f / GPS_AIRSPEED_BIAS_KI) ? -(5.0f / GPS_AIRSPEED_BIAS_KI) : airspeedErrInt;
316 
317  //There's already an airspeed sensor, so instead correct it for bias with P correction. The I correction happened earlier in the function.
318  airspeedData.TrueAirspeed = airspeed_tas_baro + airspeedErr * GPS_AIRSPEED_BIAS_KP;
319 
320 
321  /* Note:
322  This would be a good place to change the airspeed calibration, so that it matches the GPS computed values. However,
323  this might be a bad idea, as their are two degrees of freedom here: temperature and sensor calibration. I don't
324  know how to control for temperature bias.
325  */
326  }
327  else
328  #endif
329  {
330  //...there's no airspeed sensor, so everything comes from GPS. In this
331  //case, filter the airspeed for smoother output
332  float alpha=gpsSamplePeriod_ms/(gpsSamplePeriod_ms + GPS_AIRSPEED_TIME_CONSTANT_MS); //Low pass filter.
333  airspeedData.TrueAirspeed=v_air_GPS*(alpha) + airspeedData.TrueAirspeed*(1.0f-alpha);
334 
335  //Calculate calibrated airspeed from GPS, since we're not getting it from a discrete airspeed sensor
336  int16_t groundTemperature_10;
337  float groundTemperature;
338  float positionActual_Down;
339 
340  PositionActualDownGet(&positionActual_Down);
341  HomeLocationGroundTemperatureGet(&groundTemperature_10); // Gets tenths of degrees C
342  groundTemperature = groundTemperature_10/10; // Convert into degrees C
343  groundTemperature -= BARO_TEMPERATURE_OFFSET; //Do this just because we suspect that the board heats up relative to its surroundings. THIS IS BAD(tm)
344 
345  struct AirParameters air_STP = initialize_air_structure();
346  air_STP.air_temperature_at_surface = groundTemperature + CELSIUS2KELVIN;
347 
348  airspeedData.CalibratedAirspeed = tas2cas(airspeedData.TrueAirspeed, -positionActual_Down, &air_STP);
349  }
350  }
351  #ifdef BARO_AIRSPEED_PRESENT
352  else if (airspeedData.BaroConnected==BAROAIRSPEED_BAROCONNECTED_TRUE){
353  //No GPS velocity estimate this loop, so filter true airspeed data with baro airspeed
354  float alpha=delT/(delT + BARO_TRUEAIRSPEED_TIME_CONSTANT_S); //Low pass filter.
355  airspeedData.TrueAirspeed=airspeed_tas_baro*(alpha) + airspeedData.TrueAirspeed*(1.0f-alpha);
356  }
357  #endif
358 #endif
359  //Set the UAVO
360  airspeedActualData.TrueAirspeed = airspeedData.TrueAirspeed;
361  airspeedActualData.CalibratedAirspeed = airspeedData.CalibratedAirspeed;
362  BaroAirspeedSet(&airspeedData);
363  AirspeedActualSet(&airspeedActualData);
364  }
365 }
