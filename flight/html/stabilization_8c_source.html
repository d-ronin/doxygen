218 void stabilization_failsafe_checks(StabilizationDesiredData *stab_desired,
219  ActuatorDesiredData *actuator_desired,
220  SystemSettingsAirframeTypeOptions airframe_type,
221  float *input, uint8_t *mode)
222 {
223  float *rate = &stab_desired->Roll;
224  for(int i = 0; i < MAX_AXES; i++)
225  {
226  mode[i] = stab_desired->StabilizationMode[i];
227  input[i] = rate[i];
228 
229  if (mode[i] == STABILIZATIONDESIRED_STABILIZATIONMODE_FAILSAFE) {
230  // Everything except planes should drop straight down
231  if ((airframe_type != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWING) &&
232  (airframe_type != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWINGELEVON) &&
233  (airframe_type != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWINGVTAIL)) {
234  actuator_desired->Thrust = 0.0f;
235  switch (i) {
236  case 0: /* Roll */
237  mode[i] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
238  input[i] = 0;
239  break;
240  case 1:
241  default:
242  mode[i] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
243  input[i] = 0;
244  break;
245  case 2:
246  mode[i] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
247  input[i] = 0;
248  break;
249  }
250  }
251  else
252  {
253  actuator_desired->Thrust = -1.0f;
254 
255  switch (i) {
256  case 0: /* Roll */
257  mode[i] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
258  input[i] = -10;
259  break;
260  case 1:
261  default:
262  mode[i] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
263  input[i] = 0;
264  break;
265  case 2:
266  mode[i] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
267  input[i] = -5;
268  break;
269  }
270  }
271  }
272  }

295  float *rateDesiredAxis = &rateDesired.Roll;
296  float horizonRateFraction = 0.0f;
297 
298  smoothcontrol_initialize(&rc_smoothing);
299 
300  // Connect callbacks
301  FlightStatusConnectCallbackCtx(UAVObjCbSetFlag, &flightStatusUpdated);
302  SystemSettingsConnectCallbackCtx(UAVObjCbSetFlag, &systemSettingsUpdated);
303  StabilizationSettingsConnectCallbackCtx(UAVObjCbSetFlag, &settings_flag);
304  VbarSettingsConnectCallbackCtx(UAVObjCbSetFlag, &settings_flag);
305  SubTrimSettingsConnectCallbackCtx(UAVObjCbSetFlag, &settings_flag);

307  smoothcontrol_initialize(&rc_smoothing);
308  ManualControlCommandConnectCallbackCtx(UAVObjCbSetFlag, smoothcontrol_get_ringer(rc_smoothing));
309 
310  uint32_t iteration = 0;
311  float dT_measured = 0;
312 
313  uint8_t ident_shift = 5;
314 
315  float dT_expected = 0.001; // assume 1KHz if we don't know.
316 
317  uint16_t samp_rate = PIOS_SENSORS_GetSampleRate(PIOS_SENSOR_GYRO);
318 
319  if (samp_rate) {
320  dT_expected = 1.0f / samp_rate;
321  }
322 
323  smoothcontrol_update_dT(rc_smoothing, dT_expected);
324 
325  if (dT_expected < 0.0004f) {
326  // For future 3.2KHz-- 640ms period
327  ident_shift = 8;
328  } else if (dT_expected < 0.0008f) {
329  // 2KHz - 512ms period
330  // 1.6KHz - 640ms period
331  ident_shift = 7;
332  } else if (dT_expected < 0.0014f) {
333  // 1.2KHz - 427ms
334  // 1KHz - 512ms period
335  // 800Hz - 640ms period
336  // 750Hz - 683ms period
337  ident_shift = 6;
338  } else {
339  // 700Hz - 365ms period
340  // 500Hz - 512ms period
341  // 333Hz - 768ms period
342  ident_shift = 5;
343  }
344 
345  ident_wiggle_points = (1 << (ident_shift + 3));
346  uint32_t ident_mask = ident_wiggle_points - 1;
347 
348  zero_pids();
349 
350  // Main task loop
351  while(1) {
352  iteration++;
353 
354  PIOS_WDG_UpdateFlag(PIOS_WDG_STABILIZATION);
355 
356  if (settings_flag) {
357  update_settings(dT_expected);
358 
359  // Default 350ms.
360  // 175ms to 39.3% of response
361  // 350ms to 63.2% of response
362  // 700ms to 86.4% of response
363  max_rate_alpha = expf(-dT_expected / settings.AcroDynamicTau);
364 
365  // Compute time constant for vbar decay term
366  if (vbar_settings.VbarTau < 0.001f) {
367  vbar_decay = 0;
368  } else {
369  vbar_decay = expf(-dT_expected / vbar_settings.VbarTau);
370  }
371 
372  settings_flag = false;
373  }
374 
375  // Wait until the AttitudeRaw object is updated, if a timeout then go to failsafe
376  if (PIOS_Queue_Receive(queue, &ev, FAILSAFE_TIMEOUT_MS) != true)
377  {
378  AlarmsSet(SYSTEMALARMS_ALARM_STABILIZATION,SYSTEMALARMS_ALARM_WARNING);
379  continue;
380  }
381 
382  static bool frequency_wrong = false;
383 
384  float dT = PIOS_DELAY_DiffuS(timeval) * 1.0e-6f;
385  timeval = PIOS_DELAY_GetRaw();
386 
387  if (iteration < 100) {
388  dT_measured = 0;
389  } else if (iteration < 2100) {
390  dT_measured += dT;
391  } else if (iteration == 2100) {
392  dT_measured /= 2000;
393 
394  /* Other modules-- attitude, etc, -- rely on us having
395  * done this test and set an alarm here. Do not remove
396  * without verifying those places
397  */
398  if ((dT_measured > dT_expected * 1.15f) ||
399  (dT_measured < dT_expected * 0.85f)) {
400  frequency_wrong = true;
401 #ifdef SIM_POSIX
402  printf("Stabilization: frequency wrong. dT_measured=%f, expected=%f\n", dT_measured, dT_expected);
403 #endif
404  }
405  }
406 
407  bool error = frequency_wrong;
408 
409  if (flightStatusUpdated) {
410  FlightStatusGet(&flightStatus);
411  flightStatusUpdated = false;
412  }
413 
414  if (systemSettingsUpdated) {
415  SystemSettingsAirframeTypeGet(&airframe_type);
416  systemSettingsUpdated = false;
417  }
418 
419  StabilizationDesiredGet(&stabDesired);
420  AttitudeActualGet(&attitudeActual);
421  GyrosGet(&gyrosData);
422 
423  actuatorDesired.Thrust = stabDesired.Thrust;
424 
425  // Re-project axes if necessary prior to running stabilization algorithms.
426  uint8_t reprojection = stabDesired.ReprojectionMode;
427  static uint8_t previous_reprojection = 255;
428 
429  if (reprojection == STABILIZATIONDESIRED_REPROJECTIONMODE_CAMERAANGLE) {
430  float camera_tilt_angle = settings.CameraTilt;
431  if (camera_tilt_angle) {
432  float roll = stabDesired.Roll;
433  float yaw = stabDesired.Yaw;
434  // The roll input should be the cosine of the camera angle multiplied by the roll,
435  // added to the sine of camera angle multiplied by yaw.
436  stabDesired.Roll = (cosf(DEG2RAD * camera_tilt_angle) * roll +
437  (sinf(DEG2RAD * camera_tilt_angle) * yaw));
438  // Yaw is similar but uses the negative sine of the camera angle, multiplied by roll,
439  // added to the cosine of the camera angle, times the yaw
440  stabDesired.Yaw = (-1 * sinf(DEG2RAD * camera_tilt_angle) * roll) +
441  (cosf(DEG2RAD * camera_tilt_angle) * yaw);
442  }
443  } else if (reprojection == STABILIZATIONDESIRED_REPROJECTIONMODE_HEADFREE) {
444  static float reference_yaw;
445 
446  if (previous_reprojection != reprojection) {
447  reference_yaw = attitudeActual.Yaw;
448  }
449 
450  float rotation_angle = attitudeActual.Yaw - reference_yaw;
451  float roll = stabDesired.Roll;
452  float pitch = stabDesired.Pitch;
453 
454  stabDesired.Roll = cosf(DEG2RAD * rotation_angle) * roll
455  + sinf(DEG2RAD * rotation_angle) * pitch;
456  stabDesired.Pitch = cosf(DEG2RAD * rotation_angle) * pitch
457  + sinf(DEG2RAD * rotation_angle) * -roll;
458  }
459 
460  previous_reprojection = reprojection;
461 
462 #if defined(RATEDESIRED_DIAGNOSTICS)
463  RateDesiredGet(&rateDesired);
464 #endif
465  // raw_input will contain desired stabilization or the failsafe overrides.
466  float raw_input[MAX_AXES];
467  uint8_t axis_mode[MAX_AXES];
468  stabilization_failsafe_checks(&stabDesired, &actuatorDesired, airframe_type,
469  raw_input, axis_mode);
470 
471  // Do this before attitude error calc, so it benefits from it.
472  for(int i = 0; i < 3; i++)
473  smoothcontrol_run(rc_smoothing, i, &raw_input[i], settings.ManualRate[i]);

475  struct TrimmedAttitudeSetpoint {
476  float Roll;
477  float Pitch;
478  float Yaw;
479  } trimmedAttitudeSetpoint;
480 
481  // Mux in level trim values, and saturate the trimmed attitude setpoint.
482  trimmedAttitudeSetpoint.Roll = bound_min_max(
483  raw_input[ROLL] + subTrim.Roll,
484  -settings.RollMax + subTrim.Roll,
485  settings.RollMax + subTrim.Roll);
486  trimmedAttitudeSetpoint.Pitch = bound_min_max(
487  raw_input[PITCH] + subTrim.Pitch,
488  -settings.PitchMax + subTrim.Pitch,
489  settings.PitchMax + subTrim.Pitch);
490  trimmedAttitudeSetpoint.Yaw = raw_input[YAW];
491 
492  // For horizon mode we need to compute the desire attitude from an unscaled value and apply the
493  // trim offset. Also track the stick with the most deflection to choose rate blending.
494  horizonRateFraction = 0.0f;
495  if (axis_mode[ROLL] == STABILIZATIONDESIRED_STABILIZATIONMODE_HORIZON) {
496  trimmedAttitudeSetpoint.Roll = bound_min_max(
497  raw_input[ROLL] * settings.RollMax + subTrim.Roll,
498  -settings.RollMax + subTrim.Roll,
499  settings.RollMax + subTrim.Roll);
500  horizonRateFraction = fabsf(raw_input[ROLL]);
501  }
502  if (axis_mode[PITCH] == STABILIZATIONDESIRED_STABILIZATIONMODE_HORIZON) {
503  trimmedAttitudeSetpoint.Pitch = bound_min_max(
504  raw_input[PITCH] * settings.PitchMax + subTrim.Pitch,
505  -settings.PitchMax + subTrim.Pitch,
506  settings.PitchMax + subTrim.Pitch);
507  horizonRateFraction = MAX(horizonRateFraction, fabsf(raw_input[PITCH]));
508  }
509  if (axis_mode[YAW] == STABILIZATIONDESIRED_STABILIZATIONMODE_HORIZON) {
510  trimmedAttitudeSetpoint.Yaw = raw_input[YAW] * settings.YawMax;
511  horizonRateFraction = MAX(horizonRateFraction, fabsf(raw_input[YAW]));
512  }
513 
514  // For weak leveling mode the attitude setpoint is the trim value (drifts back towards "0")
515  if (axis_mode[ROLL] == STABILIZATIONDESIRED_STABILIZATIONMODE_WEAKLEVELING) {
516  trimmedAttitudeSetpoint.Roll = subTrim.Roll;
517  }
518  if (axis_mode[PITCH] == STABILIZATIONDESIRED_STABILIZATIONMODE_WEAKLEVELING) {
519  trimmedAttitudeSetpoint.Pitch = subTrim.Pitch;
520  }
521  if (axis_mode[YAW] == STABILIZATIONDESIRED_STABILIZATIONMODE_WEAKLEVELING) {
522  trimmedAttitudeSetpoint.Yaw = 0;
523  }
524 
525  // Note we divide by the maximum limit here so the fraction ranges from 0 to 1 depending on
526  // how much is requested.
527  horizonRateFraction = bound_sym(horizonRateFraction, HORIZON_MODE_MAX_BLEND) / HORIZON_MODE_MAX_BLEND;
528 
529  // Calculate the errors in each axis. The local error is used in the following modes:
530  // ATTITUDE, HORIZON, WEAKLEVELING
531  float local_attitude_error[MAX_AXES];
532  local_attitude_error[ROLL] = trimmedAttitudeSetpoint.Roll - attitudeActual.Roll;
533  local_attitude_error[PITCH] = trimmedAttitudeSetpoint.Pitch - attitudeActual.Pitch;
534  local_attitude_error[YAW] = trimmedAttitudeSetpoint.Yaw - attitudeActual.Yaw;
535 
536  // Wrap yaw error to [-180,180]
537  local_attitude_error[YAW] = circular_modulus_deg(local_attitude_error[YAW]);
538 
539  float *gyro_filtered = &gyrosData.x;
540 
541  /* Maintain a second-order, lower cutof freq variant for
542  * dynamic flight modes.
543  */
544 
545  static float max_rate_filtered[MAX_AXES];
546 
547  // A flag to track which stabilization mode each axis is in
548  static uint8_t previous_mode[MAX_AXES] = {255,255,255};
549 
550  actuatorDesired.SystemIdentCycle = 0xffff;
551 
552  uint16_t max_safe_rate = PIOS_SENSORS_GetMaxGyro() * 0.9f;
553 
554  //Run the selected stabilization algorithm on each axis:
555  for(uint8_t i=0; i< MAX_AXES; i++)
556  {
557  // Check whether this axis mode needs to be reinitialized
558  bool reinit = (axis_mode[i] != previous_mode[i]);
559 
560  if(reinit && previous_mode[i] != 255) {
561  // Disable the integrator this round. And only do so on real mode switches.
562  smoothcontrol_reinit(rc_smoothing, i, raw_input[i]);
563  }
564 
565  previous_mode[i] = axis_mode[i];
566 
567  // Apply the selected control law
568  switch(axis_mode[i])
569  {
570  case STABILIZATIONDESIRED_STABILIZATIONMODE_FAILSAFE:
571  PIOS_Assert(0); /* Shouldn't happen, per above */
572  break;
573 
574  case STABILIZATIONDESIRED_STABILIZATIONMODE_RATE:
575  if(reinit) {
576  pids[PID_GROUP_RATE + i].iAccumulator = 0;
577  }
578 
579  // Store to rate desired variable for storing to UAVO
580  rateDesiredAxis[i] = bound_sym(raw_input[i], settings.ManualRate[i]);
581 
582  // Compute the inner loop
583  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
584  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i],1.0f);
585 
586  break;
587 
588  case STABILIZATIONDESIRED_STABILIZATIONMODE_ACRODYNE:
589  if(reinit) {
590  pids[PID_GROUP_RATE + i].iAccumulator = 0;
591  max_rate_filtered[i] = settings.ManualRate[i];
592  }
593 
594  float curve_cmd = expoM(raw_input[i],
595  settings.RateExpo[i],
596  settings.RateExponent[i]*0.1f);
597 
598  const float break_point = settings.AcroDynamicTransition[i]/100.0f;
599 
600  uint16_t calc_max_rate = settings.ManualRate[i];
601 
602  float abs_cmd = fabsf(raw_input[i]);
603 
604  uint16_t acro_dynrate = settings.AcroDynamicRate[i];
605 
606  if (!acro_dynrate) {
607  acro_dynrate = settings.ManualRate[i] + settings.ManualRate[i] / 2;
608  }
609 
610  if (acro_dynrate > max_safe_rate) {
611  acro_dynrate = max_safe_rate;
612  }
613 
614  // Could precompute much of this...
615  if (abs_cmd > break_point) {
616  calc_max_rate = (settings.ManualRate[i] * (abs_cmd - 1.0f) * (2 * break_point - abs_cmd - 1.0f) + acro_dynrate * powf(break_point - abs_cmd, 2.0f)) / powf(break_point - 1.0f, 2.0f);
617  }
618 
619  calc_max_rate = MIN(calc_max_rate,
620  acro_dynrate);
621 
622  max_rate_filtered[i] = max_rate_filtered[i] * max_rate_alpha + calc_max_rate * (1 - max_rate_alpha);
623 
624  // Store to rate desired variable for storing to UAVO
625  rateDesiredAxis[i] = bound_sym(curve_cmd * max_rate_filtered[i], max_rate_filtered[i]);
626 
627  // Compute the inner loop
628  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
629  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i], 1.0f);
630 
631  break;
632 
633  case STABILIZATIONDESIRED_STABILIZATIONMODE_ACROPLUS:
634  // this implementation is based on the Openpilot/Librepilot Acro+ flightmode
635  // and our previous MWRate flightmodes
636  if(reinit) {
637  pids[PID_GROUP_RATE + i].iAccumulator = 0;
638  }
639 
640  // The factor for gyro suppression / mixing raw stick input into the output; scaled by raw stick input
641  float factor = fabsf(raw_input[i]) * settings.AcroInsanityFactor / 100.0f;
642 
643  // Store to rate desired variable for storing to UAVO
644  rateDesiredAxis[i] = bound_sym(raw_input[i] * settings.ManualRate[i], settings.ManualRate[i]);
645 
646  // Zero integral for aggressive maneuvers
647  if ((i < 2 && fabsf(gyro_filtered[i]) > settings.AcroZeroIntegralGyro) ||
648  (i == 0 && fabsf(raw_input[i]) > settings.AcroZeroIntegralStick / 100.0f)) {
649  pids[PID_GROUP_RATE + i].iAccumulator = 0;
650  }
651 
652  // Compute the inner loop
653  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
654  actuatorDesiredAxis[i] = factor * raw_input[i] + (1.0f - factor) * actuatorDesiredAxis[i];
655  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i], 1.0f);
656 
657  break;
658 
659  case STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE:
660  if(reinit) {
661  pids[PID_GROUP_ATT + i].iAccumulator = 0;
662  pids[PID_GROUP_RATE + i].iAccumulator = 0;
663  }
664 
665  // Compute the outer loop
666  rateDesiredAxis[i] = pid_apply(&pids[PID_GROUP_ATT + i], local_attitude_error[i]);
667  rateDesiredAxis[i] = bound_sym(rateDesiredAxis[i], settings.MaximumRate[i]);
668 
669  // Compute the inner loop
670  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
671  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i],1.0f);
672 
673  break;
674 
675  case STABILIZATIONDESIRED_STABILIZATIONMODE_VIRTUALBAR:
676  // Store for debugging output
677  rateDesiredAxis[i] = raw_input[i];
678 
679  // Run a virtual flybar stabilization algorithm on this axis
680  stabilization_virtual_flybar(gyro_filtered[i], rateDesiredAxis[i], &actuatorDesiredAxis[i], dT_expected, reinit, i, &pids[PID_GROUP_VBAR + i], &vbar_settings);
681 
682  break;
683 
684  case STABILIZATIONDESIRED_STABILIZATIONMODE_WEAKLEVELING:
685  {
686  if (reinit) {
687  pids[PID_GROUP_RATE + i].iAccumulator = 0;
688  }
689 
690  float weak_leveling = local_attitude_error[i] * weak_leveling_kp;
691  weak_leveling = bound_sym(weak_leveling, weak_leveling_max);
692 
693  // Compute desired rate as input biased towards leveling
694  rateDesiredAxis[i] = raw_input[i] + weak_leveling;
695  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
696  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i],1.0f);
697 
698  break;
699  }
700  case STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK:
701  if (reinit) {
702  pids[PID_GROUP_RATE + i].iAccumulator = 0;
703  }
704 
705  if (fabsf(raw_input[i]) > max_axislock_rate) {
706  // While getting strong commands act like rate mode
707  rateDesiredAxis[i] = bound_sym(raw_input[i], settings.ManualRate[i]);
708 
709  // Reset accumulator
710  axis_lock_accum[i] = 0;
711  } else {
712  // For weaker commands or no command simply lock (almost) on no gyro change
713  axis_lock_accum[i] += (raw_input[i] - gyro_filtered[i]) * dT_expected;
714  axis_lock_accum[i] = bound_sym(axis_lock_accum[i], max_axis_lock);
715 
716  // Compute the inner loop
717  float tmpRateDesired = pid_apply(&pids[PID_GROUP_ATT + i], axis_lock_accum[i]);
718  rateDesiredAxis[i] = bound_sym(tmpRateDesired, settings.MaximumRate[i]);
719  }
720 
721  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
722  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i],1.0f);
723 
724  break;
725 
726  case STABILIZATIONDESIRED_STABILIZATIONMODE_HORIZON:
727  if(reinit) {
728  pids[PID_GROUP_RATE + i].iAccumulator = 0;
729  }
730 
731  // Do not allow outer loop integral to wind up in this mode since the controller
732  // is often disengaged.
733  pids[PID_GROUP_ATT + i].iAccumulator = 0;
734 
735  // Compute the outer loop for the attitude control
736  float rateDesiredAttitude = pid_apply(&pids[PID_GROUP_ATT + i], local_attitude_error[i]);
737  // Compute the desire rate for a rate control
738  float rateDesiredRate = raw_input[i] * settings.ManualRate[i];
739 
740  // Blend from one rate to another. The maximum of all stick positions is used for the
741  // amount so that when one axis goes completely to rate the other one does too. This
742  // prevents doing flips while one axis tries to stay in attitude mode.
743  rateDesiredAxis[i] = rateDesiredAttitude * (1.0f-horizonRateFraction) + rateDesiredRate * horizonRateFraction;
744  rateDesiredAxis[i] = bound_sym(rateDesiredAxis[i], settings.ManualRate[i]);
745 
746  // Compute the inner loop
747  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
748  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i],1.0f);
749 
750  break;
751  case STABILIZATIONDESIRED_STABILIZATIONMODE_SYSTEMIDENT:
752  case STABILIZATIONDESIRED_STABILIZATIONMODE_SYSTEMIDENTRATE:
753  ;
754  static bool measuring;
755  static uint32_t enter_time;
756 
757  static uint32_t measure_remaining;
758 
759  // Takes 1250ms + the time to reach
760  // the '0th measurement'
761  // (could be the ~600ms period time)
762  const uint32_t PREPARE_TIME = 1250000;
763 
764  if (reinit) {
765  pids[PID_GROUP_ATT + i].iAccumulator = 0;
766  pids[PID_GROUP_RATE + i].iAccumulator = 0;
767 
768  if (i == 0) {
769  enter_time = timeval;
770 
771  measuring = false;
772  }
773  }
774 
775  if ((i == 0) &&
776  (!measuring) &&
777  ((timeval - enter_time) > PREPARE_TIME)) {
778  if (!(iteration & ident_mask)) {
779  measuring = true;
780  measure_remaining = 60 / dT_expected;
781  // Round down to an integer
782  // number of ident cycles.
783  measure_remaining &= ~ident_mask;
784  }
785  }
786 
787  if (flightStatus.Armed != FLIGHTSTATUS_ARMED_ARMED) {
788  measuring = false;
789  }
790 
791  if (axis_mode[i] == STABILIZATIONDESIRED_STABILIZATIONMODE_SYSTEMIDENT) {
792  // Compute the outer loop
793  rateDesiredAxis[i] = pid_apply(&pids[PID_GROUP_ATT + i], local_attitude_error[i]);
794  rateDesiredAxis[i] = bound_sym(rateDesiredAxis[i], settings.MaximumRate[i]);
795 
796  // Compute the inner loop
797  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
798  } else {
799  // Get the desired rate. yaw is always in rate mode in system ident.
800  rateDesiredAxis[i] = bound_sym(raw_input[i], settings.ManualRate[i]);
801 
802  // Compute the inner loop only for yaw
803  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
804  }
805 
806  const float scale = settings.AutotuneActuationEffort[i];
807 
808  uint32_t ident_iteration =
809  iteration >> ident_shift;
810 
811  if (measuring && measure_remaining) {
812  if (i == 2) {
813  // Only adjust the
814  // counter on one axis
815  measure_remaining--;
816  }
817 
818  actuatorDesired.SystemIdentCycle = (iteration & ident_mask);
819 
820  switch (ident_iteration & 0x07) {
821  case 0:
822  if (i == 2) {
823  actuatorDesiredAxis[i] += scale;
824  }
825  break;
826  case 1:
827  if (i == 0) {
828  actuatorDesiredAxis[i] += scale;
829  }
830  break;
831  case 2:
832  if (i == 2) {
833  actuatorDesiredAxis[i] -= scale;
834  }
835  break;
836  case 3:
837  if (i == 0) {
838  actuatorDesiredAxis[i] -= scale;
839  }
840  break;
841  case 4:
842  if (i == 2) {
843  actuatorDesiredAxis[i] += scale;
844  }
845  break;
846  case 5:
847  if (i == 1) {
848  actuatorDesiredAxis[i] += scale;
849  }
850  break;
851  case 6:
852  if (i == 2) {
853  actuatorDesiredAxis[i] -= scale;
854  }
855  break;
856  case 7:
857  if (i == 1) {
858  actuatorDesiredAxis[i] -= scale;
859  }
860  break;
861  }
862 
863  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i],1.0f);
864  }
865 
866  break;
867 
868  case STABILIZATIONDESIRED_STABILIZATIONMODE_COORDINATEDFLIGHT:
869  switch (i) {
870  case YAW:
871  if (reinit) {
872  pids[PID_COORDINATED_FLIGHT_YAW].iAccumulator = 0;
873  pids[PID_RATE_YAW].iAccumulator = 0;
874  axis_lock_accum[YAW] = 0;
875  }
876 
877  //If we are not in roll attitude mode, trigger an error
878  if (axis_mode[ROLL] != STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE)
879  {
880  error = true;
881  break ;
882  }
883 
884  if (fabsf(stabDesired.Yaw) < COORDINATED_FLIGHT_MAX_YAW_THRESHOLD) { //If yaw is within the deadband...
885  if (fabsf(stabDesired.Roll) > COORDINATED_FLIGHT_MIN_ROLL_THRESHOLD) { // We're requesting more roll than the threshold
886  float accelsDataY;
887  AccelsyGet(&accelsDataY);
888 
889  //Reset integral if we have changed roll to opposite direction from rudder. This implies that we have changed desired turning direction.
890  if ((stabDesired.Roll > 0 && actuatorDesiredAxis[YAW] < 0) ||
891  (stabDesired.Roll < 0 && actuatorDesiredAxis[YAW] > 0)){
892  pids[PID_COORDINATED_FLIGHT_YAW].iAccumulator = 0;
893  }
894 
895  // Coordinate flight can simply be seen as ensuring that there is no lateral acceleration in the
896  // body frame. As such, we use the (noisy) accelerometer data as our measurement. Ideally, at
897  // some point in the future we will estimate acceleration and then we can use the estimated value
898  // instead of the measured value.
899  float errorSlip = -accelsDataY;
900 
901  float command = pid_apply(&pids[PID_COORDINATED_FLIGHT_YAW], errorSlip);
902  actuatorDesiredAxis[YAW] = bound_sym(command ,1.0);
903 
904  // Reset axis-lock integrals
905  pids[PID_RATE_YAW].iAccumulator = 0;
906  axis_lock_accum[YAW] = 0;
907  } else if (fabsf(stabDesired.Roll) <= COORDINATED_FLIGHT_MIN_ROLL_THRESHOLD) { // We're requesting less roll than the threshold
908  // Axis lock on no gyro change
909  axis_lock_accum[YAW] += (0 - gyro_filtered[YAW]) * dT_expected;
910 
911  rateDesiredAxis[YAW] = pid_apply(&pids[PID_ATT_YAW], axis_lock_accum[YAW]);
912  rateDesiredAxis[YAW] = bound_sym(rateDesiredAxis[YAW], settings.MaximumRate[YAW]);
913 
914  actuatorDesiredAxis[YAW] = pid_apply_setpoint(&pids[PID_RATE_YAW], NULL, rateDesiredAxis[YAW], gyro_filtered[YAW]);
915  actuatorDesiredAxis[YAW] = bound_sym(actuatorDesiredAxis[YAW],1.0f);
916 
917  // Reset coordinated-flight integral
918  pids[PID_COORDINATED_FLIGHT_YAW].iAccumulator = 0;
919  }
920  } else { //... yaw is outside the deadband. Pass the manual input directly to the actuator.
921  actuatorDesiredAxis[YAW] = bound_sym(raw_input[YAW], 1.0);
922 
923  // Reset all integrals
924  pids[PID_COORDINATED_FLIGHT_YAW].iAccumulator = 0;
925  pids[PID_RATE_YAW].iAccumulator = 0;
926  axis_lock_accum[YAW] = 0;
927  }
928  break;
929  case ROLL:
930  case PITCH:
931  default:
932  //Coordinated Flight has no effect in these modes. Trigger a configuration error.
933  error = true;
934  break;
935  }
936 
937  break;
938 
939  case STABILIZATIONDESIRED_STABILIZATIONMODE_POI:
940  // The sanity check enforces this is only selectable for Yaw
941  // for a gimbal you can select pitch too.
942  if(reinit) {
943  pids[PID_GROUP_ATT + i].iAccumulator = 0;
944  pids[PID_GROUP_RATE + i].iAccumulator = 0;
945  }
946 
947  float angle_error = 0;
948  float angle;
949  if (CameraDesiredHandle()) {
950  switch(i) {
951  case PITCH:
952  CameraDesiredDeclinationGet(&angle);
953  angle_error = circular_modulus_deg(angle - attitudeActual.Pitch);
954  break;
955  case ROLL:
956  {
957  uint8_t roll_fraction = 0;
958 
959  // For ROLL POI mode we track the FC roll angle (scaled) to
960  // allow keeping some motion
961  CameraDesiredRollGet(&angle);
962  angle *= roll_fraction / 100.0f;
963  angle_error = circular_modulus_deg(angle - attitudeActual.Roll);
964  }
965  break;
966  case YAW:
967  CameraDesiredBearingGet(&angle);
968  angle_error = circular_modulus_deg(angle - attitudeActual.Yaw);
969  break;
970  }
971  } else
972  error = true;
973 
974  // Compute the outer loop
975  rateDesiredAxis[i] = pid_apply(&pids[PID_GROUP_ATT + i], angle_error);
976  rateDesiredAxis[i] = bound_sym(rateDesiredAxis[i], settings.PoiMaximumRate[i]);
977 
978  // Compute the inner loop
979  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_GROUP_RATE + i], get_deadband(i), rateDesiredAxis[i], gyro_filtered[i]);
980  actuatorDesiredAxis[i] = bound_sym(actuatorDesiredAxis[i],1.0f);
981 
982  break;
983  case STABILIZATIONDESIRED_STABILIZATIONMODE_DISABLED:
984  actuatorDesiredAxis[i] = 0.0;
985  break;
986  case STABILIZATIONDESIRED_STABILIZATIONMODE_MANUAL:
987  actuatorDesiredAxis[i] = bound_sym(raw_input[i],1.0f);
988  break;
989  default:
990  error = true;
991  break;
992  }
993  }
994 
995  // Run the smoothing over the throttle stick.
996  smoothcontrol_run_thrust(rc_smoothing, &actuatorDesired.Thrust);
997 
998  // Register loop.
999  smoothcontrol_next(rc_smoothing);
1000 
1001  if (vbar_settings.VbarPiroComp == VBARSETTINGS_VBARPIROCOMP_TRUE)
1002  stabilization_virtual_flybar_pirocomp(gyro_filtered[YAW], dT_expected);
1003 
1004 #if defined(RATEDESIRED_DIAGNOSTICS)
1005  RateDesiredSet(&rateDesired);
1006 #endif
1007 
1008  // Save dT
1009  actuatorDesired.UpdateTime = dT * 1000;
1010 
1011  ActuatorDesiredSet(&actuatorDesired);
1012 
1013  if(flightStatus.Armed != FLIGHTSTATUS_ARMED_ARMED ||
1014  (lowThrottleZeroIntegral && get_throttle(&stabDesired, &airframe_type) < 0))
1015  {
1016  // Force all axes to reinitialize when engaged
1017  for(uint8_t i=0; i< MAX_AXES; i++)
1018  previous_mode[i] = 255;
1019  }
1020 
1021  // Clear or set alarms. Done like this to prevent toggling each cycle
1022  // and hammering system alarms
1023  if (error)
1024  AlarmsSet(SYSTEMALARMS_ALARM_STABILIZATION, SYSTEMALARMS_ALARM_ERROR);
1025  else
1026  AlarmsClear(SYSTEMALARMS_ALARM_STABILIZATION);
1027  }
1028 }
