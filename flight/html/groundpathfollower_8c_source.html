161 static void groundPathFollowerTask(void *parameters)
162 {
163  SystemSettingsData systemSettings;
164  FlightStatusData flightStatus;
165 
166  uint32_t lastUpdateTime;
167 
168  GroundPathFollowerSettingsConnectCallback(SettingsUpdatedCb);
169  PathDesiredConnectCallback(SettingsUpdatedCb);
170 
171  GroundPathFollowerSettingsGet(&guidanceSettings);
172  PathDesiredGet(&pathDesired);
173 
174  // Main task loop
175  lastUpdateTime = PIOS_Thread_Systime();
176  while (1) {
177 
178  // Conditions when this runs:
179  // 1. Must have GROUND type airframe
180  // 2. Flight mode is PositionHold and PathDesired.Mode is Endpoint OR
181  // FlightMode is PathPlanner and PathDesired.Mode is Endpoint or Path
182 
183  SystemSettingsGet(&systemSettings);
184  if ( (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_GROUNDVEHICLECAR) &&
185  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_GROUNDVEHICLEDIFFERENTIAL) &&
186  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_GROUNDVEHICLEMOTORCYCLE) )
187  {
188  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_WARNING);
189  PIOS_Thread_Sleep(1000);
190  continue;
191  }
192 
193  // Continue collecting data if not enough time
194  PIOS_Thread_Sleep_Until(&lastUpdateTime, guidanceSettings.UpdatePeriod);
195 
196  // Convert the accels into the NED frame
197  updateNedAccel();
198 
199  FlightStatusGet(&flightStatus);
200 
201  // Check the combinations of flightmode and pathdesired mode
202  switch(flightStatus.FlightMode) {
203  /* This combination of RETURNTOHOME and HOLDPOSITION looks strange but
204  * is correct. RETURNTOHOME mode uses HOLDPOSITION with the position
205  * set to home */
206  case FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME:
207  if (pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
208  updateEndpointVelocity();
209  updateGroundDesiredAttitude();
210  } else {
211  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
212  }
213  break;
214  case FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD:
215  if (pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
216  updateEndpointVelocity();
217  updateGroundDesiredAttitude();
218  } else {
219  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
220  }
221  break;
222  case FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER:
223  if (pathDesired.Mode == PATHDESIRED_MODE_ENDPOINT ||
224  pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
225  updateEndpointVelocity();
226  updateGroundDesiredAttitude();
227  } else if (pathDesired.Mode == PATHDESIRED_MODE_VECTOR ||
228  pathDesired.Mode == PATHDESIRED_MODE_CIRCLELEFT ||
229  pathDesired.Mode == PATHDESIRED_MODE_CIRCLERIGHT) {
230  updatePathVelocity();
231  updateGroundDesiredAttitude();
232  } else {
233  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
234  }
235  break;
236  default:
237  // Be cleaner and get rid of global variables
238  northVelIntegral = 0;
239  eastVelIntegral = 0;
240  northPosIntegral = 0;
241  eastPosIntegral = 0;
242 
243  // Track throttle before engaging this mode. Cheap system ident
244  StabilizationDesiredData stabDesired;
245  StabilizationDesiredGet(&stabDesired);
246  throttleOffset = stabDesired.Thrust;
247 
248  break;
249  }
250 
251  AlarmsClear(SYSTEMALARMS_ALARM_PATHFOLLOWER);
252 
253  }
254 }

262 static void updatePathVelocity()
263 {
264  PositionActualData positionActual;
265  PositionActualGet(&positionActual);
266 
267  float cur[3] = {positionActual.North, positionActual.East, positionActual.Down};
268  struct path_status progress;
269 
270  path_progress(&pathDesired, cur, &progress);
271 
272  // Update the path status UAVO
273  PathStatusData pathStatus;
274  PathStatusGet(&pathStatus);
275  pathStatus.fractional_progress = progress.fractional_progress;
276  if (pathStatus.fractional_progress < 1)
277  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
278  else
279  pathStatus.Status = PATHSTATUS_STATUS_COMPLETED;
280 
281  pathStatus.Waypoint = pathDesired.Waypoint;
282 
283  PathStatusSet(&pathStatus);
284 
285  float groundspeed = pathDesired.StartingVelocity +
286  (pathDesired.EndingVelocity - pathDesired.StartingVelocity) * progress.fractional_progress;
287  if(progress.fractional_progress > 1)
288  groundspeed = 0;
289 
290  VelocityDesiredData velocityDesired;
291  velocityDesired.North = progress.path_direction[0] * groundspeed;
292  velocityDesired.East = progress.path_direction[1] * groundspeed;
293 
294  float error_speed = progress.error * guidanceSettings.HorizontalPosPI[GROUNDPATHFOLLOWERSETTINGS_HORIZONTALPOSPI_KP];
295  float correction_velocity[2] = {progress.correction_direction[0] * error_speed,
296  progress.correction_direction[1] * error_speed};
297 
298  float total_vel = sqrtf(powf(correction_velocity[0],2) + powf(correction_velocity[1],2));
299  float scale = 1;
300  if(total_vel > guidanceSettings.HorizontalVelMax)
301  scale = guidanceSettings.HorizontalVelMax / total_vel;
302 
303  // Currently not apply a PID loop for horizontal corrections
304  velocityDesired.North += progress.correction_direction[0] * error_speed * scale;
305  velocityDesired.East += progress.correction_direction[1] * error_speed * scale;
306 
307  // No altitude control on a ground vehicle
308  velocityDesired.Down = 0;
309 
310  VelocityDesiredSet(&velocityDesired);
311 }

319 void updateEndpointVelocity()
320 {
321  float dT = guidanceSettings.UpdatePeriod / 1000.0f;
322 
323  PositionActualData positionActual;
324  VelocityDesiredData velocityDesired;
325 
326  PositionActualGet(&positionActual);
327  VelocityDesiredGet(&velocityDesired);
328 
329  float northError;
330  float eastError;
331  float northCommand;
332  float eastCommand;
333 
334  float northPos = positionActual.North;
335  float eastPos = positionActual.East;
336 
337  // Compute desired north command velocity from position error
338  northError = pathDesired.End[PATHDESIRED_END_NORTH] - northPos;
339  northCommand = pid_apply(&ground_pids[NORTH_POSITION], northError, dT);
340 
341  // Compute desired east command velocity from position error
342  eastError = pathDesired.End[PATHDESIRED_END_EAST] - eastPos;
343  eastCommand = pid_apply(&ground_pids[EAST_POSITION], eastError, dT);
344 
345  // Limit the maximum velocity any direction (not north and east separately)
346  float total_vel = sqrtf(powf(northCommand,2) + powf(eastCommand,2));
347  float scale = 1;
348  if(total_vel > guidanceSettings.HorizontalVelMax)
349  scale = guidanceSettings.HorizontalVelMax / total_vel;
350 
351  velocityDesired.North = northCommand * scale;
352  velocityDesired.East = eastCommand * scale;
353 
354  // No altitude control on a ground vehicle
355  velocityDesired.Down = 0;
356 
357  VelocityDesiredSet(&velocityDesired);
358 
359  // Indicate whether we are in radius of this endpoint
360  uint8_t path_status = PATHSTATUS_STATUS_INPROGRESS;
361  float distance2 = powf(northError, 2) + powf(eastError, 2);
362  if (distance2 < (guidanceSettings.EndpointRadius * guidanceSettings.EndpointRadius)) {
363  path_status = PATHSTATUS_STATUS_COMPLETED;
364  }
365  PathStatusStatusSet(&path_status);
366 }

375 static void updateGroundDesiredAttitude()
376 {
377  float dT = guidanceSettings.UpdatePeriod / 1000.0f;
378 
379  VelocityDesiredData velocityDesired;
380  VelocityActualData velocityActual;
381  StabilizationDesiredData stabDesired;
382  AttitudeActualData attitudeActual;
383  NedAccelData nedAccel;
384  GroundPathFollowerSettingsData guidanceSettings;
385  StabilizationSettingsData stabSettings;
386  SystemSettingsData systemSettings;
387 
388  SystemSettingsGet(&systemSettings);
389  GroundPathFollowerSettingsGet(&guidanceSettings);
390  VelocityActualGet(&velocityActual);
391  VelocityDesiredGet(&velocityDesired);
392  StabilizationDesiredGet(&stabDesired);
393  VelocityDesiredGet(&velocityDesired);
394  AttitudeActualGet(&attitudeActual);
395  StabilizationSettingsGet(&stabSettings);
396  NedAccelGet(&nedAccel);
397 
398  float northVel = velocityActual.North;
399  float eastVel = velocityActual.East;
400 
401  // Calculate direction from velocityDesired and set stabDesired.Yaw
402  stabDesired.Yaw = atan2f( velocityDesired.East, velocityDesired.North ) * RAD2DEG;
403 
404  // Calculate throttle and set stabDesired.Thrust
405  float velDesired = sqrtf(powf(velocityDesired.East,2) + powf(velocityDesired.North,2));
406  float velActual = sqrtf(powf(eastVel,2) + powf(northVel,2));
407  ManualControlCommandData manualControlData;
408  ManualControlCommandGet(&manualControlData);
409  switch (guidanceSettings.ThrottleControl) {
410  case GROUNDPATHFOLLOWERSETTINGS_THROTTLECONTROL_MANUAL:
411  {
412  stabDesired.Thrust = manualControlData.Throttle;
413  break;
414  }
415  case GROUNDPATHFOLLOWERSETTINGS_THROTTLECONTROL_PROPORTIONAL:
416  {
417  float velRatio = velDesired / guidanceSettings.HorizontalVelMax;
418  stabDesired.Thrust = guidanceSettings.MaxThrottle * velRatio;
419  if (guidanceSettings.ManualOverride == GROUNDPATHFOLLOWERSETTINGS_MANUALOVERRIDE_TRUE) {
420  stabDesired.Thrust = stabDesired.Thrust * manualControlData.Throttle;
421  }
422  break;
423  }
424  case GROUNDPATHFOLLOWERSETTINGS_THROTTLECONTROL_AUTO:
425  {
426  float velError = velDesired - velActual;
427  stabDesired.Thrust = pid_apply(&ground_pids[VELOCITY], velError, dT) + velDesired * guidanceSettings.VelocityFeedforward;
428  if (guidanceSettings.ManualOverride == GROUNDPATHFOLLOWERSETTINGS_MANUALOVERRIDE_TRUE) {
429  stabDesired.Thrust = stabDesired.Thrust * manualControlData.Throttle;
430  }
431  break;
432  }
433  default:
434  {
435  PIOS_Assert(0);
436  break;
437  }
438  }
439 
440  // Limit throttle as per settings
441  stabDesired.Thrust = bound_min_max(stabDesired.Thrust, 0, guidanceSettings.MaxThrottle);
442 
443  // Set StabilizationDesired object
444  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
445  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
446  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
447  StabilizationDesiredSet(&stabDesired);
448 }

453 static void updateNedAccel()
454 {
455  float accel[3];
456  float q[4];
457  float Rbe[3][3];
458  float accel_ned[3];
459 
460  // Collect downsampled attitude data
461  AccelsData accels;
462  AccelsGet(&accels);
463  accel[0] = accels.x;
464  accel[1] = accels.y;
465  accel[2] = accels.z;
466 
467  //rotate avg accels into earth frame and store it
468  AttitudeActualData attitudeActual;
469  AttitudeActualGet(&attitudeActual);
470  q[0]=attitudeActual.q1;
471  q[1]=attitudeActual.q2;
472  q[2]=attitudeActual.q3;
473  q[3]=attitudeActual.q4;
474  Quaternion2R(q, Rbe);
475  for (uint8_t i=0; i<3; i++){
476  accel_ned[i]=0;
477  for (uint8_t j=0; j<3; j++)
478  accel_ned[i] += Rbe[j][i]*accel[j];
479  }
480  accel_ned[2] += GRAVITY;
481 
482  NedAccelData accelData;
483  NedAccelGet(&accelData);
484  accelData.North = accel_ned[0];
485  accelData.East = accel_ned[1];
486  accelData.Down = accel_ned[2];
487  NedAccelSet(&accelData);
488 }
