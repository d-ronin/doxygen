216 static void radioRxTask( __attribute__ ((unused))
217  void *parameters)
218 {
219  // Task loop
220  while (1) {
221 #ifdef PIOS_INCLUDE_WDG
222  PIOS_WDG_UpdateFlag(PIOS_WDG_RADIORX);
223 #endif
224  if (PIOS_COM_RADIOBRIDGE &&
225  PIOS_COM_Available(PIOS_COM_RADIOBRIDGE)) {
226  uint8_t serial_data[32];
227  uint16_t bytes_to_process =
228  PIOS_COM_ReceiveBuffer(PIOS_COM_RADIOBRIDGE,
229  serial_data,
230  sizeof(serial_data),
231  MAX_PORT_DELAY);
232  if (bytes_to_process > 0) {
233  // Pass the data through the UAVTalk parser.
234  for (uint8_t i = 0;
235  i < bytes_to_process; i++) {
236  ProcessRadioStream(data->radioUAVTalkCon,
237  data->telemUAVTalkCon,
238  serial_data[i]);
239  }
240  }
241 
242  /* XXX periodically inject ComBridgeStats to downstream */
243  /* XXX periodically inject sessionmanaging to downstream */
244  } else {
245  /* hand port to telemetry */
246  data->have_port = false;
247  PIOS_Thread_Sleep(5);
248  telemetry_set_inhibit(false);
249  PIOS_Thread_Sleep(5);
250  }
251  }

259 static void telemetryRxTask( __attribute__ ((unused))
260  void *parameters)
261 {
262  // Task loop
263  while (1) {
264  uint32_t inputPort = PIOS_COM_TELEMETRY;
265 #ifdef PIOS_INCLUDE_WDG
266  PIOS_WDG_UpdateFlag(PIOS_WDG_TELEMETRY);
267 #endif
268 
269  if ((!PIOS_COM_Available(PIOS_COM_RADIOBRIDGE)) ||
270  (!data->have_port)) {
271  PIOS_Thread_Sleep(5);
272  continue;
273  }
274 #if defined(PIOS_INCLUDE_USB)
275  // Determine output port (USB takes priority over telemetry port)
276  if (PIOS_COM_Available(PIOS_COM_TELEM_USB)) {
277  inputPort = PIOS_COM_TELEM_USB;
278  }
279 #endif /* PIOS_INCLUDE_USB */
280  if (inputPort) {
281  uint8_t serial_data[32];
282  uint16_t bytes_to_process =
283  PIOS_COM_ReceiveBuffer(inputPort, serial_data,
284  sizeof(serial_data),
285  MAX_PORT_DELAY);
286 
287  if (data->have_port && (bytes_to_process > 0)) {
288  PIOS_ANNUNC_Toggle(PIOS_LED_RX);
289  for (uint8_t i = 0; i < bytes_to_process;
290  i++) {
291  ProcessLocalStream(
292  data->telemUAVTalkCon,
293  data->radioUAVTalkCon,
294  serial_data[i]);
295  }
296  }
297  } else {
298  PIOS_Thread_Sleep(5);
299  }
300  }

365 static void ProcessLocalStream(UAVTalkConnection inConnectionHandle,
366  UAVTalkConnection outConnectionHandle,
367  uint8_t rxbyte)
368 {
369  // Keep reading until we receive a completed packet.
370  UAVTalkRxState state =
371  UAVTalkProcessInputStreamQuiet(inConnectionHandle, rxbyte);
372 
373  if (state == UAVTALK_STATE_COMPLETE) {
374  // all packets are transparently relayed to the remote modem
375  // Incidentally this means that we fail requests for the
376  // radio stats and only telemeter them, but I consider this
377  // okee-dokee.
378  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
379  }
380 }

389 static void ProcessRadioStream(UAVTalkConnection inConnectionHandle,
390  UAVTalkConnection outConnectionHandle,
391  uint8_t rxbyte)
392 {
393  // Keep reading until we receive a completed packet.
394  UAVTalkRxState state =
395  UAVTalkProcessInputStreamQuiet(inConnectionHandle, rxbyte);
396 
397  if (!data->have_port) {
398  telemetry_set_inhibit(true);
399  data->have_port = true;
400  }
401 
402  if (state == UAVTALK_STATE_COMPLETE) {
403  // We only want to unpack certain objects from the remote modem
404  // Similarly we only want to relay certain objects to the telemetry port
405  uint32_t objId = UAVTalkGetPacketObjId(inConnectionHandle);
406  switch (objId) {
407  case HWTAULINK_OBJID:
408  case MetaObjectId(HWTAULINK_OBJID):
409  // Ignore object...
410  // These objects are shadowed by the modem and are not transmitted to the telemetry port
411  // - RFM22BSTATUS_OBJID : ground station will receive the OPLM link status instead
412  // - HWTAULINK_OBJID : ground station will read and write the OPLM settings instead
413  break;
414  case FLIGHTBATTERYSTATE_OBJID:
415  case FLIGHTSTATUS_OBJID:
416  case POSITIONACTUAL_OBJID:
417  case VELOCITYACTUAL_OBJID:
418  case BAROALTITUDE_OBJID:
419  // process / store locally for relaying to taranis
420  UAVTalkReceiveObject(inConnectionHandle);
421  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
422  break;
423  default:
424  // all other packets are relayed to the telemetry port
425  UAVTalkRelayPacket(inConnectionHandle,
426  outConnectionHandle);
427  break;
428  }
429  }
