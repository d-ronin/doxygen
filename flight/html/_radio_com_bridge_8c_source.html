135 static int32_t RadioComBridgeStart(void)
136 {
137  if (data) {
138  // Check if this is the coordinator modem
139  data->isCoordinator = PIOS_RFM22B_IsCoordinator(PIOS_COM_RFM22B);
140 
141  // Configure our UAVObjects for updates.
142  UAVObjConnectQueue(UAVObjGetByID(RFM22BSTATUS_OBJID), data->uavtalkEventQueue,
143  EV_UPDATED | EV_UPDATED_MANUAL);
144  UAVObjConnectQueue(UAVObjGetByID(OBJECTPERSISTENCE_OBJID), data->uavtalkEventQueue,
145  EV_UPDATED | EV_UPDATED_MANUAL);
146  if (data->isCoordinator) {
147  UAVObjConnectQueue(UAVObjGetByID(RFM22BRECEIVER_OBJID), data->radioEventQueue,
148  EV_UPDATED | EV_UPDATED_MANUAL);
149  } else {
150  UAVObjConnectQueue(UAVObjGetByID(RFM22BRECEIVER_OBJID), data->uavtalkEventQueue,
151  EV_UPDATED | EV_UPDATED_MANUAL);
152  }
153 
154  if (data->isCoordinator) {
155  registerObject(RadioComBridgeStatsHandle());
156  }
157  // Configure the UAVObject callbacks
158  ObjectPersistenceConnectCallback(&objectPersistenceUpdatedCb);
159 
160  // Watchdog must be registered before starting tasks
161 #ifdef PIOS_INCLUDE_WDG
162  PIOS_WDG_RegisterFlag(PIOS_WDG_TELEMETRYRX);
163  PIOS_WDG_RegisterFlag(PIOS_WDG_RADIOTX);
164  PIOS_WDG_RegisterFlag(PIOS_WDG_RADIORX);
165 #endif
166 
167  // Start the primary tasks for receiving/sending UAVTalk packets from the GCS.
168  data->telemetryTxTaskHandle = PIOS_Thread_Create(telemetryTxTask, "telemetryTxTask", STACK_SIZE_BYTES, NULL, TASK_PRIORITY);
169  data->telemetryRxTaskHandle = PIOS_Thread_Create(telemetryRxTask, "telemetryRxTask", STACK_SIZE_BYTES, NULL, TASK_PRIORITY);
170 
171  if (PIOS_PPM_RECEIVER != 0) {
172 #ifdef PIOS_INCLUDE_WDG
173  PIOS_WDG_RegisterFlag(PIOS_WDG_PPMINPUT);
174 #endif
175  data->PPMInputTaskHandle = PIOS_Thread_Create(PPMInputTask, "PPMInputTask",STACK_SIZE_BYTES, NULL, TASK_PRIORITY);
176  }
177  data->radioTxTaskHandle = PIOS_Thread_Create(radioTxTask, "radioTxTask", STACK_SIZE_BYTES, NULL, TASK_PRIORITY);
178  data->radioRxTaskHandle = PIOS_Thread_Create(radioRxTask, "radioRxTask", STACK_SIZE_BYTES, NULL, TASK_PRIORITY);
179 
180  return 0;
181  }
182 
183  return -1;
184 }

311 static void telemetryTxTask( __attribute__ ((unused))
312  void *parameters)
313 {
314  UAVObjEvent ev;
315 
316  updateSettings();
317 
318  PIOS_WDG_RegisterFlag(PIOS_WDG_TELEMETRYTX);
319 
320  // Loop forever
321  while (1) {
322 #ifdef PIOS_INCLUDE_WDG
323  PIOS_WDG_UpdateFlag(PIOS_WDG_TELEMETRYTX);
324 #endif
325  // Wait for queue message
326  if (PIOS_Queue_Receive(data->uavtalkEventQueue, &ev, MAX_PORT_DELAY)) {
327  if (ev.obj == RadioComBridgeStatsHandle()) {
328  updateRadioComBridgeStats();
329  }
330  // Send update (with retries)
331  int32_t ret = -1;
332  uint32_t retries = 0;
333  while (retries <= MAX_RETRIES && ret == -1) {
334  ret = UAVTalkSendObject(data->telemUAVTalkCon, ev.obj, ev.instId, 0);
335  if (ret == -1) {
336  ++retries;
337  }
338  }
339  // Update stats
340  data->telemetryTxRetries += retries;
341  }
342  }

553 static void ProcessTelemetryStream(UAVTalkConnection inConnectionHandle,
554  UAVTalkConnection outConnectionHandle,
555  uint8_t rxbyte)
556 {
557  // Keep reading until we receive a completed packet.
558  UAVTalkRxState state =
559  UAVTalkProcessInputStreamQuiet(inConnectionHandle, rxbyte);
560 
561  if (state == UAVTALK_STATE_COMPLETE) {
562  // We only want to unpack certain telemetry objects
563  uint32_t objId = UAVTalkGetPacketObjId(inConnectionHandle);
564  switch (objId) {
565  case HWTAULINK_OBJID:
566  case RFM22BRECEIVER_OBJID:
567  case MetaObjectId(HWTAULINK_OBJID):
568  case MetaObjectId(RFM22BRECEIVER_OBJID):
569  case MetaObjectId(RFM22BSTATUS_OBJID):
570 
571  // These objects are received here and only here
572  UAVTalkReceiveObject(inConnectionHandle);
573  break;
574 
575  case OBJECTPERSISTENCE_OBJID:
576  case MetaObjectId(OBJECTPERSISTENCE_OBJID):
577  // Handle saving settings on modem
578  UAVTalkReceiveObject(inConnectionHandle);
579 
580  ObjectPersistenceData objectPersistence;
581  ObjectPersistenceGet(&objectPersistence);
582  if (objectPersistence.ObjectID != HWTAULINK_OBJID &&
583  objectPersistence.ObjectID != MetaObjectId(HWTAULINK_OBJID)) {
584  // relay packet to remote modem except for requests to save
585  // the settings which happens locally
586  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
587  }
588 
589  break;
590 
591  case RFM22BSTATUS_OBJID:
592  {
593  uint32_t inst_id = UAVTalkGetPacketInstId(inConnectionHandle);
594  if (inst_id == 0) {
595  // dealing with local modem
596  UAVTalkReceiveObject(inConnectionHandle);
597  } else {
598  // for remote modem
599  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
600  }
601  }
602  break;
603  default:
604  // all other packets are transparently relayed to the remote modem
605  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
606  break;
607  }
608  }
609 }

618 static void ProcessRadioStream(UAVTalkConnection inConnectionHandle,
619  UAVTalkConnection outConnectionHandle,
620  uint8_t rxbyte)
621 {
622  // Keep reading until we receive a completed packet.
623  UAVTalkRxState state =
624  UAVTalkProcessInputStreamQuiet(inConnectionHandle, rxbyte);
625 
626  if (state == UAVTALK_STATE_COMPLETE) {
627  // We only want to unpack certain objects from the remote modem
628  // Similarly we only want to relay certain objects to the telemetry port
629  uint32_t objId = UAVTalkGetPacketObjId(inConnectionHandle);
630  switch (objId) {
631  case HWTAULINK_OBJID:
632  case MetaObjectId(RFM22BSTATUS_OBJID):
633  case MetaObjectId(HWTAULINK_OBJID):
634  // Ignore object...
635  // These objects are shadowed by the modem and are not transmitted to the telemetry port
636  // - RFM22BSTATUS_OBJID : ground station will receive the OPLM link status instead
637  // - HWTAULINK_OBJID : ground station will read and write the OPLM settings instead
638  break;
639  case RFM22BRECEIVER_OBJID:
640  case MetaObjectId(RFM22BRECEIVER_OBJID):
641  // Receive object locally
642  // These objects are received by the modem and are not transmitted to the telemetry port
643  // - RFM22BRECEIVER_OBJID : sent periodically from flight controller, not needed to echo
644  // some objects will send back a response to the remote modem
645  UAVTalkReceiveObject(inConnectionHandle);
646  break;
647  case FLIGHTBATTERYSTATE_OBJID:
648  case FLIGHTSTATUS_OBJID:
649  case POSITIONACTUAL_OBJID:
650  case VELOCITYACTUAL_OBJID:
651  case BAROALTITUDE_OBJID:
652 
653  // process the battery voltage locally for relaying to taranis
654  UAVTalkReceiveObject(inConnectionHandle);
655  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
656  break;
657  case RFM22BSTATUS_OBJID:
658  {
659  uint32_t inst_id = UAVTalkGetPacketInstId(inConnectionHandle);
660  if (inst_id == 0) {
661  // instance 0 is from modem. do not pass this version
662  } else {
663  // process the remote link state locally for relaying to taranis
664  UAVTalkReceiveObject(inConnectionHandle);
665 
666  // for remote modem
667  UAVTalkRelayPacket(inConnectionHandle, outConnectionHandle);
668  }
669 
670  }
671  break;
672 
673  default:
674  // all other packets are relayed to the telemetry port
675  UAVTalkRelayPacket(inConnectionHandle,
676  outConnectionHandle);
677  break;
678  }
679  }
680 }
