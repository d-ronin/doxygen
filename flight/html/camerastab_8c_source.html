180 static void attitudeUpdated(UAVObjEvent* ev, void *ctx, void *obj, int len)
181 {
182  (void) ev; (void) ctx; (void) obj; (void) len;
183  if (ev->obj != AttitudeActualHandle())
184  return;
185 
186  float accessories[MANUALCONTROLCOMMAND_ACCESSORY_NUMELEM];
187 
188  ManualControlCommandAccessoryGet(accessories);
189 
190  CameraStabSettingsData *settings = &csd->settings;
191 
192  // Check how long since last update, time delta between calls in ms
193  uint32_t thisSysTime = PIOS_Thread_Systime();
194  float dT_ms = thisSysTime - csd->lastSysTime;
195  csd->lastSysTime = thisSysTime;
196 
197  if (dT_ms <= 0)
198  return;
199 
200  float attitude;
201  float output;
202 
203  for (uint8_t i = 0; i < MAX_AXES; i++) {
204 
205  // Get the current attitude from the selected axis
206  switch (i) {
207  case ROLL:
208  AttitudeActualRollGet(&attitude);
209  break;
210  case PITCH:
211  AttitudeActualPitchGet(&attitude);
212  break;
213  case YAW:
214  AttitudeActualYawGet(&attitude);
215  break;
216  }
217  float rt_ms = (float)settings->AttitudeFilter;
218  csd->attitude_filtered[i] = (rt_ms / (rt_ms + dT_ms)) * csd->attitude_filtered[i] + (dT_ms / (rt_ms + dT_ms)) * attitude;
219  attitude = csd->attitude_filtered[i];
220 
221  // Compute new smoothed setpoint
222  if (settings->Input[i] != CAMERASTABSETTINGS_INPUT_NONE && settings->Input[i] != CAMERASTABSETTINGS_INPUT_POI) {
223  int idx = settings->Input[i] - CAMERASTABSETTINGS_INPUT_ACCESSORY0;
224 
225  if ((idx >= 0) && (idx < MANUALCONTROLCOMMAND_ACCESSORY_NUMELEM)) {
226  float input;
227  float input_rate;
228  rt_ms = (float) settings->InputFilter;
229  switch (settings->StabilizationMode[i]) {
230  case CAMERASTABSETTINGS_STABILIZATIONMODE_ATTITUDE:
231  input = accessories[idx] * settings->InputRange[i];
232  csd->inputs[i] = (rt_ms / (rt_ms + dT_ms)) * csd->inputs[i] + (dT_ms / (rt_ms + dT_ms)) * input;
233  break;
234  case CAMERASTABSETTINGS_STABILIZATIONMODE_AXISLOCK:
235  input_rate = accessories[idx] * settings->InputRate[i];
236  if (fabsf(input_rate) > settings->MaxAxisLockRate)
237  csd->inputs[i] = bound_sym(csd->inputs[i] + input_rate * dT_ms / 1000.0f, settings->InputRange[i]);
238  break;
239  default:
240  input = 0;
241  }
242  }
243  switch(i) {
244  case PITCH:
245  CameraDesiredDeclinationSet(&csd->inputs[i]);
246  break;
247  case YAW:
248  CameraDesiredBearingSet(&csd->inputs[i]);
249  break;
250  default:
251  break;
252  }
253  }
254 #if defined(CAMERASTAB_POI_MODE)
255  else if (settings->Input[i] == CAMERASTABSETTINGS_INPUT_POI) {
256  // Process any updates of the tablet location if it wants to
257  // be the POI
258  tablet_info_process();
259 
260  PositionActualData positionActual;
261  PositionActualGet(&positionActual);
262  PoiLocationData poi;
263  PoiLocationGet(&poi);
264 
265  float dLoc[3];
266 
267  dLoc[0] = poi.North - positionActual.North;
268  dLoc[1] = poi.East - positionActual.East;
269  dLoc[2] = poi.Down - positionActual.Down;
270 
271  // Compute the pitch and yaw to the POI location, assuming UAVO is level facing north
272  float distance = sqrtf(powf(dLoc[0], 2) + powf(dLoc[1], 2));
273  float pitch = atan2f(-dLoc[2], distance) * RAD2DEG;
274  float yaw = atan2f(dLoc[1], dLoc[0]) * RAD2DEG;
275  if (yaw < 0.0f)
276  yaw += 360.0f;
277 
278  // Only try and track objects more than 2 m away
279  if (distance > 2) {
280  switch (i) {
281  case CAMERASTABSETTINGS_INPUT_ROLL:
282  // Does not make sense to use position to control yaw
283  break;
284  case CAMERASTABSETTINGS_INPUT_PITCH:
285  // Store the absolute declination relative to UAV
286  CameraDesiredDeclinationSet(&pitch);
287  csd->inputs[CAMERASTABSETTINGS_INPUT_PITCH] = -pitch;
288  break;
289  case CAMERASTABSETTINGS_INPUT_YAW:
290  CameraDesiredBearingSet(&yaw);
291  csd->inputs[CAMERASTABSETTINGS_INPUT_YAW] = yaw;
292  break;
293  }
294  }
295  }
296 #endif /* CAMERASTAB_POI_MODE */
297 
298  // Add Servo FeedForward
299  applyFF(i, dT_ms, &attitude, settings);
300 
301  // Set output channels
302  output = bound_sym((attitude + csd->inputs[i]) / settings->OutputRange[i], 1.0f);
303  if (thisSysTime > LOAD_DELAY) {
304  switch (i) {
305  case ROLL:
306  CameraDesiredRollSet(&output);
307  break;
308  case PITCH:
309  CameraDesiredPitchSet(&output);
310  break;
311  case YAW:
312  CameraDesiredYawSet(&output);
313  break;
314  }
315  }
316  }
317 
318  // Send a message over CAN, if include
319  gimbal_can_message();
320 }

329 static void applyFF(uint8_t index, float dT_ms, float *attitude, CameraStabSettingsData* cameraStab)
330 {
331  float accumulator = csd->FFfilterAccumulator[index];
332 
333  accumulator += (*attitude - csd->FFlastAttitude[index]) * cameraStab->FeedForward[index];
334  csd->FFlastAttitude[index] = *attitude;
335  *attitude += accumulator;
336 
337  float filter = (float) cameraStab->FeedForwardTime / dT_ms;
338  if (filter > 1)
339  accumulator -= accumulator / filter;
340  else
341  accumulator = 0;
342 
343  // For non-zero filter times store accumulation
344  csd->FFfilterAccumulator[index] = accumulator;
345  *attitude += accumulator;
346 
347  //acceleration and deceleration limit
348  float delta = *attitude - csd->FFlastFilteredAttitude[index];
349  float maxDelta = cameraStab->MaxAccel * dT_ms / 1000.0f;
350  if(fabsf(delta) > maxDelta) //we are accelerating too hard
351  {
352  *attitude = csd->FFlastFilteredAttitude[index] + (delta > 0 ? maxDelta : - maxDelta);
353  }
354  csd->FFlastFilteredAttitude[index] = *attitude;
355 }

386 static void tablet_info_process() {
387 
388  // Only process new information
389  if (tablet_info_updated == false)
390  return;
391  tablet_info_updated = false;
392 
393  TabletInfoData tablet;
394  TabletInfoGet(&tablet);
395  if (tablet.Connected == TABLETINFO_CONNECTED_TRUE && tablet.POI == TABLETINFO_POI_TRUE)
396  {
397  HomeLocationData homeLocation;
398  HomeLocationGet(&homeLocation);
399 
400  PoiLocationData poi;
401 
402  float lat, alt;
403  lat = homeLocation.Latitude / 10.0e6f * DEG2RAD;
404  alt = homeLocation.Altitude;
405 
406  float T[3];
407  T[0] = alt+6.378137E6f;
408  T[1] = cosf(lat)*(alt+6.378137E6f);
409  T[2] = -1.0f;
410 
411  float dL[3] = {(tablet.Latitude - homeLocation.Latitude) / 10.0e6f * DEG2RAD,
412  (tablet.Longitude - homeLocation.Longitude) / 10.0e6f * DEG2RAD,
413  (tablet.Altitude)};
414 
415  poi.North = T[0] * dL[0];
416  poi.East = T[1] * dL[1];
417  poi.Down = T[2] * dL[2];
418  PoiLocationSet(&poi);
419  }
420 }
