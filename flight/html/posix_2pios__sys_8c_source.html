281 static void go_realtime() {
282 #ifdef __linux__
283  /* First, pin all our memory. We don't want stuff we need
284  * to get faulted out. */
285  int rc = mlockall(MCL_CURRENT | MCL_FUTURE);
286 
287  if (rc) {
288  perror("mlockall");
289  exit(1);
290  }
291 
292  /* We always run on the same processor-- why migrate when
293  * you never yield?
294  */
295 
296  cpu_set_t allowable_cpus;
297 
298  CPU_ZERO(&allowable_cpus);
299 
300  CPU_SET(0, &allowable_cpus);
301 
302  rc = sched_setaffinity(0, sizeof(allowable_cpus), &allowable_cpus);
303 
304  if (rc) {
305  perror("sched_setaffinity");
306  exit(1);
307  }
308 
309  /* Next, let's go hard realtime. */
310 
311  struct sched_param sch_p = {
312  .sched_priority = 10
313  };
314 
315  rc = sched_setscheduler(0, SCHED_RR, &sch_p);
316 
317  if (rc) {
318  perror("sched_setscheduler");
319  exit(1);
320  }
321 
322  are_realtime = true;
323 #else
324  printf("Only can do realtime stuff on Linux\n");
325  exit(1);
326 #endif
327 }

332 void PIOS_SYS_Args(int argc, char *argv[]) {
333  saved_argc = argc;
334  saved_argv = argv;
335 
336  int opt;
337 
338  bool first_arg = true;
339 
340  while ((opt = getopt(argc, argv, "frl:s:d:S:")) != -1) {
341  switch (opt) {
342  case 'f':
343  debug_fpe = true;
344  break;
345  case 'r':
346  if (!first_arg) {
347  printf("Realtime must be first arg\n");
348  exit(1);
349  }
350 
351  go_realtime();
352  break;
353  case 'l':
354  {
355  uintptr_t tmp;
356  if (PIOS_FILEOUT_Init(&tmp,
357  optarg, "w")) {
358  printf("Couldn't open logfile %s\n",
359  optarg);
360  exit(1);
361  }
362 
363  if (PIOS_COM_Init(&PIOS_COM_OPENLOG,
364  &pios_fileout_com_driver,
365  tmp,
366  0,
367  PIOS_FILEOUT_TX_BUFFER_SIZE)) {
368  printf("Couldn't init fileout com layer\n");
369  exit(1);
370  }
371  break;
372  }
373 #ifdef PIOS_INCLUDE_SERIAL
374  case 'S':
375  if (handle_serial_device(optarg)) {
376  printf("Coudln't init device\n");
377  exit(1);
378  }
379  break;
380 #endif
381 #ifdef PIOS_INCLUDE_SPI
382  case 'd':
383  if (handle_device(optarg)) {
384  printf("Couldn't init device\n");
385  exit(1);
386  }
387  break;
388  case 's':
389  {
390  struct pios_spi_cfg *spi_cfg;
391 
392  spi_cfg = PIOS_malloc(sizeof(*spi_cfg));
393 
394  strncpy(spi_cfg->base_path, optarg,
395  sizeof(spi_cfg->base_path));
396 
397  spi_cfg->base_path[sizeof(spi_cfg->base_path)-1] = 0;
398 
399  int ret = PIOS_SPI_Init(spi_devs + num_spi,
400  spi_cfg);
401 
402  if (ret < 0) {
403  printf("Couldn't init SPI\n");
404  exit(1);
405  }
406 
407  num_spi++;
408  break;
409  }
410 #endif
411 
412  default:
413  Usage(argv[0]);
414  break;
415  }
416 
417  first_arg = false;
418  }
419 
420  if (optind < argc) {
421  Usage(argv[0]);
422  }
423 }

491 int32_t PIOS_SYS_Reset(void)
492 {
493  char *argv[128];
494 
495  /* Sigh, ensure NULL termination of array like execvpe(3) expects*/
496  int num = saved_argc;
497  int i = 0;
498 
499  if (num > 128) {
500  num = 127;
501  }
502 
503  for (i = 0; i < num; i++) {
504  argv[i] = saved_argv[i];
505  }
506 
507  argv[i] = 0;
508 
509  // Close out all fds-- open or not.
510  // We don't want our existing listening socket leaking, etc.
511  // Note that we should really be using FD_CLOEXEC, but it's really
512  // tricky.. and even if we do, we should leave this to be safe.
513  for (i = STDERR_FILENO+1; i < 1024; i++) {
514  close(i);
515  }
516 
517  execvp(argv[0], argv);
518 
519  abort();
520 
521  return -1;
522 }
