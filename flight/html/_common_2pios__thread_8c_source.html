58 static uint8_t * align8_alloc(uint32_t size)
59 {
60  // round size up to at nearest multiple of 8 + 4 bytes to guarantee
61  // sufficient size within. This is because PIOS_malloc only guarantees
62  // uintptr_t alignment which is 4 bytes.
63  size = size + sizeof(uintptr_t);
64  uint8_t *wap = PIOS_malloc(size);
65 
66  // shift start point to nearest 8 byte boundary.
67  uint32_t pad = ((uint32_t) wap) % sizeof(stkalign_t);
68  wap = wap + pad;
69 
70  return wap;
71 }

176 void PIOS_Thread_Sleep_Until(uint32_t *previous_ms, uint32_t increment_ms)
177 {
178  // Do the math in the millisecond domain.
179  *previous_ms += increment_ms;
180 
181  systime_t future = CVT_MS2ST(*previous_ms);
182  systime_t increment_st = CVT_MS2ST(increment_ms);
183 
184  chSysLock();
185 
186  systime_t now = chTimeNow();
187  systime_t sleep_time = future - now;
188 
189  if (sleep_time > increment_st) {
190  // OK, the calculated sleep time is much longer than
191  // the desired interval. There's two possibilities:
192  // 1) We were already late! If so just don't sleep.
193  // 2) The timer wrapped. (at 49 days) Just don't sleep.
194 
195  if ((now - future) > increment_st) {
196  // However, in this particular case we need to fix up
197  // the clock. If we're very late, OR WRAPPED,
198  // don't try and keep on the previous timebase.
199  *previous_ms = CVT_ST2MS(now);
200  }
201  } else if (sleep_time > 0) {
202  // Be sure not to request a sleep_time of 0, as that's
203  // IMMEDIATE and makes chTdSleepS panic.
204  chThdSleepS(sleep_time);
205  }
206 
207  chSysUnlock();
208 }
