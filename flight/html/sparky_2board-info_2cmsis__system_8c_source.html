252 void SystemCoreClockUpdate (void)
253 {
254  uint32_t tmp = 0, pllmull = 0, pllsource = 0, prediv1factor = 0;
255 
256  /* Get SYSCLK source -------------------------------------------------------*/
257  tmp = RCC->CFGR & RCC_CFGR_SWS;
258 
259  switch (tmp)
260  {
261  case 0x00: /* HSI used as system clock */
262  SystemCoreClock = HSI_VALUE;
263  break;
264  case 0x04: /* HSE used as system clock */
265  SystemCoreClock = HSE_VALUE;
266  break;
267  case 0x08: /* PLL used as system clock */
268  /* Get PLL clock source and multiplication factor ----------------------*/
269  pllmull = RCC->CFGR & RCC_CFGR_PLLMULL;
270  pllsource = RCC->CFGR & RCC_CFGR_PLLSRC;
271  pllmull = ( pllmull >> 18) + 2;
272 
273  if (pllsource == 0x00)
274  {
275  /* HSI oscillator clock divided by 2 selected as PLL clock entry */
276  SystemCoreClock = (HSI_VALUE >> 1) * pllmull;
277  }
278  else
279  {
280  prediv1factor = (RCC->CFGR2 & RCC_CFGR2_PREDIV1) + 1;
281  /* HSE oscillator clock selected as PREDIV1 clock entry */
282  SystemCoreClock = (HSE_VALUE / prediv1factor) * pllmull;
283  }
284  break;
285  default: /* HSI used as system clock */
286  SystemCoreClock = HSI_VALUE;
287  break;
288  }
289  /* Compute HCLK clock frequency ----------------*/
290  /* Get HCLK prescaler */
291  tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
292  /* HCLK clock frequency */
293  SystemCoreClock >>= tmp;
294 }
