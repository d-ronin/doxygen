158 #define PIOS_SYSCLK 168000000
159 // Peripherals that belongs to APB1 are:
160 // DAC |PWR |CAN1,2
161 // I2C1,2,3 |UART4,5 |USART3,2
162 // I2S3Ext |SPI3/I2S3 |SPI2/I2S2
163 // I2S2Ext |IWDG |WWDG
164 // RTC/BKP reg
165 // TIM2,3,4,5,6,7,12,13,14
166 
167 // Calculated as SYSCLK / APBPresc * (APBPre == 1 ? 1 : 2)
168 // Default APB1 Prescaler = 4
169 #define PIOS_PERIPHERAL_APB1_CLOCK (PIOS_SYSCLK / 4)
170 
171 // Peripherals belonging to APB2
172 // SDIO |EXTI |SYSCFG |SPI1
173 // ADC1,2,3
174 // USART1,6
175 // TIM1,8,9,10,11
176 //
177 // Default APB2 Prescaler = 2
178 //
179 #define PIOS_PERIPHERAL_APB2_CLOCK (PIOS_SYSCLK / 2)
