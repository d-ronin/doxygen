204 static int BankSetup_OneShot(struct timer_bank *bank, int32_t max_tim_clock)
205 {
206  /*
207  * Ensure a dead time of 2% + 10 us, e.g. 15us for 250us
208  * long pulses, 50 us for 2000us long pulses
209  */
210  float period = 1.02f * bank->max_pulse + 10.0f;
211  float num_ticks = period / 1e6f * max_tim_clock;
212  // assume 16-bit timer
213  bank->prescaler = num_ticks / 0xffff + 0.5f;
214  bank->clk_rate = max_tim_clock / (bank->prescaler + 1);
215  bank->period = period / 1e6f * bank->clk_rate;
216 
217  // select one pulse mode for SyncPWM
218  TIM_SelectOnePulseMode(bank->timer, TIM_OPMode_Single);
219  // Need to invert output polarity for one-pulse mode
220  uint16_t inverted_polarity = (servo_cfg->tim_oc_init.TIM_OCPolarity == TIM_OCPolarity_Low) ?
221  TIM_OCPolarity_High : TIM_OCPolarity_Low;
222  TIM_OC1PolarityConfig(bank->timer, inverted_polarity);
223  TIM_OC2PolarityConfig(bank->timer, inverted_polarity);
224  TIM_OC3PolarityConfig(bank->timer, inverted_polarity);
225  TIM_OC4PolarityConfig(bank->timer, inverted_polarity);
226 
227  return 0;
228 }

230 static int BankSetup_PWM(struct timer_bank *bank, int32_t max_tim_clock,
231  uint16_t rate)
232 {
233  // assume 16-bit timer
234  if (servo_cfg->force_1MHz) {
235  bank->prescaler = max_tim_clock / 1000000 - 1;
236  } else {
237  float num_ticks = (float)max_tim_clock / (float)rate;
238 
239  bank->prescaler = num_ticks / 0xffff + 0.5f;
240  }
241 
242  bank->clk_rate = max_tim_clock / (bank->prescaler + 1);
243  bank->period = (float)bank->clk_rate / (float)rate;
244 
245  // de-select one pulse mode in case SyncPWM was previously used
246  TIM_SelectOnePulseMode(bank->timer, TIM_OPMode_Repetitive);
247 
248  // and make sure the output is enabled
249  TIM_Cmd(bank->timer, ENABLE);
250 
251  // restore polarity in case one-pulse was used
252  TIM_OC1PolarityConfig(bank->timer, servo_cfg->tim_oc_init.TIM_OCPolarity);
253  TIM_OC2PolarityConfig(bank->timer, servo_cfg->tim_oc_init.TIM_OCPolarity);
254  TIM_OC3PolarityConfig(bank->timer, servo_cfg->tim_oc_init.TIM_OCPolarity);
255  TIM_OC4PolarityConfig(bank->timer, servo_cfg->tim_oc_init.TIM_OCPolarity);
256 
257  return 0;

273 int PIOS_Servo_SetMode(const uint16_t *out_rate, const int banks, const uint16_t *channel_max, const uint16_t *channel_min)
274 {
275  if (!servo_cfg || banks > PIOS_SERVO_MAX_BANKS) {
276  return -10;
277  }
278 
279  dshot_in_use = false;
280 
281  struct timer_bank timer_banks[PIOS_SERVO_MAX_BANKS];
282 
283  PIOS_TIM_InitAllTimerPins(servo_tim_id);
284 
285  memset(&timer_banks, 0, sizeof(timer_banks));
286  int banks_found = 0;
287 
288  // find max pulse length for each bank
289  for (int i = 0; i < servo_cfg->num_channels && banks_found < banks; i++) {
290  int bank = -1;
291  const struct pios_tim_channel *chan = &servo_cfg->channels[i];
292  for (int j = 0; j < banks_found; j++) {
293  if (timer_banks[j].timer == chan->timer)
294  bank = j;
295  }
296 
297  if (bank < 0)
298  bank = banks_found++;
299 
300  timer_banks[bank].timer = chan->timer;
301  timer_banks[bank].max_pulse = MAX(timer_banks[bank].max_pulse, channel_max[i]);
302  timer_banks[bank].max_pulse = MAX(timer_banks[bank].max_pulse, channel_min[i]);
303  }
304 
305  // configure timers/banks
306  TIM_TimeBaseInitTypeDef TIM_TimeBaseStructure = servo_cfg->tim_base_init;
307  TIM_TimeBaseStructure.TIM_ClockDivision = TIM_CKD_DIV1;
308  TIM_TimeBaseStructure.TIM_CounterMode = TIM_CounterMode_Up;
309  for (int i = 0; i < banks_found; i++) {
310  // Skip the bank if no outputs are configured
311  if (timer_banks[i].max_pulse == 0)
312  continue;
313 
314  /* Calculate the maximum clock frequency for the timer */
315  uint32_t max_tim_clock = max_timer_clock(timer_banks[i].timer);
316  // check if valid timer
317  if (!max_tim_clock) {
318  return -20;
319  }
320 
321  uint16_t rate = out_rate[i];
322 
323  if (servo_cfg->force_1MHz && (rate == SHOT_ONESHOT)) {
324  /* We've been asked for syncPWM but are in a config
325  * where we can't do it. This means CC3D + 333Hz.
326  * Getting fast output is functionally identical
327  * to oneshot on a target like this. Pick a period
328  * that has a lot of deadtime and call it good.
329  */
330 
331  /* Works out to 2500Hz at normal 250us max pulse,
332  * 400Hz at 2000us.
333  *
334  * Put one more way, with "oneshot", 400us of variable
335  * delay on 3300us control period.
336  */
337 
338  rate = 1000000 / (timer_banks[i].max_pulse +
339  timer_banks[i].max_pulse / 5 +
340  100);
341  }
342 
343  int ret;
344 
345  // output rate of 0 means SyncPWM
346  switch (rate) {
347  case SHOT_ONESHOT:
348  ret = BankSetup_OneShot(&timer_banks[i], max_tim_clock);
349 
350  if (ret) {
351  return ret;
352  }
353 
354  TIM_TimeBaseStructure.TIM_Prescaler = timer_banks[i].prescaler;
355  TIM_TimeBaseStructure.TIM_Period = timer_banks[i].period;
356 
357  // Configure this timer appropriately.
358  TIM_TimeBaseInit(timer_banks[i].timer, &TIM_TimeBaseStructure);
359  break;
360 
361  default:
362  ret = BankSetup_PWM(&timer_banks[i], max_tim_clock, rate);
363 
364  if (ret) {
365  return ret;
366  }
367 
368  TIM_TimeBaseStructure.TIM_Prescaler = timer_banks[i].prescaler;
369  TIM_TimeBaseStructure.TIM_Period = timer_banks[i].period;
370 
371  // Configure this timer appropriately.
372  TIM_TimeBaseInit(timer_banks[i].timer, &TIM_TimeBaseStructure);
373  break;
374 
375  case SHOT_DSHOT300:
376  case SHOT_DSHOT600:
377  case SHOT_DSHOT1200:
378  dshot_in_use = true;
379 
380  ret = BankSetup_DShot(&timer_banks[i]);
381  break;
382  }
383 
384 
385  /* Configure frequency scaler for all channels that use the same timer */
386  for (uint8_t j = 0; j < servo_cfg->num_channels; j++) {
387  const struct pios_tim_channel *chan = &servo_cfg->channels[j];
388  if (timer_banks[i].timer == chan->timer) {
389  /* save the frequency for these channels */
390  switch (rate) {
391  case SHOT_DSHOT300:
392  case SHOT_DSHOT600:
393  case SHOT_DSHOT1200:
394  ChannelSetup_DShot(j, rate);
395  break;
396  case SHOT_ONESHOT:
397  output_channels[j].i.pwm_res = timer_banks[i].clk_rate;
398  output_channels[j].mode = SYNC_PWM;
399  break;
400  default:
401  output_channels[j].i.pwm_res = timer_banks[i].clk_rate;
402  if (channel_max[j] >= channel_min[j]) {
403  output_channels[j].mode = REGULAR_PWM;
404  } else {
405  output_channels[j].mode = REGULAR_INVERTED_PWM;
406  }
407  break;
408  }
409  }
410  }
411  }
412 
413  return 0;
414 }

441 // Fraction: 0.16
442 // max_val, min_val: microseconds (seconds * 1000000)
443 void PIOS_Servo_SetFraction(uint8_t servo, uint16_t fraction,
444  uint16_t max_val, uint16_t min_val)
445 {
446  PIOS_Assert(max_val >= min_val);
447 
448  /* Make sure servo exists */
449  if (!servo_cfg || servo >= servo_cfg->num_channels ||
450  output_channels[servo].mode == UNCONFIGURED) {
451  return;
452  }
453 
454  switch (output_channels[servo].mode) {
455  case SYNC_DSHOT_300:
456  case SYNC_DSHOT_600:
457  case SYNC_DSHOT_1200:
458  /* Expect a 0 to 2047 range!
459  * Don't bother with min/max tomfoolery here.
460  */
461  output_channels[servo].i.dshot.value = fraction >> 5;
462  return;
463  default:
464  break;
465  };
466 
467  // Seconds * 1000000 : 16.0
468  uint16_t spread = max_val - min_val;
469 
470  uint64_t val = min_val;
471  val = val << 16;
472  // Seconds * 1000000 : 16.16
473 
474  val += spread * fraction;
475 
476  // Multiply by ticks/second to get: Ticks * 1000000: 36.16
477  val *= output_channels[servo].i.pwm_res;
478 
479  // Ticks: 16.16
480  val /= 1000000;
481 
482  // Round to nearest, 16.0
483  val += 32767;
484  val = val >> 16;
485 
486  if (resetting && (fraction < 0xff00)) {
487  /* If we're resetting and the value isn't driven to the maximum
488  * value, drive low with no pulses. If it's at the max value,
489  * it could be driving brushed motors and forced low is bad.
490  */
491  val = 0;
492  }
493 
494  PIOS_Servo_SetRaw(servo, val);
495 }

502 void PIOS_Servo_Set(uint8_t servo, float position)
503 {
504  /* Make sure servo exists */
505  if (!servo_cfg || servo >= servo_cfg->num_channels ||
506  output_channels[servo].mode == UNCONFIGURED) {
507  return;
508  }
509 
510  switch (output_channels[servo].mode) {
511  case SYNC_DSHOT_300:
512  case SYNC_DSHOT_600:
513  case SYNC_DSHOT_1200:
514  /* Expect a 0 to 2047 range! */
515  if (position > 2047) {
516  output_channels[servo].i.dshot.value = 2047;
517  } else if (position < 0) {
518  output_channels[servo].i.dshot.value = 0;
519  } else {
520  output_channels[servo].i.dshot.value = position;
521  }
522 
523  return;
524  default:
525  break;
526  };
527 
528 
529  /* recalculate the position value based on timer clock rate */
530  /* position is in us. */
531  float us_to_count = output_channels[servo].i.pwm_res / 1000000.0f;
532  position = position * us_to_count;
533 
534  if (resetting && (output_channels[servo].mode != REGULAR_INVERTED_PWM)) {
535  /* Don't nail inverted channels low during reset, because they
536  * could be driving brushed motors.
537  */
538  position = 0;
539  }
540 
541  PIOS_Servo_SetRaw(servo, position);
542 }

550 static int DSHOT_Update()
551 {
552  uint16_t dshot_bits[DS_GPIO_NUMVALS] = { 0 };
553  uint16_t dshot_msg_zeroes[16][DS_GPIO_NUMVALS] = { { 0 } };
554 
555  enum channel_mode dshot_mode = UNCONFIGURED;
556 
557  for (int i = 0; i < servo_cfg->num_channels; i++) {
558  struct output_channel *chan = &output_channels[i];
559 
560  switch (chan->mode) {
561  case SYNC_DSHOT_300:
562  case SYNC_DSHOT_600:
563  case SYNC_DSHOT_1200:
564  break;
565  default:
566  continue; // Continue enclosing loop if not ds
567  }
568 
569  if (dshot_mode == UNCONFIGURED) {
570  dshot_mode = chan->mode;
571  } else if (dshot_mode != chan->mode) {
572  /* Multiple dshot rates not supported */
573  return -1;
574  }
575 
576  struct dshot_info *info = &chan->i.dshot;
577 
578  PIOS_Assert(info->gpio >= 0);
579  PIOS_Assert(info->gpio < DS_GPIO_NUMVALS);
580 
581  dshot_bits[info->gpio] |= info->pin;
582 
583  PIOS_Assert(info->value < 2048);
584 
585  uint16_t message = info->value << 5;
586  /* Don't set telem req bit for now */
587 
588  message |=
589  ((message >> 4 ) & 0xf) ^
590  ((message >> 8 ) & 0xf) ^
591  ((message >> 12) & 0xf);
592 
593  for (int j = 0; j < 16; j++) {
594  if (! (message & 0x8000)) {
595  dshot_msg_zeroes[j][info->gpio] |= info->pin;
596  }
597 
598  message <<= 1;
599  }
600 
601  info->value = 0; /* turn off motor if not updated */
602  }
603 
604  uint16_t time_0, time_1, time_tot;
605 
606  /* As specified by original document:
607  * DShot 300 - 1250ns 0, 2500ns 1, 3333ns total
608  * DShot 600 - 625ns 0, 1250ns 1, 1667ns total
609  * DShot 1200- 312ns 0, 625ns 1, 833ns total
610  *
611  * Below timings are as used by Betaflight, which blheli_s seems to
612  * demand. Meh.
613  * DShot 300 - 1000ns 0, 2166ns 1, 3333ns total
614  * DShot 600 - 500ns 0, 1083ns 1, 1667ns total
615  * DShot 1200- 250ns 0, 541ns 1, 833ns total
616  */
617 
618  switch (dshot_mode) {
619  case UNCONFIGURED:
620  return -1;
621  case SYNC_DSHOT_300:
622  time_0 = PIOS_INLINEDELAY_NsToCycles(1000);
623  time_1 = PIOS_INLINEDELAY_NsToCycles(2166);
624  time_tot = PIOS_INLINEDELAY_NsToCycles(3333);
625  break;
626  case SYNC_DSHOT_600:
627  time_0 = PIOS_INLINEDELAY_NsToCycles(500);
628  time_1 = PIOS_INLINEDELAY_NsToCycles(1083);
629  time_tot = PIOS_INLINEDELAY_NsToCycles(1667);
630  break;
631  case SYNC_DSHOT_1200:
632  time_0 = PIOS_INLINEDELAY_NsToCycles(250);
633  time_1 = PIOS_INLINEDELAY_NsToCycles(541);
634  time_tot = PIOS_INLINEDELAY_NsToCycles(833);
635  break;
636  default:
637  PIOS_Assert(0);
638  break;
639  }
640 
641  if (dshot_mode == UNCONFIGURED) {
642  return -1;
643  }
644 
645  PIOS_IRQ_Disable(); // Necessary for critical timing below
646 
647  register uint16_t a = dshot_bits[DS_GPIOA];
648  register uint16_t b = dshot_bits[DS_GPIOB];
649  register uint16_t c = dshot_bits[DS_GPIOC];
650 
651  uint32_t start_cycle = PIOS_INLINEDELAY_GetCycleCnt() + 30;
652 
653  /* Time to get down to business of bitbanging this out. */
654  for (int i = 0; i < 16; i++) {
655  // Always delay at the beginning for consistent timing
656  PIOS_INLINEDELAY_TillCycleCnt(start_cycle);
657 
658 #if defined(STM32F40_41xxx) || defined(STM32F446xx)
659 #define CLEAR_BITS(gpio, bits) do { (gpio)->BSRRH = bits; } while (0)
660 #define SET_BITS(gpio, bits) do { (gpio)->BSRRL = bits; } while (0)
661 #else
662  // Many STM32F* do not allow halfword writes to BSRR (or have
663  // BSRRL/H definitions), so do it this way:
664 
665 #define CLEAR_BITS(gpio, bits) do { (gpio)->BSRR = ((uint32_t) bits) << 16; } while (0)
666 #define SET_BITS(gpio, bits) do { (gpio)->BSRR = bits; } while (0)
667 #endif
668 
669  // Start pulse by writing to BSRRL's
670  SET_BITS(GPIOA, a);
671  SET_BITS(GPIOB, b);
672  SET_BITS(GPIOC, c);
673 
674  // Calculate falling edge time for 0's
675  uint32_t next_tick = start_cycle + time_0;
676 
677  // And preload next values before waiting
678  a = dshot_msg_zeroes[i][DS_GPIOA];
679  b = dshot_msg_zeroes[i][DS_GPIOB];
680  c = dshot_msg_zeroes[i][DS_GPIOC];
681 
682  PIOS_INLINEDELAY_TillCycleCnt(next_tick);
683 
684  // Generate falling edges for zeroes (shorter pulses)
685  CLEAR_BITS(GPIOA, a);
686  CLEAR_BITS(GPIOB, b);
687  CLEAR_BITS(GPIOC, c);
688 
689  // Calculate falling edge time for 1's
690  next_tick = start_cycle + time_1;
691 
692  // Adjust next cycle time for when next bit should begin
693  start_cycle += time_tot;
694

698  b = dshot_bits[DS_GPIOB];
699  c = dshot_bits[DS_GPIOC];

702 
703  // Falling edges for all remaining bits.
704  CLEAR_BITS(GPIOA, a);
705  CLEAR_BITS(GPIOB, b);
706  CLEAR_BITS(GPIOC, c);
707  }
708

710
