161 int32_t PIOS_MPU9250_Init(uint32_t i2c_id, uint8_t i2c_addr, bool use_mag, enum pios_mpu9250_gyro_filter gyro_filter,
162  enum pios_mpu9250_accel_filter accel_filter, const struct pios_mpu9250_cfg * cfg)
163 {
164  dev = PIOS_MPU9250_alloc(use_mag);
165  if (dev == NULL)
166  return -1;
167 
168  dev->i2c_id = i2c_id;
169  dev->i2c_addr = i2c_addr;
170  dev->cfg = cfg;
171 
172  /* Configure the MPU9250 Sensor */
173  if (PIOS_MPU9250_Config(cfg, use_mag, gyro_filter, accel_filter) != 0)
174  return -2;
175 
176  /* Set up EXTI line */
177  PIOS_EXTI_Init(cfg->exti_cfg);
178 
179  // Wait 5 ms for data ready interrupt and make sure it happens
180  // twice
181  if ((PIOS_Semaphore_Take(dev->data_ready_sema, 5) != true) ||
182  (PIOS_Semaphore_Take(dev->data_ready_sema, 5) != true)) {
183  return -10;
184  }
185 
186  dev->TaskHandle = PIOS_Thread_Create(
187  PIOS_MPU9250_Task, "pios_mpu9250", MPU9250_TASK_STACK, NULL, MPU9250_TASK_PRIORITY);
188  PIOS_Assert(dev->TaskHandle != NULL)
189 
190 
191  PIOS_SENSORS_Register(PIOS_SENSOR_ACCEL, dev->accel_queue);
192  PIOS_SENSORS_Register(PIOS_SENSOR_GYRO, dev->gyro_queue);
193 
194  if (use_mag)
195  PIOS_SENSORS_Register(PIOS_SENSOR_MAG, dev->mag_queue);
196 
197  return 0;
198 }

206 static int32_t PIOS_MPU9250_Config(struct pios_mpu9250_cfg const * cfg, bool use_mag,
207  enum pios_mpu9250_gyro_filter gyro_filter, enum pios_mpu9250_accel_filter accel_filter)
208 {
209  // Reset chip
210  if (PIOS_MPU9250_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, PIOS_MPU60X0_PWRMGMT_IMU_RST) != 0)
211  return -1;
212 
213  // Give chip some time to initialize
214  PIOS_DELAY_WaitmS(50);
215  PIOS_WDG_Clear();
216 
217  //Power management configuration
218  PIOS_MPU9250_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, cfg->Pwr_mgmt_clk);
219 
220  // User control
221  PIOS_MPU9250_SetReg(PIOS_MPU60X0_USER_CTRL_REG, cfg->User_ctl & ~0x40);
222 
223  // Digital low-pass filter and scale
224  // set this before sample rate else sample rate calculation will fail
225  PIOS_MPU9250_SetGyroLPF(gyro_filter);
226  PIOS_MPU9250_SetAccelLPF(accel_filter);
227 
228  // Sample rate
229  PIOS_MPU9250_SetSampleRate(cfg->default_samplerate);
230 
231  // Set the gyro scale
232  PIOS_MPU9250_SetGyroRange(PIOS_MPU60X0_SCALE_500_DEG);
233 
234  // Set the accel scale
235  PIOS_MPU9250_SetAccelRange(PIOS_MPU60X0_ACCEL_8G);
236 
237  // Interrupt configuration
238  // XXX why 0x02??? this is a reserved bit
239  PIOS_MPU9250_SetReg(PIOS_MPU60X0_INT_CFG_REG, cfg->interrupt_cfg | 0x02);
240 
241  if (use_mag) {
242  // To enable access to the mag on auxillary i2c we must set bit 0x02 in register 0x37
243  // and clear bit 0x40 in register 0x6a (PIOS_MPU60X0_USER_CTRL_REG, default condition)
244 
245  // Disable mag first
246  if (PIOS_MPU9250_Mag_SetReg(MPU9250_MAG_CNTR, 0x00) != 0)
247  return -1;
248  PIOS_DELAY_WaitmS(20);
249  PIOS_WDG_Clear();
250  // Clear status registers
251  PIOS_MPU9250_Mag_GetReg(MPU9250_MAG_STATUS);
252  PIOS_MPU9250_Mag_GetReg(MPU9250_MAG_STATUS2);
253  PIOS_MPU9250_Mag_GetReg(MPU9250_MAG_XH);
254 
255  // Trigger first measurement
256  if (PIOS_MPU9250_Mag_SetReg(MPU9250_MAG_CNTR, 0x01) != 0)
257  return -1;
258  }
259 
260  // Interrupt enable
261  PIOS_MPU9250_SetReg(PIOS_MPU60X0_INT_EN_REG, cfg->interrupt_en);
262 
263  return 0;
264 }

314 int32_t PIOS_MPU9250_SetSampleRate(uint16_t samplerate_hz)
315 {
316  uint16_t filter_frequency = 1000;
317 
318  if ((dev->gyro_filter == PIOS_MPU9250_GYRO_LOWPASS_250_HZ) || (dev->gyro_filter == PIOS_MPU9250_GYRO_LOWPASS_3600_HZ))
319  // the sample rate divisor cannot be used in this case..
320  return 0;
321 
322  // limit samplerate to filter frequency
323  if (samplerate_hz > filter_frequency)
324  samplerate_hz = filter_frequency;
325 
326  // calculate divisor, round to nearest integeter
327  int32_t divisor = (int32_t)(((float)filter_frequency / samplerate_hz) + 0.5f) - 1;
328 
329  // limit resulting divisor to register value range
330  if (divisor < 0)
331  divisor = 0;
332 
333  if (divisor > 0xff)
334  divisor = 0xff;
335 
336  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_ACCEL, filter_frequency / (1+divisor));
337  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_GYRO, filter_frequency / (1+divisor));
338 
339  return PIOS_MPU9250_SetReg(PIOS_MPU60X0_SMPLRT_DIV_REG, (uint8_t)divisor);
340 }

675 static void PIOS_MPU9250_Task(void *parameters)
676 {
677  while (1) {
678  //Wait for data ready interrupt
679  if (PIOS_Semaphore_Take(dev->data_ready_sema, PIOS_SEMAPHORE_TIMEOUT_MAX) != true)
680  continue;
681  enum {
682  IDX_ACCEL_XOUT_H = 0,
683  IDX_ACCEL_XOUT_L,
684  IDX_ACCEL_YOUT_H,
685  IDX_ACCEL_YOUT_L,
686  IDX_ACCEL_ZOUT_H,
687  IDX_ACCEL_ZOUT_L,
688  IDX_TEMP_OUT_H,
689  IDX_TEMP_OUT_L,
690  IDX_GYRO_XOUT_H,
691  IDX_GYRO_XOUT_L,
692  IDX_GYRO_YOUT_H,
693  IDX_GYRO_YOUT_L,
694  IDX_GYRO_ZOUT_H,
695  IDX_GYRO_ZOUT_L,
696  BUFFER_SIZE,
697  };
698 
699 
700  uint8_t mpu9250_rec_buf[BUFFER_SIZE];
701 
702  if (PIOS_MPU9250_Read(PIOS_MPU60X0_ACCEL_X_OUT_MSB, mpu9250_rec_buf, sizeof(mpu9250_rec_buf)) < 0) {
703  continue;
704  }
705 
706  // Rotate the sensor to OP convention. The datasheet defines X as towards the right
707  // and Y as forward. OP convention transposes this. Also the Z is defined negatively
708  // to our convention
709 
710  // Currently we only support rotations on top so switch X/Y accordingly.
711  struct pios_sensor_accel_data accel_data;
712  struct pios_sensor_gyro_data gyro_data;
713
