173 static void PIOS_OpenLRS_Rcvr_Supervisor(uintptr_t openlrs_rcvr_id)
174 {
175  /* Recover our device context */
176  struct pios_openlrs_rcvr_dev *openlrs_rcvr_dev =
177  (struct pios_openlrs_rcvr_dev *)openlrs_rcvr_id;
178 
179  if (!PIOS_OpenLRS_Rcvr_Validate(openlrs_rcvr_dev)) {
180  /* Invalid device specified */
181  return;
182  }
183 
184  /*
185  * RTC runs at 625Hz.
186  */
187  if (++(openlrs_rcvr_dev->supv_timer) <
188  (PIOS_OPENLRS_RCVR_TIMEOUT_MS * 1000 / 625)) {
189  return;
190  }
191  openlrs_rcvr_dev->supv_timer = 0;
192 
193  if (!openlrs_rcvr_dev->fresh) {
194  for (int32_t i = 0; i < RFM22BRECEIVER_CHANNEL_NUMELEM;
195  i++) {
196  openlrs_rcvr_dev->channels[i] = PIOS_RCVR_TIMEOUT;
197  }
198 
199  }
200 
201  openlrs_rcvr_dev->fresh = false;
202 }

204 static void openlrs_rcvr_update_uavo(struct pios_openlrs_rcvr_dev *rcvr_dev)
205 {
206  // Also store the received data in a UAVO for easy
207  // debugging. However this is not what is used in
208  // ManualControl (it fetches directly from this driver)
209  RFM22BReceiverData rcvr;
210  for (uint8_t i = 0; i < OPENLRS_PPM_NUM_CHANNELS; i++) {
211  if (i < RFM22BRECEIVER_CHANNEL_NUMELEM)
212  rcvr.Channel[i] = rcvr_dev->channels[i];
213  }
214  for (int i = OPENLRS_PPM_NUM_CHANNELS - 1; i < RFM22BRECEIVER_CHANNEL_NUMELEM; i++)
215  rcvr.Channel[i] = PIOS_RCVR_INVALID;
216  RFM22BReceiverSet(&rcvr);
217 }
