222 static void PIOS_HMC5843_Config(PIOS_HMC5843_ConfigTypeDef * HMC5843_Config_Struct)
223 {
224  uint8_t CRTLA = 0x00;
225  uint8_t CRTLB = 0x00;
226  uint8_t MODE = 0x00;
227 
228  CRTLA |= (uint8_t) (HMC5843_Config_Struct->M_ODR | HMC5843_Config_Struct->Meas_Conf);
229  CRTLB |= (uint8_t) (HMC5843_Config_Struct->Gain);
230  MODE |= (uint8_t) (HMC5843_Config_Struct->Mode);
231 
232  // CRTL_REGA
233  while (!PIOS_HMC5843_Write(PIOS_HMC5843_CONFIG_REG_A, CRTLA)) ;
234 
235  // CRTL_REGB
236  while (!PIOS_HMC5843_Write(PIOS_HMC5843_CONFIG_REG_B, CRTLB)) ;
237 
238  // Mode register
239  while (!PIOS_HMC5843_Write(PIOS_HMC5843_MODE_REG, MODE)) ;
240 }

245 void PIOS_HMC5843_ReadMag(int16_t out[3])
246 {
247  uint8_t buffer[6];
248  uint8_t crtlB;
249 
250  pios_hmc5843_data_ready = false;
251 
252  while (!PIOS_HMC5843_Read(PIOS_HMC5843_CONFIG_REG_B, &crtlB, 1)) ;
253  while (!PIOS_HMC5843_Read(PIOS_HMC5843_DATAOUT_XMSB_REG, buffer, 6)) ;
254 
255  switch (crtlB & 0xE0) {
256  case 0x00:
257  for (int i = 0; i < 3; i++)
258  out[i] = ((int16_t) ((uint16_t) buffer[2 * i] << 8)
259  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_0_7Ga;
260  break;
261  case 0x20:
262  for (int i = 0; i < 3; i++)
263  out[i] = ((int16_t) ((uint16_t) buffer[2 * i] << 8)
264  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_1Ga;
265  break;
266  case 0x40:
267  for (int i = 0; i < 3; i++)
268  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)
269  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_1_5Ga;
270  break;
271  case 0x60:
272  for (int i = 0; i < 3; i++)
273  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)
274  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_2Ga;
275  break;
276  case 0x80:
277  for (int i = 0; i < 3; i++)
278  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)

280  break;
281  case 0xA0:
