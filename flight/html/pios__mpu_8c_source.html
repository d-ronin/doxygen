242 static int32_t PIOS_MPU_Config(struct pios_mpu_cfg const *cfg)
243 {
244  // reset chip
245  if (PIOS_MPU_WriteReg(PIOS_MPU_PWR_MGMT_REG, PIOS_MPU_PWRMGMT_IMU_RST) != 0)
246  return -PIOS_MPU_ERROR_WRITEFAILED;
247 
248  // give chip some time to initialize
249  PIOS_DELAY_WaitmS(50);
250 
251  // power management config
252  if (PIOS_MPU_WriteReg(PIOS_MPU_PWR_MGMT_REG, PIOS_MPU_PWRMGMT_PLL_X_CLK) != 0)
253  return -PIOS_MPU_ERROR_WRITEFAILED;
254 
255  // user control
256  if (mpu_dev->com_driver_type == PIOS_MPU_COM_SPI) {
257  if (PIOS_MPU_WriteReg(PIOS_MPU_USER_CTRL_REG, PIOS_MPU_USERCTL_DIS_I2C | PIOS_MPU_USERCTL_I2C_MST_EN) != 0)
258  return -PIOS_MPU_ERROR_WRITEFAILED;
259  } else {
260  if (PIOS_MPU_WriteReg(PIOS_MPU_USER_CTRL_REG, PIOS_MPU_USERCTL_I2C_MST_EN) != 0)
261  return -PIOS_MPU_ERROR_WRITEFAILED;
262  }
263 
264  // Digital low-pass filter and scale
265  // set this before sample rate else sample rate calculation will fail
266  PIOS_MPU_SetGyroBandwidth(184);
267  PIOS_MPU_SetAccelBandwidth(184);
268 
269  // Sample rate
270  if (PIOS_MPU_SetSampleRate(cfg->default_samplerate) != 0)
271  return -PIOS_MPU_ERROR_SAMPLERATE;
272 
273  // Set the gyro scale
274  PIOS_MPU_SetGyroRange(PIOS_MPU_SCALE_1000_DEG);
275 
276  // Set the accel scale
277  PIOS_MPU_SetAccelRange(PIOS_MPU_SCALE_8G);
278 
279  // Interrupt configuration
280  PIOS_MPU_WriteReg(PIOS_MPU_INT_CFG_REG, PIOS_MPU_INT_CLR_ANYRD);
281 
282  // Interrupt enable
283  PIOS_MPU_WriteReg(PIOS_MPU_INT_EN_REG, PIOS_MPU_INTEN_DATA_RDY);
284 
285  return 0;
286 }

295 static int32_t PIOS_MPU_Mag_WriteReg(uint8_t reg, uint8_t data)
296 {
297  // we will use I2C SLV4 to manipulate with AK8963 control registers
298  if (PIOS_MPU_WriteReg(PIOS_MPU_SLV4_REG_REG, reg) != 0)
299  return -1;
300  PIOS_MPU_WriteReg(PIOS_MPU_SLV4_REG_REG, reg);
301  PIOS_MPU_WriteReg(PIOS_MPU_SLV4_ADDR_REG, PIOS_MPU_AK89XX_ADDR);
302  PIOS_MPU_WriteReg(PIOS_MPU_SLV4_DO_REG, data);
303  PIOS_MPU_WriteReg(PIOS_MPU_SLV4_CTRL_REG, PIOS_MPU_I2CSLV_EN | 9);
304 
305  // wait for I2C transaction done, use simple safety
306  // escape counter to prevent endless loop in case
307  // MPU is broken
308  uint8_t status = 0;
309  uint32_t timeout = 0;
310  do {
311  if (timeout++ > 50)
312  return -2;
313 
314  status = PIOS_MPU_ReadReg(PIOS_MPU_I2C_MST_STATUS_REG);
315  } while ((status & PIOS_MPU_I2C_MST_SLV4_DONE) == 0);
316 
317  if (status & PIOS_MPU_I2C_MST_SLV4_NACK)
318  return -3;
319 
320  return 0;
321 }

423 static int32_t PIOS_MPU_Common_Init(void)
424 {
425  /* Configure the MPU Sensor */
426  if (PIOS_MPU_Config(mpu_dev->cfg) != 0)
427  return -PIOS_MPU_ERROR_NOCONFIG;
428 
429 #ifdef PIOS_INCLUDE_MPU_MAG
430  /* Probe for mag */
431  if (mpu_dev->cfg->use_internal_mag && PIOS_MPU_Mag_Probe() == 0) {
432  PIOS_MPU_Mag_Alloc(mpu_dev);
433  if (mpu_dev->mpu_type == PIOS_MPU60X0)
434  mpu_dev->mpu_type = PIOS_MPU9150;
435  if (PIOS_MPU_Mag_Config() != 0)
436  return -PIOS_MPU_ERROR_MAGFAILED;
437  } else {
438  mpu_dev->use_mag = false;
439  if (mpu_dev->mpu_type == PIOS_MPU9250)
440  return -PIOS_MPU_ERROR_MAGFAILED;
441  }
442 #endif // PIOS_INCLUDE_MPU_MAG
443 
444  /* Set up EXTI line */
445  PIOS_EXTI_Init(mpu_dev->cfg->exti_cfg);
446 
447  /* Wait 20 ms for data ready interrupt and make sure it happens twice */
448  if (!mpu_dev->cfg->skip_startup_irq_check) {
449  for (int i=0; i<2; i++) {
450  uint32_t ref_val = mpu_dev->interrupt_count;
451  uint32_t raw_start = PIOS_DELAY_GetRaw();
452 
453  while (mpu_dev->interrupt_count == ref_val) {
454  if (PIOS_DELAY_DiffuS(raw_start) > 20000) {
455  PIOS_EXTI_DeInit(mpu_dev->cfg->exti_cfg);
456  return -PIOS_MPU_ERROR_NOIRQ;
457  }
458  }
459  }
460  }
461 
462  mpu_dev->task_handle = PIOS_Thread_Create(
463  PIOS_MPU_Task, "pios_mpu", PIOS_MPU_TASK_STACK, NULL, PIOS_MPU_TASK_PRIORITY);
464  PIOS_Assert(mpu_dev->task_handle != NULL);
465  TaskMonitorAdd(TASKINFO_RUNNING_IMU, mpu_dev->task_handle);
466 
467  PIOS_SENSORS_Register(PIOS_SENSOR_ACCEL, mpu_dev->accel_queue);
468  PIOS_SENSORS_Register(PIOS_SENSOR_GYRO, mpu_dev->gyro_queue);
469 #ifdef PIOS_INCLUDE_MPU_MAG
470  if (mpu_dev->use_mag)
471  PIOS_SENSORS_Register(PIOS_SENSOR_MAG, mpu_dev->mag_queue);
472 #endif // PIOS_INCLUDE_MPU_MAG
473 
474  mpu_dev->accel_range = PIOS_MPU_SCALE_8G;
475  mpu_dev->gyro_range = PIOS_MPU_SCALE_1000_DEG;
476 
477  return 0;
478 }

655 int32_t PIOS_MPU_SetSampleRate(uint16_t samplerate_hz)
656 {
657  // TODO: think about supporting >1 khz, aliasing/noise issues though..
658  uint16_t internal_rate = 1000;
659 
660  // limit samplerate to filter frequency
661  if (samplerate_hz > internal_rate)
662  samplerate_hz = internal_rate;
663 
664  // calculate divisor, round to nearest integer
665  int32_t divisor = (int32_t)(((float)internal_rate / samplerate_hz) + 0.5f) - 1;
666 
667  // limit resulting divisor to register value range
668  if (divisor < 0)
669  divisor = 0;
670 
671  if (divisor > 0xff)
672  divisor = 0xff;
673 
674  // calculate true sample rate
675  samplerate_hz = internal_rate / (1 + divisor);
676 
677  int32_t retval = PIOS_MPU_WriteReg(PIOS_MPU_SMPLRT_DIV_REG, (uint8_t)divisor);
678 
679  if (retval == 0) {
680  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_ACCEL, samplerate_hz);
681  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_GYRO, samplerate_hz);
682 #ifdef PIOS_INCLUDE_MPU_MAG
683  if (mpu_dev->use_mag) {
684  if (mpu_dev->mpu_type == PIOS_MPU9150) {
685  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_MAG,
686  samplerate_hz / 10);
687  } else {
688  PIOS_SENSORS_SetSampleRate(PIOS_SENSOR_MAG, 100);
689  }
690  }
691 #endif // PIOS_INCLUDE_MPU_MAG
692  }
693 
694  return retval;
695 }

921 static void PIOS_MPU_Task(void *parameters)
922 {
923  (void)parameters;
924 
925  enum {
926  IDX_SPI_DUMMY_BYTE = 0,
927  IDX_ACCEL_XOUT_H,
928  IDX_ACCEL_XOUT_L,
929  IDX_ACCEL_YOUT_H,
930  IDX_ACCEL_YOUT_L,
931  IDX_ACCEL_ZOUT_H,
932  IDX_ACCEL_ZOUT_L,
933  IDX_TEMP_OUT_H,
934  IDX_TEMP_OUT_L,
935  IDX_GYRO_XOUT_H,
936  IDX_GYRO_XOUT_L,
937  IDX_GYRO_YOUT_H,
938  IDX_GYRO_YOUT_L,
939  IDX_GYRO_ZOUT_H,
940  IDX_GYRO_ZOUT_L,
941 #ifdef PIOS_INCLUDE_MPU_MAG
942  IDX_MAG_ST1,
943  IDX_MAG_XOUT_L,
944  IDX_MAG_XOUT_H,
945  IDX_MAG_YOUT_L,
946  IDX_MAG_YOUT_H,
947  IDX_MAG_ZOUT_L,
948  IDX_MAG_ZOUT_H,
949  IDX_MAG_ST2,
950 #endif // PIOS_INCLUDE_MPU_MAG
951  BUFFER_SIZE
952  };
953 
954  uint8_t mpu_rec_buf[BUFFER_SIZE];
955 
956 #ifdef PIOS_INCLUDE_MPU_MAG
957  bool do_full_mag = false;
958  int mag_age = 0;
959 #endif
960 
961 #ifdef PIOS_INCLUDE_SPI
962  uint8_t mpu_tx_buf[BUFFER_SIZE] = {PIOS_MPU_ACCEL_X_OUT_MSB | 0x80, };
963 #endif // PIOS_INCLUDE_SPI
964 
965  while (true) {
966 #ifdef PIOS_INCLUDE_MPU_MAG
967  mag_age++;
968 
969  uint8_t transfer_size;
970  /* By default we only transfer the magnetometer status
971  * information, and leave the actual mag data invalid. If the
972  * status info says DRDY, then next time around we transfer it
973  * all.
974  */
975  if (mpu_dev->use_mag) {
976  if (do_full_mag) {
977  transfer_size = BUFFER_SIZE;
978  } else {
979  /* If mag_age is low, could consider not
980  * even polling */
981  transfer_size = IDX_MAG_ST1 + 1;
982  }
983  } else {
984  transfer_size = IDX_GYRO_ZOUT_L + 1;
985  }
986 #else
987  uint8_t transfer_size = BUFFER_SIZE;
988 #endif // PIOS_INCLUDE_MPU_MAG
989 
990  //Wait for data ready interrupt
991  if (PIOS_Semaphore_Take(mpu_dev->data_ready_sema, PIOS_SEMAPHORE_TIMEOUT_MAX) != true)
992  continue;
993 
994 #if defined(PIOS_INCLUDE_SPI)
995  if (mpu_dev->com_driver_type == PIOS_MPU_COM_SPI) {
996  // claim bus in high speed mode
997  if (PIOS_MPU_ClaimBus(false) != 0)
998  continue;
999 
1000  if (PIOS_SPI_TransferBlock(mpu_dev->com_driver_id, mpu_tx_buf, mpu_rec_buf, transfer_size) < 0) {
1001  PIOS_MPU_ReleaseBus(false);
1002  continue;
1003  }
1004 
1005  PIOS_MPU_ReleaseBus(false);
1006  }
1007 #endif // defined(PIOS_INCLUDE_SPI)
1008 
1009 #if defined(PIOS_INCLUDE_I2C)
1010  if (mpu_dev->com_driver_type == PIOS_MPU_COM_I2C) {
1011  // we skip the SPI dummy byte at the beginning of the buffer here
1012  if (PIOS_MPU_I2C_Read(PIOS_MPU_ACCEL_X_OUT_MSB, &mpu_rec_buf[IDX_ACCEL_XOUT_H], transfer_size - 1) < 0)
1013  continue;
1014  }
1015 #endif // defined(PIOS_INCLUDE_I2C)
1016 
1017  struct pios_sensor_accel_data accel_data;
1018  struct pios_sensor_gyro_data gyro_data;
1019 
1020  float accel_x = (int16_t)(mpu_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu_rec_buf[IDX_ACCEL_XOUT_L]);
1021  float accel_y = (int16_t)(mpu_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu_rec_buf[IDX_ACCEL_YOUT_L]);
1022  float accel_z = (int16_t)(mpu_rec_buf[IDX_ACCEL_ZOUT_H] << 8 | mpu_rec_buf[IDX_ACCEL_ZOUT_L]);
1023  float gyro_x = (int16_t)(mpu_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu_rec_buf[IDX_GYRO_XOUT_L]);
1024  float gyro_y = (int16_t)(mpu_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu_rec_buf[IDX_GYRO_YOUT_L]);
1025  float gyro_z = (int16_t)(mpu_rec_buf[IDX_GYRO_ZOUT_H] << 8 | mpu_rec_buf[IDX_GYRO_ZOUT_L]);
1026 
1027 #ifdef PIOS_INCLUDE_MPU_MAG
1028  struct pios_sensor_mag_data mag_data;
1029 
1030  float mag_x = (int16_t)(mpu_rec_buf[IDX_MAG_XOUT_H] << 8 | mpu_rec_buf[IDX_MAG_XOUT_L]);
1031  float mag_y = (int16_t)(mpu_rec_buf[IDX_MAG_YOUT_H] << 8 | mpu_rec_buf[IDX_MAG_YOUT_L]);
1032  float mag_z = (int16_t)(mpu_rec_buf[IDX_MAG_ZOUT_H] << 8 | mpu_rec_buf[IDX_MAG_ZOUT_L]);
1033 #endif // PIOS_INCLUDE_MPU_MAG

1036  * Rotate the sensor to our convention (x forward, y right, z down).
1037  * Sensor orientation for all supported Invensense variants is
1038  * x right, y forward, z up.
1039  * The embedded AK8xxx magnetometer in MPU9x50 variants matches our convention.
1040  * See flight/Doc/imu_orientation.md for further detail
1041  */
1042  switch (mpu_dev->cfg->orientation) {

1045  accel_data.y = accel_x;

1058  accel_data.y = accel_y;
1059  accel_data.z = -accel_z;

1065  mag_data.y = mag_x;
1066  mag_data.z = mag_z;
1067 #endif // PIOS_INCLUDE_MPU_MAG
1068  break;

1071  accel_data.y = -accel_x;
1072  accel_data.z = -accel_z;

1081  break;

1085  accel_data.z = -accel_z;
