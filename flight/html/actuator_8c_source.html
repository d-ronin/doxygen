227 static void actuator_task(void* parameters)
228 {
229  float dT = 0.0f;
230 
231  ActuatorCommandData command;
232  ActuatorDesiredData desired;
233  MixerStatusData mixerStatus;
234  FlightStatusData flightStatus;
235  ManualControlCommandData manual_control_command;
236 
237  SystemSettingsAirframeTypeOptions airframe_type;
238 
239  // Connect update callbacks
240  FlightStatusConnectCallbackCtx(UAVObjCbSetFlag, &flightStatusUpdated);
241  ManualControlCommandConnectCallbackCtx(UAVObjCbSetFlag, &manualControlCommandUpdated);
242 
243  // Main task loop
244  uint32_t last_systime = PIOS_Thread_Systime();
245 
246  bool rc = false;
247 
248  while (1) {
249  if (actuator_settings_updated) {
250  actuator_settings_updated = false;
251  ActuatorSettingsGet(&actuatorSettings);
252  actuator_set_servo_mode();
253  }
254  if (mixer_settings_updated) {
255  mixer_settings_updated = false;
256  MixerSettingsGet(&mixerSettings);
257  SystemSettingsAirframeTypeGet(&airframe_type);
258  }
259 
260  if (rc != true) {
261  /* Update of ActuatorDesired timed out,
262  * or first iteration. Go to failsafe */
263  set_failsafe();
264  }
265 
266  PIOS_WDG_UpdateFlag(PIOS_WDG_ACTUATOR);
267 
268  UAVObjEvent ev;
269 
270  // Wait until the ActuatorDesired object is updated
271  rc = PIOS_Queue_Receive(queue, &ev, FAILSAFE_TIMEOUT_MS);
272 
273  /* If we timed out, go to top of loop, which sets failsafe
274  * and waits again. */
275  if (rc != true) {
276  continue;
277  }
278 
279  // Check how long since last update
280  uint32_t this_systime = PIOS_Thread_Systime();
281  if (this_systime > last_systime) // reuse dt in case of wraparound
282  dT = (this_systime - last_systime) / 1000.0f;
283  last_systime = this_systime;
284 
285  ActuatorDesiredGet(&desired);
286  ActuatorCommandGet(&command);
287 
288  if (flightStatusUpdated) {
289  FlightStatusGet(&flightStatus);
290  flightStatusUpdated = false;
291  }
292 
293  if (manualControlCommandUpdated) {
294  ManualControlCommandGet(&manual_control_command);
295  manualControlCommandUpdated = false;
296  }
297 
298  int nMixers = 0;
299 
300  for (int ct = 0; ct < MAX_MIX_ACTUATORS; ct++) {
301  if (get_mixer_type(ct) != MIXERSETTINGS_MIXER1TYPE_DISABLED) {
302  nMixers++;
303  }
304  }
305  if ((nMixers < 2) && !ActuatorCommandReadOnly()) {
306  // Nothing can fly with less than two mixers.
307  set_failsafe();
308  continue;
309  }
310 
311  bool armed = flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMED;
312  bool spin_while_armed = actuatorSettings.MotorsSpinWhileArmed == ACTUATORSETTINGS_MOTORSSPINWHILEARMED_TRUE;
313 
314  float throttle_source = -1;
315  if (airframe_type == SYSTEMSETTINGS_AIRFRAMETYPE_HELICP) {
316  // Helis set throttle from manual control's throttle value,
317  // unless in failsafe.
318  if (flightStatus.FlightMode != FLIGHTSTATUS_FLIGHTMODE_FAILSAFE) {
319  throttle_source = manual_control_command.Throttle;
320  }
321  } else {
322  throttle_source = desired.Thrust;
323  }
324 
325  bool stabilize_now = armed && (throttle_source > 0.0f);
326 
327  static uint32_t last_pos_throttle_time = 0;
328 
329  if (stabilize_now) {
330  if (actuatorSettings.LowPowerStabilizationMaxTime) {
331  last_pos_throttle_time = this_systime;
332  }
333 
334  // Could consider stabilizing on a positive arming edge,
335  // but this seems problematic.
336  } else if (last_pos_throttle_time) {
337  if ((this_systime - last_pos_throttle_time) <
338  1000.0f * actuatorSettings.LowPowerStabilizationMaxTime) {
339  stabilize_now = true;
340  throttle_source = 0.0f;
341  } else {
342  last_pos_throttle_time = 0;
343  }
344  }
345 
346  float curve1 = throt_curve(throttle_source, mixerSettings.ThrottleCurve1, MIXERSETTINGS_THROTTLECURVE1_NUMELEM);
347 
348  //The source for the secondary curve is selectable
349  float curve2 = collective_curve(
350  get_curve2_source(&desired, airframe_type, mixerSettings.Curve2Source),
351  mixerSettings.ThrottleCurve2,
352  MIXERSETTINGS_THROTTLECURVE2_NUMELEM);
353 
354  float * status = (float *)&mixerStatus; //access status objects as an array of floats
355 
356  float min_chan = INFINITY;
357  float max_chan = -INFINITY;
358  float neg_clip = 0;
359  int num_motors = 0;
360 
361  for (int ct = 0; ct < MAX_MIX_ACTUATORS; ct++) {
362  status[ct] = mix_channel(ct, &desired, curve1, curve2);
363 
364  if (get_mixer_type(ct) == MIXERSETTINGS_MIXER1TYPE_MOTOR) {
365  min_chan = fminf(min_chan, status[ct]);
366  max_chan = fmaxf(max_chan, status[ct]);
367 
368  if (status[ct] < 0.0f) {
369  neg_clip += status[ct];
370  }
371 
372  num_motors++;
373  }
374  }
375 
376  float gain = 1.0f;
377  float offset = 0.0f;
378 
379  /* This is a little dubious. Scale down command ranges to
380  * fit. It may cause some cross-axis coupling, though
381  * generally less than if we were to actually let it clip.
382  */
383  if ((max_chan - min_chan) > 1.0f) {
384  gain = 1.0f / (max_chan - min_chan);
385 
386  max_chan *= gain;
387  min_chan *= gain;
388  }
389 
390  /* Sacrifice throttle because of clipping */
391  if (max_chan > 1.0f) {
392  offset = 1.0f - max_chan;
393  } else if (min_chan < 0.0f) {
394  /* Low-side clip management-- how much power are we
395  * willing to add??? */
396 
397  neg_clip /= num_motors;
398 
399  /* neg_clip is now the amount of throttle "already added." by
400  * clipping...
401  *
402  * Find the "highest possible value" of offset.
403  * if neg_clip is -15%, and maxpoweradd is 10%, we need to add
404  * -5% to all motors.
405  * if neg_clip is 5%, and maxpoweradd is 10%, we can add up to
406  * 5% to all motors to further fix clipping.
407  */
408  offset = neg_clip + actuatorSettings.LowPowerStabilizationMaxPowerAdd;
409 
410  /* Add the lesser of--
411  * A) the amount the lowest channel is out of range.
412  * B) the above calculated offset.
413  */
414  offset = MIN(-min_chan, offset);
415  }
416 
417  for (int ct = 0; ct < MAX_MIX_ACTUATORS; ct++) {
418  // Motors have additional protection for when to be on
419  if (get_mixer_type(ct) == MIXERSETTINGS_MIXER1TYPE_MOTOR) {
420  if (!armed) {
421  status[ct] = -1; //force min throttle
422  } else if (!stabilize_now) {
423  if (!spin_while_armed) {
424  status[ct] = -1;
425  } else {
426  status[ct] = 0;
427  }
428  } else {
429  status[ct] = status[ct] * gain + offset;
430 
431  if (status[ct] > 0) {
432  // Apply curve fitting, mapping the input to the propeller output.
433  status[ct] = powapprox(status[ct], actuatorSettings.MotorInputOutputCurveFit);
434  } else {
435  status[ct] = 0;
436  }
437  }
438  }
439 
440  command.Channel[ct] = scale_channel(status[ct], ct);
441  }
442 
443  // Store update time
444  command.UpdateTime = 1000.0f*dT;
445  if (1000.0f*dT > command.MaxUpdateTime)
446  command.MaxUpdateTime = 1000.0f*dT;
447 
448  // Update output object
449  if (!ActuatorCommandReadOnly()) {
450  ActuatorCommandSet(&command);
451  } else {
452  // it's read only during servo configuration--
453  // so GCS takes precedence.
454  ActuatorCommandGet(&command);
455  }
456 
457 #if defined(MIXERSTATUS_DIAGNOSTICS)
458  MixerStatusSet(&mixerStatus);
459 #endif
460  for (int n = 0; n < ACTUATORCOMMAND_CHANNEL_NUMELEM; ++n) {
461  PIOS_Servo_Set(n, command.Channel[n]);
462  }
463 
464  PIOS_Servo_Update();
465 
466  AlarmsClear(SYSTEMALARMS_ALARM_ACTUATOR);
467  }
468 }

473 float process_mixer(const int index, const float curve1, const float curve2,
474  ActuatorDesiredData *desired)
475 {
476  // Taking the pointer to the array preserves type information so smart compilers
477  // can detect accesses past the end.
478  typeof(mixerSettings.Mixer1Vector) *vector = get_mixer_vec(index);
479 
480  float result = (((*vector)[MIXERSETTINGS_MIXER1VECTOR_THROTTLECURVE1] * curve1) +
481  ((*vector)[MIXERSETTINGS_MIXER1VECTOR_THROTTLECURVE2] * curve2) +
482  ((*vector)[MIXERSETTINGS_MIXER1VECTOR_ROLL] * desired->Roll) +
483  ((*vector)[MIXERSETTINGS_MIXER1VECTOR_PITCH] * desired->Pitch) +
484  ((*vector)[MIXERSETTINGS_MIXER1VECTOR_YAW] * desired->Yaw)) * (1.0f / MULTIROTOR_MIXER_UPPER_BOUND);
485 
486  return (result);
487 }

524 static float scale_channel(float value, int idx)
525 {
526  float max = actuatorSettings.ChannelMax[idx];
527  float min = actuatorSettings.ChannelMin[idx];
528  float neutral = actuatorSettings.ChannelNeutral[idx];
529 
530  float valueScaled;
531  // Scale
532  if (value >= 0.0f) {
533  valueScaled = value*(max-neutral) + neutral;
534  } else {
535  valueScaled = value*(neutral-min) + neutral;
536  }
537 
538  if (max>min) {
539  if (valueScaled > max) valueScaled = max;
540  if (valueScaled < min) valueScaled = min;
541  } else {
542  if (valueScaled < max) valueScaled = max;
543  if (valueScaled > min) valueScaled = min;
544  }
545 
546  return valueScaled;
547 }

599 static float mix_channel(int ct, ActuatorDesiredData *desired,
600  float curve1, float curve2)
601 {
602  MixerSettingsMixer1TypeOptions type = get_mixer_type(ct);
603 
604  switch (type) {
605  case MIXERSETTINGS_MIXER1TYPE_DISABLED:
606  // Set to minimum if disabled. This is not the same as saying PWM pulse = 0 us
607  return -1;
608  break;
609 
610  case MIXERSETTINGS_MIXER1TYPE_SERVO:
611  return process_mixer(ct, curve1, curve2, desired);
612  break;
613 
614  case MIXERSETTINGS_MIXER1TYPE_MOTOR:
615  (void) 0; // nil statement
616  float val = process_mixer(ct, curve1, curve2, desired);
617 
618  return val;
619  break;
620  // If an accessory channel is selected for direct bypass mode
621  // In this configuration the accessory channel is scaled and
622  // mapped directly to output.
623  // Note: THERE IS NO SAFETY CHECK HERE FOR ARMING
624  // these also will not be updated in failsafe mode. I'm not sure
625  // what the correct behavior is since it seems domain specific.
626  // I don't love this code
627  case MIXERSETTINGS_MIXER1TYPE_ACCESSORY0:
628  case MIXERSETTINGS_MIXER1TYPE_ACCESSORY1:
629  case MIXERSETTINGS_MIXER1TYPE_ACCESSORY2:
630  case MIXERSETTINGS_MIXER1TYPE_ACCESSORY3:
631  case MIXERSETTINGS_MIXER1TYPE_ACCESSORY4:
632  case MIXERSETTINGS_MIXER1TYPE_ACCESSORY5:
633  (void) 0;
634 
635  AccessoryDesiredData accessory;
636  if (AccessoryDesiredInstGet(type - MIXERSETTINGS_MIXER1TYPE_ACCESSORY0,&accessory) == 0)
637  return accessory.AccessoryVal;
638  else
639  return -1;
640  break;
641  case MIXERSETTINGS_MIXER1TYPE_CAMERAPITCH:
642  case MIXERSETTINGS_MIXER1TYPE_CAMERAROLL:
643  case MIXERSETTINGS_MIXER1TYPE_CAMERAYAW:
644  (void) 0;
645 
646  CameraDesiredData cameraDesired;
647  if (CameraDesiredHandle() && CameraDesiredGet(&cameraDesired) == 0) {
648  switch (type) {
649  case MIXERSETTINGS_MIXER1TYPE_CAMERAROLL:
650  return cameraDesired.Roll;
651  case MIXERSETTINGS_MIXER1TYPE_CAMERAPITCH:
652  return cameraDesired.Pitch;
653  case MIXERSETTINGS_MIXER1TYPE_CAMERAYAW:
654  return cameraDesired.Yaw;
655  default:
656  break;
657  }
658  }
659  return -1;
660  /* No default-- make sure we handle all types positively here */
661  }
662 
663  /* Can't get here. */
664  return -1;
