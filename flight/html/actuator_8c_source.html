141 static bool actuator_get_dshot_command(bool armed) {
142  /* Finish off any current command. */
143  if (cur_cmd.num_to_send) {
144  cur_cmd.num_to_send--;
145  return true;
146  }
147 
148  if (armed) {
149  return false;
150  }
151 
152  uint32_t value = pending_cmd.value;
153 
154  if (!value) {
155  return false;
156  }
157 
158  pending_cmd.value = 0;
159  cur_cmd.value = value;
160 
161  if (cur_cmd.num_to_send) {
162  cur_cmd.num_to_send--;
163  return true;
164  }
165 
166  return false;
167 }

391 static void post_process_scale_and_commit(float *motor_vect,
392  float *desired_vect, float dT,
393  bool armed, bool spin_while_armed, bool stabilize_now,
394  float *maxpoweradd_bucket)
395 {
396  float min_chan = INFINITY;
397  float max_chan = -INFINITY;
398  float neg_clip = 0;
399  int num_motors = 0;
400  ActuatorCommandData command;
401 
402  const float hangtime_leakybucket_timeconstant = 0.3f;
403 
404  /* Hangtime maximum power add is now a "leaky bucket" system, ensuring
405  * that the average added power in the long term is the configured value
406  * but allowing higher values briefly.
407  *
408  * The intention is to allow more aggressive maneuvers than hangtime
409  * previously did, while still providing similar safety properties.
410  * A secondary motivation is to prevent tumbling when throttle is
411  * chopped during fast-forward-flight and more than hangtime power
412  * levels are needed for stabilization (because of aerodynamic forces).
413  *
414  * The "maximum" stored is based on recent throttle history-- it decays
415  * with time; at high throttle it corresponds to 300ms of the current
416  * power; at lower throttle it corresponds to 300ms of double the
417  * configured value.
418  */
419  float maxpoweradd_softlimit = MAX(
420  2 * actuatorSettings.LowPowerStabilizationMaxPowerAdd,
421  fabsf(desired_vect[MIXERSETTINGS_MIXER1VECTOR_THROTTLECURVE1]))
422  * hangtime_leakybucket_timeconstant;
423 
424  /* If we're under the limit, add this tick's hangtime power allotment */
425  if (*maxpoweradd_bucket < maxpoweradd_softlimit) {
426  *maxpoweradd_bucket += actuatorSettings.LowPowerStabilizationMaxPowerAdd * dT;
427  } else {
428  /* Otherwise, decay towards the current limit on a 300ms
429  * time constant.
430  */
431  float alpha = dT / (dT + hangtime_leakybucket_timeconstant);
432 
433  *maxpoweradd_bucket = alpha * maxpoweradd_softlimit +
434  (1-alpha) * (*maxpoweradd_bucket);
435  }
436 
437  /* The maximum power add is what would spend the current allotment in
438  * 300ms. In other words, in the absence of recent high-throttle,
439  * start from double the hangtime configured percentage and decay on
440  * a 300ms time constant IF IT IS ACTUALLY USED.
441  *
442  * This is separate from the above decay, so we could actually be
443  * decaying twice as fast if both are in play.
444  */
445  float maxpoweradd = (*maxpoweradd_bucket) / hangtime_leakybucket_timeconstant;
446 
447  bool neg_throttle = desired_vect[MIXERSETTINGS_MIXER1VECTOR_THROTTLECURVE1] < 0.0f;
448 
449  for (int ct = 0; ct < MAX_MIX_ACTUATORS; ct++) {
450  switch (types_mixer[ct]) {
451  case MIXERSETTINGS_MIXER1TYPE_DISABLED:
452  // Set to minimum if disabled.
453  // This is not the same as saying
454  // PWM pulse = 0 us
455  motor_vect[ct] = -1;
456  break;
457 
458  case MIXERSETTINGS_MIXER1TYPE_SERVO:
459  break;
460 
461  case MIXERSETTINGS_MIXER1TYPE_MOTOR:
462  if (neg_throttle) {
463  /* We'll reverse this later! */
464  motor_vect[ct] = -motor_vect[ct];
465  }
466 
467  min_chan = fminf(min_chan, motor_vect[ct]);
468  max_chan = fmaxf(max_chan, motor_vect[ct]);
469 
470  if (motor_vect[ct] < 0.0f) {
471  neg_clip += motor_vect[ct];
472  }
473 
474  num_motors++;
475  break;
476  case MIXERSETTINGS_MIXER1TYPE_CAMERAPITCH:
477  if (CameraDesiredHandle()) {
478  CameraDesiredPitchGet(
479  &motor_vect[ct]);
480  } else {
481  motor_vect[ct] = -1;
482  }
483  break;
484  case MIXERSETTINGS_MIXER1TYPE_CAMERAROLL:
485  if (CameraDesiredHandle()) {
486  CameraDesiredRollGet(
487  &motor_vect[ct]);
488  } else {
489  motor_vect[ct] = -1;
490  }
491  break;
492  case MIXERSETTINGS_MIXER1TYPE_CAMERAYAW:
493  if (CameraDesiredHandle()) {
494  CameraDesiredRollGet(
495  &motor_vect[ct]);
496  } else {
497  motor_vect[ct] = -1;
498  }
499  break;
500  default:
501  set_failsafe();
502  PIOS_Assert(0);
503  }
504  }
505 
506  float gain = 1.0f;
507  float offset = 0.0f;
508 
509  /* This is a little dubious. Scale down command ranges to
510  * fit. It may cause some cross-axis coupling, though
511  * generally less than if we were to actually let it clip.
512  */
513  if ((max_chan - min_chan) > 1.0f) {
514  gain = 1.0f / (max_chan - min_chan);
515 
516  max_chan *= gain;
517  min_chan *= gain;
518  }
519 
520  /* Sacrifice throttle because of clipping */
521  if (max_chan > 1.0f) {
522  offset = 1.0f - max_chan;
523  } else if (min_chan < 0.0f) {
524  /* Low-side clip management-- how much power are we
525  * willing to add??? */
526 
527  neg_clip /= num_motors;
528 
529  /* neg_clip is now the amount of throttle "already added." by
530  * clipping...
531  *
532  * Find the "highest possible value" of offset.
533  * if neg_clip is -15%, and maxpoweradd is 10%, we need to add
534  * -5% to all motors.
535  * if neg_clip is 5%, and maxpoweradd is 10%, we can add up to
536  * 5% to all motors to further fix clipping.
537  */
538  offset = neg_clip + maxpoweradd;
539 
540  /* Add the lesser of--
541  * A) the amount the lowest channel is out of range.
542  * B) the above calculated offset.
543  */
544  offset = MIN(-min_chan, offset);
545 
546  /* The amount actually added is the above offset, plus the
547  * amount that came from negative clipping. (It's negative
548  * though, so subtract instead of add). Spend this from
549  * the leaky bucket.
550  */
551  *maxpoweradd_bucket -= (offset - neg_clip) * dT;
552  }
553 
554  bool active_command = false;
555 
556  active_command = actuator_get_dshot_command(armed);
557 
558  for (int ct = 0; ct < MAX_MIX_ACTUATORS; ct++) {
559  // Motors have additional protection for when to be on
560  if (types_mixer[ct] == MIXERSETTINGS_MIXER1TYPE_MOTOR) {
561  if (!armed) {
562  motor_vect[ct] = NAN; // don't spin
563  } else if (!stabilize_now) {
564  if (!spin_while_armed) {
565  /* No spin */
566  motor_vect[ct] = NAN;
567  } else {
568  /* Min spin in our direction */
569  if (neg_throttle) {
570  motor_vect[ct] = -ACTUATOR_EPSILON;
571  } else {
572  motor_vect[ct] = ACTUATOR_EPSILON;
573  }
574  }
575  } else {
576  motor_vect[ct] = motor_vect[ct] * gain + offset;
577 
578  if (motor_vect[ct] > 0) {
579  // Apply curve fitting, mapping the input to the propeller output.
580 
581  motor_vect[ct] = powapprox(motor_vect[ct], actuatorSettings.MotorInputOutputCurveFit);
582  } else {
583  /* Clip to minimum spin in this direction */
584  motor_vect[ct] = ACTUATOR_EPSILON;
585  }
586 
587  if (neg_throttle) {
588  motor_vect[ct] = -motor_vect[ct];
589  }
590  }
591  }
592 
593  command.Channel[ct] = scale_channel(motor_vect[ct], ct,
594  active_command);
595  }
596 
597  // Store update time
598  command.UpdateTime = 1000.0f*dT;
599 
600  ActuatorCommandMaxUpdateTimeGet(&command.MaxUpdateTime);
601 
602  if (command.UpdateTime > command.MaxUpdateTime)
603  command.MaxUpdateTime = 1000.0f*dT;
604 
605  // Update output object
606  if (!ActuatorCommandReadOnly()) {
607  ActuatorCommandSet(&command);
608  } else {
609  // it's read only during servo configuration--
610  // so GCS takes precedence.
611  ActuatorCommandGet(&command);
612  }
613 
614  for (int n = 0; n < MAX_MIX_ACTUATORS; ++n) {
615  PIOS_Servo_Set(n, command.Channel[n]);
616  }
617 
618  PIOS_Servo_Update();
619 }

621 static void normalize_input_data(uint32_t this_systime,
622  float (*desired_vect)[MIXERSETTINGS_MIXER1VECTOR_NUMELEM],
623  bool *armed, bool *spin_while_armed, bool *stabilize_now)
624 {
625  static float manual_throt = -1;
626  float throttle_val = 0;
627  ActuatorDesiredData desired;
628 
629  static FlightStatusData flightStatus;
630 
631  ActuatorDesiredGet(&desired);
632 
633  if (flight_status_updated) {
634  FlightStatusGet(&flightStatus);
635  flight_status_updated = false;
636  }
637 
638  if (manual_control_cmd_updated) {
639  // just pull out the throttle_val... and accessory0-2 and
640  // fill direct into the vect
641  ManualControlCommandThrottleGet(&manual_throt);
642  manual_control_cmd_updated = false;
643  ManualControlCommandAccessoryGet(
644  &(*desired_vect)[MIXERSETTINGS_MIXER1VECTOR_ACCESSORY0]);
645  }
646 
647  *armed = flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMED;
648  *spin_while_armed = actuatorSettings.MotorsSpinWhileArmed == ACTUATORSETTINGS_MOTORSSPINWHILEARMED_TRUE;
649 
650  if (airframe_type == SYSTEMSETTINGS_AIRFRAMETYPE_HELICP) {
651  // Helis set throttle from manual control's throttle value,
652  // unless in failsafe.
653  // TODO: Audit and determine whether this check is even required
654  if (flightStatus.FlightMode != FLIGHTSTATUS_FLIGHTMODE_FAILSAFE) {
655  throttle_val = manual_throt;
656  }
657  } else {
658  throttle_val = desired.Thrust;
659  }
660 
661  if (!*armed) {
662  throttle_val = 0.0f;
663  }
664 
665  *stabilize_now = throttle_val != 0.0f;
666 
667  float val1 = throttle_val;
668 
669  //The source for the secondary curve is selectable
670  float val2 = collective_curve(
671  get_curve2_source(&desired, airframe_type, curve2_src,
672  throttle_val),
673  curve2, MIXERSETTINGS_THROTTLECURVE2_NUMELEM);
674 
675  fill_desired_vector(&desired, val1, val2, desired_vect);
676 }

710 static void actuator_task(void* parameters)
711 {
712  // Connect update callbacks
713  FlightStatusConnectCallbackCtx(UAVObjCbSetFlag, &flight_status_updated);
714  ManualControlCommandConnectCallbackCtx(UAVObjCbSetFlag, &manual_control_cmd_updated);
715 
716  /* Prime dshot channels with 12 of the 'disarm' command */
717  actuator_send_dshot_command_now(0, 12, 0xffff);
718 
719  // Ensure the initial state of actuators is safe.
720  actuator_settings_update();
721  set_failsafe();
722 
723  /* This is out here because not everything may change each time */
724  uint32_t last_systime = PIOS_Thread_Systime();
725  float desired_vect[MIXERSETTINGS_MIXER1VECTOR_NUMELEM] = { 0 };
726  float dT = 0.0f;
727 
728  float maxpoweradd_bucket = 0.0f;
729 
730  bool prev_armed = false;
731 
732  // Main task loop
733  while (1) {
734  /* If settings objects have changed, update our internal
735  * state appropriately.
736  */
737  if (actuator_settings_updated) {
738  actuator_settings_update();
739  }
740 
741  if (mixer_settings_updated) {
742  mixer_settings_updated = false;
743  SystemSettingsAirframeTypeGet(&airframe_type);
744 
745  compute_mixer();
746 
747  MixerSettingsThrottleCurve2Get(curve2);
748  MixerSettingsCurve2SourceGet(&curve2_src);
749  }
750 
751  PIOS_WDG_UpdateFlag(PIOS_WDG_ACTUATOR);
752 
753  UAVObjEvent ev;
754 
755  // Wait until the ActuatorDesired object is updated
756  if (!PIOS_Queue_Receive(queue, &ev, FAILSAFE_TIMEOUT_MS)) {
757  // If we hit a timeout, set the actuator failsafe and
758  // try again.
759  set_failsafe();
760  continue;
761  }
762 
763  uint32_t this_systime = PIOS_Thread_Systime();
764 
765  /* Check how long since last update; this is stored into the
766  * UAVO to allow analysis of actuation jitter.
767  */
768  if (this_systime > last_systime) {
769  dT = (this_systime - last_systime) / 1000.0f;
770  /* (Otherwise, the timer has wrapped [rare] and we should
771  * just reuse dT)
772  */
773  }
774 
775  last_systime = this_systime;
776 
777  if (actuator_interlock != ACTUATOR_INTERLOCK_OK) {
778  /* Chosen because: 50Hz does 4-6 updates in 100ms */
779  uint32_t exp_time = this_systime + 100;
780 
781  while (actuator_interlock != ACTUATOR_INTERLOCK_OK) {
782  /* Simple state machine. If someone has asked us to
783  * stop, set actuator failsafe for a short while.
784  * Then, set the flag to STOPPED.
785  *
786  * Setting to STOPPED isn't atomic, so we rely on
787  * anyone who has stopped us to waitfor STOPPED
788  * before putting us back to OK.
789  */
790  if (actuator_interlock == ACTUATOR_INTERLOCK_STOPREQUEST) {
791  set_failsafe();
792 
793  this_systime = PIOS_Thread_Systime();
794 
795  if ((exp_time - this_systime) > 100) {
796  actuator_interlock = ACTUATOR_INTERLOCK_STOPPED;
797  }
798  }
799 
800  PIOS_Thread_Sleep(3);
801  PIOS_WDG_UpdateFlag(PIOS_WDG_ACTUATOR);
802  }
803 
804  PIOS_Servo_SetMode(actuatorSettings.TimerUpdateFreq,
805  ACTUATORSETTINGS_TIMERUPDATEFREQ_NUMELEM,
806  actuatorSettings.ChannelMax,
807  actuatorSettings.ChannelMin);
808  continue;
809  }
810 
811 
812  float motor_vect[MAX_MIX_ACTUATORS];
813 
814  bool armed, spin_while_armed, stabilize_now;
815 
816  /* Receive manual control and desired UAV objects. Perform
817  * arming / hangtime checks; form a vector with desired
818  * axis actions.
819  */
820  normalize_input_data(this_systime, &desired_vect, &armed,
821  &spin_while_armed, &stabilize_now);
822 
823  /* Multiply the actuators x desired matrix by the
824  * desired x 1 column vector. */
825  matrix_mul_check(motor_mixer, desired_vect, motor_vect,
826  MAX_MIX_ACTUATORS,
827  MIXERSETTINGS_MIXER1VECTOR_NUMELEM,
828  1);
829 
830  /* At arming time, knock all 3d actuators into 3D mode.
831  * Note we never "take them out" of 3d mode.
832  */
833  if (armed != prev_armed) {
834  if (armed && desired_3d_mask) {
835  if (!actuator_send_dshot_command_now(
836  DSHOT_COMMAND_3DMODE,
837  DSHOT_COUNT_EXCESSIVE,
838  desired_3d_mask)) {
839  prev_armed = armed;
840  }
841  } else {
842  prev_armed = armed;
843  }
844  }
845 
846  /* Perform clipping adjustments on the outputs, along with
847  * state-related corrections (spin while armed, disarmed, etc).
848  *
849  * Program the actual values to the timer subsystem.
850  */
851  post_process_scale_and_commit(motor_vect, desired_vect,
852  dT, armed, spin_while_armed, stabilize_now,
853  &maxpoweradd_bucket);
854 
855  /* If we got this far, everything is OK. */
856  AlarmsClear(SYSTEMALARMS_ALARM_ACTUATOR);
857  }
858 }

875 static float scale_channel_dshot(float value, int idx, bool active_command)
876 {
877  /* If a command is pending, send it */
878  if (active_command) {
879  if (cur_cmd.channel_mask & (1 << idx)) {
880  return cur_cmd.cmd_id;
881  }
882  }
883 
884  if (!isfinite(value)) {
885  /* Universal-- don't spin */
886  return 0;
887  }
888 
889  bool threed = false;
890  bool reversed = false;
891 
892 #define DSHOT_DEADBAND_NORMAL 0
893 #define DSHOT_DEADBAND_3D 1
894 #define DSHOT_DEADBAND_3DREV 2
895 
896  switch (actuatorSettings.ChannelDeadband[idx]) {
897  case DSHOT_DEADBAND_NORMAL:
898  break;
899  case DSHOT_DEADBAND_3DREV:
900  /* reverse 3D */
901  reversed = true;
902  /* falls through */
903  case DSHOT_DEADBAND_3D:
904  threed = true;
905  break;
906  default:
907  /* Invalid value-- don't spin */
908  return 0;
909  }
910 
911  float neutral = actuatorSettings.ChannelNeutral[idx];
912 
913  if (!threed) {
914  /* return neutral..2047 */
915  int16_t val = roundf((2047 - neutral) * value) + neutral;
916 
917  if (val > 2047) {
918  val = 2047;
919  }
920 
921  if (val < 48) {
922  /* Shouldn't happen, unless neutral val is invalid */
923  val = 0;
924  }
925 
926  return val;
927  }
928 
929  /* OK, this is more tricky. Convert from a unipolar neutral value
930  * to a "real one" as sent */
931 
932  neutral -= 48;
933  neutral /= 2;
934 
935  bool negative;
936 
937  if (value < 0) {
938  value = -value;
939  negative = !reversed;
940  } else {
941  negative = reversed;
942  }
943 
944  /* come up with a value between real neutral and 999 */
945 
946  int16_t val = roundf((999 - neutral) * value + neutral);
947 
948  if (val > 999) {
949  val = 999;
950  }
951 
952  if (val < 0) {
953  /* Shouldn't happen, unless neutral val is invalid */
954  val = 0;
955  }
956 
957  if (negative) {
958  /* 1048 + real neutral .. 2047 */
959  val += 1048;
960  } else {
961  /* 48 + real neutral .. 1047 */
962  val += 48;
963  }
964 
965  return val;
966 }

971 static float scale_channel(float value, int idx, bool active_cmd)
972 {
973  if (PIOS_Servo_IsDshot(idx)) {
974  return scale_channel_dshot(value, idx, active_cmd);
975  }
976 
977  float max = actuatorSettings.ChannelMax[idx];
978  float min = actuatorSettings.ChannelMin[idx];
979  float neutral = actuatorSettings.ChannelNeutral[idx];
980  float deadband = actuatorSettings.ChannelDeadband[idx] / 2.0f;
981 
982  float valueScaled;
983 
984  if (!isfinite(value)) {
985  if (deadband) {
986  /* 3D motor mode */
987  /* NaN signals us to not spin-- e.g. neutral value */
988 
989  return neutral;
990  } else {
991  /* Non-3D motor mode. */
992  /* NaN signals us to not spin-- e.g. minimum value */
993  return min;
994  }
995  }
996 
997  if (min > max) {
998  deadband = -deadband;
999  }
1000 
1001  if (value >= 0.0f) {
1002  valueScaled = value * (max - neutral - deadband)
1003  + neutral + deadband;
1004  } else {
1005  valueScaled = value * (neutral - min - deadband)
1006  + neutral - deadband;
1007  }
1008 
1009  if (max > min) {
1010  if (valueScaled > max) valueScaled = max;
1011  if (valueScaled < min) valueScaled = min;
1012  } else {
1013  if (valueScaled < max) valueScaled = max;
1014  if (valueScaled > min) valueScaled = min;
1015  }
1016 
1017  return valueScaled;
1018 }
