142 void PIOS_Board_Init(void) {
143  // Drive USB re-enumerate pin low (so it doesn't pull DP down!)
144  GPIO_InitTypeDef GPIO_InitStructure;
145  GPIO_StructInit(&GPIO_InitStructure);
146  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_12;
147  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
148  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
149  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_NOPULL;
150  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz;
151  GPIO_Init(GPIOB, &GPIO_InitStructure);
152  GPIO_ResetBits(GPIOB, GPIO_Pin_12);
153 
154  /* Delay system */
155  PIOS_DELAY_Init();
156 
157 #if defined(PIOS_INCLUDE_ANNUNC)
158  PIOS_ANNUNC_Init(&pios_annunc_cfg);
159 #endif /* PIOS_INCLUDE_ANNUNC */
160 
161 #if defined(PIOS_INCLUDE_FLASH)
162  /* Inititialize all flash drivers */
163  if (PIOS_Flash_Internal_Init(&pios_internal_flash_id, &flash_internal_cfg) != 0)
164  PIOS_HAL_CriticalError(PIOS_LED_ALARM, PIOS_HAL_PANIC_FLASH);
165 
166  /* Register the partition table */
167  PIOS_FLASH_register_partition_table(pios_flash_partition_table_internal, NELEMENTS(pios_flash_partition_table_internal));
168 
169  /* Mount all filesystems */
170  if (PIOS_FLASHFS_Logfs_Init(&pios_uavo_settings_fs_id, &flashfs_settings_internal_cfg, FLASH_PARTITION_LABEL_SETTINGS) != 0)
171  PIOS_HAL_CriticalError(PIOS_LED_ALARM, PIOS_HAL_PANIC_FILESYS);
172 #endif /* PIOS_INCLUDE_FLASH */
173 
174  /* Initialize UAVObject libraries */
175  UAVObjInitialize();
176 
177  HwPlayUavOsdInitialize();
178  ModuleSettingsInitialize();
179 
180 #if defined(PIOS_INCLUDE_RTC)
181  /* Initialize the real-time clock and its associated tick */
182  PIOS_RTC_Init(&pios_rtc_main_cfg);
183 #endif
184 
185  /* Initialize watchdog as early as possible to catch faults during init
186  * but do it only if there is no debugger connected
187  */
188  if ((CoreDebug->DHCSR & CoreDebug_DHCSR_C_DEBUGEN_Msk) == 0) {
189  //PIOS_WDG_Init();
190  }
191 
192  /* Initialize the alarms library */
193  AlarmsInitialize();
194  PIOS_RESET_Clear();
195 
196  /* Initialize the task monitor library */
197  TaskMonitorInitialize();
198 
199  /* Set up pulse timers */
200  //inputs
201  PIOS_TIM_InitClock(&tim_8_cfg);
202  PIOS_TIM_InitClock(&tim_12_cfg);
203  //outputs
204  PIOS_TIM_InitClock(&tim_5_cfg);
205 
206  /* IAP System Setup */
207  PIOS_IAP_Init();
208  uint16_t boot_count = PIOS_IAP_ReadBootCount();
209  if (boot_count < 3) {
210  PIOS_IAP_WriteBootCount(++boot_count);
211  AlarmsClear(SYSTEMALARMS_ALARM_BOOTFAULT);
212  } else {
213  /* Too many failed boot attempts, force hw config to defaults */
214  HwPlayUavOsdSetDefaults(HwPlayUavOsdHandle(), 0);
215  ModuleSettingsSetDefaults(ModuleSettingsHandle(),0);
216  AlarmsSet(SYSTEMALARMS_ALARM_BOOTFAULT, SYSTEMALARMS_ALARM_CRITICAL);
217  }
218 
219 #if defined(PIOS_INCLUDE_USB)
220  /* Initialize board specific USB data */
221  PIOS_USB_BOARD_DATA_Init();
222 
223  /* Flags to determine if various USB interfaces are advertised */
224  bool usb_hid_present = false;
225  bool usb_cdc_present = false;
226 
227 #if defined(PIOS_INCLUDE_USB_CDC)
228  if (PIOS_USB_DESC_HID_CDC_Init()) {
229  PIOS_Assert(0);
230  }
231  usb_hid_present = true;
232  usb_cdc_present = true;
233 #else
234  if (PIOS_USB_DESC_HID_ONLY_Init()) {
235  PIOS_Assert(0);
236  }
237  usb_hid_present = true;
238 #endif
239 
240  uintptr_t pios_usb_id;
241  PIOS_USB_Init(&pios_usb_id, &pios_usb_main_cfg);
242 
243 #if defined(PIOS_INCLUDE_USB_CDC)
244 
245  uint8_t hw_usb_vcpport;
246  /* Configure the USB VCP port */
247  HwPlayUavOsdUSB_VCPPortGet(&hw_usb_vcpport);
248 
249  if (!usb_cdc_present) {
250  /* Force VCP port function to disabled if we haven't advertised VCP in our USB descriptor */
251  hw_usb_vcpport = HWPLAYUAVOSD_USB_VCPPORT_DISABLED;
252  }
253 
254  PIOS_HAL_ConfigureCDC(hw_usb_vcpport, pios_usb_id, &pios_usb_cdc_cfg);
255 
256 #endif /* PIOS_INCLUDE_USB_CDC */
257 
258 #if defined(PIOS_INCLUDE_USB_HID)
259  /* Configure the usb HID port */
260  uint8_t hw_usb_hidport;
261  HwPlayUavOsdUSB_HIDPortGet(&hw_usb_hidport);
262 
263  if (!usb_hid_present) {
264  /* Force HID port function to disabled if we haven't advertised HID in our USB descriptor */
265  hw_usb_hidport = HWPLAYUAVOSD_USB_HIDPORT_DISABLED;
266  }
267 
268  PIOS_HAL_ConfigureHID(hw_usb_hidport, pios_usb_id, &pios_usb_hid_cfg);
269 
270 #endif /* PIOS_INCLUDE_USB_HID */
271 
272  if (usb_hid_present || usb_cdc_present) {
273  PIOS_USBHOOK_Activate();
274  }
275 #endif /* PIOS_INCLUDE_USB */
276 
277  /* Main Port */
278  uint8_t hw_mainport;
279  HwPlayUavOsdMainPortGet(&hw_mainport);
280 
281  PIOS_HAL_ConfigurePort(hw_mainport, // port type protocol
282  &pios_usart_main_cfg, // usart_port_cfg
283  &pios_usart_com_driver, // com_driver
284  NULL, // i2c_id
285  NULL, // i2c_cfg
286  NULL, // ppm_cfg
287  NULL, // pwm_cfg
288  PIOS_LED_ALARM, // led_id
289  NULL, // dsm_cfg
290  0, // dsm_mode
291  NULL); // sbus_cfg
292 
293 
294  //I2C is slow, sensor init as well, reset watchdog to prevent reset here
295  PIOS_WDG_Clear();
296 
297 #if defined(PIOS_INCLUDE_SPI)
298  if (PIOS_SPI_Init(&pios_spi_max7456_id, &pios_spi_max7456_cfg))
299  PIOS_DEBUG_Assert(0);
300 
301  GPIO_StructInit(&GPIO_InitStructure);
302  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_0;
303  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
304  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
305  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_NOPULL;
306  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz;
307  GPIO_Init(GPIOC, &GPIO_InitStructure);
308  GPIO_SetBits(GPIOC, GPIO_Pin_0);
309  GPIO_Init(GPIOA, &GPIO_InitStructure);
310  GPIO_ResetBits(GPIOA, GPIO_Pin_0);
311 
312 #if defined (PIOS_INCLUDE_MAX7456)
313  max7456_dev_t max7456;
314  if (PIOS_MAX7456_init(&max7456, pios_spi_max7456_id, 0))
315  PIOS_HAL_CriticalError(PIOS_LED_ALARM, PIOS_HAL_PANIC_OSD);
316 #endif // defined (PIOS_INCLUDE_MAX7456)
317 
318 #endif // defined(PIOS_INCLUDE_SPI)
319 
320 #if defined(PIOS_INCLUDE_VIDEO)
321  // make sure the mask pin is low
322  GPIO_Init(pios_video_cfg.mask.miso.gpio, (GPIO_InitTypeDef*)&pios_video_cfg.mask.miso.init);
323  GPIO_ResetBits(pios_video_cfg.mask.miso.gpio, pios_video_cfg.mask.miso.init.GPIO_Pin);
324 
325  // Initialize settings
326  OnScreenDisplaySettingsInitialize();
327 
328  uint8_t osd_state;
329  OnScreenDisplaySettingsOSDEnabledGet(&osd_state);
330  if (osd_state == ONSCREENDISPLAYSETTINGS_OSDENABLED_ENABLED) {
331  OSD_configure_bw_levels();
332  }
333 #endif // defined(PIOS_INCLUDE_VIDEO)
334 
335  /* Make sure we have at least one telemetry link configured or else fail initialization */
336  PIOS_Assert(pios_com_telem_serial_id || pios_com_telem_usb_id);
337 
338  PIOS_BOARD_Init_Uavos();
339 }
