151  if (dev == NULL)
152  return -1;
153 
154  dev->i2c_id = i2c_device;
155 
156  dev->oversampling = cfg->oversampling;
157  dev->temperature_interleaving = (cfg->temperature_interleaving) == 0 ? 1 : cfg->temperature_interleaving;
158  dev->cfg = cfg;

160  uint8_t data[22];
161  if (PIOS_BMP085_Read(BMP085_CALIB_ADDR, data, 2) != 0)
162  return -2;

164  if (PIOS_BMP085_Read(BMP085_CALIB_ADDR, data, 2) != 0)

169  return -2;

171  dev->AC1 = (data[0] << 8) | data[1];
172  dev->AC2 = (data[2] << 8) | data[3];
173  dev->AC3 = (data[4] << 8) | data[5];
174  dev->AC4 = (data[6] << 8) | data[7];

176  dev->AC6 = (data[10] << 8) | data[11];

178  /* Parameters B1, B2 */
179  dev->B1 = (data[12] << 8) | data[13];
180  dev->B2 = (data[14] << 8) | data[15];
181 
182  /* Parameters MB, MC, MD */
183  dev->MB = (data[16] << 8) | data[17];
184  dev->MC = (data[18] << 8) | data[19];

227 static int32_t PIOS_BMP085_StartADC(enum conversion_type type)
228 {
229  if (PIOS_BMP085_Validate(dev) != 0)
230  return -1;
231 
232  /* Start the conversion */
233  switch(type) {
234  case TEMPERATURE_CONV:
235  while (PIOS_BMP085_WriteCommand(BMP085_CTRL_ADDR, BMP085_TEMP_ADDR) != 0)
236  continue;
237  break;
238  case PRESSURE_CONV:
239  while (PIOS_BMP085_WriteCommand(BMP085_CTRL_ADDR, BMP085_PRES_ADDR + (dev->oversampling << 6)) != 0)
240  continue;
241  break;
242  default:
243  return -1;
244  }
245 
246  dev->current_conversion_type = type;
247 
248  return 0;

277 static int32_t PIOS_BMP085_ReadADC(void)
278 {
279  if (PIOS_BMP085_Validate(dev) != 0)
280  return -1;
281 
282  uint8_t data[3];
283 
284  /* Read and store the 16bit result */
285  if (dev->current_conversion_type == TEMPERATURE_CONV) {
286  uint32_t raw_temperature;
287  /* Read the temperature conversion */
288  if (PIOS_BMP085_Read(BMP085_ADC_MSB, data, 2) != 0)
289  return -1;
290 
291  raw_temperature = ((data[0] << 8) | data[1]);
292 
293  X1 = (raw_temperature - dev->AC6) * dev->AC5 >> 15;
294  X2 = ((int32_t)dev->MC << 11) / (X1 + dev->MD);
295  B5 = X1 + X2;
296  dev->temperature_unscaled = (B5 + 8) >> 4;//temperature;
297  } else {
298  uint32_t raw_pressure;
299 
300  /* Read the pressure conversion */
301  if (PIOS_BMP085_Read(BMP085_ADC_MSB, data, 3) != 0)
302  return -1;
303 
304  raw_pressure = ((data[0] << 16) | (data[1] << 8) | data[2]) >> (8 - dev->oversampling);
305 
306  B6 = B5 - 4000;
307  X1 = (dev->B2 * (B6 * B6 >> 12)) >> 11;
308  X2 = dev->AC2 * B6 >> 11;
309  X3 = X1 + X2;
310  B3 = ((((int32_t)dev->AC1 * 4 + X3) << dev->oversampling) + 2) >> 2;
311  X1 = dev->AC3 * B6 >> 13;
312  X2 = (dev->B1 * (B6 * B6 >> 12)) >> 16;
313  X3 = ((X1 + X2) + 2) >> 2;
314  B4 = (dev->AC4 * (uint32_t)(X3 + 32768)) >> 15;
315  B7 = ((uint32_t)raw_pressure - B3) * (50000 >> dev->oversampling);
316  P = B7 < 0x80000000 ? (B7 * 2) / B4 : (B7 / B4) * 2;
317  X1 = (P >> 8) * (P >> 8);
318  X1 = (X1 * 3038) >> 16;
319  X2 = (-7357 * P) >> 16;
320  dev->pressure_unscaled = P + ((X1 + X2 + 3791) >> 4);
321  }
322  return 0;
323 }

425 static void PIOS_BMP085_Task(void *parameters)
426 {
427  int32_t temp_press_interleave_count = dev->temperature_interleaving;
428  int32_t read_adc_result = 0;
429 
430  while (1) {
431 
432  temp_press_interleave_count --;
433  if(temp_press_interleave_count <= 0)
434  {
435  // Update the temperature data
436  PIOS_BMP085_ClaimDevice();
437  PIOS_BMP085_StartADC(TEMPERATURE_CONV);
438  PIOS_Thread_Sleep(5);
439  read_adc_result = PIOS_BMP085_ReadADC();
440  PIOS_BMP085_ReleaseDevice();
441  temp_press_interleave_count = dev->temperature_interleaving;
442  }
443  // Update the pressure data
444  PIOS_BMP085_ClaimDevice();
445  PIOS_BMP085_StartADC(PRESSURE_CONV);
446  PIOS_Thread_Sleep(PIOS_BMP085_GetDelay());
447  read_adc_result = PIOS_BMP085_ReadADC();
448  PIOS_BMP085_ReleaseDevice();
449 
450  // Compute the altitude from the pressure and temperature and send it out
451  struct pios_sensor_baro_data data;
452  data.temperature = ((float) dev->temperature_unscaled) / 10.0f;
453  data.pressure = ((float) dev->pressure_unscaled) / 1000.0f;
454  data.altitude = 44330.0f * (1.0f - powf(data.pressure / BMP085_P0, (1.0f / 5.255f)));
455 
456  if (read_adc_result == 0) {
457  PIOS_Queue_Send(dev->queue, (void*)&data, 0);
458  }
459  }
460 }
