123 static bool frsky_encode_rssi(struct frsky_settings *frsky, uint32_t *value, bool test_presence_only, uint32_t arg)
124 {
125  uint8_t local_link_quality, local_link_connected;
126 
127  RFM22BStatusLinkStateGet(&local_link_connected);
128  RFM22BStatusLinkQualityGet(&local_link_quality);
129 
130  RFM22BStatusData rfm22bStatus;
131  RFM22BStatusInstGet(1, &rfm22bStatus);
132 
133  if (local_link_connected == RFM22BSTATUS_LINKSTATE_CONNECTED) {
134  // report whichever link quality is worse
135  *value = (rfm22bStatus.LinkQuality < local_link_quality) ? rfm22bStatus.LinkQuality : local_link_quality;
136 
137  // Rescale to values that match Taranis
138  if (*value < 64) {
139  *value = 0;
140  } else {
141  *value = (*value - 64) * 100 / 64;
142  }
143  } else {
144  *value = 0;
145  }
146 
147  return true;
148 }

170 static void frsky_schedule_next_item(void)
171 {
172  uint32_t i = 0;
173  int32_t max_exp_time = INT32_MIN;
174  int32_t most_exp_item = -1;
175  for (i = 0; i < NELEMENTS(frsky_value_items); i++) {
176  if (frsky_value_items[i].encode_value(&frsky->frsky_settings, 0, true, frsky_value_items[i].fn_arg)) {
177  int32_t exp_time = PIOS_DELAY_GetuSSince(frsky->item_last_triggered[i]) -
178  (frsky_value_items[i].period_ms * 1000);
179  if (exp_time > max_exp_time) {
180  max_exp_time = exp_time;
181  most_exp_item = i;
182  }
183  }
184  }
185  frsky->scheduled_item = most_exp_item;
186 }
191 static bool frsky_send_scheduled_item(void)
192 {
193  int32_t item = frsky->scheduled_item;
194  if ((item >= 0) && (item < NELEMENTS(frsky_value_items))) {
195  frsky->item_last_triggered[item] = PIOS_DELAY_GetuS();
196  uint32_t value = 0;
197  if (frsky_value_items[item].encode_value(&frsky->frsky_settings, &value, false,
198  frsky_value_items[item].fn_arg)) {
199  frsky_send_frame(frsky->com, (uint16_t)(frsky_value_items[item].id), value, true);
200  return true;
201  }
202  }
203 
204  return false;
205 }

256  frsky->last_poll_time = PIOS_DELAY_GetuS();
257  frsky->ignore_rx_chars = 0;
258  frsky->scheduled_item = -1;
259  frsky->com = sport_com;
260 
261  uint8_t i;
262  for (i = 0; i < NELEMENTS(frsky_value_items); i++)
263  frsky->item_last_triggered[i] = PIOS_DELAY_GetuS();
264  PIOS_COM_ChangeBaud(frsky->com, FRSKY_SPORT_BAUDRATE);
265  module_enabled = true;
266  return 0;
267  }
268 
269  module_enabled = true;
270 
271  return 0;
272  }
273

283 static void uavoTaranisTask(void *parameters)
284 {
285  while(1) {
286 
287  if (true) {
288 
289  // for some reason, only first four messages are sent.
290  for (uint32_t i = 0; i < sizeof(frsky_value_items) / sizeof(frsky_value_items[0]); i++) {
291  frsky->scheduled_item = i;
292  frsky_send_scheduled_item();
293  PIOS_Thread_Sleep(5);
294  }
295 
296  }
297 
298  if (false) {
299 
300  // fancier schedlued message sending. doesn't appear to work
301  // currently.
302  PIOS_Thread_Sleep(1);
303  frsky_schedule_next_item();
304  frsky_send_scheduled_item();
305  }
306 
307  }
