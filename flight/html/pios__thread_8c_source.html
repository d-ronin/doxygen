232 static uint8_t * align8_alloc(uint32_t size)
233 {
234  // round size up to at nearest multiple of 8 + 4 bytes to guarantee
235  // sufficient size within. This is because PIOS_malloc only guarantees
236  // uintptr_t alignment which is 4 bytes.
237  size = size + sizeof(uintptr_t);
238  uint8_t *wap = PIOS_malloc(size);
239 
240  // shift start point to nearest 8 byte boundary.
241  uint32_t pad = ((uint32_t) wap) % sizeof(stkalign_t);
242  wap = wap + pad;
243 
244  return wap;
245 }

356 void PIOS_Thread_Sleep_Until(uint32_t *previous_ms, uint32_t increment_ms)
357 {
358  // Do the math in the millisecond domain.
359  *previous_ms += increment_ms;
360 
361  systime_t future = CVT_MS2ST(*previous_ms);
362  systime_t increment_st = CVT_MS2ST(increment_ms);
363 
364  chSysLock();
365 
366  systime_t now = chTimeNow();
367  systime_t sleep_time = future - now;
368 
369  if (sleep_time > increment_st) {
370  // OK, the calculated sleep time is much longer than
371  // the desired interval. There's two possibilities:
372  // 1) We were already late! If so just don't sleep.
373  // 2) The timer wrapped. (at 49 days) Just don't sleep.
374 
375  if ((now - future) > increment_st) {
376  // However, in this particular case we need to fix up
377  // the clock. If we're very late, OR WRAPPED,
378  // don't try and keep on the previous timebase.
379  *previous_ms = CVT_ST2MS(now);
380  }
381  } else if (sleep_time > 0) {
382  // Be sure not to request a sleep_time of 0, as that's
383  // IMMEDIATE and makes chTdSleepS panic.
384  chThdSleepS(sleep_time);
385  }
386 
387  chSysUnlock();
388 }
