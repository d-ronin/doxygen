143 void PIOS_DMAShot_WriteValue(const struct pios_tim_channel *servo_channel, uint16_t throttle)
144 {
145  // Fail hard if trying to write values without configured DMAShot. We shouldn't be getting to
146  // this point in that case.
147  PIOS_Assert(dmashot_cfg);
148 
149  // If we can't find a timer for this, something's wrong. We shouldn't be getting to this point.
150  // Fail hard.
151  struct servo_timer *s_timer = PIOS_DMAShot_GetServoTimer(servo_channel);
152  PIOS_Assert(s_timer);
153 
154  int shift = (servo_channel->timer_chan - s_timer->low_channel) >> 2;
155  int channels = PIOS_DMAShot_GetNumChannels(s_timer);
156 
157  if (throttle > 2047)
158  throttle = 2047;
159 
160  throttle <<= 5;
161  throttle |=
162  ((throttle >> 4 ) & 0xf) ^
163  ((throttle >> 8 ) & 0xf) ^
164  ((throttle >> 12) & 0xf);
165 
166  // Leading zero, trailing zero.
167  for (int i = DMASHOT_MESSAGE_PAUSE; i < DMASHOT_MESSAGE_WIDTH+DMASHOT_MESSAGE_PAUSE; i++) {
168  int addr = i * channels + shift;
169  if (PIOS_DMAShot_HalfWord(s_timer)) {
170  s_timer->buffer.hw[addr] = throttle & 0x8000 ? s_timer->duty_cycle_1 : s_timer->duty_cycle_0;
171  } else {
172  s_timer->buffer.fw[addr] = throttle & 0x8000 ? s_timer->duty_cycle_1 : s_timer->duty_cycle_0;
173  }
174  throttle <<= 1;
175  }
176 }

178 bool PIOS_DMAShot_RegisterTimer(TIM_TypeDef *timer, uint32_t clockrate, uint32_t dshot_freq)
179 {
180  // No configuration, push for GPIO in upstream.
181  if (!dmashot_cfg || !servo_timers)
182  return false;
183 
184  // Check whether the timer is configured for DMA first. If not, bail and
185  // tell upstairs that DMA DShot is a no-go.
186  const struct pios_dmashot_timer_cfg *dma_config = NULL;
187 
188  for (int i = 0; i < dmashot_cfg->num_timers; i++) {
189  if (dmashot_cfg->timer_cfg[i].timer == timer) {
190  dma_config = &dmashot_cfg->timer_cfg[i];
191  }
192  }
193 
194  if (!dma_config)
195  return false;
196 
197  // Check whether there's an existing config from a previous registration,
198  // since we can't clean up due to a lack of free(). Timer/pin stuff is
199  // static anyway.
200  struct servo_timer *s_timer = NULL;
201 
202  bool found = false;
203  for (int i = 0; i < MAX_TIMERS; i++) {
204  if (servo_timers[i] && servo_timers[i]->dma->timer == timer) {
205  s_timer = servo_timers[i];
206  found = true;
207  break;
208  }
209  }
210 
211  // Nothing found? Allocate a new slot for the timer.
212  if (!found) {
213  for (int i = 0; i < MAX_TIMERS; i++) {
214  if (!servo_timers[i]) {
215  s_timer = PIOS_malloc_no_dma(sizeof(*s_timer));
216  // No point in returning false and falling back to GPIO, because that
217  // also needs to allocate stuff later on, so we might just fail here.
218  PIOS_Assert(s_timer);
219  memset(s_timer, 0, sizeof(*s_timer));
220  s_timer->low_channel = TIM_Channel_4;
221  s_timer->high_channel = TIM_Channel_1;
222  servo_timers[i] = s_timer;
223  break;
224  }
225  }
226  }
227 
228  // Why are we trying to register more than MAX_TIMERS, anyway?
229  PIOS_Assert(s_timer);
230 
231  s_timer->sysclock = clockrate;
232  s_timer->dshot_freq = dshot_freq;
233  s_timer->dma = dma_config;
234 
235  return true;
236 }

238 bool PIOS_DMAShot_RegisterServo(const struct pios_tim_channel *servo_channel)
239 {
240  // No configuration? kthxbye!

242  return false;
243 
244  struct servo_timer *s_timer = PIOS_DMAShot_GetServoTimer(servo_channel);
245 
246  // No timer found, tell upstream to GPIO!
247  if (!s_timer)
248  return false;
249 
250  if (s_timer->dma->master_timer && (PIOS_DMAShot_GetNumChannels(s_timer) > 0) &&
251  (s_timer->low_channel != servo_channel->timer_chan)) {
252  // I'm sorry, Dave, I'm afraid I cannot do that!
253 
254  // If DMA'ing directly to CCR, can only do one channel per timer.
255  // Tell upstream to GPIO instead.
256  return false;
257  }
258 
259  if (servo_channel->timer_chan < s_timer->low_channel) s_timer->low_channel = servo_channel->timer_chan;
260  if (servo_channel->timer_chan > s_timer->high_channel) s_timer->high_channel = servo_channel->timer_chan;
261 
262  s_timer->servo_channels[TIMC_TO_INDEX(servo_channel->timer_chan)] = servo_channel;
263 
264  return true;

310 static void PIOS_DMAShot_TimerSetup(struct servo_timer *s_timer, uint32_t sysclock, uint32_t dshot_freq, TIM_OCInitTypeDef *ocinit, bool master)
311 {
312  TIM_TypeDef *timer = master ? s_timer->dma->master_timer : s_timer->dma->timer;
313  TIM_TimeBaseInitTypeDef timerdef;
314 
315  TIM_Cmd(timer, DISABLE);
316 
317  TIM_TimeBaseStructInit(&timerdef);
318 
319  timerdef.TIM_Prescaler = 0;
320  timerdef.TIM_Period = sysclock / dshot_freq;
321  timerdef.TIM_ClockDivision = TIM_CKD_DIV1;
322  timerdef.TIM_RepetitionCounter = 0;
323  timerdef.TIM_CounterMode = TIM_CounterMode_Up;
324 
325  TIM_TimeBaseInit(timer, &timerdef);
326 
327  // TIM_Channels are spread apart by 4 in stm32f4xx_tim.h
328  for(int i = s_timer->low_channel; i <= s_timer->high_channel; i+=4)
329  {
330  switch(i)
331  {
332  case TIM_Channel_1:
333  TIM_OC1Init(timer, ocinit);
334  TIM_OC1PreloadConfig(timer, TIM_OCPreload_Enable);
335  break;
336  case TIM_Channel_2:
337  TIM_OC2Init(timer, ocinit);
338  TIM_OC2PreloadConfig(timer, TIM_OCPreload_Enable);
339  break;
340  case TIM_Channel_3:
341  TIM_OC3Init(timer, ocinit);
342  TIM_OC3PreloadConfig(timer, TIM_OCPreload_Enable);
343  break;
344  case TIM_Channel_4:
345  TIM_OC4Init(timer, ocinit);
346  TIM_OC4PreloadConfig(timer, TIM_OCPreload_Enable);
347  break;
348  }
349  }
350 
351  // Do this, in case SyncPWM was configured before.
352  TIM_SelectOnePulseMode(timer, TIM_OPMode_Repetitive);
353 }

356 {

387 static void PIOS_DMAShot_DMASetup(struct servo_timer *s_timer)
388 {
389  DMA_Cmd(s_timer->dma->stream, DISABLE);
390  while (DMA_GetCmdStatus(s_timer->dma->stream) == ENABLE) ;
391 
392  DMA_DeInit(s_timer->dma->stream);
393 
394  DMA_InitTypeDef dma;
395  DMA_StructInit(&dma);
396 
397  dma.DMA_Channel = s_timer->dma->channel;
398  dma.DMA_Memory0BaseAddr = s_timer->buffer.ptr;
399  if (s_timer->dma->master_timer) {
400  // The DMABase shift is in count of 32-bit registers. 16-bit registers are padded with 16-bit reserved ones,
401  // and thus defacto 32-bit.
402  dma.DMA_PeripheralBaseAddr = ((uint32_t)&s_timer->dma->timer->CR1) + ((s_timer->dma->master_config & 0x00FF) << 2);
403  } else {
404  dma.DMA_PeripheralBaseAddr = (uint32_t)(&s_timer->dma->timer->DMAR);
405  }
406 
407  if (PIOS_DMAShot_HalfWord(s_timer)) {
408  dma.DMA_MemoryDataSize = DMA_MemoryDataSize_HalfWord;
409  dma.DMA_PeripheralDataSize = DMA_PeripheralDataSize_HalfWord;
410  } else {
411  dma.DMA_MemoryDataSize = DMA_MemoryDataSize_Word;
412  dma.DMA_PeripheralDataSize = DMA_PeripheralDataSize_Word;
413  }
414 
415  dma.DMA_MemoryInc = DMA_MemoryInc_Enable;
416  dma.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
417 
418  dma.DMA_DIR = DMA_DIR_MemoryToPeripheral;
419  dma.DMA_Mode = DMA_Mode_Normal;
420 
421  dma.DMA_BufferSize = PIOS_DMAShot_GetNumChannels(s_timer) * DMASHOT_STM32_BUFFER;
422 
423  dma.DMA_Priority = DMA_Priority_VeryHigh;
424  dma.DMA_MemoryBurst = DMA_MemoryBurst_INC4;
425  dma.DMA_PeripheralBurst = DMA_PeripheralBurst_Single;
426 
427  dma.DMA_FIFOMode = DMA_FIFOMode_Enable;
428  dma.DMA_FIFOThreshold = DMA_FIFOThreshold_Full;
429 
430  DMA_Init(s_timer->dma->stream, &dma);
431 
432  // Don't do interrupts.
433  DMA_ITConfig(s_timer->dma->stream, DMA_IT_TC, DISABLE);
434 }

436 // Allocates an aligned buffer for DMA burst transfers. Returns uint32_t, since
437 // that's the pointer type for Mem0 base address in DMA_InitTypeDef.
438 static uint32_t PIOS_DMAShot_AllocateBuffer(uint16_t size)
439 {
440  size += 12;
441  uint32_t ptr = (uint32_t)PIOS_malloc(size);
442 
443  // Blow up if we didn't get a buffer.
444  PIOS_Assert(ptr);
445 
446  memset((void*)ptr, 0, size);
447 
448  // Align to 16-byte boundary.
449  uint32_t shift = 16 - (ptr % 16);
450  if(shift != 16) ptr += shift;
451 
452  return ptr;
453 }

477 void PIOS_DMAShot_TriggerUpdate()
478 {
479  // If there's nothing setup, fail hard. We shouldn't be getting here.
480  PIOS_Assert(dmashot_cfg && servo_timers);
481 
482  for (int i = 0; i < MAX_TIMERS; i++) {
483  struct servo_timer *s_timer = servo_timers[i];
484  if (!s_timer || !s_timer->sysclock)
485  continue;
486 
487  // Wait for DMA to finish.
488  if(s_timer->dma_started) {
489  while(DMA_GetFlagStatus(s_timer->dma->stream, s_timer->dma->tcif) != SET) ;
490  }
491 
492  DMA_Cmd(s_timer->dma->stream, DISABLE);
493  while (DMA_GetCmdStatus(s_timer->dma->stream) == ENABLE) ;
494 
495  if (s_timer->dma->master_timer) {
496  TIM_DMACmd(s_timer->dma->master_timer,
497  // This results in a typical event source value
498  s_timer->dma->master_config & 0xFF00,
499  DISABLE);
500  TIM_Cmd(s_timer->dma->master_timer, DISABLE);
501  TIM_SetCounter(s_timer->dma->master_timer, 0);
502  } else {
503  TIM_DMACmd(s_timer->dma->timer, TIM_DMA_Update, DISABLE);
504  }
505 
506  TIM_Cmd(s_timer->dma->timer, DISABLE);
507  TIM_SetCounter(s_timer->dma->timer, 0);
508 
509  DMA_ClearFlag(s_timer->dma->stream, s_timer->dma->tcif);
510  DMA_SetCurrDataCounter(s_timer->dma->stream, PIOS_DMAShot_GetNumChannels(s_timer) * DMASHOT_STM32_BUFFER);
511  }
512 
513  // Re-enable the timers and DMA in a separate loop, to make the signals line up better.
514  for (int i = 0; i < MAX_TIMERS; i++) {
515  struct servo_timer *s_timer = servo_timers[i];
516  if (!s_timer || !s_timer->sysclock)
517  continue;
518 
519  if (s_timer->dma->master_timer) {
520  TIM_DMACmd(s_timer->dma->master_timer,
521  // This results in a typical event source value
522  s_timer->dma->master_config & 0xFF00,
523  ENABLE);
524  TIM_Cmd(s_timer->dma->timer, ENABLE);
525  TIM_Cmd(s_timer->dma->master_timer, ENABLE);
526  } else {
527  int offset = s_timer->low_channel >> 2;
528  int transfers = (s_timer->high_channel - s_timer->low_channel) >> 2;
529  TIM_DMAConfig(s_timer->dma->timer, TIM_DMABase_CCR1 + offset, transfers << 8);
530 
531  TIM_DMACmd(s_timer->dma->timer, TIM_DMA_Update, ENABLE);
532  TIM_Cmd(s_timer->dma->timer, ENABLE);
533  }
534 
535  DMA_Cmd(s_timer->dma->stream, ENABLE);
536  s_timer->dma_started = 1;
537  }
538 }
