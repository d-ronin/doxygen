592 void PIOS_HAL_ConfigurePort(HwSharedPortTypesOptions port_type,
593  const struct pios_usart_cfg *usart_port_cfg,

595  uint32_t *i2c_id,

599  uint32_t led_id,
600 /* TODO: future work to factor most of these away */
601  const struct pios_dsm_cfg *dsm_cfg,

603  const struct pios_sbus_cfg *sbus_cfg)
604 {
605  uintptr_t port_driver_id;
606  uintptr_t *target = NULL, *target2 = NULL;
607 
608  struct pios_usart_params usart_port_params;
609 
610  usart_port_params.init.USART_BaudRate = 57600;
611  usart_port_params.init.USART_WordLength = USART_WordLength_8b;
612  usart_port_params.init.USART_Parity = USART_Parity_No;
613  usart_port_params.init.USART_StopBits = USART_StopBits_1;
614  usart_port_params.init.USART_HardwareFlowControl = USART_HardwareFlowControl_None;
615  usart_port_params.init.USART_Mode = USART_Mode_Rx | USART_Mode_Tx;
616 
617  usart_port_params.rx_invert = false; // Only used by F3 targets
618  usart_port_params.tx_invert = false; // Only used by F3 targets
619  usart_port_params.rxtx_swap = false; // Only used by F3 targets
620  usart_port_params.single_wire = false; // Only used by F3 targets
621 
622  // If there is a hardware inverter for this port
623  if (sbus_cfg != NULL) {
624  // Enable inverter gpio clock
625  if (sbus_cfg->gpio_clk_func != NULL)
626  (*sbus_cfg->gpio_clk_func)(sbus_cfg->gpio_clk_periph, ENABLE);
627 
628  // Initialize hardware inverter GPIO pin
629  GPIO_Init(sbus_cfg->inv.gpio, (GPIO_InitTypeDef*)&sbus_cfg->inv.init);
630 
631  // Enable Inverter for Sbus
632  if (port_type == HWSHARED_PORTTYPES_SBUS)
633  GPIO_WriteBit(sbus_cfg->inv.gpio, sbus_cfg->inv.init.GPIO_Pin, sbus_cfg->gpio_inv_enable);
634  else
635  GPIO_WriteBit(sbus_cfg->inv.gpio, sbus_cfg->inv.init.GPIO_Pin, sbus_cfg->gpio_inv_disable);
636  }
637 
638  switch (port_type) {
639 
640  case HWSHARED_PORTTYPES_DISABLED:
641  break;
642 
643  case HWSHARED_PORTTYPES_COMBRIDGE:
644  PIOS_HAL_ConfigureCom(usart_port_cfg, &usart_port_params, PIOS_COM_BRIDGE_RX_BUF_LEN, PIOS_COM_BRIDGE_TX_BUF_LEN, com_driver, &port_driver_id);
645  target = &pios_com_bridge_id;
646  //module is enabled by the USB port config rather than here
647  break;
648 
649  case HWSHARED_PORTTYPES_DEBUGCONSOLE:
650 #if defined(PIOS_INCLUDE_DEBUG_CONSOLE)
651  PIOS_HAL_ConfigureCom(usart_port_cfg, &usart_port_params, 0, PIOS_COM_DEBUGCONSOLE_TX_BUF_LEN, com_driver, &port_driver_id);
652  target = &pios_com_debug_id;
653 #endif /* PIOS_INCLUDE_DEBUG_CONSOLE */
654  break;
655 
656  case HWSHARED_PORTTYPES_DSM:
657 #if defined(PIOS_INCLUDE_DSM)
658  if (dsm_cfg && usart_port_cfg) {
659  usart_port_params.init.USART_BaudRate = 115200;
660  usart_port_params.init.USART_WordLength = USART_WordLength_8b;
661  usart_port_params.init.USART_Parity = USART_Parity_No;
662  usart_port_params.init.USART_StopBits = USART_StopBits_1;
663  usart_port_params.init.USART_HardwareFlowControl = USART_HardwareFlowControl_None;
664  usart_port_params.init.USART_Mode = USART_Mode_Rx;

669  break;
