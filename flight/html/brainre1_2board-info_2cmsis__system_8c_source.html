330 void SystemCoreClockUpdate(void)
331 {
332  uint32_t tmp = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;
333 #if defined(STM32F446xx)
334  uint32_t pllr = 2;
335 #endif /* STM32F446xx */
336  /* Get SYSCLK source -------------------------------------------------------*/
337  tmp = RCC->CFGR & RCC_CFGR_SWS;
338 
339  switch (tmp)
340  {
341  case 0x00: /* HSI used as system clock source */
342  SystemCoreClock = HSI_VALUE;
343  break;
344  case 0x04: /* HSE used as system clock source */
345  SystemCoreClock = HSE_VALUE;
346  break;
347  case 0x08: /* PLL P used as system clock source */
348  /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
349  SYSCLK = PLL_VCO / PLL_P
350  */
351  pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
352  pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
353 
354 #if defined(STM32F40_41xxx) || defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F401xx) || defined(STM32F446xx)
355  if (pllsource != 0)
356  {
357  /* HSE used as PLL clock source */
358  pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
359  }
360  else
361  {
362  /* HSI used as PLL clock source */
363  pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
364  }
365 #elif defined(STM32F411xE)
366 #if defined(USE_HSE_BYPASS)
367  if (pllsource != 0)
368  {
369  /* HSE used as PLL clock source */
370  pllvco = (HSE_BYPASS_INPUT_FREQUENCY / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
371  }
372 #else
373  if (pllsource == 0)
374  {
375  /* HSI used as PLL clock source */
376  pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
377  }
378 #endif /* USE_HSE_BYPASS */
379 #endif /* STM32F40_41xxx || STM32F427_437xx || STM32F429_439xx || STM32F401xx || STM32F446xx */
380  pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >>16) + 1 ) *2;
381  SystemCoreClock = pllvco/pllp;
382  break;
383 #if defined(STM32F446xx)
384  case 0x0C: /* PLL R used as system clock source */
385  /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
386  SYSCLK = PLL_VCO / PLL_R
387  */
388  pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
389  pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
390  if (pllsource != 0)
391  {
392  /* HSE used as PLL clock source */
393  pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
394  }
395  else
396  {
397  /* HSI used as PLL clock source */
398  pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
399  }
400 
401  pllr = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLR) >>28) + 1 ) *2;
402  SystemCoreClock = pllvco/pllr;
403  break;
404 #endif /* STM32F446xx */
405  default:
406  SystemCoreClock = HSI_VALUE;
407  break;
408  }
409  /* Compute HCLK frequency --------------------------------------------------*/
410  /* Get HCLK prescaler */
411  tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
412  /* HCLK frequency */
413  SystemCoreClock >>= tmp;
414 }
