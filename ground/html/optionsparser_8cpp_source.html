78 void OptionsParser::checkForPluginOption(QStringList pluginOptions)
79 {
80  foreach (QString option, pluginOptions) {
81  QString simplified = option.simplified().replace(" ", "");
82  QString argument;
83  QString value;
84  if(option.contains("=")) {
85  if(simplified.split("=").length() != 2) {
86  m_errorStrings.append(QCoreApplication::translate(("PluginManager"),
87  "Wrong plugin options syntax: %0").arg(option));
88  continue;
89  }
90  argument = simplified.split("=").at(0);
91  value = simplified.split("=").at(1);
92  }
93  else {
94  argument = option;
95  }
96  bool requiresParameter;
97  PluginSpec *spec = m_pmPrivate->pluginForOption(argument, &requiresParameter);
98  if (!spec) {
99  m_errorStrings.append(QCoreApplication::translate("PluginManager",
100  "No Plugin was found for given argument: %0").arg(argument));
101  continue;
102  }
103  spec->addArgument(argument);
104  if(!value.isEmpty())
105  spec->addArgument(value);
106  }
107  return;
