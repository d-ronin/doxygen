222 QJsonObject UAVObject::getJsonRepresentation()
223 {
224 
225  QJsonObject obj, fieldMap;
226 
227  obj["id"] = static_cast<qint64>(getObjID());
228  obj["name"] = getName();
229 
230  foreach (UAVObjectField *field, fields) {
231  quint32 nelem = field->getNumElements();
232 
233  if (nelem > 1) {
234  QVariantList vals;
235  for (unsigned int n = 0; n < nelem; ++n) {
236  vals.append(field->getValue(n));
237  }
238 
239  fieldMap[field->getName()] = QJsonArray::fromVariantList(vals);
240  } else {
241  fieldMap[field->getName()] = QJsonValue::fromVariant(field->getValue(0));
242  }
243  }
244 
245  obj["fields"] = fieldMap;
246 
247  return obj;
248 }

254 UAVObjectField *UAVObject::getField(const QString &name)
255 {
256  // Look for field
257  for (int n = 0; n < fields.length(); ++n) {
258  if (name.compare(fields[n]->getName()) == 0) {
259  return fields[n];

261  }
