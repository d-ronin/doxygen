206 QJsonObject UAVObject::getJsonRepresentation()
207 {
208 
209  QJsonObject obj, fieldMap;
210 
211  obj["id"] = static_cast<qint64>(getObjID());
212  obj["name"] = getName();
213 
214  foreach (UAVObjectField *field, fields) {
215  quint32 nelem = field->getNumElements();
216 
217  if (nelem > 1) {
218  QVariantList vals;
219  for (unsigned int n = 0; n < nelem; ++n) {
220  vals.append(field->getValue(n));
221  }
222 
223  fieldMap[field->getName()] = QJsonArray::fromVariantList(vals);
224  } else {
225  fieldMap[field->getName()] = QJsonValue::fromVariant(field->getValue(0));
226  }
227  }
228 
229  obj["fields"] = fieldMap;
230 
231  return obj;
232 }

238 UAVObjectField *UAVObject::getField(const QString &name)
239 {
240  // Look for field
241  for (int n = 0; n < fields.length(); ++n) {
242  if (name.compare(fields[n]->getName()) == 0) {
243  return fields[n];
244  }
245  }
246  // If this point is reached then the field was not found
247  qWarning() << "UAVObject::getField Non existant field " << name

249  "non-debug code.";
