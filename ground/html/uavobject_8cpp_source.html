224 QJsonObject UAVObject::getJsonRepresentation() {
225 
226  QJsonObject obj, fieldMap;
227 
228  obj["id"] = static_cast <qint64> (getObjID());
229  obj["name"] = getName();
230 
231  foreach (UAVObjectField* field, fields) {
232  quint32 nelem = field->getNumElements();
233 
234  if (nelem > 1) {
235  QVariantList vals;
236  for (unsigned int n = 0; n < nelem; ++n) {
237  vals.append(field->getValue(n));
238  }
239 
240  fieldMap[field->getName()] = QJsonArray::fromVariantList(vals);
241  } else {
242  fieldMap[field->getName()] = QJsonValue::fromVariant(field->getValue(0));
243  }
244  }
245 
246  obj["fields"] = fieldMap;
247 
248  return obj;
249 }
