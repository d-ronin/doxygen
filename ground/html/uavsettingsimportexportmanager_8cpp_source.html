110 bool UAVSettingsImportExportManager::importUAVSettings(const QByteArray &settings, bool quiet)
111 {
112  QDomDocument doc("UAVObjects");
113 
114  if (!doc.setContent(settings)) {
115  QMessageBox msgBox;
116  msgBox.setText(tr("File Parsing Failed."));
117  msgBox.setInformativeText(tr("This file is not a correct XML file"));
118  msgBox.setStandardButtons(QMessageBox::Ok);
119  msgBox.exec();
120  return false;
121  }
122  emit importAboutToBegin();
123  qDebug() << "Import about to begin";
124 
125  // find the root of settings subtree
126  QDomElement root = doc.documentElement();
127  if (root.tagName() == "uavobjects") {
128  root = root.firstChildElement("settings");
129  }
130  if (root.isNull() || (root.tagName() != "settings")) {
131  QMessageBox msgBox;
132  msgBox.setText(tr("Wrong file contents"));
133  msgBox.setInformativeText(tr("This file does not contain correct UAVSettings"));
134  msgBox.setStandardButtons(QMessageBox::Ok);
135  msgBox.exec();
136  return false;
137  }
138 
139  // We are now ok: setup the import summary dialog & update it as we
140  // go along.
141  ImportSummaryDialog swui((QWidget *)Core::ICore::instance()->mainWindow(), quiet);
142 
143  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
144  UAVObjectManager *boardObjManager = pm->getObject<UAVObjectManager>();
145  UAVObjectManager *importedObjectManager = new UAVObjectManager();
146 
147  swui.show();
148 
149  QDomNode node = root.firstChild();
150  while (!node.isNull()) {
151  QDomElement e = node.toElement();
152  if (e.tagName() == "object") {
153 
154  // - Read each object
155  QString uavObjectName = e.attribute("name");
156  uint uavObjectID = e.attribute("id").toUInt(NULL, 16);
157 
158  // Sanity Check:
159  UAVObject *obj = boardObjManager->getObject(uavObjectName);
160  UAVDataObject *dobj = dynamic_cast<UAVDataObject *>(obj);
161  if (obj == NULL) {
162  // This object is unknown!
163  qDebug() << "Object unknown:" << uavObjectName << uavObjectID;
164  swui.addLine(uavObjectName, "Error (Object unknown)", false);
165  } else if (dobj && !dobj->getIsPresentOnHardware()) {
166  swui.addLine(uavObjectName, "Error (Object not present on hw)", false);
167  } else {
168  // - Update each field
169  // - Issue and "updated" command
170  UAVDataObject *newObj = dobj->clone();
171 
172  bool error = false;
173  bool setError = false;
174  QDomNode field = node.firstChild();
175  while (!field.isNull()) {
176  QDomElement f = field.toElement();
177  if (f.tagName() == "field") {
178  UAVObjectField *uavfield = newObj->getField(f.attribute("name"));
179  if (uavfield) {
180  QStringList list = f.attribute("values").split(",");
181  if (list.length() == 1) {
182  if (false == uavfield->checkValue(f.attribute("values"))) {
183  qDebug() << "checkValue returned false on: " << uavObjectName
184  << f.attribute("values");
185  setError = true;
186  } else {
187  uavfield->setValue(f.attribute("values"));
188  }
189  } else {
190  // This is an enum:
191  int i = 0;
192  QStringList list = f.attribute("values").split(",");
193  foreach (QString element, list) {
194  if (false == uavfield->checkValue(element, i)) {
195  qDebug() << "checkValue(list) returned false on: "
196  << uavObjectName << list;
197  setError = true;
198  } else {
199  uavfield->setValue(element, i);
200  }
201  i++;
202  }
203  }
204  } else {
205  error = true;
206  }
207  }
208  field = field.nextSibling();
209  }
210  newObj->updated();
211 
212  if (error) {
213  swui.addLine(uavObjectName, "Warning (Object field unknown)", true);
214  } else if (uavObjectID != newObj->getObjID()) {
215  qDebug() << "Mismatch for Object " << uavObjectName << uavObjectID << " - "
216  << newObj->getObjID();
217  swui.addLine(uavObjectName, "Warning (ObjectID mismatch)", true);
218  } else if (setError) {
219  swui.addLine(uavObjectName, "Warning (Objects field value(s) invalid)", false);
220  } else {
221  swui.addLine(uavObjectName, "OK", true);
222  }
223  importedObjectManager->registerObject(newObj);
224  }
225  }
226  node = node.nextSibling();
227  }
228  qDebug() << "End import";
229 
230  if (swui.numLines() < 1) {
231  QMessageBox::critical(nullptr, tr("Unable to import settings"),
232  tr("No settings found in XML dump; the flight controller settings may have been blank."), QMessageBox::Ok);
233  /* This is a termination condition-- no settings found. */
234  return true;
235  }
236 
237  swui.setUAVOSettings(importedObjectManager);
238  swui.exec();
239 
240  return swui.result() == QDialog::Accepted;
241 }

265 QString UAVSettingsImportExportManager::createXMLDocument(const enum storedData what,
266  const bool fullExport)
267 {
268  // generate an XML first (used for all export formats as a formatted data source)
269  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
270  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
271 
272  // create an XML root
273  QDomDocument doc("UAVObjects");
274  QDomElement root = doc.createElement("uavobjects");
275  doc.appendChild(root);
276 
277  // add hardware, firmware and GCS version info
278  QDomElement versionInfo = doc.createElement("version");
279  root.appendChild(versionInfo);
280 
281  UAVObjectUtilManager *utilMngr = pm->getObject<UAVObjectUtilManager>();
282  deviceDescriptorStruct board;
283  utilMngr->getBoardDescriptionStruct(board);
284 
285  QDomElement hw = doc.createElement("hardware");
286  hw.setAttribute("type", QString().setNum(board.boardType, 16));
287  hw.setAttribute("revision", QString().setNum(board.boardRevision, 16));
288  hw.setAttribute("serial", QString(utilMngr->getBoardCPUSerial().toHex()));
289  versionInfo.appendChild(hw);
290 
291  QDomElement fw = doc.createElement("firmware");
292  fw.setAttribute("date", board.gitDate);
293  fw.setAttribute("hash", board.gitHash);
294  fw.setAttribute("tag", board.gitTag);
295  versionInfo.appendChild(fw);
296 
297  QString gcsRevision = QString::fromLatin1(Core::Constants::GCS_REVISION_STR);
298  QString gcsGitDate = gcsRevision.mid(gcsRevision.indexOf(" ") + 1, 14);
299  QString gcsGitHash = gcsRevision.mid(gcsRevision.indexOf(":") + 1, 8);
300  QString gcsGitTag = gcsRevision.left(gcsRevision.indexOf(":"));
301 
302  QDomElement gcs = doc.createElement("gcs");
303  gcs.setAttribute("date", gcsGitDate);
304  gcs.setAttribute("hash", gcsGitHash);
305  gcs.setAttribute("tag", gcsGitTag);
306  versionInfo.appendChild(gcs);
307 
308  // create settings and/or data elements
309  QDomElement settings = doc.createElement("settings");
310  QDomElement data = doc.createElement("data");
311 
312  switch (what) {
313  case Settings:
314  root.appendChild(settings);
315  break;
316  case Data:
317  root.appendChild(data);
318  break;
319  case Both:
320  root.appendChild(data);
321  root.appendChild(settings);
322  break;
323  }
324 
325  // iterate over settings objects
326  QVector<QVector<UAVDataObject *>> objList = objManager->getDataObjectsVector();
327  foreach (QVector<UAVDataObject *> list, objList) {
328  foreach (UAVDataObject *obj, list) {
329  if (!obj->getIsPresentOnHardware())
330  continue;
331  if (((what == Settings) && obj->isSettings()) || ((what == Data) && !obj->isSettings())
332  || (what == Both)) {
333 
334  // add each object to the XML
335  QDomElement o = doc.createElement("object");
336  o.setAttribute("name", obj->getName());
337  o.setAttribute("id",
338  QString("0x") + QString().setNum(obj->getObjID(), 16).toUpper());
339  if (fullExport) {
340  QDomElement d = doc.createElement("description");
341  QDomText t = doc.createTextNode(
342  obj->getDescription().remove("@Ref ", Qt::CaseInsensitive));
343  d.appendChild(t);
344  o.appendChild(d);
345  }
346 
347  // iterate over fields
348  QList<UAVObjectField *> fieldList = obj->getFields();
349 
350  foreach (UAVObjectField *field, fieldList) {
351  QDomElement f = doc.createElement("field");
352 
353  // iterate over values
354  QString vals;
355  quint32 nelem = field->getNumElements();
356  for (unsigned int n = 0; n < nelem; ++n) {
357  vals.append(QString("%1,").arg(field->getValue(n).toString()));
358  }
359  vals.chop(1);
360 
361  f.setAttribute("name", field->getName());
362  f.setAttribute("values", vals);
363  if (fullExport) {
364  f.setAttribute("type", field->getTypeAsString());
365  f.setAttribute("units", field->getUnits());
366  f.setAttribute("elements", nelem);
367  if (field->getType() == UAVObjectField::ENUM) {
368  f.setAttribute("options", field->getOptions().join(","));
369  }
370  }
371  o.appendChild(f);
372  }
373 
374  // append to the settings or data element
375  if (obj->isSettings())
376  settings.appendChild(o);
377  else
378  data.appendChild(o);
379  }
380  }
381  }
382 
383  // This sorts the XML <object> children's <name=".."> attribute by alphabetical order,
384  // as well as sorting all attributes by alphabetical order (see note below for
385  // special circumstances around sorting attributes. This is particularly helpful when
386  // comparing *.uav files to each other.
387  QString preliminaryXMLDoc = doc.toString(4);
388  QString alphabetizedAttributesXMLDoc;
389  QString alphabetizedXMLDoc;
390 
391  // (New C++11 "raw string literal" string format.)
392  // This sorts all attributes by alphabetical order. This is *not* guaranteed to
393  // work. In fact, it is guaranteed *not* to work in all conditions. See
394  // http://stackoverflow.com/questions/1429991/using-xsl-to-sort-attributes.
395  // However, it seems to work so far and there's no expectation it will change
396  // in Qt's near future.
397  QString xmlAttributeSorter = R"(
398  <xsl:stylesheet version="2.0" xmlns:xsl="http://www.w3.org/1999/XSL/Transform">
399  <xsl:output method="xml" encoding="UTF-8" indent="yes"/>
400  <xsl:template match="*">
401  <xsl:copy>
402  <xsl:apply-templates select="@*">
403  <xsl:sort select="name() "/>
404  </xsl:apply-templates>
405  <xsl:apply-templates/>
406  </xsl:copy>
407  </xsl:template>
408  <xsl:template match="@*|comment()|processing-instruction() ">
409  <xsl:copy />
410  </xsl:template>
411  </xsl:stylesheet>
412  )";
413 
414  // (New C++11 "raw string literal" string format.)
415  // This XSLT template sorts all children by alphabetical order. This is guaranteed
416  // to work (unlike the attribute sorting).
417  QString xmlAlpheticalSorter = R"(
418  <xsl:stylesheet version="2.0" xmlns:xsl="http://www.w3.org/1999/XSL/Transform">
419  <xsl:output method="xml" indent="yes" omit-xml-declaration="no"/>
420  <xsl:strip-space elements="*"/>
421 
422  <xsl:template match="@* | node() ">
423  <xsl:copy>
424  <xsl:apply-templates select="@* | node() "/>
425  </xsl:copy>
426  </xsl:template>
427 
428  <xsl:template match="settings">
429  <xsl:copy>
430  <xsl:apply-templates select="@*" />
431  <xsl:apply-templates select="object">
432  <xsl:sort select="@name" data-type="text"/>
433 
434  </xsl:apply-templates>
435  </xsl:copy>
436  </xsl:template>
437 
438  </xsl:stylesheet>
439  )";
440 
441  // Generate query and set to XSLT v2.0
442  QXmlQuery query(QXmlQuery::XSLT20);
443 
444  // First sort attributes...
445  query.setFocus(preliminaryXMLDoc);
446  query.setQuery(xmlAttributeSorter);
447  query.evaluateTo(&alphabetizedAttributesXMLDoc);
448 
449  // Then sort children.
450  query.setFocus(alphabetizedAttributesXMLDoc);
451  query.setQuery(xmlAlpheticalSorter);
452  query.evaluateTo(&alphabetizedXMLDoc);
453 
454  return alphabetizedXMLDoc;
455 }

462  QString filters = tr("UAVObjects XML files (*.uav)");

468

479

485  QMessageBox::critical(nullptr, tr("UAV Settings Export"),
486  tr("Unable to save settings: ") + fileName, QMessageBox::Ok);

489 
490  QMessageBox msgBox;
491  msgBox.setText(tr("Settings saved."));
492  msgBox.setStandardButtons(QMessageBox::Ok);
