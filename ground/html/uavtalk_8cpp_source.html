198 bool UAVTalk::processInputByte(quint8 rxbyte)
199 {
200  // Update stats
201  stats.rxBytes++;
202 
203  rxPacketLength++; // update packet byte count
204 
205  if (useUDPMirror)
206  rxDataArray.append(rxbyte);
207 
208  // Receive state machine
209  switch (rxState) {
210  case STATE_SYNC:
211 
212  if (rxbyte != SYNC_VAL) {
213  UAVTALK_QXTLOG_DEBUG("UAVTalk: Sync->Sync (" + QString::number(rxbyte) + " "
214  + QString("0x%1").arg(rxbyte, 2, 16) + ")");
215  break;
216  }
217 
218  // Initialize and update CRC
219  rxCS = updateCRC(0, rxbyte);
220 
221  rxPacketLength = 1;
222 
223  if (useUDPMirror) {
224  rxDataArray.clear();
225  rxDataArray.append(rxbyte);
226  }
227 
228  rxState = STATE_TYPE;
229  UAVTALK_QXTLOG_DEBUG("UAVTalk: Sync->Type");
230  break;
231 
232  case STATE_TYPE:
233 
234  // Update CRC
235  rxCS = updateCRC(rxCS, rxbyte);
236 
237  if ((rxbyte & TYPE_MASK) != TYPE_VER) {
238  rxState = STATE_SYNC;
239  UAVTALK_QXTLOG_DEBUG("UAVTalk: Type->Sync");
240  break;
241  }
242 
243  rxType = rxbyte;
244 
245  packetSize = 0;
246 
247  rxState = STATE_SIZE;
248  UAVTALK_QXTLOG_DEBUG("UAVTalk: Type->Size");
249  rxCount = 0;
250  break;
251 
252  case STATE_SIZE:
253 
254  // Update CRC
255  rxCS = updateCRC(rxCS, rxbyte);
256 
257  if (rxCount == 0) {
258  packetSize += rxbyte;
259  rxCount++;
260  UAVTALK_QXTLOG_DEBUG("UAVTalk: Size->Size");
261  break;
262  }
263 
264  packetSize += (quint32)rxbyte << 8;
265 
266  if (packetSize < MIN_HEADER_LENGTH
267  || packetSize > MAX_HEADER_LENGTH + MAX_PAYLOAD_LENGTH) { // incorrect packet size
268  rxState = STATE_SYNC;
269  UAVTALK_QXTLOG_DEBUG("UAVTalk: Size->Sync");
270  break;
271  }
272 
273  rxCount = 0;
274  rxState = STATE_OBJID;
275  UAVTALK_QXTLOG_DEBUG("UAVTalk: Size->ObjID");
276  break;
277 
278  case STATE_OBJID:
279 
280  // Update CRC
281  rxCS = updateCRC(rxCS, rxbyte);
282 
283  rxTmpBuffer[rxCount++] = rxbyte;
284  if (rxCount < 4) {
285  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->ObjID");
286  break;
287  }
288 
289  // Search for object, if not found reset state machine
290  rxObjId = (qint32)qFromLittleEndian<quint32>(rxTmpBuffer);
291  {
292  UAVObject *rxObj = objMngr->getObject(rxObjId);
293  if (rxObj == NULL && rxType != TYPE_OBJ_REQ) {
294  stats.rxErrors++;
295  rxState = STATE_SYNC;
296  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Sync (badtype)");
297  break;
298  } else if (rxObj == NULL) {
299  // This is a non-existing object, just skip to checksum
300  // and we'll send a NACK next.
301  rxState = STATE_CS;
302  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->CSum (no obj)");
303  rxInstId = 0;
304  rxCount = 0;
305  break;
306  }
307 
308  // Determine data length
309  if (rxType == TYPE_OBJ_REQ || rxType == TYPE_ACK || rxType == TYPE_NACK) {
310  rxLength = 0;
311  } else {
312  rxLength = rxObj->getNumBytes();
313  }
314 
315  // Check length and determine next state
316  if (rxLength >= MAX_PAYLOAD_LENGTH) {
317  stats.rxErrors++;
318  rxState = STATE_SYNC;
319  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Sync (oversize)");
320  break;
321  }
322 
323  quint8 rxInstanceLength = (rxObj->isSingleInstance() ? 0 : 2);
324  if ((rxPacketLength + rxInstanceLength + rxLength)
325  != packetSize) { // packet error - mismatched packet size
326  stats.rxErrors++;
327  rxState = STATE_SYNC;
328  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Sync (length mismatch)");
329  break;
330  }
331 
332  if (rxObj->isSingleInstance()) { // Check if this is a single instance object (i.e. if
333  // the instance ID field is coming next)
334  // If there is a payload get it, otherwise receive checksum
335  if (rxLength > 0) {
336  rxState = STATE_DATA;
337  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Data (needs data)");
338  } else {
339  rxState = STATE_CS;
340  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Checksum");
341  }
342  rxInstId = 0;
343  rxCount = 0;
344  } else {
345  rxState = STATE_INSTID;
346  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->InstID");
347  rxCount = 0;
348  }
349  }
350 
351  break;
352 
353  case STATE_INSTID:
354 
355  // Update CRC
356  rxCS = updateCRC(rxCS, rxbyte);
357 
358  rxTmpBuffer[rxCount++] = rxbyte;
359  if (rxCount < 2) {
360  UAVTALK_QXTLOG_DEBUG("UAVTalk: InstID->InstID");
361  break;
362  }
363 
364  rxInstId = (qint16)qFromLittleEndian<quint16>(rxTmpBuffer);
365 
366  rxCount = 0;
367 
368  // If there is a payload get it, otherwise receive checksum
369  if (rxLength > 0) {
370  rxState = STATE_DATA;
371  UAVTALK_QXTLOG_DEBUG("UAVTalk: InstID->Data");
372  } else {
373  rxState = STATE_CS;
374  UAVTALK_QXTLOG_DEBUG("UAVTalk: InstID->CSum");
375  }
376  break;
377 
378  case STATE_DATA:
379 
380  // Update CRC
381  rxCS = updateCRC(rxCS, rxbyte);
382 
383  rxBuffer[rxCount++] = rxbyte;
384  if (rxCount < rxLength) {
385  // UAVTALK_QXTLOG_DEBUG("UAVTalk: Data->Data");
386  break;
387  }
388 
389  rxState = STATE_CS;
390  UAVTALK_QXTLOG_DEBUG("UAVTalk: Data->CSum");
391  rxCount = 0;
392  break;
393 
394  case STATE_CS:
395 
396  // The CRC byte
397  rxCSPacket = rxbyte;
398 
399  if (rxCS != rxCSPacket) { // packet error - faulty CRC
400  stats.rxErrors++;
401  rxState = STATE_SYNC;
402  UAVTALK_QXTLOG_DEBUG("UAVTalk: CSum->Sync (badcrc)");
403  break;
404  }
405 
406  if (rxPacketLength != packetSize + 1) { // packet error - mismatched packet size
407  stats.rxErrors++;
408  rxState = STATE_SYNC;
409  UAVTALK_QXTLOG_DEBUG("UAVTalk: CSum->Sync (length mismatch)");
410  break;
411  }
412 
413  receiveObject(rxType, rxObjId, rxInstId, rxBuffer, rxLength);
414  if (useUDPMirror) {
415  udpSocketTx->writeDatagram(rxDataArray, QHostAddress::LocalHost,
416  udpSocketRx->localPort());
417  }
418  stats.rxObjectBytes += rxLength;
419  stats.rxObjects++;
420 
421  rxState = STATE_SYNC;
422  UAVTALK_QXTLOG_DEBUG("UAVTalk: CSum->Sync (OK)");
423  break;
424 
425  default:
426  rxState = STATE_SYNC;
427  stats.rxErrors++;
428  UAVTALK_QXTLOG_DEBUG("UAVTalk: \?\?\?->Sync"); // Use the escape character for '?' so that
429  // the tripgraph isn't triggered.
430  }
431 
432  // Done
433  return true;
434 }

446 bool UAVTalk::receiveObject(quint8 type, quint32 objId, quint16 instId, quint8 *data, qint32 length)
447 {
448  Q_UNUSED(length);
449  UAVObject *obj = NULL;
450  bool error = false;
451  bool allInstances = (instId == ALL_INSTANCES);
452 
453  // Process message type
454  switch (type) {
455  case TYPE_OBJ: // We have received an object.
456  // All instances, not allowed for OBJ messages
457  if (!allInstances) {
458  // Get object and update its data
459  obj = updateObject(objId, instId, data);
460  if (obj == NULL) {
461  UAVTALK_QXTLOG_DEBUG(
462  QString("[uavtalk.cpp ] Received a UAVObject update for a UAVObject we don't "
463  "know about OBJID:%0 INSTID:%1")
464  .arg(QString(QString("0x") + QString::number(objId, 16).toUpper()))
465  .arg(instId));
466  error = true;
467  }
468  } else {
469  error = true;
470  }
471  break;
472  case TYPE_OBJ_ACK: // We have received an object and are asked for an ACK
473  // All instances, not allowed for OBJ_ACK messages
474  if (!allInstances) {
475  // Get object and update its data
476  obj = updateObject(objId, instId, data);
477  // Transmit ACK
478  if (obj != NULL) {
479  transmitObject(obj, TYPE_ACK, false);
480  } else {
481  UAVTALK_QXTLOG_DEBUG(QString("[uavtalk.cpp ] Received an acknowledged UAVObject "
482  "update for a UAVObject we don't know about:")
483  .arg(obj->getName()));
484  // UAVTALK Protocol update 2013.07.10 (E. Lafargue): send a NACK packet for this
485  // ObjID
486  transmitNack(objId);
487  error = true;
488  }
489  } else {
490  error = true;
491  }
492  break;
493  case TYPE_OBJ_REQ: // We are being asked for an object
494  // Get object, if all instances are requested get instance 0 of the object
495  if (allInstances) {
496  obj = objMngr->getObject(objId);
497  } else {
498  obj = objMngr->getObject(objId, instId);
499  }
500  // If object was found transmit it
501  if (obj != NULL) {
502  transmitObject(obj, TYPE_OBJ, allInstances);
503  } else {
504  // Object was not found, transmit a NACK with the
505  // objId which was not found.
506  transmitNack(objId);
507  error = true;
508  }
509  break;
510  case TYPE_NACK: // We have received a NACK for an object that does not exist on the remote end.
511  // (but should exist on our end)
512  // All instances, not allowed for NACK messages
513  if (!allInstances) {
514  // Get object
515  obj = objMngr->getObject(objId, instId);
516  // Check if object exists:
517  if (obj != NULL) {
518  UAVTALK_QXTLOG_DEBUG(
519  QString("[uavtalk.cpp ] The %0 UAVObject does not exist on the remote end, "
520  "got a Nack")
521  .arg(obj->getName()
522  + QString(QString(" 0x") + QString::number(objId, 16).toUpper())));
523  emit nackReceived(obj);
524  } else {
525  UAVTALK_QXTLOG_DEBUG(
526  QString("[uavtalk.cpp ] Critical error: Received a Nack for an unknown "
527  "UAVObject:%0")
528  .arg(QString(QString("0x") + QString::number(objId, 16).toUpper())));
529  error = true;
530  }
531  }
532  break;
533  case TYPE_ACK: // We have received a ACK, supposedly after sending an object with OBJ_ACK
534  // All instances, not allowed for ACK messages
535  if (!allInstances) {
536  // Get object
537  obj = objMngr->getObject(objId, instId);
538  UAVTALK_QXTLOG_DEBUG(
539  QString("[uavtalk.cpp ] Got ack for instance:%0 of UAVObject:%1 with ID:%2")
540  .arg(instId)
541  .arg(obj->getName())
542  .arg(QString(QString("0x") + QString::number(objId, 16).toUpper())));
543  // Check if we actually know this object (tiny chance the ObjID
544  // could be unknown and got through CRC check...)
545  if (obj != NULL) {
546  UAVTALK_QXTLOG_DEBUG(
547  QString("[uavtalk.cpp ] UAVObject name:%0").arg(obj->getName()));
548  emit ackReceived(obj);
549  } else {
550  error = true;
551  }
552  }
553  break;
554  default:
555  error = true;
556  }
557  // Done (exit value is "success", hence the "!" below)
558  return !error;
559 }

566 UAVObject *UAVTalk::updateObject(quint32 objId, quint16 instId, quint8 *data)
567 {
568  // Get object
569  UAVObject *obj = objMngr->getObject(objId, instId);
570  // If the instance does not exist create it
571  if (obj == NULL) {
572  // Get the object type
573  UAVObject *tobj = objMngr->getObject(objId);
574  if (tobj == NULL) {
575  return NULL;
576  }
577  // Make sure this is a data object
578  UAVDataObject *dobj = dynamic_cast<UAVDataObject *>(tobj);
579  if (dobj == NULL) {
580  return NULL;
581  }
582  // Create a new instance, unpack and register
583  UAVDataObject *instobj = dobj->clone(instId);
584  if (!objMngr->registerObject(instobj)) {
585  return NULL;
586  }
587  instobj->unpack(data);
588  return instobj;
589  } else {
590  // Unpack data into object instance
591  obj->unpack(data);
592  return obj;
593  }

603 bool UAVTalk::transmitObject(UAVObject *obj, quint8 type, bool allInstances)
604 {
605  // If all instances are requested on a single instance object it is an error
606  if (allInstances && obj->isSingleInstance()) {
607  allInstances = false;
608  }
609 
610  // Process message type
611  if (type == TYPE_OBJ || type == TYPE_OBJ_ACK) {
612  if (allInstances) {
613  // Get number of instances
614  quint32 numInst = objMngr->getNumInstances(obj->getObjID());
615  // Send all instances
616  for (quint32 instId = 0; instId < numInst; ++instId) {
617  UAVObject *inst = objMngr->getObject(obj->getObjID(), instId);
618  transmitSingleObject(inst, type, false);
619  }
620  return true;
621  } else {
622  return transmitSingleObject(obj, type, false);
623  }
624  } else if (type == TYPE_OBJ_REQ) {
625  return transmitSingleObject(obj, TYPE_OBJ_REQ, allInstances);
626  } else if (type == TYPE_ACK) {
627  if (!allInstances) {
628  return transmitSingleObject(obj, TYPE_ACK, false);
629  } else {
630  return false;
631  }
632  } else {
633  return false;
634  }
635 }

643 bool UAVTalk::transmitNack(quint32 objId)
644 {
645  int dataOffset = 8;
646 
647  txBuffer[0] = SYNC_VAL;
648  txBuffer[1] = TYPE_NACK;
649  qToLittleEndian<quint32>(objId, &txBuffer[4]);
650 
651  // Calculate checksum
652  txBuffer[dataOffset] = updateCRC(0, txBuffer, dataOffset);
653 
654  qToLittleEndian<quint16>(dataOffset, &txBuffer[2]);
655 
656  // Send buffer, check that the transmit backlog does not grow above limit
657  if (io && io->isWritable() && io->bytesToWrite() < TX_BUFFER_SIZE) {
658  io->write((const char *)txBuffer, dataOffset + CHECKSUM_LENGTH);
659  if (useUDPMirror) {
660  udpSocketRx->writeDatagram((const char *)txBuffer, dataOffset + CHECKSUM_LENGTH,
661  QHostAddress::LocalHost, udpSocketTx->localPort());
662  }
663  } else {
664  ++stats.txErrors;
665  return false;
666  }
667 
668  // Update stats
669  stats.txBytes += 8 + CHECKSUM_LENGTH;
670 
671  // Done
672  return true;
673 }

681 bool UAVTalk::transmitSingleObject(UAVObject *obj, quint8 type, bool allInstances)
682 {
683  qint32 length;
684  qint32 dataOffset;
685  quint32 objId;
686  quint16 instId;
687  quint16 allInstId = ALL_INSTANCES;
688 
689  // Setup type and object id fields
690  objId = obj->getObjID();
691  txBuffer[0] = SYNC_VAL;
692  txBuffer[1] = type;
693  qToLittleEndian<quint32>(objId, &txBuffer[4]);
694 
695  // Setup instance ID if one is required
696  if (obj->isSingleInstance()) {
697  dataOffset = 8;
698  } else {
