276 void ConfigGadgetWidget::onAutopilotConnect()
277 {
278 
279  QIcon *icon;
280  QWidget *qwd;
281 
282  bool hasOSD = false;
283 
284  int index = ftw->currentIndex();
285 
286  if (index != ConfigGadgetWidget::hardware) {
287  lastTabIndex = index;
288  }
289 
290  // First of all, check what Board type we are talking to, and
291  // if necessary, remove/add tabs in the config gadget:
292  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
293  UAVObjectUtilManager *utilMngr = pm->getObject<UAVObjectUtilManager>();
294  if (utilMngr) {
295  ftw->setCurrentIndex(ConfigGadgetWidget::hardware);
296  ftw->removeTab(ConfigGadgetWidget::hardware);
297 
298  // If the board provides a custom configuration widget then use it,
299  // otherwise use the default which populates the fields from the
300  // hardware UAVO
301  bool valid_board = false;
302  Core::IBoardType *board = utilMngr->getBoardType();
303  if (board) {
304  qwd = board->getBoardConfiguration();
305  if (qwd) {
306  valid_board = true;
307  } else {
308  UAVObject *settingsObj =
309  utilMngr->getObjectManager()->getObject(board->getHwUAVO());
310  if (settingsObj) {
311  qwd = new DefaultHwSettingsWidget(settingsObj, this);
312  valid_board = true;
313  }
314  }
315 
316  if (board->queryCapabilities(Core::IBoardType::BOARD_CAPABILITIES_OSD)) {
317  hasOSD = true;
318  } else if (lastTabIndex == ConfigGadgetWidget::osd) {
319  lastTabIndex = ConfigGadgetWidget::hardware;
320  }
321  }
322 
323  if (!valid_board) {
324  QLabel *txt = new QLabel(tr("<p>Board detected, but of unknown type. This could be "
325  "because either your GCS or firmware is out of date.</p>"),
326  this);
327  txt->setWordWrap(true);
328  qwd = txt;
329  }
330 
331  icon = new QIcon();
332  icon->addFile(":/configgadget/images/hardware_normal.png", QSize(), QIcon::Normal,
333  QIcon::Off);
334  icon->addFile(":/configgadget/images/hardware_selected.png", QSize(), QIcon::Selected,
335  QIcon::Off);
336  ftw->insertTab(ConfigGadgetWidget::hardware, qwd, *icon, QString(tr("Hardware")));
337  ftw->setCurrentIndex(ConfigGadgetWidget::hardware);
338  }
339 
341  ftw->removeTab(ConfigGadgetWidget::sensors);
342  icon = new QIcon();
343  icon->addFile(":/configgadget/images/ins_normal.png", QSize(), QIcon::Normal, QIcon::Off);
344  icon->addFile(":/configgadget/images/ins_selected.png", QSize(), QIcon::Selected, QIcon::Off);
345  qwd = new ConfigAttitudeWidget(this);
346  ftw->insertTab(ConfigGadgetWidget::sensors, qwd, *icon, QString("Attitude"));
347 
348  // Hide OSD if not applicable, else show
349  ftw->setHidden(ConfigGadgetWidget::osd, !hasOSD);
350 
351  ftw->setCurrentIndex(lastTabIndex);
352 
353  emit autopilotConnected();
354 }

375  tr("The tab you are leaving has unsaved changes,"
376  "if you proceed they may be lost."

378  QMessageBox::Yes, QMessageBox::No);
379  if (ans == QMessageBox::No) {
380  *proceed = false;
381  } else {
382  wid->setDirty(false);
383  }
