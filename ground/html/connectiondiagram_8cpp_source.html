194 void ConnectionDiagram::addUavoFieldElements(QStringList &elements, UAVObject *obj,
195  const QString &prefix)
196 {
197  if (!obj) {
198  Q_ASSERT(false);
199  qWarning() << "Invalid object!";
200  return;
201  }
202 
203  for (const auto field : obj->getFields()) {
204  if (!field) {
205  Q_ASSERT(false);
206  qWarning() << "Invalid object field!";
207  continue;
208  }
209  if (field->getType() != UAVObjectField::ENUM)
210  continue;
211 
212  quint32 nelements = field->getNumElements();
213  if (nelements > 1) {
214  for (quint32 i = 0; i < nelements; i++) {
215  QString val = field->getValue(i)
216  .toString()
217  .replace(QRegExp(ENUM_SPECIAL_CHARS), "")
218  .toLower();
219  elements << QString("%0%1-%2-%3-%4")
220  .arg(prefix)
221  .arg(obj->getName().toLower())
222  .arg(field->getName().toLower())
223  .arg(field->getElementName(i)
224  .replace(QRegExp(ENUM_SPECIAL_CHARS), "")
225  .toLower())
226  .arg(val);
227  }
228  } else {
229  QString val =
230  field->getValue().toString().replace(QRegExp(ENUM_SPECIAL_CHARS), "").toLower();
231  elements << QString("%0%1-%2-%3")
232  .arg(prefix)
233  .arg(obj->getName().toLower())
234  .arg(field->getName().toLower())
235  .arg(val);
236  }
237  }
238 }
