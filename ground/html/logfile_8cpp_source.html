179 
180  if(file.bytesAvailable() > 4)
181  {
182 
183  int time;
184  time = myTime.elapsed();
185 
186  //Read packets
187  while ((lastPlayTime + ((time - lastPlayTimeOffset)* playbackSpeed) > (lastTimeStamp-firstTimestamp)))
188  {
189  lastPlayTime += ((time - lastPlayTimeOffset)* playbackSpeed);
190  if(file.bytesAvailable() < 4) {
191  stopReplay();
192  return;
193  }
194 
195  file.seek(lastTimeStampPos+sizeof(lastTimeStamp));
196 
197  file.read((char *) &dataSize, sizeof(dataSize));
198 
199  if (dataSize<1 || dataSize>(1024*1024)) {
200  qDebug() << "Error: Logfile corrupted! Unlikely packet size: " << dataSize << "\n";
201  stopReplay();
202  return;
203  }
204  if(file.bytesAvailable() < dataSize) {
205  stopReplay();
206  return;
207  }
208

216  return;

233 bool LogFile::startReplay() {
234  dataBuffer.clear();
235  myTime.restart();
236  lastPlayTimeOffset = 0;
237  lastPlayTime = 0;
238  playbackSpeed = 1;
239 
240  //Read all log timestamps into array
241  timestampBuffer.clear(); //Save beginning of log for later use
242  timestampPos.clear();
243  quint64 logFileStartIdx = file.pos();
244  timestampBufferIdx = 0;
245  lastTimeStamp = 0;
246 
247  while (!file.atEnd()){
248  qint64 dataSize;
249 
250  //Get time stamp position
251  timestampPos.append(file.pos());
252 
253  //Read timestamp and logfile packet size
254  file.read((char *) &lastTimeStamp, sizeof(lastTimeStamp));
255  file.read((char *) &dataSize, sizeof(dataSize));
256 
257  //Check if dataSize sync bytes are correct.
258  //TODO: LIKELY AS NOT, THIS WILL FAIL TO RESYNC BECAUSE THERE IS TOO LITTLE INFORMATION IN THE STRING OF SIX 0x00
259  if ((dataSize & 0xFFFFFFFFFFFF0000)!=0){
260  qDebug() << "Wrong sync byte. At file location 0x" << QString("%1").arg(file.pos(),0,16) << "Got 0x" << QString("%1").arg(dataSize & 0xFFFFFFFFFFFF0000,0,16) << ", but expected 0x""00"".";
261  file.seek(timestampPos.last()+1);
262  timestampPos.pop_back();
263  continue;
264  }
265 
266  //Check if timestamps are sequential.
267  if (!timestampBuffer.isEmpty() && lastTimeStamp < timestampBuffer.last()){
268  QMessageBox msgBox;
269  msgBox.setText("Corrupted file.");
270  msgBox.setInformativeText("Timestamps are not sequential. Playback may have unexpected behavior"); //<--TODO: add hyperlink to webpage with better description.
271  msgBox.exec();
272 
273  qDebug() << "Timestamp: " << timestampBuffer.last() << " " << lastTimeStamp;
274  }
275 
276  timestampBuffer.append(lastTimeStamp);
277 
278  file.seek(timestampPos.last()+sizeof(lastTimeStamp)+sizeof(dataSize)+dataSize);
279  }
280 
281  //Check if any timestamps were successfully read
282  if (timestampBuffer.size() == 0){
283  QMessageBox msgBox;
284  msgBox.setText("Empty logfile.");
285  msgBox.setInformativeText("No log data can be found.");
286  msgBox.exec();
287 
288  stopReplay();
289  return false;
290  }
291 
292  //Reset to log beginning.
293  file.seek(logFileStartIdx+sizeof(lastTimeStamp));
294  lastTimeStampPos = timestampPos[0];
295  lastTimeStamp = timestampBuffer[0];
296  firstTimestamp = timestampBuffer[0];
297  timestampBufferIdx = 1;
298 
299  timer.setInterval(10);
300  timer.start();
301  emit replayStarted();
302  return true;
303 }
