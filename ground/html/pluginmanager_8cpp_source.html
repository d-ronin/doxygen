345  const QString &opt, const QString &parm, const QString &description,
346  int optionIndentation, int descriptionIndentation)
347 {
348  int remainingIndent = descriptionIndentation - optionIndentation - opt.size();
349  indent(str, optionIndentation);

590 bool PluginManagerPrivate::loadQueue(PluginSpec *spec, QList<PluginSpec *> &queue,
591  QList<PluginSpec *> &circularityCheckQueue)
592 {
593  if (queue.contains(spec))
594  return true;
595  // check for circular dependencies
596  if (circularityCheckQueue.contains(spec)) {
597  spec->d->hasError = true;
598  spec->d->errorString = PluginManager::tr("Circular dependency detected:\n");
599  int index = circularityCheckQueue.indexOf(spec);
600  for (int i = index; i < circularityCheckQueue.size(); ++i) {
601  spec->d->errorString.append(PluginManager::tr("%1(%2) depends on\n")
602  .arg(circularityCheckQueue.at(i)->name()).arg(circularityCheckQueue.at(i)->version()));
603  }
604  spec->d->errorString.append(PluginManager::tr("%1(%2)").arg(spec->name()).arg(spec->version()));
605  return false;
606  }
607  circularityCheckQueue.append(spec);
608  // check if we have the dependencies
609  if (spec->state() == PluginSpec::Invalid || spec->state() == PluginSpec::Read) {
610  spec->d->hasError = true;
611  spec->d->errorString += "\n";
612  spec->d->errorString += PluginManager::tr("Cannot load plugin because dependencies are not resolved");
613  return false;
614  }
615  // add dependencies
616  foreach (PluginSpec *depSpec, spec->dependencySpecs()) {
617  if (!loadQueue(depSpec, queue, circularityCheckQueue)) {
618  spec->d->hasError = true;
619  spec->d->errorString =
620  PluginManager::tr("Cannot load plugin because dependency failed to load: %1(%2)\nReason: %3")
621  .arg(depSpec->name()).arg(depSpec->version()).arg(depSpec->errorString());
622  return false;
623  }
624  }
625  // add self
626  queue.append(spec);
627  return true;
628 }

634 void PluginManagerPrivate::loadPlugin(PluginSpec *spec, PluginSpec::State destState)
635 {
636  if (spec->hasError())
637  return;
638  if (destState == PluginSpec::Running) {
639  spec->d->initializeExtensions();
640  return;
641  } else if (destState == PluginSpec::Deleted) {
642  spec->d->kill();
643  return;
644  }
645  foreach (PluginSpec *depSpec, spec->dependencySpecs()) {
646  if (depSpec->state() != destState) {
647  spec->d->hasError = true;
648  spec->d->errorString =
649  PluginManager::tr("Cannot load plugin because dependency failed to load: %1(%2)\nReason: %3")
650  .arg(depSpec->name()).arg(depSpec->version()).arg(depSpec->errorString());
651  return;
652  }
653  }
654  if (destState == PluginSpec::Loaded)
655  spec->d->loadLibrary();
656  else if (destState == PluginSpec::Initialized)
657  spec->d->initializePlugin();
658  else if (destState == PluginSpec::Stopped)
659  spec->d->stop();
660 }

676 void PluginManagerPrivate::readPluginPaths()
677 {
678  qDeleteAll(pluginSpecs);
679  pluginSpecs.clear();
680 
681  QStringList specFiles;
682  QStringList searchPaths = pluginPaths;
683  while (!searchPaths.isEmpty()) {
684  const QDir dir(searchPaths.takeFirst());
685  const QFileInfoList files = dir.entryInfoList(QStringList() << QString("*.%1").arg(extension), QDir::Files);
686  foreach (const QFileInfo &file, files)
687  specFiles << file.absoluteFilePath();
688  const QFileInfoList dirs = dir.entryInfoList(QDir::Dirs|QDir::NoDotAndDotDot);
689  foreach (const QFileInfo &subdir, dirs)
690  searchPaths << subdir.absoluteFilePath();
691  }
692  foreach (const QString &specFile, specFiles) {
693  PluginSpec *spec = new PluginSpec;
694  spec->d->read(specFile);
695  pluginSpecs.append(spec);
696  }
697  resolveDependencies();
698  // ensure deterministic plugin load order by sorting
699  qSort(pluginSpecs.begin(), pluginSpecs.end(), lessThanByPluginName);
700  emit q->pluginsChanged();
701 }
