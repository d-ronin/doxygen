56 bool optionsPageLessThan(const IOptionsPage *p1, const IOptionsPage *p2)
57 {
58  const UAVGadgetOptionsPageDecorator *gp1 = qobject_cast<const UAVGadgetOptionsPageDecorator*>(p1);
59  const UAVGadgetOptionsPageDecorator *gp2 = qobject_cast<const UAVGadgetOptionsPageDecorator*>(p2);
60  if (gp1 && (gp2 == NULL))
61  return false;
62 
63  if (gp2 && (gp1 == NULL))
64  return true;
65 
66  if (const int cc = QString::localeAwareCompare(p1->trCategory(), p2->trCategory()))
67  return cc < 0;
68 
69  return QString::localeAwareCompare(p1->trName(), p2->trName()) < 0;
70 }
