85 void SystemHealthGadgetWidget::updateAlarms(UAVObject* systemAlarm)
86 {
87  static QList<QString> warningClean;
88  // This code does not know anything about alarms beforehand, and
89  // I found no efficient way to locate items inside the scene by
90  // name, so it's just as simple to reset the scene:
91  // And add the one with the right name.
92  QGraphicsScene *m_scene = scene();
93  foreach ( QGraphicsItem* item ,background->childItems()){
94  m_scene->removeItem(item);
95  delete item; // removeItem does _not_ delete the item.
96  }
97 
98  UAVObjectField *field = systemAlarm->getField("Alarm");
99  Q_ASSERT(field);
100  if (field == NULL)
101  return;
102 
103  for (uint i = 0; i < field->getNumElements(); ++i) {
104  QString element = field->getElementNames()[i];
105  QString value = field->getValue(i).toString();
106  if (m_renderer->elementExists(element)) {
107  QMatrix blockMatrix = m_renderer->matrixForElement(element);
108  qreal startX = blockMatrix.mapRect(m_renderer->boundsOnElement(element)).x();
109  qreal startY = blockMatrix.mapRect(m_renderer->boundsOnElement(element)).y();
110  QString element2 = element + "-" + value;
111  if (m_renderer->elementExists(element2)) {
112  QGraphicsSvgItem *ind = new QGraphicsSvgItem();
113  ind->setSharedRenderer(m_renderer);
114  ind->setElementId(element2);
115  ind->setParentItem(background);
116  QTransform matrix;
117  matrix.translate(startX,startY);
118  ind->setTransform(matrix,false);
119  } else {
120  if ((value.compare("Uninitialised") != 0) && !warningClean.contains(element2))
121  {
122  qDebug() << "[SystemHealth] Warning: The SystemHealth SVG does not contain a graphical element for the " << element2 << " alarm.";
123  warningClean.append(element2);
124  }
125  }
126  } else if(!warningClean.contains(element)){
127  qDebug() << "[SystemHealth] Warning: The SystemHealth SVG does not contain a graphical element for the " << element << " alarm.";
128  warningClean.append(element);
129  }
130  }
131 }

139 void SystemHealthGadgetWidget::setSystemFile(QString dfn)
140 {
141  setBackgroundBrush(QBrush(Utils::StyleHelper::baseColor()));
142  if (QFile::exists(dfn)) {
143  m_renderer->load(dfn);
144  if(m_renderer->isValid()) {
145  fgenabled = false;
146  background->setSharedRenderer(m_renderer);
147  background->setElementId("background");
148 
149  if (m_renderer->elementExists("foreground")) {
150  foreground->setSharedRenderer(m_renderer);
151  foreground->setElementId("foreground");
152  foreground->setZValue(99);
153  fgenabled = true;
154  }

156  nolink->setSharedRenderer(m_renderer);
157  nolink->setElementId("nolink");
158  nolink->setZValue(100);
159  }
160 
161  QGraphicsScene *l_scene = scene();
162  l_scene->setSceneRect(background->boundingRect());
163  fitInView(background, Qt::KeepAspectRatio );
164 
165  // Check whether the autopilot is connected already, by the way:
166  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
167  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
168  TelemetryManager* telMngr = pm->getObject<TelemetryManager>();
169  if (telMngr->isConnected()) {
170  onAutopilotConnect();
171  SystemAlarms* obj = SystemAlarms::GetInstance(objManager);
172  updateAlarms(obj);
173  }
174  }
175  }
176  else
177  { qDebug() <<"SystemHealthGadget: no file"; }
178 }

209 void SystemHealthGadgetWidget::mousePressEvent ( QMouseEvent * event )
210 {
211  QGraphicsScene *graphicsScene = scene();
212  if(graphicsScene){
213  QPoint point = event->pos();
214  bool haveAlarmItem = false;
215  foreach(QGraphicsItem* sceneItem, items(point)){
216  QGraphicsSvgItem *clickedItem = dynamic_cast<QGraphicsSvgItem*>(sceneItem);
217 
218  if(clickedItem){
219  if((clickedItem != foreground) && (clickedItem != background)){
220  // Clicked an actual alarm. We need to set haveAlarmItem to true
221  // as two of the items in this loop will always be foreground and
222  // background. Without this flag, at some point in the loop we
223  // would always call showAllAlarmDescriptions...
224  haveAlarmItem = true;
225  QString itemId = clickedItem->elementId();
226  if(itemId.contains("OK")){
227  // No alarm set for this item
228  showAlarmDescriptionForItemId("AlarmOK", event->globalPos());
229  }else{
230  // Warning, error or critical alarm
231  showAlarmDescriptionForItemId(itemId, event->globalPos());
232  }
233  }
234  }
235  }
236  if(!haveAlarmItem){
237  // Clicked foreground or background
238  showAllAlarmDescriptions(event->globalPos());
239  }
240  }
241 }

251 void SystemHealthGadgetWidget::showAllAlarmDescriptions(const QPoint& location){
252  QGraphicsScene *graphicsScene = scene();
253  if(graphicsScene){
254  QString alarmsText;
255  // Loop through all items in the scene looking for svg items that represent alarms
256  foreach(QGraphicsItem* curItem, graphicsScene->items()){
257  QGraphicsSvgItem* curSvgItem = dynamic_cast<QGraphicsSvgItem*>(curItem);
258  if(curSvgItem && (curSvgItem != foreground) && (curSvgItem != background)){
259  QString elementId = curSvgItem->elementId();
260  if(!elementId.contains("OK")){
261  // Found an alarm, get its corresponding alarm html file contents
262  // and append to the cumulative string for all alarms.
263  QFile alarmDescription(getAlarmDescriptionFileName(elementId));
264  if(alarmDescription.open(QIODevice::ReadOnly | QIODevice::Text)){
265  QTextStream textStream(&alarmDescription);
266  alarmsText.append(textStream.readAll());
267  alarmDescription.close();
268  }
269  }
270  }
271  }
272  // Show alarms text if we have any
273  if(alarmsText.length() > 0){
274  QWhatsThis::showText(location, alarmsText);
275  }
276  }
277 }
