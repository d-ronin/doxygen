143 void FGSimulator::transmitUpdate()
144 {
145  ActuatorDesired::DataFields actData;
146  FlightStatus::DataFields flightStatusData = flightStatus->getData();
147  float ailerons = -1;
148  float elevator = -1;
149  float rudder = -1;
150  float throttle = -1;
151 
152  // Read ActuatorDesired from autopilot
153  actData = actDesired->getData();
154 
155  // if we're in any mode other than manual, or if we're in manual and we're armed, update outputs from ActuatorDesired
156  if(flightStatus->getFlightMode() != FlightStatus::FLIGHTMODE_MANUAL || flightStatusData.Armed == FlightStatus::ARMED_ARMED)
157  {
158  ailerons = actData.Roll;
159  elevator = -actData.Pitch;
160  rudder = actData.Yaw;
161  throttle = actData.Thrust;
162  }
163 
164  int allowableDifference = 10;
165 
166  //qDebug() << "UDP sent:" << udpCounterGCSsend << " - UDP Received:" << udpCounterFGrecv;
167 
168  if(udpCounterFGrecv == udpCounterGCSsend)
169  udpCounterGCSsend = 0;
170 
171  if((udpCounterGCSsend < allowableDifference) || (udpCounterFGrecv==0) ) //FG udp queue is not delayed
172  {
173  udpCounterGCSsend++;
174 
175  // Send update to FlightGear
176  QString cmd;
177  cmd = QString("%1,%2,%3,%4,%5\n")
178  .arg(ailerons) //ailerons
179  .arg(elevator) //elevator
180  .arg(rudder) //rudder
181  .arg(throttle) //throttle
182  .arg(udpCounterGCSsend); //UDP packet counter delay
183 
184  QByteArray data = cmd.toLatin1();
185 
186  if(outSocket->writeDatagram(data, QHostAddress(settings.remoteAddress), settings.outPort) == -1)
187  {
188  emit processOutput("Error sending UDP packet to FG: " + outSocket->errorString() + "\n");
189  }
190  }
191  else
192  {
193  // don't send new packet. Flightgear cannot process UDP fast enough.
194  // V1.9.1 reads udp packets at set frequency and will get delayed if packets are sent too fast
195  // V2.0 does not currently work with --generic-protocol
196  }
197 
198  if(settings.manualControlEnabled)
199  {
200  actData.Roll = ailerons;
201  actData.Pitch = -elevator;
202  actData.Yaw = rudder;
203  actData.Thrust = throttle;
204  actDesired->setData(actData);
205  }
206 }
