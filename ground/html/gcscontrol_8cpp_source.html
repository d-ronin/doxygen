98 bool GCSControl::beginGCSControl()
99 {
100  if (hasControl)
101  return false;
102  dataBackup = manControlSettingsUAVO->getData();
103  metaBackup = manControlSettingsUAVO->getMetadata();
104  ManualControlSettings::Metadata meta = manControlSettingsUAVO->getDefaultMetadata();
105  UAVObject::SetGcsAccess(meta, UAVObject::ACCESS_READWRITE);
106 
107  // No need to set flight mode, we leave that at none and directly change
108  // the setting
109  manControlSettingsUAVO->setChannelGroups(ManualControlSettings::CHANNELGROUPS_FLIGHTMODE,
110  ManualControlSettings::CHANNELGROUPS_NONE);
111 
112  // Only throttle, roll, pitch and yaw need to be configured for this
113  // mode to work
114  const int NUM_CHANNELS = 4;
115  const quint8 channels[NUM_CHANNELS] = { ManualControlSettings::CHANNELGROUPS_THROTTLE,
116  ManualControlSettings::CHANNELGROUPS_ROLL,
117  ManualControlSettings::CHANNELGROUPS_PITCH,
118  ManualControlSettings::CHANNELGROUPS_YAW };
119 
120  for (quint8 i = 0; i < NUM_CHANNELS; ++i) {
121  quint8 x = channels[i];
122 
123  // Assign this channel to GCS control
124  manControlSettingsUAVO->setChannelGroups(x, ManualControlSettings::CHANNELGROUPS_GCS);
125 
126  // Set the ranges to match what the widget produces
127  manControlSettingsUAVO->setChannelNumber(x, x + 1);
128  manControlSettingsUAVO->setChannelMax(x, CHANNEL_MAX);
129 
130  if (x == ManualControlSettings::CHANNELGROUPS_THROTTLE) {
131  manControlSettingsUAVO->setChannelNeutral(x, CHANNEL_THROTNEUTRAL);
132  } else {
133  manControlSettingsUAVO->setChannelNeutral(x, CHANNEL_NEUTRAL);
134  }
135 
136  manControlSettingsUAVO->setChannelMin(x, CHANNEL_MIN);
137  }
138 
139  manControlSettingsUAVO->setDeadband(0);
140  manControlSettingsUAVO->setFlightModeNumber(1);
141  manControlSettingsUAVO->setFlightModePosition(
142  0, ManualControlSettings::FLIGHTMODEPOSITION_STABILIZED1);
143  manControlSettingsUAVO->updated();
144  connect(manControlSettingsUAVO, &UAVObject::objectUpdated, this, &GCSControl::objectsUpdated);
145  hasControl = true;
146  for (quint8 x = 0; x < GCSReceiver::CHANNEL_NUMELEM; ++x)
147  setChannel(x, 0);
148  receiverActivity.start();
149  return true;
150 }

200 bool GCSControl::setChannel(quint8 channel, float value)
201 {
202  if (value > 1 || value < -1 || channel >= GCSReceiver::CHANNEL_NUMELEM || !hasControl)
203  return false;
204  quint16 pwmValue;
205  if (value >= 0)
206  pwmValue = (value * (float)(CHANNEL_MAX - CHANNEL_NEUTRAL)) + (float)CHANNEL_NEUTRAL;
207  else
208  pwmValue = (value * (float)(CHANNEL_NEUTRAL - CHANNEL_MIN)) + (float)CHANNEL_NEUTRAL;
209  m_gcsReceiver->setChannel(channel, pwmValue);
210  m_gcsReceiver->updated();
211  return true;
212 }
