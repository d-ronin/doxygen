214 void SpectrogramScopeConfig::loadConfiguration(ScopeGadgetWidget *scopeGadgetWidget)
215 {
216  preparePlot(scopeGadgetWidget);
217  scopeGadgetWidget->setScope(this);
218  scopeGadgetWidget->startTimer(m_refreshInterval);
219 
220  // There should be only one spectrogram per plot //TODO: Upgrade this to handle multiple
221  // spectrograms on a single axis
222  if (m_spectrogramSourceConfigs.length() != 1)
223  return;
224 
225  Plot3dCurveConfiguration *spectrogramSourceConfigs = m_spectrogramSourceConfigs.front();
226  QString uavObjectName = spectrogramSourceConfigs->uavObjectName;
227  QString uavFieldName = spectrogramSourceConfigs->uavFieldName;
228 
229  // Get and store the units
230  units = getUavObjectFieldUnits(uavObjectName, uavFieldName);
231 
232  SpectrogramData *spectrogramData = new SpectrogramData(
233  uavObjectName, uavFieldName, samplingFrequency, windowWidth, timeHorizon);
234  spectrogramData->setXMinimum(0);
235  spectrogramData->setXMaximum(samplingFrequency / 2);
236  spectrogramData->setYMinimum(0);
237  spectrogramData->setYMaximum(timeHorizon);
238  spectrogramData->setZMaximum(zMaximum);
239  spectrogramData->setScalePower(spectrogramSourceConfigs->yScalePower);
240  spectrogramData->setMeanSamples(spectrogramSourceConfigs->yMeanSamples);
241  spectrogramData->setMathFunction(spectrogramSourceConfigs->mathFunction);
242 
243  // Generate the waterfall name
244  QString waterfallName =
245  (spectrogramData->getUavoName()) + "." + (spectrogramData->getUavoFieldName());
246  if (spectrogramData->getHaveSubFieldFlag())
247  waterfallName = waterfallName.append("." + spectrogramData->getUavoSubFieldName());
248 
249  // Get the uav object
250  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
251  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
252  UAVDataObject *obj =
253  dynamic_cast<UAVDataObject *>(objManager->getObject((spectrogramData->getUavoName())));
254  if (!obj) {
255  qDebug() << "Object " << spectrogramData->getUavoName() << " is missing";
256  return;
257  }
258 
259  // Get the units
260  QString units =
261  getUavObjectFieldUnits(spectrogramData->getUavoName(), spectrogramData->getUavoFieldName());
262 
263  // Generate name with scaling factor appeneded
264  QString waterfallNameScaled;
265  if (spectrogramSourceConfigs->yScalePower == 0)
266  waterfallNameScaled = waterfallName + "(" + units + ")";
267  else
268  waterfallNameScaled = waterfallName + "(x10^"
269  + QString::number(spectrogramSourceConfigs->yScalePower) + " " + units + ")";
270 
271  // Create the waterfall plot
272  QwtPlotSpectrogram *plotSpectrogram = new QwtPlotSpectrogram(waterfallNameScaled);
273  plotSpectrogram->setRenderThreadCount(0); // use system specific thread count
274  plotSpectrogram->setRenderHint(QwtPlotItem::RenderAntialiased);
275  plotSpectrogram->setColorMap(new ColorMap(colorMapType));
276 
277  // Initial raster data
278 
279  QDateTime NOW =
280  QDateTime::currentDateTime(); // TODO: Upgrade this to show UAVO time and not system time
281  for (uint i = 0; i < timeHorizon; i++) {
282  spectrogramData->timeDataHistory->append(NOW.toTime_t() + NOW.time().msec() / 1000.0 + i);
283  }
284 
285  if (((double)windowWidth) * timeHorizon < (double)10000000.0
286  * sizeof(spectrogramData->zDataHistory->front())) { // Don't exceed 10MB for memory
287  for (uint i = 0; i < windowWidth * timeHorizon; i++) {
288  spectrogramData->zDataHistory->append(0);
289  }
290  } else {
291  qDebug() << "For some reason, we're trying to allocate a gigantic spectrogram. This "
292  "probably represents a problem in the configuration file. TimeHorizion: "
293  << timeHorizon << ", windowWidth: " << windowWidth;
294  Q_ASSERT(0);
295  return;
296  }
297 
298  // Set up colorbar on right axis
299  spectrogramData->rightAxis = scopeGadgetWidget->axisWidget(QwtPlot::yRight);
300  spectrogramData->rightAxis->setTitle("Intensity");
301  spectrogramData->rightAxis->setColorBarEnabled(true);
302  spectrogramData->rightAxis->setColorMap(QwtInterval(0, zMaximum), new ColorMap(colorMapType));
303  scopeGadgetWidget->setAxisScale(QwtPlot::yRight, 0, zMaximum);
304  scopeGadgetWidget->enableAxis(QwtPlot::yRight);
305 
306  plotSpectrogram->setData(spectrogramData->getRasterData());
307 
308  plotSpectrogram->attach(scopeGadgetWidget);
309  spectrogramData->setSpectrogram(plotSpectrogram);
310 
311  // Keep the curve details for later
312  scopeGadgetWidget->insertDataSources(waterfallNameScaled, spectrogramData);
313 
314  // Connect the UAVO
315  scopeGadgetWidget->connectUAVO(obj);
316 
317  scopeGadgetWidget->replot();
318 }
