204 void XplaneSimulator::processUpdate(const QByteArray& dataBuf)
205 {
206  float altitude_msl = 0;
207  float altitude_agl = 0;
208  float latitude = 0;
209  float longitude = 0;
210  float airspeed_keas = 0;
211  float airspeed_ktas = 0;
212  float groundspeed_ktgs = 0;
213  float alpha_D=0;
214  float beta_D=0;
215  float pitch = 0;
216  float roll = 0;
217  float heading = 0;
218  float pressure = 0;
219  float temperature = 0;
220  float velX = 0;
221  float velY = 0;
222  float velZ = 0;
223  float dstX = 0;
224  float dstY = 0;
225  float dstZ = 0;
226  float accX = 0;
227  float accY = 0;
228  float accZ = 0;
229  float rollRate_rad=0;
230  float pitchRate_rad=0;
231  float yawRate_rad=0;
232 
233  // QString str;
234  QByteArray& buf = const_cast<QByteArray&>(dataBuf);
235  QString data(buf);
236 
237  if(data.left(4) == "DATA") // check type of packet
238  {
239  buf.remove(0,5);
240  if(dataBuf.size() % 36)
241  {
242  qDebug() << "XPlaneSimulator: incorrect length of UDP packet: " << buf;
243  return; // incorrect length of struct
244  }
245  // check correctness of data length, length must be multiple of (id_size+8*float_size)=4+8*4=36
246  int channelCounter = dataBuf.size() / 36;
247  do
248  {
249  switch(buf[0]) // switch by id
250  {
251  case XplaneSimulator::LatitudeLongitudeAltitude:
252  latitude = *((float*)(buf.data()+4*1));
253  longitude = *((float*)(buf.data()+4*2));
254  altitude_msl = *((float*)(buf.data()+4*3)) * FEET2MILES;
255  altitude_agl = *((float*)(buf.data()+4*4)) * FEET2MILES;
256  break;
257 
258  case XplaneSimulator::Speed:
259  airspeed_keas = *((float*)(buf.data()+4*2));
260  airspeed_ktas = *((float*)(buf.data()+4*3));
261  groundspeed_ktgs = *((float*)(buf.data()+4*4));
262  break;
263 
264  case XplaneSimulator::AoA:
265  alpha_D = *((float*)(buf.data()+4*1));
266  beta_D = *((float*)(buf.data()+4*2));
267  break;
268 
269  case XplaneSimulator::PitchRollHeading:
270  pitch = *((float*)(buf.data()+4*1));
271  roll = *((float*)(buf.data()+4*2));
272  heading = *((float*)(buf.data()+4*3));
273  break;
274 
275  /*
276  case XplaneSimulator::SystemPressures:
277  pressure = *((float*)(buf.data()+4*1));
278  break;
279  */
280 
281  case XplaneSimulator::AtmosphereWeather:
282  pressure = *((float*)(buf.data()+4*1)) * INCHES_MERCURY2KPA;
283  temperature = *((float*)(buf.data()+4*2));
284  break;
285 
286  case XplaneSimulator::LocVelDistTraveled:
287  dstX = *((float*)(buf.data()+4*1));
288  dstY = - *((float*)(buf.data()+4*3));
289  dstZ = *((float*)(buf.data()+4*2));
290  velX = *((float*)(buf.data()+4*4));
291  velY = - *((float*)(buf.data()+4*6));
292  velZ = *((float*)(buf.data()+4*5));
293  break;
294 
295  case XplaneSimulator::AngularVelocities: //In [rad/s]
296  pitchRate_rad = *((float*)(buf.data()+4*1));
297  rollRate_rad = *((float*)(buf.data()+4*2));
298  yawRate_rad = *((float*)(buf.data()+4*3));
299  break;
300 
301  case XplaneSimulator::Gload:
302  accX = *((float*)(buf.data()+4*6)) * GRAVITY;
303  accY = *((float*)(buf.data()+4*7)) * GRAVITY;
304  accZ = *((float*)(buf.data()+4*5)) * GRAVITY;
305  break;
306 
307  default:
308  break;
309  }
310  channelCounter--;
311  buf.remove(0,36);
312  } while (channelCounter);
313

316  // Output formatting
318  Output2Hardware out;
319  memset(&out, 0, sizeof(Output2Hardware));
320 
321  // Update GPS Position objects
322  out.latitude = latitude * 1e7;
323  out.longitude = longitude * 1e7;
324  out.altitude = altitude_msl;
325  out.agl = altitude_agl;
326  out.groundspeed = groundspeed_ktgs*1.15*1.6089/3.6; //Convert from [kts] to [m/s]

329  out.trueAirspeed = airspeed_ktas*1.15*1.6089/3.6; //Convert from [kts] to [m/s]
330  out.angleOfAttack=alpha_D;

332 
333  // Update BaroAltitude object
