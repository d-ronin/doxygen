136 
137  // TODO: Replace the second and third [in eraseDone()] pop-up dialogs with a progress indicator,

146  ObjectPersistence *objper = ObjectPersistence::GetInstance(getObjectManager());
147  Q_ASSERT(objper);

178 void ConfigPlugin::eraseDone(UAVObject *obj)
179 {
180  Q_UNUSED(obj)
181  QMessageBox msgBox;
182  ObjectPersistence *objper = ObjectPersistence::GetInstance(getObjectManager());
183  ObjectPersistence::DataFields data = objper->getData();
184  Q_ASSERT(obj->getInstID() == objper->getInstID());
185 
186  if (data.Operation != ObjectPersistence::OPERATION_COMPLETED) {
187  return;
188  }
189 
190  disconnect(objper, &UAVObject::objectUpdated, this, &ConfigPlugin::eraseDone);
191  if (data.Operation == ObjectPersistence::OPERATION_COMPLETED) {
192  settingsErased = true;
193  msgBox.setText(tr("Settings are now erased."));
194  msgBox.setInformativeText(tr("Please ensure that the status LED is flashing regularly and "
195  "then power-cycle your board."));
196  } else {
197  msgBox.setText(tr("Error trying to erase settings."));
198  msgBox.setInformativeText(tr(
199  "Power-cycle your board after removing all blades. Settings might be inconsistent."));
200  }
201  msgBox.setStandardButtons(QMessageBox::Ok);
202  msgBox.setDefaultButton(QMessageBox::Ok);
203  msgBox.exec();
204 }
