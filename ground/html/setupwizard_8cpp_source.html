76 int SetupWizard::nextId() const
77 {
78  switch (currentId()) {
79  case PAGE_START:
80  return PAGE_CONTROLLER;
81 
82  case PAGE_CONTROLLER: {
83  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
84  Q_ASSERT(pm);
85  UAVObjectUtilManager *utilMngr = pm->getObject<UAVObjectUtilManager>();
86  Q_ASSERT(utilMngr);
87 
88  const QString fwHash = utilMngr->getFirmwareHash();
89  const QString gcsHash = utilMngr->getGcsHash();
90  if (!fwHash.length() || fwHash != gcsHash) {
91  qobject_cast<BoardtypeUnknown *>(page(PAGE_BOARDTYPE_UNKNOWN))
92  ->setFailureType(BoardtypeUnknown::UNKNOWN_FIRMWARE);
93  return PAGE_BOARDTYPE_UNKNOWN;
94  }
95 
96  Core::IBoardType *type = getControllerType();
97  if (type == NULL) {
98  qobject_cast<BoardtypeUnknown *>(page(PAGE_BOARDTYPE_UNKNOWN))
99  ->setFailureType(BoardtypeUnknown::UNKNOWN_BOARD);
100  return PAGE_BOARDTYPE_UNKNOWN;
101  }
102 
103  if (type->isInputConfigurationSupported())
104  return PAGE_INPUT;
105  else
106  return PAGE_INPUT_NOT_SUPPORTED;
107  }
108  case PAGE_VEHICLES: {
109  switch (getVehicleType()) {
110  case VEHICLE_MULTI:
111  return PAGE_MULTI;
112 
113  case VEHICLE_FIXEDWING:
114  return PAGE_FIXEDWING;
115 
116  case VEHICLE_HELI:
117  return PAGE_HELI;
118 
119  case VEHICLE_SURFACE:
120  return PAGE_SURFACE;
121 
122  default:
123  return PAGE_NOTYETIMPLEMENTED;
124  }
125  }
126  case PAGE_MULTI:
127  return PAGE_OUTPUT;
128 
129  case PAGE_INPUT:
130  if (isRestartNeeded()) {
131  saveHardwareSettings();
132  return PAGE_REBOOT;
133  } else {
134  return PAGE_VEHICLES;
135  }
136  case PAGE_REBOOT:
137  return PAGE_VEHICLES;
138 
139  case PAGE_OUTPUT:
140  return PAGE_SUMMARY;
141 
142  case PAGE_BIAS_CALIBRATION:
143  return PAGE_OUTPUT_CALIBRATION;
144 
145  case PAGE_OUTPUT_CALIBRATION:
146  return PAGE_SAVE;
147 
148  case PAGE_SUMMARY: {
149  Core::IBoardType *type = getControllerType();
150  if (type != NULL)
151  return PAGE_BIAS_CALIBRATION;
152  else
153  return PAGE_NOTYETIMPLEMENTED;
154  }
155  case PAGE_SAVE:
156  return PAGE_END;
157 
158  case PAGE_INPUT_NOT_SUPPORTED:
159  // The user is informed on this page to configure manually and restart the wizard
160  // but if they advance assume they have configured the input.
161  return PAGE_VEHICLES;
162 
163  case PAGE_BOARDTYPE_UNKNOWN:
164  case PAGE_NOTYETIMPLEMENTED:
165  return PAGE_END;
166 
167  default:
168  return -1;
169  }
170 }
