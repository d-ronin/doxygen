478 static bool isDesktopFileManagerDrop(const QMimeData *d, QStringList *files = nullptr)
479 {
480  if (files)
481  files->clear();
482  // Extract dropped files from Mime data.
483  if (!d->hasFormat(QLatin1String(uriListMimeFormatC)))
484  return false;
485  const QList<QUrl> urls = d->urls();
486  if (urls.empty())
487  return false;
488  // Try to find local files
489  bool hasFiles = false;
490  const QList<QUrl>::const_iterator cend = urls.constEnd();
491  for (QList<QUrl>::const_iterator it = urls.constBegin(); it != cend; ++it) {
492  const QString fileName = it->toLocalFile();
493  if (!fileName.isEmpty()) {
494  hasFiles = true;
495  if (files) {
496  files->push_back(fileName);
497  } else {
498  break; // No result list, sufficient for checking
499  }
500  }
501  }
502  return hasFiles;
503 }

1118 
1119  Core::UAVGadgetManager *uavGadgetManager;
1120 
1121  // If diffOnly is true, we only add/remove the number of workspaces
1122  // that has changed,
1123  // otherwise a complete reload of workspaces is done
1124  int toRemoveFirst = m_uavGadgetManagers.count();
1125  int newWorkspacesNo = m_workspaceSettings->numberOfWorkspaces();
1126  if (diffOnly && m_uavGadgetManagers.count() > newWorkspacesNo)
1127  toRemoveFirst = m_uavGadgetManagers.count() - newWorkspacesNo;
1128  else
1129  toRemoveFirst = 0;
1130 
1131  int removed = 0;
1132 
1133  while (!m_uavGadgetManagers.isEmpty() && (toRemoveFirst > removed)) {
1134  int index = takeLeastPriorityUavGadgetManager(m_uavGadgetManagers);
1135  uavGadgetManager = m_uavGadgetManagers.takeAt(index);
1136  uavGadgetManager->removeAllSplits();
1137  pm->removeObject(uavGadgetManager);
1138  delete uavGadgetManager;
1139  removed++;
1140  }
1141 
1142  int start = 0;
1143  if (diffOnly) {
1144  start = m_uavGadgetManagers.count();
1145  } else {
1146  m_uavGadgetManagers.clear();
1147  }
1148  for (int i = start; i < newWorkspacesNo; ++i) {
1149 
1150  const QString name = m_workspaceSettings->name(i);
1151  const QString iconName = m_workspaceSettings->iconName(i);
1152  const QString modeName = m_workspaceSettings->modeName(i);
1153  uavGadgetManager = new Core::UAVGadgetManager(CoreImpl::instance(), name, QIcon(iconName),
1154  90 - i + 1, modeName, this);
