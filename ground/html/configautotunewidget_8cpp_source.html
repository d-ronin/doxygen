564 void AutotuneSlidersPage::compute()
565 {
566  // These three parameters define the desired response properties
567  // - rate scale in the fraction of the natural speed of the system
568  // to strive for.
569  // - damp is the amount of damping in the system. higher values
570  // make oscillations less likely
571  // - ghf is the amount of high frequency gain and limits the influence
572  // of noise
573 
574  const double ghf = rateNoise->value() / 1000.0;
575  const double damp = rateDamp->value() / 100.0;
576 
577  tuneState->damping = damp;
578  tuneState->noiseSens = ghf;
579 
580  /* Average roll and pitch tau for now. */
581  double tau = (tuneState->tau[0] + tuneState->tau[1]) / 2.0;
582  double beta_roll = tuneState->beta[0];
583  double beta_pitch = tuneState->beta[1];
584  double beta_yaw = tuneState->beta[2];
585 
586  // First clear out warnings..
587  lblWarnings->setText("");
588 
589  if (beta_yaw < 6.8) {
590  lblWarnings->setText(tr("Unable to auto-calculate yaw gains for this craft."));
591  cbUseYaw->setChecked(false);
592  cbUseYaw->setEnabled(false);
593  }
594 
595  bool doYaw = cbUseYaw->isChecked();
596  bool doOuterKi = cbUseOuterKi->isChecked();
597 
598  double wn = 1 / tau, wn_last = 1 / tau + 10;
599  double tau_d = 0, tau_d_last = 1000;
600 
601  const int iteration_limit = 100, stability_limit = 5;
602  bool converged = false;
603  int iterations = 0;
604  int stable_iterations = 0;
605 
606  while (!converged && (++iterations <= iteration_limit)) {
607  double tau_d_roll =
608  (2 * damp * tau * wn - 1) / (4 * tau * damp * damp * wn * wn - 2 * damp * wn
609  - tau * wn * wn + exp(beta_roll) * ghf);
610  double tau_d_pitch =
611  (2 * damp * tau * wn - 1) / (4 * tau * damp * damp * wn * wn - 2 * damp * wn
612  - tau * wn * wn + exp(beta_pitch) * ghf);
613 
614  // Select the slowest filter property
615  tau_d = (tau_d_roll > tau_d_pitch) ? tau_d_roll : tau_d_pitch;
616  wn = (tau + tau_d) / (tau * tau_d) / (2 * damp + 2);
617 
618  // check for convergence
619  if (fabs(tau_d - tau_d_last) <= 0.00001 && fabs(wn - wn_last) <= 0.00001) {
620  if (++stable_iterations >= stability_limit)
621  converged = true;
622  } else {
623  stable_iterations = 0;
624  }
625  tau_d_last = tau_d;
626  wn_last = wn;
627  }
628 
629  tuneState->iterations = iterations;
630  tuneState->converged = converged;
631 
632  tuneState->derivativeCutoff = 1 / (2 * M_PI * tau_d);
633  tuneState->naturalFreq = wn / 2 / M_PI;
634 
635  // Set the real pole position. The first pole is quite slow, which
636  // prevents the integral being too snappy and driving too much
637  // overshoot.
638  const double a = ((tau + tau_d) / tau / tau_d - 2 * damp * wn) / 20.0;
639  const double b = ((tau + tau_d) / tau / tau_d - 2 * damp * wn - a);
640 
641  CONF_ATUNE_QXTLOG_DEBUG("ghf: ", ghf);
642  CONF_ATUNE_QXTLOG_DEBUG("wn: ", wn, "tau_d: ", tau_d);
643  CONF_ATUNE_QXTLOG_DEBUG("a: ", a, " b: ", b);
644 
645  // Calculate the gain for the outer loop by approximating the
646  // inner loop as a single order lpf. Set the outer loop to be
647  // critically damped;
648  const double zeta_o = 1.3;
649  tuneState->outerKp = 1 / 4.0 / (zeta_o * zeta_o) / (1 / wn);
650 
651  // Except, if this is very high, we may be slew rate limited and pick
652  // up oscillation that way. Fix it with very soft clamping.
653  // MaximumRate defaults to 350, 6.5 corresponds to where we begin
654  // clamping rate ourselves. ESCs, etc, it depends upon gains
655  // and any pre-emphasis they do. Still give ourselves partial credit
656  // for inner loop bandwidth.
657  if (tuneState->outerKp > 6.5) {
658  tuneState->outerKp = 6.5 - sqrt(6.5) + sqrt(tuneState->outerKp);
659  }
660 
661  if (doOuterKi) {
662  tuneState->outerKp *= 0.95f; // Pick up some margin.
663  // Add a zero at 1/15th the innermost bandwidth.
664  tuneState->outerKi = 0.75 * tuneState->outerKp / (2 * M_PI * tau * 15.0);
665  } else {
666  tuneState->outerKi = 0;
667  }
668 
669  for (int i = 0; i < 2; i++) {
670  double beta = exp(tuneState->beta[i]);
671 
672  double ki;
673  double kp;
674  double kd;
675 
676  ki = a * b * wn * wn * tau * tau_d / beta;
677  kp = tau * tau_d * ((a + b) * wn * wn + 2 * a * b * damp * wn) / beta - ki * tau_d;
678  kd = (tau * tau_d * (a * b + wn * wn + (a + b) * 2 * damp * wn) - 1) / beta - kp * tau_d;
679 
680  tuneState->kp[i] = kp;
681  tuneState->ki[i] = ki;
682  tuneState->kd[i] = kd;
683  }
684 
685  if (doYaw) {
686  // Don't take yaw beta completely seriously. Why?
687  // 1) It's got two different time constants and magnitudes of
688  // effect (reaction wheel vs. drag). Don't want to overcontrol.
689  // 2) Far better to be undertuned on yaw than to get into weird
690  // scenarios from coupling between axes. If yaw is far less
691  // powerful than other axes, even a small amount of nonlinearity
692  // or cross-axis coupling will excite pitch and roll.
693  double scale = exp(0.6 * (tuneState->beta[0] - tuneState->beta[2]));
694  tuneState->kp[2] = tuneState->kp[0] * scale;
695  tuneState->ki[2] = 0.8 * tuneState->ki[0] * scale;
696  tuneState->kd[2] = 0.8 * tuneState->kd[0] * scale;
697  } else {
698  tuneState->kp[2] = -1;
699  tuneState->ki[2] = -1;
700  tuneState->kd[2] = -1;
701  }
702 
703  // handle non-convergence case. Takes precedence over all else.
704  if (!converged) {
705  lblWarnings->setText(tr("<span style=\"color: red\">Error:</span> Tune didn't converge! "
706  "Check noise and damping sliders."));
707  }
708 
709  emit completeChanged();
710 
711  setText(rollRateKp, tuneState->kp[0], 5);
712  setText(rollRateKi, tuneState->ki[0], 5);
713  setText(rollRateKd, tuneState->kd[0], 6);
714 
715  setText(pitchRateKp, tuneState->kp[1], 5);
716  setText(pitchRateKi, tuneState->ki[1], 5);
717  setText(pitchRateKd, tuneState->kd[1], 6);
718 
719  setText(yawRateKp, tuneState->kp[2], 5);
720  setText(yawRateKi, tuneState->ki[2], 5);
721  setText(yawRateKd, tuneState->kd[2], 6);
722 
723  setText(lblOuterKp, tuneState->outerKp, 2);
724  setText(lblOuterKi, tuneState->outerKi, 2);
725  setText(derivativeCutoff, tuneState->derivativeCutoff, 1);
726  setText(this->wn, tuneState->naturalFreq, 1);
727  setText(lblDamp, damp, 2);
728  lblNoise->setText(QString::number(ghf * 100, 'f', 1) + " %");
729 }

877 /* Run a Butterworth biquad filter on a circular buffer. First go around once
878  * to "prime" the filter, then actually filter in place. This is derived from
879  * @glowtape's excellent flight implementation
880  */
881 void AutotuneBeginningPage::biquadFilter(float cutoff, int pts,
882  QVector<float> &data)
883 {
884  float f = 1.0f / tan(M_PI * cutoff);
885  float q = 1.4142f;
886 
887  float y2 = 0, y1 = 0, x2 = 0, x1 = 0;
888 
889  float b0 = 1.0f / (1.0f + q * f + f * f);
890  float a1 = 2.0f * (f * f - 1.0f) * b0;
891  float a2 = -(1.0f - q * f + f * f) * b0;
892 
893  for (int i = 0; i < pts; i++) {
894  float y = b0 * (data[i] + 2.0f * x1 + x2) + a1 * y1 + a2 * y2;
895 
896  y2 = y1;
897  y1 = y;
898 
899  x2 = x1;
900  x1 = data[i];
901  }
902 
903  for (int i = 0; i < pts; i++) {
904  float y = b0 * (data[i] + 2.0f * x1 + x2) + a1 * y1 + a2 * y2;
905 
906  y2 = y1;
907  y1 = y;
908 
909  x2 = x1;
910  x1 = data[i];
911 
912  data[i] = y;
913  }
914 }

916 /* Returns number of samples of delay between series */
917 float AutotuneBeginningPage::getSampleDelay(int pts,
918  const QVector<float> &delayed, const QVector<float> &orig,
919  int seriesCutoff)
920 {
921  ffft::FFTReal<float> fft(pts);
922 
923  /* Convert to frequency domain */
924  QVector<float> delayed_fft(pts);
925  fft.do_fft(delayed_fft.data(), delayed.data());
926 
927  QVector<float> orig_fft(pts);
928  fft.do_fft(orig_fft.data(), orig.data());
929 
930  /* Now perform a correlation by multiplying -orig_fft* by delayed_fft.
931  * The types are all floats here, so we need to do the heavy lifting
932  * ourselves. gfft = x+yi, dfft = u+vi, dfft* = u-vi,
933  * -dfft* = -u + vi
934  *
935  * -dfft* x gfft = (-ux - vy) + (vx - uy)i
936  */
937 
938  QVector<float> product(pts);
939 
940  int fpts = pts / 2;
941 
942  // Memory layout here is annoyin'. All reals, then all imaginaries
943  for (int i = 0; i < fpts; i++) {
944  float x = delayed_fft[i];
945  float y = delayed_fft[i + fpts];
946  float u = orig_fft[i];
947  float v = orig_fft[i + fpts];
948 
949  product[i] = -(u * x) - (v * y);
950  product[i + fpts] = (v * x) - (u * y);
951  }
952 
953  /* Inverse FFT converts this to the time domain */
954  QVector<float> prod_time(pts);
955 
956  fft.do_ifft(product.data(), prod_time.data());
957 
958  /* And we take magnitudes to find tau. */
959  QVector<float> mags(fpts);
960 
961  int max_idx = 0;
962  float max_val = 0;
963 
964  for (int i = 0; i < fpts / seriesCutoff; i++) {
965  float real = prod_time[i];
966  float imag = prod_time[i + fpts];
967  mags[i] = sqrt(real * real + imag * imag);
968 
969  if (mags[i] > max_val) {
970  max_val = mags[i];
971  max_idx = i;
972  }
973 
974  // cout << i << "\t" << mags[i] << "\n";
975  }
976 
977  // cout << "Maximum mags[" << max_idx << "] = " << max_val << "\n";
978 
979  // TODO / optional: interpolate/find a better peak around max_idx.
980 
981  return max_idx;
982 }

984 bool AutotuneBeginningPage::processAutotuneData()
985 {
986  QByteArray &loadedFile = tuneState->data;
987 
988  const at_flash *flash_data = reinterpret_cast<const at_flash *>((const void *)loadedFile);
989 
990  unsigned int size = loadedFile.size();
991 
992  /* Determine whether we have a sane amount of data, etc. */
993  if ((size < sizeof(at_flash)) || (flash_data->hdr.magic != ATFLASH_MAGIC)) {
994  return false;
995  }
996 
997  unsigned int size_expected = sizeof(at_flash)
998  + sizeof(at_measurement) * flash_data->hdr.wiggle_points + flash_data->hdr.aux_data_len;
999 
1000  if (size < size_expected) {
1001  return false;
1002  }
1003 
1004  float duration = (float)flash_data->hdr.wiggle_points / flash_data->hdr.sample_rate;
1005 
1006  if ((duration < 0.25f) || (duration > 5.0f)) {
1007  return false;
1008  }
1009 
1010  int pts = flash_data->hdr.wiggle_points;
1011 
1012  for (int axis = 0; axis < 3; axis++) {
1013  QVector<float> gyro_deriv(pts);
1014  QVector<float> actu_desired(pts);
1015 
1016  for (int i = 0; i < pts; i++) {
1017  actu_desired[i] = flash_data->data[i].u[axis];
1018  }
1019 
1020  // Differentiate the gyro data
1021  for (int i = 1; i < pts; i++) {
1022  gyro_deriv[i] = flash_data->data[i].y[axis] - flash_data->data[i - 1].y[axis];
1023  }
1024 
1025  gyro_deriv[0] = flash_data->data[0].y[axis] - flash_data->data[pts - 1].y[axis];
1026 
1027  float sample_tau = getSampleDelay(pts, gyro_deriv, actu_desired,
1028  (axis == 2) ? 8 : 4);
1029 
1030  float tau = sample_tau / flash_data->hdr.sample_rate;
1031 
1032  biquadFilter(1 / (sample_tau * M_PI * 1.414), pts, actu_desired);
1033 
1034  QVector<float> gyro_sorted = gyro_deriv;
1035  QVector<float> actu_sorted = actu_desired;
1036 
1037  std::sort(gyro_sorted.begin(), gyro_sorted.end());
1038  std::sort(actu_sorted.begin(), actu_sorted.end());
1039 
1040  int low_idx = pts * 0.05 + 0.5;
1041  int high_idx = pts - 1 - low_idx;
1042 
1043  float gyro_span = gyro_sorted[high_idx] - gyro_sorted[low_idx];
1044  float actu_span = actu_sorted[high_idx] - actu_sorted[low_idx];
1045 
1046  float gain = gyro_span / actu_span * flash_data->hdr.sample_rate;
1047 
1048  float avg = std::accumulate(gyro_deriv.begin(), gyro_deriv.end(), 0.0f) / pts;
1049 
1050  for (int i = 0; i < pts; i++) {
1051  gyro_deriv[i] = gyro_deriv[i] - avg;
1052  }
1053 
1054  float avg_act = std::accumulate(actu_desired.begin(), actu_desired.end(), 0.0f) / pts;
1055 
1056  for (int i = 0; i < pts; i++) {
1057  actu_desired[i] = (actu_desired[i] - avg_act) * (gain / flash_data->hdr.sample_rate);
1058  }
1059 
1060  tuneState->model[axis] = new QLineSeries(this);
1061  tuneState->actual[axis] = new QLineSeries(this);
1062 
1063  for (int i = 0; i < pts; i++) {
1064  int tm = (i * 1000) / flash_data->hdr.sample_rate;
1065 
1066  tuneState->model[axis]->append(tm, actu_desired[i]);
1067  tuneState->actual[axis]->append(tm, gyro_deriv[i]);
1068  }
1069 
1070  float bias = avg - avg_act * (gain / flash_data->hdr.sample_rate);
1071 
1072  double noise = 0;
1073 
1074  for (int i = 0; i < pts; i++) {
1075  noise += (actu_desired[i] - gyro_deriv[i]) * (actu_desired[i] - gyro_deriv[i]);
1076  }
1077 
1078  noise = sqrt(noise / pts);
1079 
1080  qDebug() << "Series " << axis << ": tau=" << tau << "; gain=" << gain << " (" << log(gain)
1081  << "); bias=" << bias << " noise=" << noise << "";
1082 
1083  tuneState->tau[axis] = tau;
1084  tuneState->beta[axis] = log(gain);
1085  tuneState->bias[axis] = bias;
1086  tuneState->noise[axis] = noise;
1087  }
1088 
1089  tuneState->valid = true;
1090 
1091  return true;
1092 }
