123 QString ConfigAutotuneWidget::systemIdentValid(SystemIdent::DataFields &data,
124  bool *okToContinue)
125 {
126  if (data.Tau == 0) {
127  // Invalid / no tune.
128 
129  *okToContinue = false;
130  return tr("<span style=\"color: red\">It doesn't appear an autotune was successfully completed and saved; we are unable to continue.</span>");
131  }
132 
133  QString retVal;
134 
135  *okToContinue = true;
136 
137  if (data.NumAfPredicts < 5000) {
138  // Wayyyyy too few points to be plausible. CC3D is expected to do
139  // 333 * 60 = 20000 give or take a bit, with most other platforms way
140  // more. If you don't get this many, it doesn't make sense.
141 
142  retVal.append(tr("Error: Too few measurements were obtained to successfully autotune this craft."));
143  retVal.append("<br/>");
144  *okToContinue = false;
145  }
146 
147  if (data.NumSpilledPts > (data.NumAfPredicts / 8)) {
148  // Greater than 12.5% points spilled / not processed. You're not going
149  // to have a good time.
150 
151  retVal.append(tr("Error: Many measurements were lost because the flight controller was unable to keep up with the tuning process."));
152  retVal.append("<br/>");
153  *okToContinue = false;
154  } else if (data.NumSpilledPts > 10) {
155  // We shouldn't spill points at all. Warn!
156  retVal.append(tr("Warning: Some measurements were lost because the flight controller was unable to keep up with the tuning process."));
157  retVal.append("<br/>");
158  }
159 
160  if (data.Tau < -5.3) {
161  // Too low of tau to be plausible. (5ms)
162  retVal.append(tr("Error: Autotune did not measure valid values for this craft (low tau). Consider slightly lowering the starting roll/pitch rate P values or slightly decreasing Motor Input/Output Curve Fit on the output pane."));
163  retVal.append("<br/>");
164  *okToContinue = false;
165  } else if (data.Tau < -4.9) {
166  // Probably too low to be real-- 7.4ms-- warn!
167  retVal.append(tr("Warning: The tau value measured for this craft is very low."));
168  retVal.append("<br/>");
169  } else if (data.Tau > -1.4) {
170  // Too high of a tau to be plausible / accurate (247ms)-- warn!
171  retVal.append(tr("Warning: The tau value measured for this craft is very high."));
172  retVal.append("<br/>");
173  }
174 
175  // Lowest valid gains seen have been 7.9, with most values in the range 9..11
176  if (data.Beta[SystemIdent::BETA_ROLL] < 7.25) {
177  retVal.append(tr("Error: Autotune did not measure valid values for this craft (low roll gain)."));
178  retVal.append("<br/>");
179  *okToContinue = false;
180  }
181 
182  if (data.Beta[SystemIdent::BETA_PITCH] < 7.25) {
183  retVal.append(tr("Error: Autotune did not measure valid values for this craft (low pitch gain)."));
184  retVal.append("<br/>");
185  *okToContinue = false;
186  }
187 
188  retVal.replace(QRegExp("(\\w+:)"), "<span style=\"color: red\"><b>\\1</b></span>");
189 
190  if (*okToContinue) {
191  if (retVal.isEmpty()) {
192  retVal.append(tr("Everything checks out, and we're ready to proceed!"));
193  } else {
194  retVal.append(tr("<br/>These warnings may result in an invalid tune. Proceed with caution."));
195  }
196  } else {
197  retVal.append(tr("<br/>Unable to complete the autotune process because of the above error(s)."));
198  }
199 
200  return retVal;
201 }

581 void AutotuneSlidersPage::compute()
582 {
583  // These three parameters define the desired response properties
584  // - rate scale in the fraction of the natural speed of the system
585  // to strive for.
586  // - damp is the amount of damping in the system. higher values
587  // make oscillations less likely
588  // - ghf is the amount of high frequency gain and limits the influence
589  // of noise
590 
591  const double ghf = rateNoise->value() / 1000.0;
592  const double damp = rateDamp->value() / 100.0;
593 
594  av->damping = damp;
595  av->noiseSens = ghf;
596 
597  double tau = exp(sysIdent.Tau);
598  double beta_roll = sysIdent.Beta[SystemIdent::BETA_ROLL];
599  double beta_pitch = sysIdent.Beta[SystemIdent::BETA_PITCH];
600  double beta_yaw = sysIdent.Beta[SystemIdent::BETA_YAW];
601 
602  // First clear out warnings..
603  lblWarnings->setText("");
604 
605  if (beta_yaw < 6.8) {
606  lblWarnings->setText(tr("Unable to auto-calculate yaw gains for this craft."));
607  cbUseYaw->setChecked(false);
608  cbUseYaw->setEnabled(false);
609  }
610 
611  bool doYaw = cbUseYaw->isChecked();
612  bool doOuterKi = cbUseOuterKi->isChecked();
613 
614  double wn = 1/tau, wn_last = 1/tau + 10;
615  double tau_d = 0, tau_d_last = 1000;
616 
617  const int iteration_limit = 100, stability_limit = 5;
618  bool converged = false;
619  int iterations = 0;
620  int stable_iterations = 0;
621 
622  while (!converged && (++iterations <= iteration_limit)) {
623  double tau_d_roll = (2*damp*tau*wn - 1)/(4*tau*damp*damp*wn*wn - 2*damp*wn - tau*wn*wn + exp(beta_roll)*ghf);
624  double tau_d_pitch = (2*damp*tau*wn - 1)/(4*tau*damp*damp*wn*wn - 2*damp*wn - tau*wn*wn + exp(beta_pitch)*ghf);
625 
626  // Select the slowest filter property
627  tau_d = (tau_d_roll > tau_d_pitch) ? tau_d_roll : tau_d_pitch;
628  wn = (tau + tau_d) / (tau*tau_d) / (2 * damp + 2);
629 
630  // check for convergence
631  if (fabs(tau_d - tau_d_last) <= 0.00001 && fabs(wn - wn_last) <= 0.00001) {
632  if (++stable_iterations >= stability_limit)
633  converged = true;
634  } else {
635  stable_iterations = 0;
636  }
637  tau_d_last = tau_d;
638  wn_last = wn;
639  }
640 
641  av->iterations = iterations;
642  av->converged = converged;
643 
644  av->derivativeCutoff = 1 / (2*M_PI*tau_d);
645  av->naturalFreq = wn / 2 / M_PI;
646 
647  // Set the real pole position. The first pole is quite slow, which
648  // prevents the integral being too snappy and driving too much
649  // overshoot.
650  const double a = ((tau+tau_d) / tau / tau_d - 2 * damp * wn) / 20.0;
651  const double b = ((tau+tau_d) / tau / tau_d - 2 * damp * wn - a);
652 
653  CONF_ATUNE_QXTLOG_DEBUG("ghf: ", ghf);
654  CONF_ATUNE_QXTLOG_DEBUG("wn: ", wn, "tau_d: ", tau_d);
655  CONF_ATUNE_QXTLOG_DEBUG("a: ", a, " b: ", b);
656 
657  // Calculate the gain for the outer loop by approximating the
658  // inner loop as a single order lpf. Set the outer loop to be
659  // critically damped;
660  const double zeta_o = 1.3;
661  av->outerKp = 1 / 4.0 / (zeta_o * zeta_o) / (1/wn);
662 
663  // Except, if this is very high, we may be slew rate limited and pick
664  // up oscillation that way. Fix it with very soft clamping.
665  // MaximumRate defaults to 350, 6.5 corresponds to where we begin
666  // clamping rate ourselves. ESCs, etc, it depends upon gains
667  // and any pre-emphasis they do. Still give ourselves partial credit
668  // for inner loop bandwidth.
669  if (av->outerKp > 6.5) {
670  av->outerKp = 6.5 - sqrt(6.5) + sqrt(av->outerKp);
671  }
672 
673  if (doOuterKi) {
674  av->outerKp *= 0.95; // Pick up some margin.
675  // Add a zero at 1/15th the innermost bandwidth.
676  av->outerKi = 0.75 * av->outerKp / (2 * M_PI * tau * 15.0);
677  } else {
678  av->outerKi = 0;
679  }
680 
681  for (int i = 0; i < 2; i++) {
682  double beta = exp(sysIdent.Beta[i]);
683 
684  double ki;
685  double kp;
686  double kd;
687 
688  ki = a * b * wn * wn * tau * tau_d / beta;
689  kp = tau * tau_d * ((a+b)*wn*wn + 2*a*b*damp*wn) / beta - ki*tau_d;
690  kd = (tau * tau_d * (a*b + wn*wn + (a+b)*2*damp*wn) - 1) / beta - kp * tau_d;
691 
692  av->kp[i] = kp;
693  av->ki[i] = ki;
694  av->kd[i] = kd;
695  }
696 
697  if (doYaw) {
698  // Don't take yaw beta completely seriously. Why?
699  // 1) It's got two different time constants and magnitudes of
700  // effect (reaction wheel vs. drag). Don't want to overcontrol.
701  // 2) Far better to be undertuned on yaw than to get into weird
702  // scenarios from coupling between axes. If yaw is far less
703  // powerful than other axes, even a small amount of nonlinearity
704  // or cross-axis coupling will excite pitch and roll.
705  double scale = exp(0.6 * (sysIdent.Beta[0] - sysIdent.Beta[2]));
706  av->kp[2] = av->kp[0] * scale;
707  av->ki[2] = 0.8 * av->ki[0] * scale;
708  av->kd[2] = 0.8 * av->kd[0] * scale;
709  } else {
710  av->kp[2] = -1; av->ki[2] = -1; av->kd[2] = -1;
711  }
712 
713  // handle non-convergence case. Takes precedence over all else.
714  if (!converged) {
715  lblWarnings->setText(tr("<span style=\"color: red\">Error:</span> Tune didn't converge! Check noise and damping sliders."));
716  }
717 
718  emit completeChanged();
719 
720  setText(rollRateKp, av->kp[0], 5);
721  setText(rollRateKi, av->ki[0], 5);
722  setText(rollRateKd, av->kd[0], 6);
723 
724  setText(pitchRateKp, av->kp[1], 5);
725  setText(pitchRateKi, av->ki[1], 5);
726  setText(pitchRateKd, av->kd[1], 6);
727 
728  setText(yawRateKp, av->kp[2], 5);
729  setText(yawRateKi, av->ki[2], 5);
730  setText(yawRateKd, av->kd[2], 6);
731 
732  setText(lblOuterKp, av->outerKp, 2);
733  setText(lblOuterKi, av->outerKi, 2);
734  setText(derivativeCutoff, av->derivativeCutoff, 1);
735  setText(this->wn, av->naturalFreq, 1);
736  setText(lblDamp, damp, 2);
737  lblNoise->setText(QString::number(ghf * 100, 'f', 1) + " %");
738 }
