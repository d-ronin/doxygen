174 bool UAVConfigInfo::askToAbort(int compat, QString message)
175 {
176  QMessageBox msgBox;
177  msgBox.setInformativeText(tr("Do you want to continue the import?"));
178  msgBox.setStandardButtons(QMessageBox::Ok | QMessageBox::Cancel);
179 
180  int result = 0;
181 
182  switch (compat) {
183 
184  case FullyCompatible:
185  return false;
186 
187  case MinorLossOfConfiguration:
188  msgBox.setText(tr("INFO: ") + message + TEXT_MINOR_LOSS_OF_CONFIGURATION);
189  msgBox.setDefaultButton(QMessageBox::Ok);
190  result = msgBox.exec();
191  break;
192 
193  case MissingConfiguration:
194  msgBox.setText(tr("WARNING: ") + message + TEXT_MISSING_CONFIGURATION);
195  msgBox.setDefaultButton(QMessageBox::Ok);
196  result = msgBox.exec();
197  break;
198 
199  case MajorLossOfConfiguration:
200  msgBox.setText(tr("ERROR: ") + message + TEXT_MAJOR_LOSS_OF_CONFIGURATION);
201  msgBox.setDefaultButton(QMessageBox::Cancel);
202  result = msgBox.exec();
203  break;
204 
205  case NotCompatible:
206  msgBox.setText("ERROR: " + message + TEXT_NOT_COMPATIBLE);
207  msgBox.setInformativeText(tr(""));
208  msgBox.setStandardButtons(QMessageBox::Ok);
209  msgBox.exec();
210  return true;
211 
212  default:
213  msgBox.setText("INTERNAL ERROR: " + message + tr("Unknown compatibility level: ")
214  + QString::number(compat));
215  }
216  if (result == QMessageBox::Ok)
217  return false;
218  else
219  return true;
220 }

229 int UAVConfigInfo::checkCompatibilityWith(UAVConfigVersion programVersion)
230 {
231  if (m_version.majorNr != programVersion.majorNr)
232  return NotCompatible;
233  if (m_version.minorNr < programVersion.minorNr)
234  return MissingConfiguration;
235  if (m_version.minorNr > programVersion.minorNr)
236  return MajorLossOfConfiguration;
237  if (m_version.patchNr > programVersion.patchNr)
238  return MinorLossOfConfiguration;
239 
240  return FullyCompatible;
241 }

255 UAVConfigVersion::UAVConfigVersion(QString versionString)
256 {
257  int begin;
258  int end = 0;
259 
260  begin = end;
261  end = versionString.indexOf(".", begin);
262  majorNr = versionString.mid(begin, end - begin).toInt();
263 
264  begin = end + 1;
265  end = versionString.indexOf(".", begin);
266  minorNr = versionString.mid(begin, end - begin).toInt();
267 
268  begin = end + 1;
