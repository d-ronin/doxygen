75 void KmlExportPlugin::exportToKML()
76 {
77  // Get input file
78  QString inputFileName = QFileDialog::getOpenFileName(NULL, tr("Open file"), QString(""),
79  tr("dRonin Log Files (*.drlog *.tll)"));
80  if (inputFileName.isEmpty())
81  return;
82 
83  // Set up input filter.
84  QString filters = tr("Keyhole Markup Language (compressed) (*.kmz);; Keyhole Markup Language "
85  "(uncompressed) (*.kml)");
86  bool proceed_flag = false;
87  QString outputFileName;
88  QString localizedOutputFileName;
89 
90  // Get output file. Suggest to user that output have same base name and location as input file.
91  while (proceed_flag == false) {
92  outputFileName = QFileDialog::getSaveFileName(
93  NULL, tr("Export log"), inputFileName.split(".", QString::SkipEmptyParts).at(0),
94  filters);
95 
96  if (outputFileName.isEmpty()) {
97  qDebug() << "No KML file name given.";
98  return;
99  } else if (QFileInfo(outputFileName).suffix() == NULL) {
100  qDebug() << "No KML file extension: " << outputFileName;
101  QMessageBox::critical(new QWidget(), "No file extension",
102  "Filename must have .kml or .kmz extension.");
103  } else if (QFileInfo(outputFileName).suffix().toLower() != "kml"
104  && QFileInfo(outputFileName).suffix().toLower() != "kmz") {
105  qDebug() << "Incorrect KML file extension: " << QFileInfo(outputFileName).suffix();
106  QMessageBox::critical(new QWidget(), "Incorrect file extension",
107  "Filename must have .kml or .kmz extension.");
108  } else if (outputFileName.toLocal8Bit() == (QByteArray)NULL) {
109  qDebug() << "Unsupported characters in path: " << outputFileName;
110  QMessageBox::critical(new QWidget(), "Unsupported characters",
111  "Not all uni-code characters are supported. Please choose a path "
112  "and file name that uses only the standard latin alphabet.");
113  } else {
114  proceed_flag = true;
115 
116  // Due to limitations in the KML library, we must convert the output file name into
117  // local 8-bit characters.
118  localizedOutputFileName = outputFileName.toLocal8Bit();
119  }
120  }
121 
122  // Create kmlExport instance, and trigger export
123  KmlExport kmlExport(inputFileName, localizedOutputFileName);
124  kmlExport.exportToKML();
125 }
