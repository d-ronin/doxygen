83 void GeneralSettings::fillLanguageBox() const
84 {
85  const QString currentLocale = language();
86 
87  m_page->languageBox->addItem(tr("<System Language>"), QString());
88  // need to add this explicitly, since there is no qm file for English
89  m_page->languageBox->addItem(QLatin1String("English"), QLatin1String("C"));
90  if (currentLocale == QLatin1String("C"))
91  m_page->languageBox->setCurrentIndex(m_page->languageBox->count() - 1);
92 
93  const QString creatorTrPath =
94  Core::ICore::instance()->resourcePath() + QLatin1String("/translations");
95  const QStringList languageFiles =
96  QDir(creatorTrPath).entryList(QStringList(QLatin1String("dronin*.qm")));
97 
98  Q_FOREACH (const QString &languageFile, languageFiles) {
99  int start = languageFile.indexOf(QLatin1Char('_')) + 1;
100  int end = languageFile.lastIndexOf(QLatin1Char('.'));
101  const QString locale = languageFile.mid(start, end - start);
102  // no need to show a language that creator will not load anyway
103  if (hasQmFilesForLocale(locale, creatorTrPath)) {
104  m_page->languageBox->addItem(QLocale::languageToString(QLocale(locale).language()),
105  locale);
106  if (locale == currentLocale)
107  m_page->languageBox->setCurrentIndex(m_page->languageBox->count() - 1);
108  }
109  }
110 }
