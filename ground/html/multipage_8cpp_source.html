154 void MultiPage::updateImageAndDescription()

175  case SetupWizard::MULTI_ROTOR_HEXA_COAX_Y:
176  elementId = "hexa-coax";
177  break;
178  case SetupWizard::MULTI_ROTOR_HEXA_H:

180  break;
181  case SetupWizard::MULTI_ROTOR_OCTO:
182  elementId = "quad-octo";

184  case SetupWizard::MULTI_ROTOR_OCTO_COAX_X:
185  elementId = "octo-coax-X";
