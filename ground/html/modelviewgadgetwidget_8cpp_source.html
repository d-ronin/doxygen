191 void ModelViewGadgetWidget::CreateScene()
192 {
193  // put a black background if the 3D model is invalid or if the background image is also invalid
194  if (acFilename == fallbackAcFilename || !QFile::exists(bgFilename)) {
195  bgFilename = fallbackBgFilename;
196  }
197 
198  try {
199  m_GlView.loadBackGroundImage(bgFilename);
200  } catch (GLC_Exception e) {
201  qDebug("ModelView: background image file loading failed.");
202  }
203 
204  try {
205  if (QFile::exists(acFilename)) {
206  QFile aircraft(acFilename);
207  m_World = GLC_Factory::instance()->createWorldFromFile(aircraft);
208  m_ModelBoundingBox = m_World.boundingBox();
209  m_GlView.reframe(m_ModelBoundingBox); // center 3D model in the scene
210  } else {
211  qDebug() << "ModelView: aircraft file not found:" << acFilename;
212  }
213  } catch (GLC_Exception e) {
214  qDebug("ModelView: aircraft file loading failed.");
215  }

327 void ModelViewGadgetWidget::updateAttitude()
328 {
329  AttitudeActual::DataFields data = attState->getData(); // get attitude data
330  GLC_StructOccurence *rootObject = m_World.rootOccurence(); // get the full 3D model
331  double x = data.q3;
332  double y = data.q2;
333  double z = data.q4;
334  double w = data.q1;
335 
336  if (w == 0.0) {
337  w = 1.0;
338  }
339  // create and gives the product of 2 4x4 matrices to get the rotation of the 3D model's matrix
340  QMatrix4x4 m1;
341  m1.setRow(0, QVector4D(w, z, -y, x));
342  m1.setRow(1, QVector4D(-z, w, x, y));
343  m1.setRow(2, QVector4D(y, -x, w, z));
344  m1.setRow(3, QVector4D(-x, -y, -z, w));
345  QMatrix4x4 m2;
346  m2.setRow(0, QVector4D(w, z, -y, -x));
347  m2.setRow(1, QVector4D(-z, w, x, -y));
348  m2.setRow(2, QVector4D(y, -x, w, -z));
349  m2.setRow(3, QVector4D(x, y, z, w));
350  QMatrix4x4 m0 = m1 * m2;
351  // convert QMatrix4x4 to GLC_Matrix4x4
352  GLC_Matrix4x4 rootObjectRotation(m0.data());
353  rootObject->structInstance()->setMatrix(rootObjectRotation);
354  rootObject->updateChildrenAbsoluteMatrix();
355  updateGL();
356 }
