222 void HistogramScopeConfig::loadConfiguration(ScopeGadgetWidget *scopeGadgetWidget)
223 {
224  preparePlot(scopeGadgetWidget);
225  scopeGadgetWidget->setScope(this);
226  scopeGadgetWidget->startTimer(m_refreshInterval);
227 
228  // Configure each data source
229  foreach (Plot2dCurveConfiguration *histogramDataSourceConfig, m_HistogramSourceConfigs) {
230  QRgb color = histogramDataSourceConfig->color;
231 
232  // Get and store the units
233  units = getUavObjectFieldUnits(histogramDataSourceConfig->uavObjectName,
234  histogramDataSourceConfig->uavFieldName);
235 
236  HistogramData *histogramData;
237  histogramData =
238  new HistogramData(histogramDataSourceConfig->uavObjectName,
239  histogramDataSourceConfig->uavFieldName, binWidth, maxNumberOfBins);
240 
241  histogramData->setScalePower(histogramDataSourceConfig->yScalePower);
242  histogramData->setMeanSamples(histogramDataSourceConfig->yMeanSamples);
243  histogramData->setMathFunction(histogramDataSourceConfig->mathFunction);
244 
245  // Generate the curve name
246  QString curveName =
247  (histogramData->getUavoName()) + "." + (histogramData->getUavoFieldName());
248  if (histogramData->getHaveSubFieldFlag())
249  curveName = curveName.append("." + histogramData->getUavoSubFieldName());
250 
251  // Get the uav object
252  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
253  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
254  UAVDataObject *obj =
255  dynamic_cast<UAVDataObject *>(objManager->getObject((histogramData->getUavoName())));
256  if (!obj) {
257  qDebug() << "Object " << histogramData->getUavoName() << " is missing";
258  return;
259  }
260 
261  // Get the units
262  QString units =
263  getUavObjectFieldUnits(histogramData->getUavoName(), histogramData->getUavoFieldName());
264 
265  // Generate name with scaling factor appeneded
266  QString histogramNameScaled;
267  if (histogramDataSourceConfig->yScalePower == 0)
268  histogramNameScaled = curveName + "(" + units + ")";
269  else
270  histogramNameScaled = curveName + "(x10^"
271  + QString::number(histogramDataSourceConfig->yScalePower) + " " + units + ")";
272 
273  while (scopeGadgetWidget->getDataSources().keys().contains(histogramNameScaled))
274  histogramNameScaled = histogramNameScaled + "*";
275 
276  // Create the histogram
277  QwtPlotHistogram *plotHistogram = new QwtPlotHistogram(histogramNameScaled);
278  plotHistogram->setStyle(QwtPlotHistogram::Columns);
279  plotHistogram->setBrush(QBrush(QColor(color)));
280  plotHistogram->setData(histogramData->getIntervalSeriesData());
281 
282  plotHistogram->attach(scopeGadgetWidget);
283  histogramData->setHistogram(plotHistogram);
284 
285  // Keep the curve details for later
286  scopeGadgetWidget->insertDataSources(histogramNameScaled, histogramData);
287 
288  // Connect the UAVO
289  scopeGadgetWidget->connectUAVO(obj);
290  }
291  scopeGadgetWidget->replot();
292 }
