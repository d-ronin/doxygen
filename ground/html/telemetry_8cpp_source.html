221 void Telemetry::updateObject(UAVObject *obj, quint32 eventType)
222 {
223  // Get metadata
224  UAVObject::Metadata metadata = obj->getMetadata();
225  UAVObject::UpdateMode updateMode = UAVObject::GetGcsTelemetryUpdateMode(metadata);
226 
227  // Setup object depending on update mode
228  qint32 eventMask;
229  if (updateMode == UAVObject::UPDATEMODE_PERIODIC) {
230  // Set update period
231  setUpdatePeriod(obj, metadata.gcsTelemetryUpdatePeriod);
232  // Connect signals for all instances
233  eventMask = EV_UPDATED_MANUAL | EV_UPDATE_REQ | EV_UPDATED_PERIODIC;
234  eventMask |= EV_UNPACKED; // we also need to act on remote updates (unpack events)
235  connectToObjectInstances(obj, eventMask);
236  } else if (updateMode == UAVObject::UPDATEMODE_ONCHANGE) {
237  // Set update period
238  setUpdatePeriod(obj, 0);
239  // Connect signals for all instances
240  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
241  eventMask |= EV_UNPACKED; // we also need to act on remote updates (unpack events)
242  connectToObjectInstances(obj, eventMask);
243  } else if (updateMode == UAVObject::UPDATEMODE_THROTTLED) {
244  // If we received a periodic update, we can change back to update on change
245  if ((eventType == EV_UPDATED_PERIODIC) || (eventType == EV_NONE)) {
246  // Set update period
247  if (eventType == EV_NONE)
248  setUpdatePeriod(obj, metadata.gcsTelemetryUpdatePeriod);
249  // Connect signals for all instances
250  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ | EV_UPDATED_PERIODIC;
251  } else {
252  // Otherwise, we just received an object update, so switch to periodic for the timeout
253  // period to prevent more updates
254  // Connect signals for all instances
255  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
256  }
257  eventMask |= EV_UNPACKED; // we also need to act on remote updates (unpack events)
258  connectToObjectInstances(obj, eventMask);
259  } else if (updateMode == UAVObject::UPDATEMODE_MANUAL) {
260  // Set update period
261  setUpdatePeriod(obj, 0);
262  // Connect signals for all instances
263  eventMask = EV_UPDATED_MANUAL | EV_UPDATE_REQ;
264  eventMask |= EV_UNPACKED; // we also need to act on remote updates (unpack events)
265  connectToObjectInstances(obj, eventMask);
266  }

405 QByteArray *Telemetry::downloadFile(quint32 fileId, quint32 maxSize,
406  std::function<void(quint32)>progressCb)
407 {
408  quint32 curOffset = 0;
409 
410  QByteArray *result = new QByteArray();
411 
412  quint32 sizeGuess = 32 * 1024;
413 
414  if (maxSize < sizeGuess) {
415  sizeGuess = maxSize;
416  }
417 
418  result->reserve(maxSize);
419 
420  bool newReqNeeded = false;
421  bool completed = false;
422  int inactivityCount = 0;
423  int failCount = 0;
424 
425  QEventLoop loop;
426  QTimer timeStep;
427 
428  timeStep.setSingleShot(false);
429  timeStep.start(150);
430 
431  connect(&timeStep, &QTimer::timeout, &loop, &QEventLoop::quit);
432 
433  connect(utalk, &UAVTalk::fileDataReceived, &loop,
434  [&](quint32 recvFileId, quint32 offset, quint8 *data, quint32 dataLen,
435  bool eof, bool lastInSeq) {
436  if (recvFileId != fileId) {
437  return;
438  }
439 
440  if (offset != curOffset) {
441  if (lastInSeq) {
442  newReqNeeded = true;
443  loop.exit();
444  }
445 
446  return;
447  }
448 
449  result->append((const char *) data, dataLen);
450 
451  curOffset += dataLen;
452 
453  if (eof) {
454  completed = true;
455  loop.exit();
456  } else if (lastInSeq) {
457  newReqNeeded = true;
458  loop.exit();
459  }
460 
461  inactivityCount = 0;
462  failCount = 0;
463 
464  if (progressCb) {
465  progressCb(curOffset);
466  }
467  }
468  );
469 
470  while (curOffset < maxSize && (!completed)) {
471  utalk->requestFile(fileId, curOffset);
472 
473  newReqNeeded = false;
474  inactivityCount = 0;
475 
476  do {
477  if ((inactivityCount++) > 10) {
478  qDebug() << "Retrying file transfer because of inactivity";
479  failCount++;
480  break;
481  }
482 
483  loop.exec();
484 
485  } while (curOffset < maxSize && (!completed) && (!newReqNeeded));
486 
487  if (failCount > 5) {
488  qDebug() << "Aborting file transfer";
489  delete result;
490  return NULL;
491  }
492  }
493 
494  return result;
495 }

559 void Telemetry::processObjectQueue()
560 {
561  if (objQueue.length() > 1) {
562  TELEMETRY_QXTLOG_DEBUG(
563  "[telemetry.cpp] **************** Object Queue above 1 in backlog ****************");
564  }
565  // Get object information from queue (first the priority and then the regular queue)
566  ObjectQueueInfo objInfo;
567  if (!objPriorityQueue.isEmpty()) {
568  objInfo = objPriorityQueue.dequeue();
569  } else if (!objQueue.isEmpty()) {
570  objInfo = objQueue.dequeue();
571  } else {
572  return;
573  }
574 
575  // Check if a connection has been established, only process GCSTelemetryStats updates
576  // (used to establish the connection)
577  GCSTelemetryStats::DataFields gcsStats = gcsStatsObj->getData();
578  if (gcsStats.Status != GCSTelemetryStats::STATUS_CONNECTED) {
579  objQueue.clear();
580  if (objInfo.obj->getObjID() != GCSTelemetryStats::OBJID
581  && objInfo.obj->getObjID() != HwTauLink::OBJID
582  && objInfo.obj->getObjID() != ObjectPersistence::OBJID) {
583  // If Telemetry is not connected, then all transactions fail except
584  // - GCSTelemetryStats (to establish connection)
585  // - HwTauLink (to configure the modem)
586  // - ObjectPersistence (to save modem configuration)
587  objInfo.obj->emitTransactionCompleted(false);
588  objInfo.obj->emitTransactionCompleted(false, false);
589  return;
590  }
591  }
592 
593  // Setup transaction (skip if unpack event)
594  UAVObject::Metadata metadata = objInfo.obj->getMetadata();
595  UAVObject::UpdateMode updateMode = UAVObject::GetGcsTelemetryUpdateMode(metadata);
596  if ((objInfo.event != EV_UNPACKED) && ((objInfo.event != EV_UPDATED_PERIODIC)
597  || (updateMode != UAVObject::UPDATEMODE_THROTTLED))) {
598  // We are either going to send an object, or are requesting one:
599  if (transMap.contains(TransactionKey(objInfo.obj, objInfo.event == EV_UPDATE_REQ))) {
600  TELEMETRY_QXTLOG_DEBUG(QString("[telemetry.cpp] Warning: Got request for %0 for which "
601  "a request is already in progress. Not doing it")
602  .arg(objInfo.obj->getName()));
603  // We will not re-request it, then, we should wait for a timeout or success...
604  } else {
605  UAVObject::Metadata metadata = objInfo.obj->getMetadata();
606  ObjectTransactionInfo *transInfo = new ObjectTransactionInfo(this);
607  transInfo->obj = objInfo.obj;
608  transInfo->allInstances = objInfo.allInstances;
609  transInfo->retriesRemaining = MAX_RETRIES;
610  transInfo->acked = UAVObject::GetGcsTelemetryAcked(metadata);
611  if (objInfo.event == EV_UPDATED || objInfo.event == EV_UPDATED_MANUAL
612  || objInfo.event == EV_UPDATED_PERIODIC) {
613  transInfo->objRequest = false;
614  } else if (objInfo.event == EV_UPDATE_REQ) {
615  transInfo->objRequest = true;
616  }
617  transInfo->telem = this;
618  // Insert the transaction into the transaction map.
619  TransactionKey key(objInfo.obj, transInfo->objRequest);
620  transMap.insert(key, transInfo);
621  processObjectTransaction(transInfo);
622  }
623  }
624 
625  // If this is a metaobject then make necessary telemetry updates
626  // to the connections of this object to Telemetry (this) :
627  UAVMetaObject *metaobj = dynamic_cast<UAVMetaObject *>(objInfo.obj);
628  if (metaobj != NULL) {
629  updateObject(metaobj->getParentObject(), EV_NONE);
630  } else if (updateMode != UAVObject::UPDATEMODE_THROTTLED) {
631  updateObject(objInfo.obj, objInfo.event);
632  }
633 
634  // We received an "unpacked" event, check whether
635  // this is for an object we were expecting
636  if (objInfo.event == EV_UNPACKED) {
637  // TODO: Check here this is for a OBJ_REQ
638  if (transMap.contains(TransactionKey(objInfo.obj, true))) {
639  TELEMETRY_QXTLOG_DEBUG(
640  QString("[telemetry.cpp] EV_UNPACKED %0 Instance:%1")
641  .arg(objInfo.obj->getName()
642  + QString(QString("0x")
643  + QString::number(objInfo.obj->getObjID(), 16).toUpper()))
644  .arg(objInfo.obj->getInstID()));
645  transactionRequestCompleted(objInfo.obj);
646  } else {
647  processObjectQueue();
648  }
649  }

655 void Telemetry::processPeriodicUpdates()
656 {
657  // Stop timer
658  updateTimer->stop();
659 
660  // Iterate through each object and update its timer, if zero then transmit object.
661  // Also calculate smallest delay to next update (will be used for setting timeToNextUpdateMs)
662  qint32 minDelay = MAX_UPDATE_PERIOD_MS;
663  qint32 elapsedMs = 0;
664  QTime time;
665  qint32 offset;
666 
667  const QVector<ObjectTimeInfo>::iterator objInfoEnd = objList.end();
668  for (QVector<ObjectTimeInfo>::iterator objinfo = objList.begin(); objinfo != objInfoEnd;
669  ++objinfo) {
670  // If object is configured for periodic updates
671  // FixMe: This is an inefficient process as it depends on polling. It would be better to
672  // have
673  // this functionality be driven through timers.
674  if (objinfo->updatePeriodMs > 0) {
675  objinfo->timeToNextUpdateMs -= timeToNextUpdateMs;
676  // Check if time for the next update
677  if (objinfo->timeToNextUpdateMs <= 0) {
678  // Reset timer
679  offset = (-objinfo->timeToNextUpdateMs) % objinfo->updatePeriodMs;
680  objinfo->timeToNextUpdateMs = objinfo->updatePeriodMs - offset;
681  // Send object
682  time.start();
683  processObjectUpdates(objinfo->obj, EV_UPDATED_PERIODIC, true, false);
684  elapsedMs = time.elapsed();
685  // Update timeToNextUpdateMs with the elapsed delay of sending the object;
686  timeToNextUpdateMs += elapsedMs;
687  }
688  // Update minimum delay
689  if (objinfo->timeToNextUpdateMs < minDelay) {
690  minDelay = objinfo->timeToNextUpdateMs;
691  }
692  }
693  }
694 
695  // Check if delay for the next update is too short
696  if (minDelay < MIN_UPDATE_PERIOD_MS) {
697  minDelay = MIN_UPDATE_PERIOD_MS;
698  }
699 
700  // Done
701  timeToNextUpdateMs = minDelay;
702 
703  // Restart timer
704  updateTimer->start(timeToNextUpdateMs);
705 }
