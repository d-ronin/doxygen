105 void TextEditLoggerEngine::writeToTextEdit(const QString &level, const QList<QVariant> &msgs,
106  QColor color)
107 {
108  /* Message format...
109  [time] [error level] First message.....
110  second message
111  third message
112  */
113  if (msgs.isEmpty())
114  return;
115  QScrollBar *sb = m_textEdit->verticalScrollBar();
116  bool scroll = sb->value() == sb->maximum();
117  QString header = '[' + QTime::currentTime().toString("hh:mm:ss.zzz") + "] [" + level + "] ";
118  QString padding;
119  QString appendText;
120  appendText.append(header);
121  for (int i = 0; i < header.size(); i++)
122  padding.append(' ');
123  int count = 0;
124  Q_FOREACH (const QVariant &out, msgs) {
125  if (!out.isNull()) {
126  if (count != 0)
127  appendText.append(padding);
128  appendText.append(out.toString());
129  }
130  count++;
131  }
132  Q_ASSERT(m_textEdit);
133  appendText = QString("<font color=%1>%2</font>").arg(color.name()).arg(appendText);
134  m_textEdit->append(appendText);
135  if (scroll)
136  sb->setValue(sb->maximum());
137 }
