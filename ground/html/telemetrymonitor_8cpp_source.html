129 void TelemetryMonitor::startRetrievingObjects()
130 {
131  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 connectionStatus changed to CON_RETRIEVING_OBJECT").arg(Q_FUNC_INFO));
132  connectionStatus = CON_RETRIEVING_OBJECTS;
133  // Get all objects, add metaobjects, settings and data objects with OnChange update mode to the queue
134  queue.clear();
135  retries = 0;
136  objectRetrieveTimeout->start(OBJECT_RETRIEVE_TIMEOUT);
137  foreach(UAVObjectManager::ObjectMap map, objMngr->getObjects().values())
138  {
139  UAVObject* obj = map.first();
140  if(obj->getObjID() == SessionManaging::OBJID)
141  {
142  continue;
143  }
144  UAVDataObject* dobj = dynamic_cast<UAVDataObject*>(obj);
145  UAVObject::Metadata mdata = obj->getMetadata();
146  if ( dobj != NULL )
147  {
148  if(!dobj->getIsPresentOnHardware())
149  {
150  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 %1 not present on hardware, skipping").arg(Q_FUNC_INFO).arg(obj->getName()));
151  continue;
152  }
153  queue.enqueue(dobj->getMetaObject());
154  if ( dobj->isSettings() )
155  {
156  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 queing settings object %1").arg(Q_FUNC_INFO).arg(dobj->getName()));
157  queue.enqueue(obj);
158  }
159  else
160  {
161  if ( UAVObject::GetFlightTelemetryUpdateMode(mdata) == UAVObject::UPDATEMODE_ONCHANGE )
162  {
163  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 queing UPDATEMODE_ONCHANGE object %1").arg(Q_FUNC_INFO).arg(dobj->getName()));
164  queue.enqueue(obj);
165  }
166  else
167  {
168  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 skipping %1, not settings nor metadata or UPDATEMODE_ONCHANGE").arg(Q_FUNC_INFO).arg(dobj->getName()));
169  }
170  }
171  }
172  }
173  // Start retrieving
174  TELEMETRYMONITOR_QXTLOG_DEBUG(QString(tr("Starting to retrieve meta and settings objects from the autopilot (%1 objects)"))
175  .arg( queue.length()));
176  retrieveNextObject();
177 }

239  if ( queue.isEmpty() )
240  {
241  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 Object retrieval completed").arg(Q_FUNC_INFO));
242  if(isManaged)

246  }
247  else
248  {
249  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 connectionStatus set to CON_CONNECTED_MANAGED( %1 )").arg(Q_FUNC_INFO).arg(connectionStatus));

342 void TelemetryMonitor::sessionObjUnpackedCB(UAVObject *obj)
343 {
344  sessionObjRetries = 0;
345  switch(connectionStatus)
346  {
347  case CON_INITIALIZING:
348  if(sessions.contains(sessionObj->getSessionID()) && (sessions.value(sessionObj->getSessionID()).count() == sessionObj->getNumberOfObjects()))
349  {
350  sessionID = sessionObj->getSessionID();
351  numberOfObjects = sessionObj->getNumberOfObjects();
352  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 status:%1 session already known startRetrievingObjects").arg(Q_FUNC_INFO).arg(connectionStatus));
353  foreach(objStruc objs, sessions.value(sessionObj->getSessionID()))
354  {
355  UAVDataObject * dobj = dynamic_cast<UAVDataObject*>(objMngr->getObject(objs.objID));
356  if(dobj)
357  {
358  if(dobj->isSingleInstance())
359  {
360  dobj->setIsPresentOnHardware(true);
361  }
362  else
363  {
364  changeObjectInstances(objs.objID,objs.instID, true);
365  }
366  }
367  }
368  startRetrievingObjects();
369  }
370  else
371  {
372  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 new session startSessionRetrieving HWsessionID=%1 GCSsessionID=%2 HWnumberofobjs=%3 GCSnumberofobjs=%4").arg(Q_FUNC_INFO).arg(sessionObj->getSessionID()).arg(sessionID).arg(sessionObj->getNumberOfObjects()).arg(sessions.value(sessionObj->getSessionID()).count()));
373  startSessionRetrieving(NULL);
374  }
375  break;
376  case CON_CONNECTED_MANAGED:
377  qDebug()<<QString("HW=%0 GCS=%1 GCS_ADJ=%2").arg(sessionObj->getSessionID()).arg(sessionID).arg(sessionID + 1);
378  if(sessionObj->getSessionID() == sessionID && numberOfObjects == sessionObj->getNumberOfObjects())
379  return;
380  else if(sessionObj->getSessionID() == (sessionID + 1))//number of instances changed
381  {
382  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 received sessionID shift by one, this signal that a multi instance object changed number of instances")
383  .arg(Q_FUNC_INFO));
384  changeObjectInstances(sessionObj->getObjectID(),sessionObj->getObjectInstances(), false);
385  sessionID = sessionID + 1;
386  saveSession();
387  }
388  else if(!sessions.contains(sessionObj->getSessionID()))
389  {
390  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 new session during CONNECTED? startSessionRetrieving").arg(Q_FUNC_INFO));
391  startSessionRetrieving(NULL);
392  }
393  break;
394  case CON_SESSION_INITIALIZING:
395  startSessionRetrieving(obj);
396  break;
397  case CON_RETRIEVING_OBJECTS:
398  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 received sessionManaging object during object retrievel, this shouldn't happen").arg(Q_FUNC_INFO));
399  break;
400  case CON_CONNECTED_UNMANAGED:
401  case CON_DISCONNECTED:
402  break;
403  }
404 }

468 void TelemetryMonitor::startSessionRetrieving(UAVObject *session)
469 {
470  static int currentIndex = 0;
471  static int objectCount = 0;
472  static int sessionNegotiationRetries = 0;
473  if(session == NULL)
474  {
475  sessionRetrieveTimeout->start(SESSION_RETRIEVE_TIMEOUT);
476  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 NULL new session start").arg(Q_FUNC_INFO));
477  foreach(UAVObjectManager::ObjectMap map, objMngr->getObjects())
478  {
479  foreach(UAVObject* obj, map.values())
480  {
481  UAVDataObject* dobj = dynamic_cast<UAVDataObject*>(obj);
482  if(dobj)
483  dobj->setIsPresentOnHardware(false);
484  }
485  }
486  currentIndex = 0;
487  objectCount = 0;
488  sessionObjRetries = 0;
489  connectionStatus = CON_SESSION_INITIALIZING;
490  sessionObj->setSessionID(0);
491  sessionObj->updated();
492  }
493  else if(sessionObj->getSessionID() == 0)
494  {
495  objectCount = sessionObj->getNumberOfObjects();
496  if(objectCount == 0)
497  return;
498  sessionID = QDateTime::currentDateTime().toTime_t();
499  sessionObj->setSessionID(sessionID);
500  sessionObj->setObjectOfInterestIndex(0);
501  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 SESSION SETUP with numberofobjects:%1 and sessionID:%2").arg(Q_FUNC_INFO)
502  .arg(objectCount).arg(sessionID));
503  sessionObj->updated();
504  }
505  else if(sessionObj->getSessionID() == sessionID)
506  {
507  if(sessionObj->getObjectOfInterestIndex() != currentIndex)
508  {
509  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 RECEIVED WRONG INDEX RETRY current retries:%1").arg(Q_FUNC_INFO).arg(sessionNegotiationRetries));
510  if(sessionNegotiationRetries < SESSION_INIT_RETRIES)
511  {
512  sessionObj->setObjectOfInterestIndex(currentIndex);
513  ++sessionNegotiationRetries;
514  sessionObj->updated();
515  return;
516  }
517  else
518  {
519  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 EXCEEDED SESSION OBJECT FETCHING RETRIES, GOING INTO FALLBACK currentIndex was:%1").arg(Q_FUNC_INFO).arg(currentIndex));
520  sessionFallback();
521  }
522  }
523  else
524  {
525  sessionNegotiationRetries = 0;
526  }
527  UAVObject *obj = objMngr->getObject(sessionObj->getObjectID());
528  if(obj)
529  {
530  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 RECEIVED INDEX:%1 object:%2 instances:%3").arg(Q_FUNC_INFO).arg(sessionObj->getObjectOfInterestIndex())
531  .arg(obj->getName()).arg(sessionObj->getObjectInstances()));
532  UAVDataObject* dobj = dynamic_cast<UAVDataObject*>(obj);
533  if(dobj)
534  {
535  dobj->setIsPresentOnHardware(true);
536  }
537  if((sessionObj->getObjectInstances() > 1) && !obj->isSingleInstance())
538  {
539  changeObjectInstances(dobj->getObjID(),sessionObj->getObjectInstances(), true);
540  }
541  }
542  ++currentIndex;
543  if(currentIndex < objectCount)
544  {
545  TELEMETRYMONITOR_QXTLOG_DEBUG(QString("%0 ASKING BOARD FOR INDEX:%1").arg(Q_FUNC_INFO).arg(currentIndex));
546  sessionObj->setObjectOfInterestIndex(currentIndex);
547  sessionObj->updated();
548  }
549  else
550  {
551  saveSession();
552  startRetrievingObjects();
553  }
554  }

575 void TelemetryMonitor::processStatsUpdates()
576 {
577  // Get telemetry stats
578  GCSTelemetryStats::DataFields gcsStats = gcsStatsObj->getData();
579  FlightTelemetryStats::DataFields flightStats = flightStatsObj->getData();
580  Telemetry::TelemetryStats telStats = tel->getStats();
581  tel->resetStats();
582 
583  // Update stats object
584  gcsStats.RxDataRate = (float)telStats.rxBytes / ((float)statsTimer->interval()/1000.0);
585  gcsStats.TxDataRate = (float)telStats.txBytes / ((float)statsTimer->interval()/1000.0);
586  gcsStats.RxFailures += telStats.rxErrors;
587  gcsStats.TxFailures += telStats.txErrors;
588  gcsStats.TxRetries += telStats.txRetries;
589 
590  // Check for a connection timeout
591  bool connectionTimeout;
592  if ( telStats.rxObjects > 0 )
593  {
594  connectionTimer->start();
595  }
596  if ( connectionTimer->elapsed() > CONNECTION_TIMEOUT_MS )
597  {
598  connectionTimeout = true;
599  }
600  else
601  {
602  connectionTimeout = false;
603  }
604 
605  // Update connection state
606  int oldStatus = gcsStats.Status;
607  if ( gcsStats.Status == GCSTelemetryStats::STATUS_DISCONNECTED )
608  {
609  // Request connection
610  gcsStats.Status = GCSTelemetryStats::STATUS_HANDSHAKEREQ;
611  }
612  else if ( gcsStats.Status == GCSTelemetryStats::STATUS_HANDSHAKEREQ )
613  {
614  // Check for connection acknowledge
615  if ( flightStats.Status == FlightTelemetryStats::STATUS_HANDSHAKEACK )
616  {
617  gcsStats.Status = GCSTelemetryStats::STATUS_CONNECTED;
