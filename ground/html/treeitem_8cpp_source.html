78 /*
79  * Callback called periodically by the timer.
80  * This method checks for expired highlights and
81  * removes them if they are expired.
82  * Expired highlights are restored.
83  */
84 void HighLightManager::checkItemsExpired()
85 {
86  // Get a mutable iterator for the list
87  QMutableLinkedListIterator<TreeItem*> iter(m_itemsList);
88 
89  // Loop over all items, check if they expired.
90  while(iter.hasNext())
91  {
92  TreeItem* item = iter.next();
93  if(item->getHiglightExpires() < *m_currentTime)
94  {
95  // If expired, call removeHighlight
96  item->removeHighlight();
97 
98  // Remove from list since it is restored.
99  iter.remove();
100  }
101  }
102 }

197 /*
198  * Called after a value has changed to trigger highlightning of tree item.
199  */
200 void TreeItem::setHighlight(bool highlight) {
201  m_highlight = highlight;
202  m_changed = false;
203  if (highlight) {
204  // Update the expires timestamp
205  if (m_currentTime != NULL)
206  m_highlightExpires = m_currentTime->addMSecs(m_highlightTimeMs);
207  else
208  m_highlightExpires = QTime::currentTime().addMSecs(m_highlightTimeMs);
209 
210  // Add to highlightmanager
211  if(m_highlightManager->add(this))
212  {
213  // Only emit signal if it was added
214  emit updateHighlight(this);
215  }
216  }
217  else if(m_highlightManager->remove(this))
218  {
219  // Only emit signal if it was removed
220  emit updateHighlight(this);
221  }
222 
223  // If we have a parent, call recursively to update highlight status of parents.
224  // This will ensure that the root of a leaf that is changed also is highlighted.
225  // Only updates that really changes values will trigger highlight of parents.
226  if(m_parent)
227  {
228  m_parent->setHighlight(highlight);
229  }
230 }
