800 void ConfigVehicleTypeWidget::bnLevelTrim_clicked()
801 {
802  QMessageBox msgBox(QMessageBox::Question, tr("Trim level"),
803  tr("Use the transmitter trim to set the autopilot for straight and level flight? (Please see the tooltip for more information.)"),
804  QMessageBox::Yes | QMessageBox::No, this);
805  int userChoice = msgBox.exec();
806 
807  // If the user cancels, stop here.
808  if (userChoice != QMessageBox::Yes)
809  return;
810 
811  // Call bias set function
812  VehicleTrim vehicleTrim;
813  VehicleTrim::autopilotLevelBiasMessages ret;
814  ret = vehicleTrim.setAutopilotBias();
815 
816  // Process return state
817  switch (ret){
818  case VehicleTrim::AUTOPILOT_LEVEL_FAILED_DUE_TO_MISSING_RECEIVER:
819  {
820  QMessageBox msgBox(QMessageBox::Critical, tr("No receiver detected"),
821  tr("Transmitter and receiver must be powered on."), QMessageBox::Ok, this);
822  msgBox.exec();
823  break;
824  }
825  case VehicleTrim::AUTOPILOT_LEVEL_FAILED_DUE_TO_ARMED_STATE:
826  {
827  QMessageBox msgBox(QMessageBox::Critical, tr("Vehicle armed"),
828  tr("The autopilot must be disarmed first."), QMessageBox::Ok, this);
829  msgBox.exec();
830  break;
831  }
832  case VehicleTrim::AUTOPILOT_LEVEL_FAILED_DUE_TO_FLIGHTMODE:
833  {
834  QMessageBox msgBox(QMessageBox::Critical, tr("Vehicle not in Stabilized mode"),
835  tr("The autopilot must be in Leveling, Stabilized1, Stabilized2, or Stabilized3 mode."), QMessageBox::Ok, this);
836  msgBox.exec();
837  break;
838  }
839  case VehicleTrim::AUTOPILOT_LEVEL_FAILED_DUE_TO_STABILIZATIONMODE:
840  {
841  QMessageBox msgBox(QMessageBox::Critical, tr("Incorrect roll and pitch stabilization modes."),
842  tr("Both roll and pitch must be in Attitude stabilization mode."), QMessageBox::Ok, this);
843  msgBox.exec();
844  break;
845  }
846  case VehicleTrim::AUTOPILOT_LEVEL_SUCCESS:
847  QMessageBox msgBox(QMessageBox::Information, tr("Trim updated"),
848  tr("Trim successfully updated, please reset the transmitter's trim to zero and be sure to configure stabilization settings to use Attitude mode."), QMessageBox::Ok, this);
849  msgBox.exec();
850 
851  // Set tab as dirty (i.e. having unsaved changes).
852  setDirty(true);
853 
854  break;
