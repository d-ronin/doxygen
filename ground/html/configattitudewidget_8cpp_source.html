232 void ConfigAttitudeWidget::displayPlane(int position)
233 {
234  QString displayElement;
235  switch (position) {
236  case 1:
237  displayElement = "plane-horizontal";
238  break;
239  case 2:
240  displayElement = "plane-left";
241  break;
242  case 3:
243  displayElement = "plane-flip";
244  break;
245  case 4:
246  displayElement = "plane-right";
247  break;
248  case 5:
249  displayElement = "plane-up";
250  break;
251  case 6:
252  displayElement = "plane-down";
253  break;
254  default:
255  return;
256  }
257

259  m_ui->sixPointHelp->setSceneRect(paperplane->boundingRect());

261 }
