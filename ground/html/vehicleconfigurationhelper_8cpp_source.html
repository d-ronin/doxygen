196 
197  quint16 updateFrequency = ESC_FREQUENCY_LEGACY;
198  switch (m_configSource->getESCType()) {
199  case VehicleConfigurationSource::ESC_LEGACY:

204  break;

208  break;

211  break;
212  case VehicleConfigurationSource::ESC_DSHOT600:
213  updateFrequency = ESC_FREQUENCY_DSHOT600;
214  break;
215  case VehicleConfigurationSource::ESC_DSHOT1200:
216  updateFrequency = ESC_FREQUENCY_DSHOT1200;
217  break;
218  default:
219  break;
220  }
221 
222  // TOOD: vehicle specific sets of update frequencies
223  switch (m_configSource->getVehicleSubType()) {
224  case VehicleConfigurationSource::MULTI_ROTOR_TRI_Y:
225  // Ignore the servo for now.
226  max_motors = 3;
227 
228  data.MotorsSpinWhileArmed = ActuatorSettings::MOTORSSPINWHILEARMED_TRUE;
229  break;
230  case VehicleConfigurationSource::MULTI_ROTOR_QUAD_X:
231  case VehicleConfigurationSource::MULTI_ROTOR_QUAD_PLUS:
232  max_motors = 4;
233 
234  data.MotorsSpinWhileArmed = ActuatorSettings::MOTORSSPINWHILEARMED_TRUE;
235  break;
236  case VehicleConfigurationSource::MULTI_ROTOR_HEXA:
237  case VehicleConfigurationSource::MULTI_ROTOR_HEXA_COAX_Y:
238  case VehicleConfigurationSource::MULTI_ROTOR_HEXA_H:
239  max_motors = 6;
240 
241  data.MotorsSpinWhileArmed = ActuatorSettings::MOTORSSPINWHILEARMED_TRUE;
242 
243  break;
244  case VehicleConfigurationSource::MULTI_ROTOR_OCTO:
245  case VehicleConfigurationSource::MULTI_ROTOR_OCTO_COAX_X:
246  case VehicleConfigurationSource::MULTI_ROTOR_OCTO_COAX_PLUS:
247  case VehicleConfigurationSource::MULTI_ROTOR_OCTO_V:
248  max_motors = 8;
249 
250  data.MotorsSpinWhileArmed = ActuatorSettings::MOTORSSPINWHILEARMED_TRUE;
251  break;
252  default:
253  break;
254  }
255 
256  for (quint16 i = 0; i < ActuatorSettings::CHANNELMAX_NUMELEM; i++) {
257  data.ChannelType[i] = ActuatorSettings::CHANNELTYPE_PWM;
258 
259  if (i < max_motors) {
260  data.ChannelMin[i] = actuatorSettings[i].channelMin;
261  data.ChannelNeutral[i] = actuatorSettings[i].channelNeutral;
262  data.ChannelMax[i] = actuatorSettings[i].channelMax;
263 
264  // Channels are 1-indexed in boardplugin and UI. :(
265  int bankNum = boardPlugin->getBankFromOutputChannel(i+1);
266 
267  if (bankNum >= 0) {
268  data.TimerUpdateFreq[bankNum] = updateFrequency;
269  }
270  } else {
271  data.ChannelMin[i] = 0;
272  data.ChannelNeutral[i] = 0;
273  data.ChannelMax[i] = 0;
274  }
275  }
276 
277  actSettings->setData(data);
278  addModifiedObject(actSettings, tr("Writing actuator settings"));
279  break;
280  }
281  case VehicleConfigurationSource::VEHICLE_FIXEDWING:
282  case VehicleConfigurationSource::VEHICLE_HELI:
283  case VehicleConfigurationSource::VEHICLE_SURFACE:
284  // TODO: Implement settings for other vehicle types?
285  break;
286  default:
287  break;
288  }

394 void VehicleConfigurationHelper::applyManualControlDefaults()
395 {
396  ManualControlSettings *mcSettings = ManualControlSettings::GetInstance(m_uavoManager);
397 
398  Q_ASSERT(mcSettings);
399  ManualControlSettings::DataFields cData = mcSettings->getData();
400 
401  ManualControlSettings::ChannelGroupsOptions channelType = ManualControlSettings::CHANNELGROUPS_PWM;
402 
403  /* For all of these -- minVal and maxVal gathered from multiple autotune
404  * configuration shares. neutVal is the true midpoint. throtNeutVal
405  * is 4% up from min-- which is more than we usually do but conservative
406  * for auto-setup. There are exceptions / cases where this is not right,
407  * like for openLRS s.bus. But it's a decent start for people who will
408  * enter their ranges by hand.
409  */
410  qint16 minVal = 1000;
411  qint16 maxVal = 2000;
412 
413  switch (m_configSource->getInputType()) {
414  case Core::IBoardType::INPUT_TYPE_PWM:
415  channelType = ManualControlSettings::CHANNELGROUPS_PWM;
416  break;
417  case Core::IBoardType::INPUT_TYPE_PPM:
418  channelType = ManualControlSettings::CHANNELGROUPS_PPM;
419  break;
420  case Core::IBoardType::INPUT_TYPE_SBUS:
421  case Core::IBoardType::INPUT_TYPE_SBUSNONINVERTED:
422  minVal = 172;
423  maxVal = 1811;
424  channelType = ManualControlSettings::CHANNELGROUPS_SBUS;
425  break;
426  case Core::IBoardType::INPUT_TYPE_DSM:
427  // Sadly this is the worse one-- it depends upon resolution.
428  // 74-950, 64-940 seen for 10 bit. 192-1856, 23-2025 for 11 bit.
429  minVal = 70;
430  maxVal = 940;
431  channelType = ManualControlSettings::CHANNELGROUPS_DSM;
432  break;
433  case Core::IBoardType::INPUT_TYPE_HOTTSUMD:
434  case Core::IBoardType::INPUT_TYPE_HOTTSUMH:
435  minVal = 858;
436  maxVal = 2138;
437  channelType = ManualControlSettings::CHANNELGROUPS_HOTTSUM;
438  break;
439  case Core::IBoardType::INPUT_TYPE_IBUS:
440  /* Seems to be 1000..2000 */
441  channelType = ManualControlSettings::CHANNELGROUPS_IBUS;
442  break;
443  case Core::IBoardType::INPUT_TYPE_SRXL:
444  // unfortunately Weatronic variant has different scaling, this is the range for standard impl. per protocol spec.
445  minVal = 585;
446  maxVal = 3510;
447  channelType = ManualControlSettings::CHANNELGROUPS_SRXL;
448  break;
449  case Core::IBoardType::INPUT_TYPE_UNKNOWN:
450  case Core::IBoardType::INPUT_TYPE_DISABLED:
451  channelType = ManualControlSettings::CHANNELGROUPS_NONE;
452  break;
453  case Core::IBoardType::INPUT_TYPE_ANY:
454  break;
455  }
456 
457  qint16 neutralVal = minVal + (maxVal - minVal) / 2;
458  qint16 throttleNeutralVal = static_cast<qint16>(minVal + (maxVal - minVal) * 0.04);
459 
460  for (unsigned int i = 0; i < ManualControlSettings::CHANNELGROUPS_NUMELEM;
461  i++) {
462  cData.ChannelGroups[i] = channelType;
463  cData.ChannelNumber[i] = 0;
464  cData.ChannelMin[i] = minVal;
465  cData.ChannelMax[i] = maxVal;
466  cData.ChannelNeutral[i] = neutralVal;
467  }
468 
469  cData.ChannelNumber[ManualControlSettings::CHANNELGROUPS_THROTTLE] = 1;
470  cData.ChannelNeutral[ManualControlSettings::CHANNELNEUTRAL_THROTTLE] =
471  throttleNeutralVal;
472  cData.ChannelNumber[ManualControlSettings::CHANNELGROUPS_ROLL] = 2;
473  cData.ChannelNumber[ManualControlSettings::CHANNELGROUPS_YAW] = 3;
474  cData.ChannelNumber[ManualControlSettings::CHANNELGROUPS_PITCH] = 4;
475  cData.ChannelNumber[ManualControlSettings::CHANNELGROUPS_FLIGHTMODE] = 5;
476 
477  mcSettings->setData(cData);
478  addModifiedObject(mcSettings, tr("Writing manual control defaults"));
479 }

481 bool VehicleConfigurationHelper::saveChangesToController(bool save)
482 {
483  qDebug() << "Saving modified objects to controller. " << m_modifiedObjects.count() << " objects in found.";
484  const int OUTER_TIMEOUT = 3000 * 20; // 10 seconds timeout for saving all objects
485  const int INNER_TIMEOUT = 2000; // 1 second timeout on every save attempt
486 
487  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
488  Q_ASSERT(pm);
489  UAVObjectUtilManager *utilMngr = pm->getObject<UAVObjectUtilManager>();
490  Q_ASSERT(utilMngr);
491 
492  QTimer outerTimeoutTimer;
493  outerTimeoutTimer.setSingleShot(true);
494 
495  QTimer innerTimeoutTimer;
496  innerTimeoutTimer.setSingleShot(true);
497 
498  connect(utilMngr, SIGNAL(saveCompleted(int, bool)), this, SLOT(uAVOTransactionCompleted(int, bool)));
499  connect(&innerTimeoutTimer, SIGNAL(timeout()), &m_eventLoop, SLOT(quit()));
500  connect(&outerTimeoutTimer, SIGNAL(timeout()), this, SLOT(saveChangesTimeout()));
501 
502  outerTimeoutTimer.start(OUTER_TIMEOUT);
503  for (int i = 0; i < m_modifiedObjects.count(); i++) {
504  QPair<UAVDataObject *, QString> *objPair = m_modifiedObjects.at(i);
505  m_transactionOK = false;
506  UAVDataObject *obj = objPair->first;
507  QString objDescription = objPair->second;
508  if (UAVObject::GetGcsAccess(obj->getMetadata()) != UAVObject::ACCESS_READONLY && obj->isSettings()) {
509  emit saveProgress(m_modifiedObjects.count() + 1, ++m_progress, objDescription);
510 
511  m_currentTransactionObjectID = obj->getObjID();
512 
513  connect(obj, SIGNAL(transactionCompleted(UAVObject *, bool)), this, SLOT(uAVOTransactionCompleted(UAVObject *, bool)));
514  while (!m_transactionOK && !m_transactionTimeout) {
515  // Allow the transaction to take some time
516  innerTimeoutTimer.start(INNER_TIMEOUT);
517 
518  // Set object updated
519  obj->updated();
520  if (!m_transactionOK) {
521  m_eventLoop.exec();
522  }
523  innerTimeoutTimer.stop();
524  }
525  disconnect(obj, SIGNAL(transactionCompleted(UAVObject *, bool)), this, SLOT(uAVOTransactionCompleted(UAVObject *, bool)));
526  if (m_transactionOK) {
527  qDebug() << "Object " << obj->getName() << " was successfully updated.";
528  if (save) {
529  m_transactionOK = false;
530  m_currentTransactionObjectID = obj->getObjID();
531  // Try to save until success or timeout
532  while (!m_transactionOK && !m_transactionTimeout) {
533  // Allow the transaction to take some time
534  innerTimeoutTimer.start(INNER_TIMEOUT);
535 
536  // Persist object in controller
537  utilMngr->saveObjectToFlash(obj);
538  if (!m_transactionOK) {
539  m_eventLoop.exec();
540  }
541  innerTimeoutTimer.stop();
542  }
543  m_currentTransactionObjectID = -1;
544  }
545  }
546 
547  if (!m_transactionOK) {
548  qDebug() << "Transaction timed out when trying to save: " << obj->getName();
549  } else {
550  qDebug() << "Object " << obj->getName() << " was successfully saved.";
551  }
552  } else {
553  qDebug() << "Trying to save a UAVDataObject that is read only or is not a settings object.";
554  }
555  if (m_transactionTimeout) {
556  qDebug() << "Transaction timed out when trying to save " << m_modifiedObjects.count() << " objects.";
557  break;
558  }
559  }
560 
561  outerTimeoutTimer.stop();
562  disconnect(&outerTimeoutTimer, SIGNAL(timeout()), this, SLOT(saveChangesTimeout()));
563  disconnect(&innerTimeoutTimer, SIGNAL(timeout()), &m_eventLoop, SLOT(quit()));
564  disconnect(utilMngr, SIGNAL(saveCompleted(int, bool)), this, SLOT(uAVOTransactionCompleted(int, bool)));
565 
566  qDebug() << "Finished saving modified objects to controller. Success = " << m_transactionOK;
567 
568  return m_transactionOK;
569 }
