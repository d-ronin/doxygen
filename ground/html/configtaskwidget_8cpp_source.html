290 double ConfigTaskWidget::listVar(QList<double> list)
291 {
292  double mean_accum = 0;
293  double var_accum = 0;
294  double mean;
295 
296  for (int i = 0; i < list.size(); i++)
297  mean_accum += list[i];
298  mean = mean_accum / list.size();
299 
300  for (int i = 0; i < list.size(); i++)
301  var_accum += (list[i] - mean) * (list[i] - mean);
302 
303  // Use unbiased estimator
304  return var_accum / (list.size() - 1);
