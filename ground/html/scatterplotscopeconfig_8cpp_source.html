224 void Scatterplot2dScopeConfig::loadConfiguration(ScopeGadgetWidget *scopeGadgetWidget)
225 {
226  preparePlot(scopeGadgetWidget);
227  scopeGadgetWidget->setScope(this);
228  scopeGadgetWidget->startTimer(m_refreshInterval);
229 
230  // Configure each data source
231  foreach (Plot2dCurveConfiguration *plotCurveConfig, m_scatterplotSourceConfigs) {
232  QString uavObjectName = plotCurveConfig->uavObjectName;
233  QString uavFieldName = plotCurveConfig->uavFieldName;
234  QRgb color = plotCurveConfig->color;
235 
236  ScatterplotData *scatterplotData = NULL;
237 
238  switch (scatterplot2dType) {
239  case SERIES2D:
240  scatterplotData = new SeriesPlotData(uavObjectName, uavFieldName);
241  break;
242  case TIMESERIES2D:
243  scatterplotData = new TimeSeriesPlotData(uavObjectName, uavFieldName);
244  break;
245  }
246 
247  scatterplotData->setXWindowSize(timeHorizon);
248  scatterplotData->setScalePower(plotCurveConfig->yScalePower);
249  scatterplotData->setMeanSamples(plotCurveConfig->yMeanSamples);
250  scatterplotData->setMathFunction(plotCurveConfig->mathFunction);
251 
252  // Generate the curve name
253  QString curveName =
254  (scatterplotData->getUavoName()) + "." + (scatterplotData->getUavoFieldName());
255  if (scatterplotData->getHaveSubFieldFlag())
256  curveName = curveName.append("." + scatterplotData->getUavoSubFieldName());
257 
258  // Get the uav object
259  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
260  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
261  UAVDataObject *obj =
262  dynamic_cast<UAVDataObject *>(objManager->getObject((scatterplotData->getUavoName())));
263  if (!obj) {
264  qDebug() << "Object " << scatterplotData->getUavoName() << " is missing";
265  return;
266  }
267 
268  // Get the units
269  QString units = getUavObjectFieldUnits(scatterplotData->getUavoName(),
270  scatterplotData->getUavoFieldName());
271 
272  // Generate name with scaling factor appeneded
273  QString curveNameScaled;
274  if (plotCurveConfig->yScalePower == 0)
275  curveNameScaled = curveName + "(" + units + ")";
276  else
277  curveNameScaled = curveName + "(x10^" + QString::number(plotCurveConfig->yScalePower)
278  + " " + units + ")";
279 
280  QString curveNameScaledMath;
281  if (plotCurveConfig->mathFunction == "None")
282  curveNameScaledMath = curveNameScaled;
283  else if (plotCurveConfig->mathFunction == "Boxcar average") {
284  curveNameScaledMath = curveNameScaled + " (avg)";
285  } else if (plotCurveConfig->mathFunction == "Standard deviation") {
286  curveNameScaledMath = curveNameScaled + " (std)";
287  } else {
288  // Shouldn't be able to get here. Perhaps a new math function was added without
289  // updating this list?
290  Q_ASSERT(0);
291  }
292 
293  while (scopeGadgetWidget->getDataSources().keys().contains(curveNameScaledMath))
294  curveNameScaledMath = curveNameScaledMath + "*";
295 
296  // Create the curve plot
297  QwtPlotCurve *plotCurve = new QwtPlotCurve(curveNameScaledMath);
298  plotCurve->setPen(QPen(QBrush(QColor(color), Qt::SolidPattern), (qreal)1, Qt::SolidLine,
299  Qt::SquareCap, Qt::BevelJoin));
300  plotCurve->setSamples(*(scatterplotData->getXData()), *(scatterplotData->getYData()));
301  plotCurve->attach(scopeGadgetWidget);
302  scatterplotData->setCurve(plotCurve);
303 
304  // Keep the curve details for later
305  scopeGadgetWidget->insertDataSources(curveNameScaledMath, scatterplotData);
306 
307  // Connect the UAVO
308  scopeGadgetWidget->connectUAVO(obj);
309  }
310  scopeGadgetWidget->replot();
311 }
