181 bool FlightDataModel::setData(const QModelIndex &index, const QVariant &value, int role)
182 {
183  if (index.isValid() && role == Qt::EditRole) {
184  PathPlanData *row = dataStorage.at(index.row());
185 
186  // Do not allow changing any values except locked when the column is locked
187  if (row->locked && index.column() != (int)FlightDataModel::LOCKED)
188  return false;
189 
190  struct FlightDataModel::NED NED;
191  QModelIndex otherIndex;
192  switch (index.column()) {
193  case WPDESCRIPTION:
194  row->wpDescription = value.toString();
195  break;
196  case LATPOSITION:
197  row->latPosition = value.toDouble();
198  // Indicate this also changed the north
199  otherIndex = this->index(index.row(), FlightDataModel::NED_NORTH);
200  emit dataChanged(otherIndex, otherIndex);
201  break;
202  case LNGPOSITION:
203  row->lngPosition = value.toDouble();
204  // Indicate this also changed the east
205  otherIndex = this->index(index.row(), FlightDataModel::NED_EAST);
206  emit dataChanged(otherIndex, otherIndex);
207  break;
208  case ALTITUDE:
209  row->altitude = value.toDouble();
210  // Indicate this also changed the NED down
211  otherIndex = this->index(index.row(), FlightDataModel::NED_DOWN);
212  emit dataChanged(otherIndex, otherIndex);
213  break;
214  case NED_NORTH:
215  NED = getNED(row);
216  NED.North = value.toDouble();
217  setNED(row, NED);
218  // Indicate this also changed the latitude
219  otherIndex = this->index(index.row(), FlightDataModel::LATPOSITION);
220  emit dataChanged(otherIndex, otherIndex);
221  break;
222  case NED_EAST:
223  NED = getNED(index.row());
224  NED.East = value.toDouble();
225  setNED(row, NED);
226  // Indicate this also changed the longitude
227  otherIndex = this->index(index.row(), FlightDataModel::LNGPOSITION);
228  emit dataChanged(otherIndex, otherIndex);

232  NED.Down = value.toDouble();

234  // Indicate this also changed the altitude
235  otherIndex = this->index(index.row(), FlightDataModel::ALTITUDE);
236  emit dataChanged(otherIndex, otherIndex);
237  break;
238  case VELOCITY:
239  row->velocity = value.toFloat();
240  break;
241  case MODE:
242  row->mode = value.toInt();
243  break;
244  case MODE_PARAMS:

249  break;

287 {
288  PathPlanData *data;
289  beginInsertRows(QModelIndex(), row, row + count - 1);
290  for (int x = 0; x < count; ++x) {
291  // Initialize new internal representation
292  data = new PathPlanData;
293  data->latPosition = 0;
294  data->lngPosition = 0;
295 
296  // If there is a previous waypoint, initialize some of the fields to that value
297  if (rowCount() > 0) {

299  data->altitude = prevRow->altitude;

307  data->mode = Waypoint::MODE_VECTOR;
308  data->mode_params = 0;
309  data->locked = false;

459 void FlightDataModel::fixupValidationErrors()
460 {
461  struct FlightDataModel::NED prevNED = { 0.0, 0.0, 0.0 };
462 
463  // Skip validation when there's a download from firmware in process.
464  if (valPaused) {
465  return;
466  }
467 
468  for (int i = 0; i < rowCount(); i++) {
469  PathPlanData *row = dataStorage.at(i);
470 
471  bool dirty = false;
472 
473  struct FlightDataModel::NED thisNED = getNED(row);
474 
475  if (i == 0) {
476  switch (row->mode) {
477  case Waypoint::MODE_CIRCLELEFT:
478  case Waypoint::MODE_CIRCLERIGHT:
479  row->mode = Waypoint::MODE_VECTOR;
480 
481  showErrorDialog("Waypoint corrected",
482  "First waypoint may not be the endpoint of an arc");
483  dirty = true;
484 
485  break;
486  default:
487  break;
488  }
489  }
490 
491  double distance =
492  sqrt(pow(thisNED.North - prevNED.North, 2) + pow(thisNED.East - prevNED.East, 2));
493 
494  if (distance > 600) {
495  // If the distance is more than 600m, presume it's invalid.
496  // Distances larger than this begin to become problematic
497  // with local tangent plane approximation.
498  //
499  // If this happens, move the location of this waypoint to the
500  // previous location.
501 
502  thisNED = prevNED;
503  setNED(row, thisNED);
504  distance = 0;
505 
506  showErrorDialog("Waypoint corrected", "Over-long leg shortened.");
507 
508  dirty = true;
509  }
510 
511  switch (row->mode) {
512  case Waypoint::MODE_CIRCLELEFT:
513  case Waypoint::MODE_CIRCLERIGHT:
514  if (row->mode_params < (distance / 2 + 0.1f)) {
515  row->mode_params = distance / 2 + 0.2f;
516 
517  showErrorDialog("Waypoint corrected", "Radius of circle increased to minimum");
518 
519  dirty = true;
520  }
521 
522  break;
523  case Waypoint::MODE_CIRCLEPOSITIONLEFT:
524  case Waypoint::MODE_CIRCLEPOSITIONRIGHT:
525  if (row->mode_params < 0.5f) {
526  row->mode_params = 0.5f;
527 
528  showErrorDialog("Waypoint corrected", "Radius of circle increased to minimum");
529 
530  dirty = true;
531  } else if (row->mode_params > 300) {
532  row->mode_params = 300;
533 
534  showErrorDialog("Waypoint corrected", "Radius of circle decreased to maximum");
535 
536  dirty = true;
537  }
538 
539  break;
540 
541  default:
542  break;
543  }
544 
545  if (dirty) {
546  /* Let anyone listening know we changed it - Fire an event for the
547  * entire row being changed. (Because changing NED changes lots of
548  * columns) */
549  QModelIndex leftIndex, rightIndex;
550 
551  leftIndex = this->index(i, LATPOSITION);
552  rightIndex = this->index(i, LASTCOLUMN - 1);
553 
554  emit dataChanged(leftIndex, rightIndex);
555  }
556 
557  prevNED = thisNED;
558  }
559 }

565 void FlightDataModel::readFromFile(QString fileName)
566 {
567  double HomeLLA[3];
568 
569  removeRows(0, rowCount());
570  QFile file(fileName);
571  if (!file.open(QIODevice::ReadOnly)) {
572  showErrorDialog("Unable to open file", "Unable to open file");
573  return;
574  }
575 
576  QDomDocument doc("PathPlan");
577  QByteArray array = file.readAll();
578  QString error;
579 
580  file.close();
581 
582  if (!doc.setContent(array, &error)) {
583  showErrorDialog("File Parsing Failure", "This file is not a correct XML file");
584  return;
585  }
586 
587  QDomElement root = doc.documentElement();
588 
589  // First of all, read in the Home Location and reset it:
590 
591  if (root.isNull() || root.tagName() != "pathplan") {
592  showErrorDialog("Wrong file contents", "This is not a valid flight plan file");
593  return;
594  }
595 
596  PathPlanData *data = NULL;
597 
598  // First of all, find the Home location saved in the file
599  QDomNodeList hlist = root.elementsByTagName("homelocation");
600  if (hlist.length() != 1) {
601  showErrorDialog("Wrong file contents", "File format is incorrect (missing home location)");
602  return;
603  }
604 
605  QDomNode homelocField = hlist.at(0).firstChild();
606  while (!homelocField.isNull()) {
607  QDomElement field = homelocField.toElement();
608  if (field.tagName() == "field") {
609  if (field.attribute("name") == "latitude") {
610  HomeLLA[0] = field.attribute("value").toDouble();
611  } else if (field.attribute("name") == "longitude") {
612  HomeLLA[1] = field.attribute("value").toDouble();
613  } else if (field.attribute("name") == "altitude") {
614  HomeLLA[2] = field.attribute("value").toDouble();
615  }
616  }
617  homelocField = homelocField.nextSibling();
618  }
619 
620  // For now, reset home location to the location in the file.
621  // In a future revision, we should consider asking the user whether to remap the flight plan
622  // to the current home location or use the one in the flight plan
623  if (!setHomeLocation(HomeLLA)) {
624  showErrorDialog("Home location error", "Home location coordinates invalid");
625  return;
626  }
627 
628  hlist = root.elementsByTagName("waypoints");
629  if (hlist.length() != 1) {
630  showErrorDialog("Wrong file contents", "File format is incorrect (missing waypoints)");
631  }
632  QDomNode node = hlist.at(0).firstChild();
633  while (!node.isNull()) {
634  QDomElement e = node.toElement();
635  if (e.tagName() == "waypoint") {
636  QDomNode fieldNode = e.firstChild();
637  data = new PathPlanData;
638  double wpLLA[3];
639  double wpNED[3];
640  int params = 0;
641  while (!fieldNode.isNull()) {
642  QDomElement field = fieldNode.toElement();
643  if (field.tagName() == "field") {
644  if (field.attribute("name") == "down") {
645  wpNED[2] = field.attribute("value").toDouble();
646  params++;
647  } else if (field.attribute("name") == "description") {
648  data->wpDescription = field.attribute("value");
649  params++;
650  } else if (field.attribute("name") == "north") {
651  wpNED[0] = field.attribute("value").toDouble();
652  params++;
653  } else if (field.attribute("name") == "east") {
654  wpNED[1] = field.attribute("value").toDouble();
655  params++;
656  } else if (field.attribute("name") == "velocity") {
657  data->velocity = field.attribute("value").toFloat();
658  params++;
659  } else if (field.attribute("name") == "mode") {
660  data->mode = field.attribute("value").toInt();
661  params++;
662  } else if (field.attribute("name") == "mode_params") {
663  data->mode_params = field.attribute("value").toFloat();
664  params++;
665  } else if (field.attribute("name") == "is_locked") {
666  data->locked = field.attribute("value").toInt();
667  params++;
668  }
669  }
670  fieldNode = fieldNode.nextSibling();
671  }
672 
673  // We can't really check everything in the file is fully consistent, but
674  // at least we can make sure we have the right number of fields
675  if (params < 8) {
676  showErrorDialog("Waypoint error", "Waypoint coordinates invalid");
677  }
678 
679  Utils::CoordinateConversions().NED2LLA_HomeLLA(HomeLLA, wpNED, wpLLA);
680  data->latPosition = wpLLA[0];
681  data->lngPosition = wpLLA[1];
682  data->altitude = wpLLA[2];
683 
684  beginInsertRows(QModelIndex(), dataStorage.length(), dataStorage.length());
685  dataStorage.append(data);
686  endInsertRows();
687  }
688  node = node.nextSibling();
689  }
690 
691  fixupValidationErrors();
692 }
