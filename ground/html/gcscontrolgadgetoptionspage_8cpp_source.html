240 void GCSControlGadgetOptionsPage::apply()
241 {
242  m_config->setControlsMode(options_page->controlsMode->currentIndex() + 1);
243 
244  int roll = -1, pitch = -1, yaw = -1, throttle = -1;
245  for (int i = 0; i < chList.length(); i++) {
246  switch (chList.at(i)->currentIndex()) {
247  case 1:
248  roll = i;
249  break;
250  case 2:
251  pitch = i;
252  break;
253  case 3:
254  yaw = i;
255  break;
256  case 4:
257  throttle = i;
258  break;
259  }
260  }
261  m_config->setRPYTchannels(roll, pitch, yaw, throttle);
262 
263  m_config->setUDPControlSettings(options_page->udp_port->text().toInt(),
264  options_page->udp_host->text());
265 
266  m_config->setGcsReceiverMode(options_page->gcsReceiverCB->checkState());
267 
268  for (unsigned int j = 0; j < 8; j++) {
269  m_config->setbuttonSettingsAction(j, buttonActionList.at(j)->currentIndex());
270  m_config->setbuttonSettingsFunction(j, buttonFunctionList.at(j)->currentIndex());
271  m_config->setbuttonSettingsAmount(j, buttonValueList.at(j)->value());
272  m_config->setChannelReverse(j, chRevList.at(j)->isChecked());
273  }
274 }
