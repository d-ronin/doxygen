117 void PathCompiler::doAddWaypoint(waypoint newWaypointInternal, int /*position*/)
118 {
119  /* TODO: If a waypoint is inserted not at the end shift them all by one and */
120  /* add the data there */
121 
122  UAVObjectManager *objManager = getObjectManager();
123 
124  // Format the data from the map into a UAVO
125  Waypoint::DataFields newWaypoint = InternalToUavo(newWaypointInternal);
126 
127  // Search for any waypoints set to stop, because if they exist we
128  // should add the waypoint immediately after that one
129  int numWaypoints = objManager->getNumInstances(Waypoint::OBJID);
130  int i;
131  for (i = 0; i < numWaypoints; i++) {
132  Waypoint *waypoint = Waypoint::GetInstance(objManager, i);
133  Q_ASSERT(waypoint);
134  if(waypoint == NULL)
135  return;
136 
137  Waypoint::DataFields waypointData = waypoint->getData();
138  if(waypointData.Action == Waypoint::ACTION_STOP) {
139  waypointData.Action = Waypoint::ACTION_PATHTONEXT;
140  waypoint->setData(waypointData);
141  waypoint->updated();
142  break;
143  }
144  }
145 
146  if (i >= numWaypoints - 1) {
147  // We reached end of list so new waypoint needs to be registered
148 
149  Waypoint *waypoint = new Waypoint();
150  Q_ASSERT(waypoint);
151  if (waypoint) {
152  // Register a new waypoint instance
153  quint32 newInstId = objManager->getNumInstances(waypoint->getObjID());
154  waypoint->initialize(newInstId,waypoint->getMetaObject());
155  objManager->registerObject(waypoint);
156 
157  // Set the data in the new object
158  waypoint->setData(newWaypoint);
159  waypoint->updated();
160  }
161  } else {
162  Waypoint *waypoint = Waypoint::GetInstance(objManager, i + 1);
163  Q_ASSERT(waypoint);
164  if (waypoint) {
165  waypoint->setData(newWaypoint);
166  waypoint->updated();
167  }
168  }
169 }

174 void PathCompiler::doUpdateWaypoints(PathCompiler::waypoint changedWaypoint, int position)
175 {
176  int numWaypoints = getObjectManager()->getNumInstances(Waypoint::OBJID);
177  Q_ASSERT(position < numWaypoints);
178  if (position >= numWaypoints)
179  return;
180 
181  Waypoint *waypointInst = Waypoint::GetInstance(getObjectManager(), position);
182  Q_ASSERT(waypointInst);
183 
184  // Mirror over the updated position. We don't just use the changedWaypoint
185  // because things like action might need to be preserved
186  Waypoint::DataFields changedWaypointUAVO = InternalToUavo(changedWaypoint);
187  Waypoint::DataFields oldWaypointUAVO = waypointInst->getData();
188  oldWaypointUAVO.Position[0] = changedWaypointUAVO.Position[0];
189  oldWaypointUAVO.Position[1] = changedWaypointUAVO.Position[1];
190  // Don't take the altitude from the map for now
191 
192  waypointInst->setData(oldWaypointUAVO);
193  waypointInst->updated();
194 }

200 void PathCompiler::doDelWaypoint(int index)
201 {
202  // This method is awkward because there is no support
203  // on the FC for actually deleting a waypoint. We need
204  // to shift them all by one and set the new "last" waypoint
205  // to a stop action
206 
207  UAVObjectManager *objManager = getObjectManager();
208  Waypoint *waypoint = Waypoint::GetInstance(objManager);
209  Q_ASSERT(waypoint);
210  if (!waypoint)
211  return;
212 
213  int numWaypoints = objManager->getNumInstances(waypoint->getObjID());
214  for (int i = index; i < numWaypoints - 1; i++) {
215  Waypoint *waypointDest = Waypoint::GetInstance(objManager, i);
216  Q_ASSERT(waypointDest);
217 
218  Waypoint *waypointSrc = Waypoint::GetInstance(objManager, i + 1);
219  Q_ASSERT(waypointSrc);
220 
221  if (!waypointDest || !waypointSrc)
222  return;
223 
224  // Copy the data down an index
225  Waypoint::DataFields waypoint = waypointSrc->getData();
226  waypointDest->setData(waypoint);
227  waypointDest->updated();
228  }
229 
230  // Set the second to last waypoint to stop (and last for safety)
231  // the functional equivalent to deleting
232  for (int i = numWaypoints - 2; i < numWaypoints; i++) {
233  waypoint = Waypoint::GetInstance(objManager, i);
234  Q_ASSERT(waypoint);
235  if (waypoint) {
236  Waypoint::DataFields waypointData = waypoint->getData();
237  waypointData.Action = Waypoint::ACTION_STOP;
238  waypoint->setData(waypointData);
239  waypoint->updated();
240  }
241  }
242 }

272 void PathCompiler::doUpdateFromUAV(UAVObject *obj)
273 {
274  UAVObjectManager *objManager = getObjectManager();
275  if (!objManager)
276  return;
277 
278  Waypoint *waypointObj = Waypoint::GetInstance(getObjectManager());
279  Q_ASSERT(waypointObj);
280  if (waypointObj == NULL)
281  return;
282 
283  /* Get all the waypoints from the UAVO and create a representation for the visualization */
284  QList <waypoint> waypoints;
285  waypoints.clear();
286  int numWaypoints = objManager->getNumInstances(waypointObj->getObjID());
287  bool stopped = false;
288  for (int i = 0; i < numWaypoints && !stopped; i++) {
289  Waypoint *waypoint = Waypoint::GetInstance(objManager, i);
290  Q_ASSERT(waypoint);
291  if(waypoint == NULL)
292  return;
293 
294  Waypoint::DataFields waypointData = waypoint->getData();
295  waypoints.append(UavoToInternal(waypointData));
296  stopped = waypointData.Action == Waypoint::ACTION_STOP;
297  }
298 
299  if (previousWaypoints != waypoints) {
300  /* Because the waypoints have to update periodically (or we miss new ones on the FC */
301  /* side - needs telem fix) we want to filter updates to map that are simply periodic */
302  previousWaypoints = waypoints;
303 
304  /* Inform visualization */
305  emit visualizationChanged(waypoints);
306  }
307 }
