191 bool ShortcutSettings::filter(const QString &f, const QTreeWidgetItem *item)
192 {
193  if (item->childCount() == 0) {
194  if (f.isEmpty())
195  return false;
196  for (int i = 0; i < item->columnCount(); ++i) {
197  if (item->text(i).contains(f, Qt::CaseInsensitive))
198  return false;
199  }
200  return true;
201  }
202 
203  bool found = false;
204  for (int i = 0; i < item->childCount(); ++i) {
205  QTreeWidgetItem *citem = item->child(i);
206  if (filter(f, citem)) {
207  citem->setHidden(true);
208  } else {
209  citem->setHidden(false);
210  found = true;
211  }
212  }
213  return !found;
214 }

276 void ShortcutSettings::handleKeyEvent(QKeyEvent *e)
277 {
278  int nextKey = e->key();
279  if ( m_keyNum > 3 ||
280  nextKey == Qt::Key_Control ||
281  nextKey == Qt::Key_Shift ||
282  nextKey == Qt::Key_Meta ||
283  nextKey == Qt::Key_Alt )
284  return;
285 
286  nextKey |= translateModifiers(e->modifiers(), e->text());
287  switch (m_keyNum) {
288  case 0:
289  m_key[0] = nextKey;
290  break;
291  case 1:
292  m_key[1] = nextKey;
293  break;
294  case 2:
295  m_key[2] = nextKey;
296  break;
297  case 3:
298  m_key[3] = nextKey;
299  break;
300  default:
301  break;
302  }
303  m_keyNum++;
304  QKeySequence ks(m_key[0], m_key[1], m_key[2], m_key[3]);
305  m_page->shortcutEdit->setText(ks.toString());
306  e->accept();
307 }

309 int ShortcutSettings::translateModifiers(Qt::KeyboardModifiers state,
310  const QString &text)
311 {
312  int result = 0;
313  // The shift modifier only counts when it is not used to type a symbol
314  // that is only reachable using the shift key anyway
315  if ((state & Qt::ShiftModifier) && (text.size() == 0
316  || !text.at(0).isPrint()
317  || text.at(0).isLetter()
318  || text.at(0).isSpace()))
319  result |= Qt::SHIFT;
320  if (state & Qt::ControlModifier)
321  result |= Qt::CTRL;
322  if (state & Qt::MetaModifier)
323  result |= Qt::META;
324  if (state & Qt::AltModifier)
325  result |= Qt::ALT;
326  return result;
327 }
