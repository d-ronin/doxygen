196 void ConfigModuleWidget::objectUpdated(UAVObject *obj, bool success)
197 {
198  if (!obj)
199  return;
200 
201  ModuleSettings *moduleSettings =
202  qobject_cast<ModuleSettings *>(getObjectManager()->getObject(ModuleSettings::NAME));
203  Q_ASSERT(moduleSettings);
204  TaskInfo *taskInfo = qobject_cast<TaskInfo *>(getObjectManager()->getObject(TaskInfo::NAME));
205  bool enableHott = (taskInfo && taskInfo->getIsPresentOnHardware())
206  ? (taskInfo->getRunning_UAVOHoTTBridge() == TaskInfo::RUNNING_TRUE)
207  : true;
208  bool enableGps = (taskInfo && taskInfo->getIsPresentOnHardware())
209  ? (taskInfo->getRunning_GPS() == TaskInfo::RUNNING_TRUE)
210  : true;
211  enableGps &= moduleSettings->getIsPresentOnHardware();
212 
213  enableGpsTab(enableGps);
214 
215  QString objName = obj->getName();
216  if (objName.compare(AirspeedSettings::NAME) == 0) {
217  enableAirspeedTab(success
218  && moduleSettings->getAdminState_Airspeed()
219  == ModuleSettings::ADMINSTATE_ENABLED);
220  } else if (objName.compare(FlightBatterySettings::NAME) == 0) {
221  enableBatteryTab(success
222  && moduleSettings->getAdminState_Battery()
223  == ModuleSettings::ADMINSTATE_ENABLED);
224  refreshAdcNames();
225  } else if (objName.compare(HoTTSettings::NAME) == 0) {
226  enableHoTTTelemetryTab(success && enableHott);
227  } else if (objName.compare(LoggingSettings::NAME) == 0) {
228  enableLoggingTab(success
229  && moduleSettings->getAdminState_Logging()
230  == ModuleSettings::ADMINSTATE_ENABLED);
231  }
232 }
