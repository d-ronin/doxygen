73 void UAVGadgetInstanceManager::readSettings(QSettings *qs)
74 {
75  while (!m_configurations.isEmpty()) {
76  emit configurationToBeDeleted(m_configurations.takeLast());
77  }
78  qs->beginGroup("UAVGadgetConfigurations");
79  UAVConfigInfo configInfo(qs);
80  configInfo.setNameOfConfigurable("UAVGadgetConfigurations");
81  if (configInfo.version() == UAVConfigVersion()) {
82  // If version is not set, assume its a old version before readable config (1.0.0).
83  // however compatibility to 1.0.0 is broken.
84  configInfo.setVersion("1.0.0");
85  }
86 
87  if (configInfo.version() == UAVConfigVersion("1.1.0")) {
88  configInfo.notify(tr("Migrating UAVGadgetConfigurations from version 1.1.0 to ")
89  + m_versionUAVGadgetConfigurations.toString());
90  readConfigs_1_1_0(qs); // this is fully compatible with 1.2.0
91  } else if (!configInfo.standardVersionHandlingOK(m_versionUAVGadgetConfigurations)) {
92  // We are in trouble now. User wants us to quit the import, but when he saves
93  // the GCS, his old config will be lost.
94  configInfo.notify(tr("You might want to save your old config NOW since it might be "
95  "replaced by broken one when you exit the GCS!"));
96  } else {
97  readConfigs_1_2_0(qs);
98  }
99 
100  qs->endGroup();
101  createOptionsPages();
102 }

112  QStringList configs = QStringList();
113 
114  configs = qs->childGroups();

125  int idx = indexForConfig(m_configurations, classId, configName);
126  if (idx >= 0) {
127  // We should replace the config, but it might be used, so just

129  m_configurations[idx] = config;
130  } else {
131  m_configurations.append(config);
132  }
133  }
134  qs->endGroup();
135  qs->endGroup();
136  }
137 
138  // In case no configuration settings were found, try to create a default configuration
139  if (configs.count() == 0) {
140  IUAVGadgetConfiguration *config = f->createConfiguration(nullptr, nullptr);
141  // It is not mandatory for uavgadgets to have a configuration (e.g. settings),
142  // therefore we have to check that "config" exists.

144  config->setName(tr("default"));

222 void UAVGadgetInstanceManager::createOptionsPages()
223 {
224  // In case there are pages (import a configuration), remove them.
225  // Maybe they should be deleted as well (memory-leak),
226  // but this might lead to NULL-pointers?
227  while (!m_optionsPages.isEmpty()) {
228  m_pm->removeObject(m_optionsPages.takeLast());
229  }
230 
231  QMutableListIterator<IUAVGadgetConfiguration *> ite(m_configurations);
232  while (ite.hasNext()) {
233  IUAVGadgetConfiguration *config = ite.next();
234  IUAVGadgetFactory *f = factory(config->classId());
235  IOptionsPage *p = f->createOptionsPage(config);
236  if (p) {
237  emit splashMessages(QString(tr("Loading %1 plugin options page for configuration %2")
238  .arg(config->classId())
239  .arg(config->name())));
240  IOptionsPage *page =
241  new UAVGadgetOptionsPageDecorator(p, config, f->isSingleConfigurationGadget());
242  page->setIcon(f->icon());
243  m_optionsPages.append(page);
244  m_pm->addObject(page);
245  } else {
246  qWarning() << "UAVGadgetInstanceManager::createOptionsPages - failed to create options "
247  "page for configuration "
248  + config->classId() + ":" + config->name() + ", configuration will be removed.";
249  // The m_optionsPages list and m_configurations list must be in synch otherwise nasty
250  // issues happen later
251  // so if we fail to create an options page we must remove the associated configuration
252  ite.remove();
253  }
254  }

358 QString UAVGadgetInstanceManager::suggestName(QString classId, QString name)
359 {
360  QString suggestedName;
361 
362  QString last = name.split(" ").last();
363  bool ok;
364  int num = last.toInt(&ok);
365  int i = 1;
366  if (ok) {
367  QStringList n = name.split(" ");
368  n.removeLast();
369  name = n.join(" ");
370  i = num + 1;
371  }
372  do {
373  suggestedName = name + " " + QString::number(i);
374  ++i;
375  } while (m_takenNames[classId].contains(suggestedName));
376 
377  m_takenNames[classId].append(suggestedName);
378  return suggestedName;

381 void UAVGadgetInstanceManager::applyChanges(IUAVGadgetConfiguration *config)
382 {

384  m_provisionalDeletes.removeAt(m_provisionalDeletes.indexOf(config));
385  int i = m_configurations.indexOf(config);
386  if (i >= 0) {
387  emit configurationToBeDeleted(config);
388  int j = m_takenNames[config->classId()].indexOf(config->name());
389  m_takenNames[config->classId()].removeAt(j);
390  m_pm->removeObject(m_optionsPages.at(i));

392  m_optionsPages.removeAt(i); // TODO delete

394  return;

397  emit configurationNameChanged(config, config->name(), config->provisionalName());
398  config->setName(config->provisionalName());
399  }
400  if (m_configurations.contains(config)) {
401  emit configurationChanged(config);
402  } else if (m_provisionalConfigs.contains(config)) {
403  emit configurationAdded(config);
404  int i = m_provisionalConfigs.indexOf(config);
405  IOptionsPage *page = m_provisionalOptionsPages.at(i);
406  m_configurations.append(config);
407  m_optionsPages.append(page);
408  m_provisionalConfigs.removeAt(i);
409  m_provisionalOptionsPages.removeAt(i);
410  m_pm->addObject(page);
411  }

414 void UAVGadgetInstanceManager::configurationNameEdited(QString text, bool hasText)

425  }
426  }

430  qDebug() << "An identically named provisional and normal configuration: "

432  << "." << d->provisionalName();
433  disable = true;
434  }
435  }
436  }
437  foreach (IUAVGadgetConfiguration *c, m_provisionalConfigs) {
438  foreach (IUAVGadgetConfiguration *d, m_provisionalConfigs) {
439  if (c != d && c->classId() == d->classId()
440  && c->provisionalName() == d->provisionalName()) {

444  disable = true;
445  }
446  }
447  }
448  if (hasText && text == "")
449  disable = true;
450  m_settingsDialog->disableApplyOk(disable);
