131 bool ConnectionManager::connectDevice(DevListItem device)
132 {
133  if (!device.connection) {
134  return false;
135  }
136 
137  QIODevice *io_dev = device.connection->openDevice(device.device);
138  if (!io_dev) {
139  return false;
140  }
141 
142  io_dev->open(QIODevice::ReadWrite);
143 
144  // check if opening the device worked
145  if (!io_dev->isOpen()) {
146  return false;
147  }
148 
149  // The device is connected make it the one selected on the dropbox
150  for (int x = 0; x < m_availableDevList->count(); ++x) {
151  if (device.getConName() == m_availableDevList->itemData(x, Qt::ToolTipRole).toString())
152  m_availableDevList->setCurrentIndex(x);
153  }
154  // we appear to have connected to the device OK
155  // remember the connection/device details
156  m_connectionDevice = device;
157  m_ioDev = io_dev;
158 
159  connect(m_connectionDevice.connection, SIGNAL(destroyed(QObject *)), this,
160  SLOT(onConnectionDestroyed(QObject *)), Qt::QueuedConnection);
161 
162  // signal interested plugins that we connected to the device
163  emit deviceConnected(io_dev);
164  m_connectBtn->setText("Disconnect");
165  m_availableDevList->setEnabled(false);
166 
167  // tell the monitorwidget we're conneced
168  m_monitorWidget->connected();
169 
170  return true;
171 }

177 bool ConnectionManager::disconnectDevice()
178 {
179  // tell the monitor widget we're disconnected
180  m_monitorWidget->disconnect();
181 
182  if (!m_ioDev) {
183  // apparently we are already disconnected: this can
184  // happen if a plugin tries to force a disconnect whereas
185  // we are not connected. Just return.
186  return false;
187  }
188 
189  // We are connected - disconnect from the device
190 
191  // stop our timers
192  if (reconnect->isActive())
193  reconnect->stop();
194  if (reconnectCheck->isActive())
195  reconnectCheck->stop();
196 
197  // signal interested plugins that user is disconnecting his device
198  emit deviceAboutToDisconnect();
199 
200  try {
201  if (m_connectionDevice.connection) {
202  m_connectionDevice.connection->closeDevice(m_connectionDevice.getConName());
203  }
204  } catch (...) { // handle exception
205  qDebug() << "Exception: m_connectionDevice.connection->closeDevice("
206  << m_connectionDevice.getConName() << ")";
207  }
208 
209  m_connectionDevice.connection = NULL;
210  m_ioDev = NULL;
211 
212  emit deviceDisconnected();
213  m_connectBtn->setText("Connect");
214  m_availableDevList->setEnabled(true);
215 
216  return true;
217 }

407 void ConnectionManager::updateConnectionList(IConnection *connection)
408 {
409  // Get the updated list of devices
410  QList<IDevice *> availableDev = connection->availableDevices();
411 
412  // Go through the list of connections of that type. If they are not in the
413  // available device list then remove them. If they are connected, then
414  // disconnect them.
415  for (QLinkedList<DevListItem>::iterator iter = m_devList.begin(); iter != m_devList.end();) {
416  if (iter->connection != connection) {
417  ++iter;
418  continue;
419  }
420 
421  // See if device exists in the updated availability list
422  bool found = availableDev.contains(iter->device);
423  if (!found) {
424  // we are currently using the one we are about to erase
425  if (m_connectionDevice.connection && m_connectionDevice.connection == connection
426  && m_connectionDevice.device == iter->device) {
427  disconnectDevice();
428  }
429 
430  // We have to delete the IDevice in that DevListItem before getting rid
431  // of the iter itself
432  if (!iter->device.isNull())
433  iter->device->deleteLater();
434 
435  iter = m_devList.erase(iter);
436  } else
437  ++iter;
438  }
439 
440  // Add any back to list that don't exist
441  foreach (IDevice *dev, availableDev) {
442  bool found = m_devList.contains(DevListItem(connection, dev));
443  if (!found) {
444  registerDevice(connection, dev);
445  }
446  }
447 }
