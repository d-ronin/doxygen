135 void GCSControlGadget::manualControlCommandUpdated(UAVObject *obj)
136 {
137 
138  // Not sending then show updates from transmitter
139  if (enableSending)
140  return;
141 
142  double roll = obj->getField("Roll")->getDouble();
143  double pitch = obj->getField("Pitch")->getDouble();
144  double yaw = obj->getField("Yaw")->getDouble();
145  double throttle = obj->getField("Throttle")->getDouble();
146  // Remap RPYT to left X/Y and right X/Y depending on mode

150  emit sticksChangedRemotely(yaw, -pitch, roll, throttle);

152  case 2:
153  // Mode 2: LeftX = Yaw, LeftY = Throttle, RightX = Roll, RightY = Pitch
154  emit sticksChangedRemotely(yaw, throttle, roll, -pitch);
155  break;
156  case 3:
157  // Mode 3: LeftX = Roll, LeftY = Pitch, RightX = Yaw, RightY = Throttle
158  emit sticksChangedRemotely(roll, -pitch, yaw, throttle);
159  break;
160  case 4:
161  // Mode 4: LeftX = Roll, LeftY = Throttle, RightX = Yaw, RightY = Pitch;

178 void GCSControlGadget::setGcsReceiver(double leftX, double leftY, double rightX, double rightY)
179 {
180  GCSControl *ctr = getGcsControl();
181  Q_ASSERT(ctr);
182  if (ctr == NULL)
183  return;
184 
185  double newRoll = 0;
186  double newPitch = 0;
187  double newYaw = 0;
188  double newThrottle = 0;
189 
190  // Remap left X/Y and right X/Y to RPYT depending on mode
191  switch (controlsMode) {
192  case 1:
193  // Mode 1: LeftX = Yaw, LeftY = Pitch, RightX = Roll, RightY = Throttle
194  newRoll = rightX;
195  newPitch = -leftY;
196  newYaw = leftX;
197  newThrottle = rightY;
198  break;
199  case 2:
200  // Mode 2: LeftX = Yaw, LeftY = Throttle, RightX = Roll, RightY = Pitch
201  newRoll = rightX;
202  newPitch = -rightY;
203  newYaw = leftX;
204  newThrottle = leftY;
205  break;
206  case 3:
207  // Mode 3: LeftX = Roll, LeftY = Pitch, RightX = Yaw, RightY = Throttle
208  newRoll = leftX;
209  newPitch = -leftY;
210  newYaw = rightX;
211  newThrottle = rightY;
212  break;
213  case 4:
214  // Mode 4: LeftX = Roll, LeftY = Throttle, RightX = Yaw, RightY = Pitch;
215  newRoll = leftX;
216  newPitch = -rightY;
217  newYaw = rightX;
218  newThrottle = leftY;
219  break;
220  }
221  ctr->setThrottle(newThrottle);
222  ctr->setRoll(newRoll);
223  ctr->setPitch(newPitch);
224  ctr->setYaw(newYaw);
225 
226  switch (controlsMode) {
227  case 1:
228  // Mode 1: LeftX = Yaw, LeftY = Pitch, RightX = Roll, RightY = Throttle
229  emit sticksChangedRemotely(newYaw, -newPitch, newRoll, newThrottle);
230  break;
231  case 2:
232  // Mode 2: LeftX = Yaw, LeftY = Throttle, RightX = Roll, RightY = Pitch
233  emit sticksChangedRemotely(newYaw, newThrottle, newRoll, -newPitch);
234  break;
235  case 3:
236  // Mode 3: LeftX = Roll, LeftY = Pitch, RightX = Yaw, RightY = Throttle
237  emit sticksChangedRemotely(newRoll, -newPitch, newYaw, newThrottle);
238  break;
239  case 4:
240  // Mode 4: LeftX = Roll, LeftY = Throttle, RightX = Yaw, RightY = Pitch;
241  emit sticksChangedRemotely(newRoll, newThrottle, newYaw, -newPitch);
242  break;
243  }

257 void GCSControlGadget::readUDPCommand()
258 {
259  double pitch = 0, yaw = 0, roll = 0, throttle = 0;
260  while (control_sock->hasPendingDatagrams()) {
261  QByteArray datagram;
262  datagram.resize(control_sock->pendingDatagramSize());
263  control_sock->readDatagram(datagram.data(), datagram.size());
264  QDataStream readData(datagram);
265  bool badPack = false;
266  int state = 0;
267  while (!readData.atEnd() && !badPack) {
268  double buffer;
269  readData >> buffer;
270  switch (state) {
271  case 0:
272  if (buffer == 42) {
273  state = 1;
274  } else {
275  state = 0;
276  badPack = true;
277  }
278  break;
279  case 1:
280  pitch = buffer;
281  state = 2;
282  break;
283  case 2:
284  yaw = buffer;
285  state = 3;
286  break;
287  case 3:
288  roll = buffer;
289  state = 4;
290  break;
291  case 4:
292  throttle = buffer;
293  state = 5;
294  break;
295  case 5:
296  if (buffer != 36 || !readData.atEnd())
297  badPack = true;
298  break;
299  }
300  }
301  if (!badPack && ((GCSControlGadgetWidget *)m_widget)->getUDPControl()) {
302  ManualControlCommand *obj = getManualControlCommand();
303  bool update = false;
304 
305  if (pitch != obj->getField("Pitch")->getDouble()) {
306  obj->getField("Pitch")->setDouble(constrain(pitch));
307  update = true;
308  }
309  if (yaw != obj->getField("Yaw")->getDouble()) {
310  obj->getField("Yaw")->setDouble(constrain(yaw));
311  update = true;
312  }
313  if (roll != obj->getField("Roll")->getDouble()) {
314  obj->getField("Roll")->setDouble(constrain(roll));
315  update = true;
316  }
317  if (throttle != obj->getField("Throttle")->getDouble()) {
318  obj->getField("Throttle")->setDouble(constrain(throttle));
319  update = true;
320  }
321  if (update)
322  obj->updated();
323  }
324  }
325 
326  qDebug() << "Pitch: " << pitch << " Yaw: " << yaw << " Roll: " << roll
327  << " Throttle: " << throttle;
328 }

441 void GCSControlGadget::axesValues(QListInt16 values)
442 {
443  int chMax = values.length();
444  if (rollChannel >= chMax || pitchChannel >= chMax || yawChannel >= chMax
445  || throttleChannel >= chMax) {
446  qDebug()
447  << "GCSControl: configuration is inconsistent with current joystick! Aborting update.";
448  return;
449  }
450 
451  double rValue = (rollChannel > -1) ? values[rollChannel] : 0;
452  double pValue = (pitchChannel > -1) ? values[pitchChannel] : 0;
453  double yValue = (yawChannel > -1) ? values[yawChannel] : 0;
454  double tValue = (throttleChannel > -1) ? values[throttleChannel] : 0;
455  double max = 32767;
456 
457  if (rollChannel > -1)
458  if (channelReverse[rollChannel] == true)
459  rValue = -rValue;
460  if (pitchChannel > -1)
461  if (channelReverse[pitchChannel] == true)
462  pValue = -pValue;
463  if (yawChannel > -1)
464  if (channelReverse[yawChannel] == true)
465  yValue = -yValue;
466  if (throttleChannel > -1)
467  if (channelReverse[throttleChannel] == true)
468  tValue = -tValue;
469 
470  if (joystickTime.elapsed() > JOYSTICK_UPDATE_RATE) {
471  joystickTime.restart();
472  // Remap RPYT to left X/Y and right X/Y depending on mode
473  // Mode 1: LeftX = Yaw, LeftY = Pitch, RightX = Roll, RightY = Throttle
474  // Mode 2: LeftX = Yaw, LeftY = THrottle, RightX = Roll, RightY = Pitch
475  // Mode 3: LeftX = Roll, LeftY = Pitch, RightX = Yaw, RightY = Throttle
476  // Mode 4: LeftX = Roll, LeftY = Throttle, RightX = Yaw, RightY = Pitch;
477  switch (controlsMode) {
478  case 1:
479  sticksChangedLocally(yValue / max, -pValue / max, rValue / max, -tValue / max);
480  break;
481  case 2:
482  sticksChangedLocally(yValue / max, -tValue / max, rValue / max, -pValue / max);
483  break;
484  case 3:
485  sticksChangedLocally(rValue / max, -pValue / max, yValue / max, -tValue / max);
486  break;
487  case 4:
488  sticksChangedLocally(rValue / max, -tValue / max, yValue / max, -pValue / max);
489  break;
490  }
491  }
492 }
