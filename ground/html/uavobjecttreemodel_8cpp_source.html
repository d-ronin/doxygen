173 void UAVObjectTreeModel::addDataObject(UAVDataObject *obj, bool categorize)
174 {
175  // Determine if the root tree is the settings or dynamic data tree
176  TopTreeItem *root = obj->isSettings() ? m_settingsTree : m_nonSettingsTree;
177 
178  TreeItem *parent = root;
179 
180  if (categorize && obj->getCategory() != 0 && !obj->getCategory().isEmpty()) {
181  QStringList categoryPath = obj->getCategory().split('/');

283 void UAVObjectTreeModel::addSingleField(int index, UAVObjectField *field, TreeItem *parent)
284 {
285  QList<QVariant> data;
286  if (field->getNumElements() == 1)
287  data.append(field->getName());
288  else
289  data.append(QString("[%1]").arg((field->getElementNames())[index]));
290 
291  FieldTreeItem *item = NULL;
292  UAVObjectField::FieldType type = field->getType();
293  switch (type) {
294  case UAVObjectField::BITFIELD:
295  case UAVObjectField::ENUM: {
296  QStringList options = field->getOptions();
297  QVariant value = field->getValue();
298  data.append(options.indexOf(value.toString()));
299  data.append(field->getUnits());
300  item = new EnumFieldTreeItem(field, index, data);
301  break;
302  }
303  case UAVObjectField::INT8:
304  case UAVObjectField::INT16:
305  case UAVObjectField::INT32:
306  case UAVObjectField::UINT8:
307  case UAVObjectField::UINT16:
308  case UAVObjectField::UINT32:
309  data.append(field->getValue(index));
310  data.append(field->getUnits());
311  item = new IntFieldTreeItem(field, index, data);
312  break;
313  case UAVObjectField::FLOAT32:
314  data.append(field->getValue(index));
315  data.append(field->getUnits());
316  item = new FloatFieldTreeItem(field, index, data, m_useScientificFloatNotation);
317  break;
318  default:
319  Q_ASSERT(false);
320  }
321  item->setDescription(field->getDescription());
322  item->setHighlightManager(m_highlightManager);
323  parent->appendChild(item);
324 }

423 QVariant UAVObjectTreeModel::data(const QModelIndex &index, int role) const
424 {
425  if (!index.isValid())
426  return QVariant();
427 
428  if (index.column() == TreeItem::dataColumn && role == Qt::EditRole) {
429  TreeItem *item = static_cast<TreeItem *>(index.internalPointer());
430  return item->data(index.column());
431  }
432 
433  if (role == Qt::ToolTipRole) {
434  TreeItem *item = static_cast<TreeItem *>(index.internalPointer());
435  return item->description();
436  }
437 
438  TreeItem *item = static_cast<TreeItem *>(index.internalPointer());
439 
440  if (index.column() == 0 && role == Qt::BackgroundRole) {
441  if (!dynamic_cast<TopTreeItem *>(item) && item->highlighted())
442  return QVariant(m_recentlyUpdatedColor);
443  if (!dynamic_cast<TopTreeItem *>(item) && item->updatedOnly())
444  return QVariant(m_updatedOnlyColor);
445  }
446 
447  if (role == Qt::TextColorRole) {
448  if (item) {
449  if (item->getIsPresentOnHardware())
450  return QVariant(m_isPresentOnHwColor);
451  else
452  return QVariant(m_notPresentOnHwColor);
453  }
454  }
455 
456  if (role == Qt::FontRole) {
457  if (!item->isDefaultValue())
458  return m_nonDefaultValueFont;
459  return m_defaultValueFont;
460  }
461 
462  if (index.column() == TreeItem::dataColumn && role == Qt::BackgroundRole) {
463  FieldTreeItem *fieldItem = dynamic_cast<FieldTreeItem *>(item);
464  if (fieldItem && fieldItem->highlighted())
465  return QVariant(m_recentlyUpdatedColor);
466 
467  if (fieldItem && fieldItem->changed())
468  return QVariant(m_manuallyChangedColor);
469 
470  if (fieldItem && fieldItem->updatedOnly())
471  return QVariant(m_updatedOnlyColor);
472  }
473 
474  if (role != Qt::DisplayRole)
475  return QVariant();
476 
477  if (index.column() == TreeItem::dataColumn) {
478  EnumFieldTreeItem *fieldItem = dynamic_cast<EnumFieldTreeItem *>(item);
479  if (fieldItem) {
480  int enumIndex = fieldItem->data(index.column()).toInt();
481  return fieldItem->enumOptions(enumIndex);
482  }
483  IntFieldTreeItem *intFieldItem = dynamic_cast<IntFieldTreeItem *>(item);
484  if (intFieldItem)
485  return intFieldItem->formattedData();
486  }
487 
488  return item->data(index.column());
489 }
