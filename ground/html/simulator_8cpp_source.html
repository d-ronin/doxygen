207 void Simulator::receiveUpdate()
208 {

210  simTimer->setInterval(simTimeout);
211  simTimer->stop();
212  simTimer->start();
213  if ( !simConnectionStatus )

218 
219  // Process data

221  // Receive datagram

224  QHostAddress sender;

234 void Simulator::setupUAVObjects()

236  UAVObjectUtilManager* utilMngr = getObjectUtilManager();
237 
238  // Store original metadata, but only if it hasn't already been set
239  // by another class instance.

241  originalMetaData = utilMngr->readAllNonSettingsMetadata();
242  }
243 
244  quint16 slowUpdate = 5000; // in [ms]
245 
246  // Iterate over list of UAVObjects, setting all dynamic data metadata objects to slow update rate.
247  UAVObjectManager *objManager = getObjectManager();
248  QVector< QVector<UAVDataObject*> > objList = objManager->getDataObjectsVector();
249  foreach (QVector<UAVDataObject*> list, objList) {
250  foreach (UAVDataObject* obj, list) {
251  if(!obj->isSettings()) {
252  UAVObject::Metadata mdata = obj->getMetadata();
253  UAVObject::SetFlightTelemetryUpdateMode(mdata, UAVObject::UPDATEMODE_PERIODIC);
254 
255  // Default configuration for all UAVO data rates is "slowUpdate". Add a random factor
256  // so that not all UAVOs update at the same time, which otherwise congests the bus.
257  mdata.flightTelemetryUpdatePeriod = slowUpdate + (rand() % 30);
258 
259  metaDataList.insert(obj->getName(), mdata);
260  }
261  }
262  }
263 
264  //-----------------------------------------//
265  // Configure metadata for individual UAVOs //
266  //-----------------------------------------//
267

270  if (settings.simulatorId == "FG" ||

273  setupInputObject(actDesired, settings.minOutputPeriod);
274  }

277  setupOutputObject(gcsReceiver, settings.minOutputPeriod);
278  } else if (settings.manualControlEnabled) {
279  setupOutputObject(manCtrlCommand, settings.minOutputPeriod);
280  }
281

284  setupOutputObject(gpsVel, settings.gpsPosRate);
285  }
286 
287  if (settings.groundTruthEnabled){

293  setupOutputObject(accels, settings.attRawRate);

295  }
296

298  setupOutputObject(accels, settings.attRawRate);
299  setupOutputObject(gyros, settings.attRawRate);
300  }

309

312

508  static float gyro_correct_int2 = 0;

511 
512  AttitudeSettings::DataFields attitudeSettingsData = attitudeSettings->getData();
513  float accelKp = attitudeSettingsData.AccKp * 0.0001666666666666667;
514 
515  // calibrate sensors on arming
516  if (flightStatus->getData().Armed == FlightStatus::ARMED_ARMING) {
517  accelKp = 2.0;
518  }
519 
520  float gyro[3] = {out.rollRate, out.pitchRate, out.yawRate};
521  float attRawAcc[3] = {out.accX, out.accY, out.accZ};
522 
523  // code from Attitude module begin ///////////////////////////////
524  float *accels = attRawAcc;
525  float grot[3];
526  float accel_err[3];
527 
528  // XXX this is no longer accurate attitude code from the atti module.
529 
530  // Rotate gravity to body frame and cross with accels
531  grot[0] = -(2 * (q[1] * q[3] - q[0] * q[2]));
532  grot[1] = -(2 * (q[2] * q[3] + q[0] * q[1]));
533  grot[2] = -(q[0] * q[0] - q[1]*q[1] - q[2]*q[2] + q[3]*q[3]);
534 
535  // CrossProduct
536  {
537  accel_err[0] = accels[1]*grot[2] - grot[1]*accels[2];
538  accel_err[1] = grot[0]*accels[2] - accels[0]*grot[2];
539  accel_err[2] = accels[0]*grot[1] - grot[0]*accels[1];
540  }
541 
542  // Account for accel magnitude
543  float accel_mag = sqrt(accels[0] * accels[0] + accels[1] * accels[1] + accels[2] * accels[2]);
544  accel_err[0] /= accel_mag;
545  accel_err[1] /= accel_mag;
546  accel_err[2] /= accel_mag;
547 
548  // Correct rates based on error, integral component dealt with in updateSensors
549  gyro[0] += accel_err[0] * accelKp / dT;
550  gyro[1] += accel_err[1] * accelKp / dT;
551  gyro[2] += accel_err[2] * accelKp / dT + gyro_correct_int2;
552 
553  // Work out time derivative from INSAlgo writeup
554  // Also accounts for the fact that gyros are in deg/s
555  float qdot[4];
556  qdot[0] = (-q[1] * gyro[0] - q[2] * gyro[1] - q[3] * gyro[2]) * dT * M_PI / 180 / 2;
557  qdot[1] = (+q[0] * gyro[0] - q[3] * gyro[1] + q[2] * gyro[2]) * dT * M_PI / 180 / 2;
558  qdot[2] = (+q[3] * gyro[0] + q[0] * gyro[1] - q[1] * gyro[2]) * dT * M_PI / 180 / 2;
559  qdot[3] = (-q[2] * gyro[0] + q[1] * gyro[1] + q[0] * gyro[2]) * dT * M_PI / 180 / 2;
560 
561  // Take a time step
562  q[0] += qdot[0];
563  q[1] += qdot[1];
564  q[2] += qdot[2];
565  q[3] += qdot[3];
566 
567  if(q[0] < 0) {
568  q[0] = -q[0];
569  q[1] = -q[1];
570  q[2] = -q[2];
571  q[3] = -q[3];
572  }
573 
574  // Renomalize
575  float qmag = sqrt((q[0] * q[0]) + (q[1] * q[1]) + (q[2] * q[2]) + (q[3] * q[3]));
576  q[0] /= qmag;
577  q[1] /= qmag;
578  q[2] /= qmag;
579  q[3] /= qmag;
580 
581  // If quaternion has become inappropriately short or is nan reinit.
582  // THIS SHOULD NEVER ACTUALLY HAPPEN
583  if((fabs(qmag) < 1e-3) || (qmag != qmag)) {
584  q[0] = 1;
585  q[1] = 0;
586  q[2] = 0;
587  q[3] = 0;
588  }
589 
590  float rpy2[3];
591  // Quaternion2RPY
592  {
593  float q0s, q1s, q2s, q3s;
594  q0s = q[0] * q[0];
595  q1s = q[1] * q[1];
596  q2s = q[2] * q[2];
597  q3s = q[3] * q[3];
598 
599  float R13, R11, R12, R23, R33;
600  R13 = 2 * (q[1] * q[3] - q[0] * q[2]);
601  R11 = q0s + q1s - q2s - q3s;
602  R12 = 2 * (q[1] * q[2] + q[0] * q[3]);
603  R23 = 2 * (q[2] * q[3] + q[0] * q[1]);
604  R33 = q0s - q1s - q2s + q3s;
605 
606  rpy2[1] = RAD2DEG * asinf(-R13); // pitch always between -pi/2 to pi/2
607  rpy2[2] = RAD2DEG * atan2f(R12, R11);
608  rpy2[0] = RAD2DEG * atan2f(R23, R33);
609  }
610 
611  attActualData.Roll = rpy2[0];
612  attActualData.Pitch = rpy2[1];
613  attActualData.Yaw = rpy2[2];
614  attActualData.q1 = q[0];

622  }
