58 QColor StyleHelper::mergedColors(const QColor &colorA, const QColor &colorB, int factor)

265 void StyleHelper::drawArrow(QStyle::PrimitiveElement element, QPainter *painter, const QStyleOption *option)
266 {
267  // From windowsstyle but modified to enable AA
268  if (option->rect.width() <= 1 || option->rect.height() <= 1)
269  return;
270 
271  QRect r = option->rect;
272  int size = qMin(r.height(), r.width());
273  QPixmap pixmap;
274  QString pixmapName;
275  pixmapName.sprintf("arrow-%s-%d-%d-%d-%lld",
276  "$qt_ia",
277  uint(option->state), element,
278  size, option->palette.cacheKey());
279  if (!QPixmapCache::find(pixmapName, pixmap)) {
280  int border = size/5;
281  int sqsize = 2*(size/2);
282  QImage image(sqsize, sqsize, QImage::Format_ARGB32);
283  image.fill(Qt::transparent);
284  QPainter imagePainter(&image);
285  imagePainter.setRenderHint(QPainter::Antialiasing, true);
286  imagePainter.translate(0.5, 0.5);
287  QPolygon a;
288  switch (element) {
289  case QStyle::PE_IndicatorArrowUp:
290  a.setPoints(3, border, sqsize/2, sqsize/2, border, sqsize - border, sqsize/2);
291  break;
292  case QStyle::PE_IndicatorArrowDown:
293  a.setPoints(3, border, sqsize/2, sqsize/2, sqsize - border, sqsize - border, sqsize/2);
294  break;
295  case QStyle::PE_IndicatorArrowRight:
296  a.setPoints(3, sqsize - border, sqsize/2, sqsize/2, border, sqsize/2, sqsize - border);
297  break;
298  case QStyle::PE_IndicatorArrowLeft:
299  a.setPoints(3, border, sqsize/2, sqsize/2, border, sqsize/2, sqsize - border);
300  break;
301  default:
302  break;
303  }
304 
305  int bsx = 0;
306  int bsy = 0;
307 
308  if (option->state & QStyle::State_Sunken) {
309  bsx = qApp->style()->pixelMetric(QStyle::PM_ButtonShiftHorizontal);
310  bsy = qApp->style()->pixelMetric(QStyle::PM_ButtonShiftVertical);
311  }
312 
313  QRect bounds = a.boundingRect();
314  int sx = sqsize / 2 - bounds.center().x() - 1;
315  int sy = sqsize / 2 - bounds.center().y() - 1;
316  imagePainter.translate(sx + bsx, sy + bsy);
317 
318  if (!(option->state & QStyle::State_Enabled)) {
319  QColor foreGround(150, 150, 150, 150);
320  imagePainter.setBrush(option->palette.mid().color());
321  imagePainter.setPen(option->palette.mid().color());
322  } else {
323  QColor shadow(0, 0, 0, 100);
324  imagePainter.translate(0, 1);
325  imagePainter.setPen(shadow);
326  imagePainter.setBrush(shadow);
327  QColor foreGround(255, 255, 255, 210);
328  imagePainter.drawPolygon(a);
329  imagePainter.translate(0, -1);
330  imagePainter.setPen(foreGround);
331  imagePainter.setBrush(foreGround);
332  }
333  imagePainter.drawPolygon(a);
334  imagePainter.end();
335  pixmap = QPixmap::fromImage(image);
336  QPixmapCache::insert(pixmapName, pixmap);
337  }
338  int xOffset = r.x() + (r.width() - size)/2;
339  int yOffset = r.y() + (r.height() - size)/2;
340  painter->drawPixmap(xOffset, yOffset, pixmap);
341 }

377 // Draws a cached pixmap with shadow
378 void StyleHelper::drawIconWithShadow(const QIcon &icon, const QRect &rect,
379  QPainter *p, QIcon::Mode iconMode, int dipRadius, const QColor &color, const QPoint &dipOffset)
380 {
381  QPixmap cache;
382  QString pixmapName = QString::fromLatin1("icon %0 %1 %2").arg(icon.cacheKey()).arg(iconMode).arg(rect.height());
383 
384  if (!QPixmapCache::find(pixmapName, cache)) {
385  // High-dpi support: The in parameters (rect, radius, offset) are in
386  // device-independent pixels. The call to QIcon::pixmap() below might
387  // return a high-dpi pixmap, which will in that case have a devicePixelRatio
388  // different than 1. The shadow drawing caluculations are done in device
389  // pixels.
390  QPixmap px = icon.pixmap(rect.size());
391  int devicePixelRatio = qCeil(pixmapDevicePixelRatio(px));
392  int radius = dipRadius * devicePixelRatio;
393  QPoint offset = dipOffset * devicePixelRatio;
394  cache = QPixmap(px.size() + QSize(radius * 2, radius * 2));
395  cache.fill(Qt::transparent);
396 
397  QPainter cachePainter(&cache);
398  if (iconMode == QIcon::Disabled) {
399  QImage im = px.toImage().convertToFormat(QImage::Format_ARGB32);
400  for (int y=0; y<im.height(); ++y) {
401  QRgb *scanLine = (QRgb*)im.scanLine(y);
402  for (int x=0; x<im.width(); ++x) {
403  QRgb pixel = *scanLine;
404  char intensity = qGray(pixel);
405  *scanLine = qRgba(intensity, intensity, intensity, qAlpha(pixel));
406  ++scanLine;
407  }
408  }
409  px = QPixmap::fromImage(im);
410  }
411 
412  // Draw shadow
413  QImage tmp(px.size() + QSize(radius * 2, radius * 2 + 1), QImage::Format_ARGB32_Premultiplied);
414  tmp.fill(Qt::transparent);
415 
416  QPainter tmpPainter(&tmp);
417  tmpPainter.setCompositionMode(QPainter::CompositionMode_Source);
418  tmpPainter.drawPixmap(QRect(radius, radius, px.width(), px.height()), px);
419  tmpPainter.end();
420 
421  // blur the alpha channel
422  QImage blurred(tmp.size(), QImage::Format_ARGB32_Premultiplied);
423  blurred.fill(Qt::transparent);
424  QPainter blurPainter(&blurred);
425  qt_blurImage(&blurPainter, tmp, radius, false, true);
426  blurPainter.end();
427 
428  tmp = blurred;
429 
430  // blacken the image...
431  tmpPainter.begin(&tmp);
432  tmpPainter.setCompositionMode(QPainter::CompositionMode_SourceIn);
433  tmpPainter.fillRect(tmp.rect(), color);
434  tmpPainter.end();
435 
436  tmpPainter.begin(&tmp);
437  tmpPainter.setCompositionMode(QPainter::CompositionMode_SourceIn);
438  tmpPainter.fillRect(tmp.rect(), color);
439  tmpPainter.end();
440 
441  // draw the blurred drop shadow...
442  cachePainter.drawImage(QRect(0, 0, cache.rect().width(), cache.rect().height()), tmp);
443 
444  // Draw the actual pixmap...
445  cachePainter.drawPixmap(QRect(QPoint(radius, radius) + offset, QSize(px.width(), px.height())), px);
446 #if QT_VERSION > 0x050000
447  cache.setDevicePixelRatio(devicePixelRatio);
448 #endif
449  QPixmapCache::insert(pixmapName, cache);
450  }
451 
452  QRect targetRect = cache.rect();
453  targetRect.setSize(targetRect.size() / pixmapDevicePixelRatio(cache));
454  targetRect.moveCenter(rect.center() - dipOffset);
455  p->drawPixmap(targetRect, cache);
456 }

501 {
502  QPainter p(&img);
503  p.setCompositionMode(QPainter::CompositionMode_Screen);
504 
505  for (int x = 0; x < img.width(); ++x) {
506  for (int y = 0; y < img.height(); ++y) {

515  newColor.setAlpha(alpha);

517  }
