71 {
72  if (!widget)

76  if ((widget->window()->windowFlags() & Qt::WindowType_Mask) == Qt::Dialog)

364 void ManhattanStyle::drawPrimitive(PrimitiveElement element, const QStyleOption *option,
365  QPainter *painter, const QWidget *widget) const
366 {
367  if (!panelWidget(widget))
368  return QProxyStyle::drawPrimitive(element, option, painter, widget);
369 
370  bool animating = (option->state & State_Animating);
371  int state = option->state;
372  QRect rect = option->rect;
373  QRect oldRect;
374  QRect newRect;
375  if (widget && (element == PE_PanelButtonTool) && !animating) {
376  QWidget *w = const_cast<QWidget *> (widget);
377  int oldState = w->property("_q_stylestate").toInt();
378  oldRect = w->property("_q_stylerect").toRect();
379  newRect = w->rect();
380  w->setProperty("_q_stylestate", (int)option->state);
381  w->setProperty("_q_stylerect", w->rect());
382 
383  // Determine the animated transition
384  bool doTransition = ((state & State_On) != (oldState & State_On) ||
385  (state & State_MouseOver) != (oldState & State_MouseOver));
386  if (oldRect != newRect)
387  {
388  doTransition = false;
389  d->animator.stopAnimation(widget);
390  }
391 
392  if (doTransition) {
393  QImage startImage(option->rect.size(), QImage::Format_ARGB32_Premultiplied);
394  QImage endImage(option->rect.size(), QImage::Format_ARGB32_Premultiplied);
395  Animation *anim = d->animator.widgetAnimation(widget);
396  QStyleOption opt = *option;
397  opt.state = (QStyle::State)oldState;
398  opt.state |= State_Animating;
399  startImage.fill(0);
400  Transition *t = new Transition;
401  t->setWidget(w);
402  QPainter startPainter(&startImage);
403  if (!anim) {
404  drawPrimitive(element, &opt, &startPainter, widget);
405  } else {
406  anim->paint(&startPainter, &opt);
407  d->animator.stopAnimation(widget);
408  }
409  QStyleOption endOpt = *option;
410  endOpt.state |= State_Animating;
411  t->setStartImage(startImage);
412  d->animator.startAnimation(t);
413  endImage.fill(0);
414  QPainter endPainter(&endImage);
415  drawPrimitive(element, &endOpt, &endPainter, widget);
416  t->setEndImage(endImage);
417  if (oldState & State_MouseOver)
418  t->setDuration(150);
419  else
420  t->setDuration(75);
421  t->setStartTime(QTime::currentTime());
422  }
423  }
424 
425  switch (element) {
426  case PE_IndicatorDockWidgetResizeHandle:
427  painter->fillRect(option->rect, Utils::StyleHelper::borderColor());
428  break;
429  case PE_FrameDockWidget:
430  QCommonStyle::drawPrimitive(element, option, painter, widget);
431  break;
432  case PE_PanelLineEdit:
433  {
434  painter->save();
435 
436  // Fill the line edit background
437  QRect filledRect = option->rect.adjusted(1, 1, -1, -1);
438  painter->setBrushOrigin(filledRect.topLeft());
439  painter->fillRect(filledRect, option->palette.base());
440 
441  if (option->state & State_Enabled)
442  Utils::StyleHelper::drawCornerImage(d->lineeditImage, painter, option->rect, 5, 5, 5, 5);
443  else
444  Utils::StyleHelper::drawCornerImage(d->lineeditImage_disabled, painter, option->rect, 5, 5, 5, 5);
445 
446  if (option->state & State_HasFocus || option->state & State_MouseOver) {
447  QColor hover = Utils::StyleHelper::baseColor();
448  if (state & State_HasFocus)
449  hover.setAlpha(100);
450  else
451  hover.setAlpha(50);
452 
453  painter->setPen(QPen(hover, 1));
454  painter->drawRect(option->rect.adjusted(1, 1, -2 ,-2));
455  }
456  painter->restore();
457  }
458  break;
459 
460  case PE_FrameStatusBarItem:
461  break;
462 
463  case PE_PanelButtonTool: {
464  Animation *anim = d->animator.widgetAnimation(widget);
465  if (!animating && anim) {
466  anim->paint(painter, option);
467  } else {
468  bool pressed = option->state & State_Sunken || option->state & State_On;
469  QColor shadow(0, 0, 0, 30);
470  painter->setPen(shadow);
471  if (pressed) {
472  QColor shade(0, 0, 0, 40);
473  painter->fillRect(rect, shade);
474  painter->drawLine(rect.topLeft() + QPoint(1, 0), rect.topRight() - QPoint(1, 0));
475  painter->drawLine(rect.topLeft(), rect.bottomLeft());
476  painter->drawLine(rect.topRight(), rect.bottomRight());
477  // painter->drawLine(rect.bottomLeft() + QPoint(1, 0), rect.bottomRight() - QPoint(1, 0));
478  QColor highlight(255, 255, 255, 30);
479  painter->setPen(highlight);
480  }
481  else if (option->state & State_Enabled &&
482  option->state & State_MouseOver) {
483  QColor lighter(255, 255, 255, 37);
484  painter->fillRect(rect, lighter);
485  }
486  if (option->state & State_HasFocus && (option->state & State_KeyboardFocusChange)) {
487  QColor highlight = option->palette.highlight().color();
488  highlight.setAlphaF(0.4);
489  painter->setPen(QPen(highlight.lighter(), 1));
490  highlight.setAlphaF(0.3);
491  painter->setBrush(highlight);
492  painter->setRenderHint(QPainter::Antialiasing);
493  QRectF rect = option->rect;
494  rect.translate(0.5, 0.5);
495  painter->drawRoundedRect(rect.adjusted(2, 2, -3, -3), 2, 2);
496  }
497  }
498  }
499  break;
500 
501  case PE_PanelStatusBar:
502  {
503  painter->save();
504  QLinearGradient grad = Utils::StyleHelper::statusBarGradient(rect);
505  painter->fillRect(rect, grad);
506  painter->setPen(QColor(255, 255, 255, 60));
507  painter->drawLine(rect.topLeft() + QPoint(0,1),
508  rect.topRight()+ QPoint(0,1));
509  painter->setPen(Utils::StyleHelper::borderColor().darker(110));
510  painter->drawLine(rect.topLeft(), rect.topRight());
511  painter->restore();
512  }
513  break;
514 
515  case PE_IndicatorToolBarSeparator:
516  {
517  QColor separatorColor = Utils::StyleHelper::borderColor();
518  separatorColor.setAlpha(100);
519  painter->setPen(separatorColor);
520  const int margin = 6;
521  if (option->state & State_Horizontal) {
522  const int offset = rect.width()/2;
523  painter->drawLine(rect.bottomLeft().x() + offset,
524  rect.bottomLeft().y() - margin,
525  rect.topLeft().x() + offset,
526  rect.topLeft().y() + margin);
527  } else { //Draw vertical separator
528  const int offset = rect.height()/2;
529  painter->setPen(QPen(option->palette.background().color().darker(110)));
530  painter->drawLine(rect.topLeft().x() + margin ,
531  rect.topLeft().y() + offset,
532  rect.topRight().x() - margin,
533  rect.topRight().y() + offset);
534  }
535  }
536  break;
537 
538  case PE_IndicatorToolBarHandle:
539  {
540  bool horizontal = option->state & State_Horizontal;
541  painter->save();
542  QPainterPath path;
543  int x = option->rect.x() + (horizontal ? 2 : 6);
544  int y = option->rect.y() + (horizontal ? 6 : 2);
545  static const int RectHeight = 2;
546  if (horizontal) {
547  while (y < option->rect.height() - RectHeight - 6) {
548  path.moveTo(x, y);
549  path.addRect(x, y, RectHeight, RectHeight);
550  y += 6;
551  }
552  } else {
553  while (x < option->rect.width() - RectHeight - 6) {
554  path.moveTo(x, y);
555  path.addRect(x, y, RectHeight, RectHeight);
556  x += 6;
557  }
558  }
559 
560  painter->setPen(Qt::NoPen);
561  QColor dark = Utils::StyleHelper::borderColor();
562  dark.setAlphaF(0.4);
563 
564  QColor light = Utils::StyleHelper::baseColor();
565  light.setAlphaF(0.4);
566 
567  painter->fillPath(path, light);
568  painter->save();
569  painter->translate(1, 1);
570  painter->fillPath(path, dark);
571  painter->restore();
572  painter->translate(3, 3);
573  painter->fillPath(path, light);
574  painter->translate(1, 1);
575  painter->fillPath(path, dark);
576  painter->restore();
577  }
578  break;
579  case PE_IndicatorArrowUp:
580  case PE_IndicatorArrowDown:
581  case PE_IndicatorArrowRight:
582  case PE_IndicatorArrowLeft:
583  {
584  Utils::StyleHelper::drawArrow(element, painter, option);
585  }
586  break;
587 
588  default:
589  QProxyStyle::drawPrimitive(element, option, painter, widget);
590  break;
591  }

594 void ManhattanStyle::drawControl(ControlElement element, const QStyleOption *option,
595  QPainter *painter, const QWidget *widget) const
596 {
597  if (!panelWidget(widget))
598  return QProxyStyle::drawControl(element, option, painter, widget);
599 
600  switch (element) {
601  case CE_Splitter:
602  painter->fillRect(option->rect, Utils::StyleHelper::borderColor());
603  break;
604 
605  case CE_TabBarTabShape:
606  // Most styles draw a single dark outline. This looks rather ugly when combined with our
607  // single pixel dark separator so we adjust the first tab to compensate for this
608 
609  if (const QStyleOptionTab *tab = qstyleoption_cast<const QStyleOptionTab *>(option)) {
610  QStyleOptionTab adjustedTab = *tab;
611  if (tab->cornerWidgets == QStyleOptionTab::NoCornerWidgets && (
612  tab->position == QStyleOptionTab::Beginning ||
613  tab->position == QStyleOptionTab::OnlyOneTab))
614  {
615  if (option->direction == Qt::LeftToRight)
616  adjustedTab.rect = adjustedTab.rect.adjusted(-1, 0, 0, 0);
617  else
618  adjustedTab.rect = adjustedTab.rect.adjusted(0, 0, 1 ,0);
619  }
620  QProxyStyle::drawControl(element, &adjustedTab, painter, widget);
621  return;
622  }
623  break;
624 
625  case CE_MenuBarItem:
626  painter->save();
627  if (const QStyleOptionMenuItem *mbi = qstyleoption_cast<const QStyleOptionMenuItem *>(option)) {
628  QColor highlightOutline = Utils::StyleHelper::borderColor().lighter(120);
629  bool act = mbi->state & State_Sunken;
630  bool dis = !(mbi->state & State_Enabled);
631  Utils::StyleHelper::menuGradient(painter, option->rect, option->rect);
632  QStyleOptionMenuItem item = *mbi;
633  item.rect = mbi->rect;
634  QPalette pal = mbi->palette;
635  pal.setBrush(QPalette::ButtonText, dis ? Qt::gray : Qt::black);
636  item.palette = pal;
637  QCommonStyle::drawControl(element, &item, painter, widget);
638  QRect r = option->rect;
639 
640  if (act) {
641  // Fill|
642  QColor baseColor = Utils::StyleHelper::baseColor();
643  QLinearGradient grad(option->rect.topLeft(), option->rect.bottomLeft());
644  grad.setColorAt(0, baseColor.lighter(120));
645  grad.setColorAt(1, baseColor.lighter(130));
646  painter->fillRect(option->rect.adjusted(1, 1, -1, 0), grad);
647 
648  // Outline
649  painter->setPen(QPen(highlightOutline, 0));
650  painter->drawLine(QPoint(r.left(), r.top() + 1), QPoint(r.left(), r.bottom()));
