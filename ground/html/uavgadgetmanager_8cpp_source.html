396 // Removes all gadgets and splits in the workspace, except the current/active gadget.
397 void UAVGadgetManager::removeAllSplits()
398 {
399  if (m_core->modeManager()->currentMode() != this)
400  return;
401 
402  if (!m_splitterOrView->isSplitter())
403  return;
404 
405  // Use a QPointer, just in case we accidently delete the gadget we want to keep.
406  QPointer<IUAVGadget> currentGadget = m_currentGadget;
407 
408  Q_ASSERT(currentGadget);
409  QList<IUAVGadget *> gadgets = m_splitterOrView->gadgets();
410  Q_ASSERT(gadgets.count(currentGadget) == 1);
411  gadgets.removeOne(currentGadget);
412 
413  // Remove all splits and their gadgets, then create a new view with the current gadget.
414  m_splitterOrView->unsplitAll(currentGadget);
415 
416  // Zeroing the current gadget means setCurrentGadget will do something when we call it.
417  m_currentGadget = nullptr;
418  setCurrentGadget(currentGadget);
419 
420  // Remove all other gadgets from the instance manager.
421  UAVGadgetInstanceManager *im = ICore::instance()->uavGadgetInstanceManager();
422  foreach (IUAVGadget *g, gadgets) {
423  im->removeGadget(g);
424  }
425  updateUavGadgetMenus();
426 }
