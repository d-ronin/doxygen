219 void DialGadgetOptionsPage::uavoChanged(QComboBox *widget, const QString &uavo)
220 {
221  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
222  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
223 
224  UAVDataObject *obj = dynamic_cast<UAVDataObject *>(objManager->getObject(uavo));
225  if (!obj) {
226  qWarning() << "Invalid object" << uavo;
227  return;
228  }
229 
230  widget->clear();
231  for (auto field : obj->getFields()) {
232  if (field->getType() == UAVObjectField::STRING || field->getType() == UAVObjectField::ENUM)
233  continue;
234  if (field->getElementNames().count() > 1) {
235  for (const QString &elemName : field->getElementNames())
236  widget->addItem(field->getName() + "-" + elemName);
237  } else
238  widget->addItem(field->getName());
239  }
240 }
