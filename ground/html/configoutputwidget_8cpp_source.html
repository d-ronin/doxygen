335 void ConfigOutputWidget::startESCCalibration()
336 {
337  bool channelsToCalibrate[ActuatorCommand::CHANNEL_NUMELEM];
338  if (!showOutputChannelSelectWindow(channelsToCalibrate))
339  return;
340 
341  QMessageBox mbox;
342  mbox.setText(QString(tr("Starting ESC calibration.<br/><b>Please remove all propellers and "
343  "disconnect battery from ESCs.</b>")));
344  mbox.setStandardButtons(QMessageBox::Ok | QMessageBox::Cancel);
345  mbox.setDefaultButton(QMessageBox::Cancel);
346 
347  if (mbox.exec() != QMessageBox::Ok)
348  return;
349 
350  // Get access to actuator command (for setting actual value)
351  ActuatorCommand *actuatorCommand = ActuatorCommand::GetInstance(getObjectManager());
352  Q_ASSERT(actuatorCommand);
353  ActuatorCommand::DataFields actuatorCommandFields = actuatorCommand->getData();
354  UAVObject::Metadata mdata = actuatorCommand->getMetadata();
355  // Get access to actuator settings (for min / max values)
356  ActuatorSettings *actuatorSettings = ActuatorSettings::GetInstance(getObjectManager());
357  Q_ASSERT(actuatorSettings);
358  ActuatorSettings::DataFields actuatorSettingsData = actuatorSettings->getData();
359 
360  // Save previous metadata
361  accInitialData = mdata;
362 
363  // Change settings for period of calibration
364  UAVObject::SetFlightAccess(mdata, UAVObject::ACCESS_READONLY);
365  UAVObject::SetFlightTelemetryUpdateMode(mdata, UAVObject::UPDATEMODE_ONCHANGE);
366  UAVObject::SetGcsTelemetryAcked(mdata, false);
367  UAVObject::SetGcsTelemetryUpdateMode(mdata, UAVObject::UPDATEMODE_ONCHANGE);
368  mdata.gcsTelemetryUpdatePeriod = 100;
369  actuatorCommand->setMetadata(mdata);
370  actuatorCommand->updated();
371 
372  mbox.setText(QString(tr("Please wait...")));
373 
374  // Set output for motors to nil to disarm
375  // Should not be necessary; battery is supposed to be disconnected, etc.
376  // But this makes it moderately safer if the user does not obey.
377  for (unsigned int i = 0; i < ActuatorCommand::CHANNEL_NUMELEM; i++) {
378  // Check if the output channel was selected
379  if (!channelsToCalibrate[i])
380  continue;
381 
382  actuatorCommandFields.Channel[i] = 0;
383  }
384 
385  actuatorCommand->setData(actuatorCommandFields);
386 
387  QEventLoop loop;
388 
389  QTimer timeout;
390  timeout.setSingleShot(true);
391  connect(&timeout, &QTimer::timeout, &loop, &QEventLoop::quit);
392  timeout.start(1250); // 1.25 seconds
393  loop.exec();
394 
395  // Increase output for all motors
396  for (unsigned int i = 0; i < ActuatorCommand::CHANNEL_NUMELEM; i++) {
397  // Check if the output channel was selected
398  if (!channelsToCalibrate[i])
399  continue;
400 
401  actuatorCommandFields.Channel[i] = actuatorSettingsData.ChannelMax[i];
402  }
403 
404  actuatorCommand->setData(actuatorCommandFields);
405 
406  mbox.setText(QString(tr("Motors outputs were increased to maximum. "
407  "Reconnect the battery and wait for notification from ESCs that they "
408  "recognized high throttle position.<br/>"
409  "<b>Immediately after that proceed to next step.</b>")));
410  if (mbox.exec() == QMessageBox::Ok) {
411  // Decrease output for all motors
412  for (unsigned int i = 0; i < ActuatorCommand::CHANNEL_NUMELEM; i++) {
413  // Check if the output channel was selected
414  if (!channelsToCalibrate[i])
415  continue;
416 
417  // calibrate the minimum of the range
418  // to be a little off what we send at idle. Blheli takes the value
419  // we give verbatim as the minimum, and small amounts of jitter
420  // can cause the motor to tick over.
421  double range = actuatorSettingsData.ChannelMax[i] - actuatorSettingsData.ChannelMin[i];
422  actuatorCommandFields.Channel[i] = actuatorSettingsData.ChannelMin[i]
423  + std::copysign(std::min(5.0, std::abs(range) / 40.0 + 0.5), range);
424  }
425  actuatorCommand->setData(actuatorCommandFields);
426 
427  mbox.setStandardButtons(QMessageBox::Ok);
428  mbox.setDefaultButton(QMessageBox::Ok);
429  mbox.setText(QString(tr("Motors outputs were decreased to minimum.<br/>Wait for "
430  "notification from ESCs that calibration is finished.")));
431  mbox.exec();
432  }
433 
434  // Restore metadata
435  actuatorCommand->setMetadata(accInitialData);
436  actuatorCommand->updated();
437 }

510 void ConfigOutputWidget::refreshWidgetRanges()
511 {
512  QList<OutputChannelForm *> outputChannelForms = findChildren<OutputChannelForm *>();
513 
514  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
515  Q_ASSERT(pm);
516  UAVObjectUtilManager *utilMngr = pm->getObject<UAVObjectUtilManager>();
517 
518  if (utilMngr) {
519  Core::IBoardType *board = utilMngr->getBoardType();
520  if (board != NULL) {
521  QVector<QVector<qint32>> banks = board->getChannelBanks();
522 
523  for (int i = 0; i < banks.size(); i++) {
524  QVector<int> bank = banks[i];
525 
526  // Pull out the parameters relating to this bank, and
527  // calculate the maximum value.
528  //
529  QComboBox *ccmb = rateList.at(i);
530  quint32 timerFreq = timerStringToFreq(ccmb->currentText());
531 
532  // First saturate at the timer period and UAVO maximum value.
533  double maxPulseWidth, timerPeriodUs = UINT16_MAX;
534  int minPulseWidth = 0;
535  bool digital = false;
536 
537  switch (timerFreq) {
538  case RATE_SYNCPWM:
539  // only the timer period provides bounding
540  maxPulseWidth = timerPeriodUs;
541  break;
542  case RATE_DSHOT300:
543  case RATE_DSHOT600:
544  case RATE_DSHOT1200:
545  // 1-47 reserved for commands
546  minPulseWidth = 48;
547  // 11-bit value
548  maxPulseWidth = 2047;
549  // don't allow min/max to be changed
550  digital = true;
551  break;
552  default:
553  maxPulseWidth = 1000000 / timerFreq;
554 
555  // Should never happen, but here in case we ever have a
556  // < 16Hz PWM mode.
557  if (maxPulseWidth > timerPeriodUs)
558  maxPulseWidth = timerPeriodUs;
559  break;
560  }
561 
562  // For each of the channels...
563  for (auto channel : bank) {
564  // Find its form line and set the limit.
565  for (auto outputChannelForm : outputChannelForms) {
566  // Don't like the bogus index adjustments
567  if (outputChannelForm->index() != (channel - 1))
568  continue;
569  outputChannelForm->updateChannelLimits(
570  minPulseWidth, static_cast<int>(maxPulseWidth), digital);
571  }
572  }
573  }
574  }
575  }
